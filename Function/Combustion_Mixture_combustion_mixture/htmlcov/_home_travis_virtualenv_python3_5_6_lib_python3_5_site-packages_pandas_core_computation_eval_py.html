169def eval( 

170 expr, 

171 parser="pandas", 

172 engine=None, 

173 truediv=True, 

174 local_dict=None, 

175 global_dict=None, 

176 resolvers=(), 

177 level=0, 

178 target=None, 

179 inplace=False, 

180): 

181 """ 

182 Evaluate a Python expression as a string using various backends. 

183 

184 The following arithmetic operations are supported: ``+``, ``-``, ``*``, 

185 ``/``, ``**``, ``%``, ``//`` (python engine only) along with the following 

186 boolean operations: ``|`` (or), ``&`` (and), and ``~`` (not). 

187 Additionally, the ``'pandas'`` parser allows the use of :keyword:`and`, 

188 :keyword:`or`, and :keyword:`not` with the same semantics as the 

189 corresponding bitwise operators. :class:`~pandas.Series` and 

190 :class:`~pandas.DataFrame` objects are supported and behave as they would 

191 with plain ol' Python evaluation. 

192 

193 Parameters 

194 ---------- 

195 expr : str or unicode 

196 The expression to evaluate. This string cannot contain any Python 

197 `statements 

198 <https://docs.python.org/3/reference/simple_stmts.html#simple-statements>`__, 

199 only Python `expressions 

200 <https://docs.python.org/3/reference/simple_stmts.html#expression-statements>`__. 

201 parser : string, default 'pandas', {'pandas', 'python'} 

202 The parser to use to construct the syntax tree from the expression. The 

203 default of ``'pandas'`` parses code slightly different than standard 

204 Python. Alternatively, you can parse an expression using the 

205 ``'python'`` parser to retain strict Python semantics. See the 

206 :ref:`enhancing performance <enhancingperf.eval>` documentation for 

207 more details. 

208 engine : string or None, default 'numexpr', {'python', 'numexpr'} 

209 

210 The engine used to evaluate the expression. Supported engines are 

211 

212 - None : tries to use ``numexpr``, falls back to ``python`` 

213 - ``'numexpr'``: This default engine evaluates pandas objects using 

214 numexpr for large speed ups in complex expressions 

215 with large frames. 

216 - ``'python'``: Performs operations as if you had ``eval``'d in top 

217 level python. This engine is generally not that useful. 

218 

219 More backends may be available in the future. 

220 

221 truediv : bool, optional 

222 Whether to use true division, like in Python >= 3 

223 local_dict : dict or None, optional 

224 A dictionary of local variables, taken from locals() by default. 

225 global_dict : dict or None, optional 

226 A dictionary of global variables, taken from globals() by default. 

227 resolvers : list of dict-like or None, optional 

228 A list of objects implementing the ``__getitem__`` special method that 

229 you can use to inject an additional collection of namespaces to use for 

230 variable lookup. For example, this is used in the 

231 :meth:`~DataFrame.query` method to inject the 

232 ``DataFrame.index`` and ``DataFrame.columns`` 

233 variables that refer to their respective :class:`~pandas.DataFrame` 

234 instance attributes. 

235 level : int, optional 

236 The number of prior stack frames to traverse and add to the current 

237 scope. Most users will **not** need to change this parameter. 

238 target : object, optional, default None 

239 This is the target object for assignment. It is used when there is 

240 variable assignment in the expression. If so, then `target` must 

241 support item assignment with string keys, and if a copy is being 

242 returned, it must also support `.copy()`. 

243 inplace : bool, default False 

244 If `target` is provided, and the expression mutates `target`, whether 

245 to modify `target` inplace. Otherwise, return a copy of `target` with 

246 the mutation. 

247 

248 Returns 

249 ------- 

250 ndarray, numeric scalar, DataFrame, Series 

251 

252 Raises 

253 ------ 

254 ValueError 

255 There are many instances where such an error can be raised: 

256 

257 - `target=None`, but the expression is multiline. 

258 - The expression is multiline, but not all them have item assignment. 

259 An example of such an arrangement is this: 

260 

261 a = b + 1 

262 a + 2 

263 

264 Here, there are expressions on different lines, making it multiline, 

265 but the last line has no variable assigned to the output of `a + 2`. 

266 - `inplace=True`, but the expression is missing item assignment. 

267 - Item assignment is provided, but the `target` does not support 

268 string item assignment. 

269 - Item assignment is provided and `inplace=False`, but the `target` 

270 does not support the `.copy()` method 

271 

272 See Also 

273 -------- 

274 DataFrame.query 

275 DataFrame.eval 

276 

277 Notes 

278 ----- 

279 The ``dtype`` of any objects involved in an arithmetic ``%`` operation are 

280 recursively cast to ``float64``. 

281 

282 See the :ref:`enhancing performance <enhancingperf.eval>` documentation for 

283 more details. 

284 """ 

285 from pandas.core.computation.expr import Expr 

286 

287 inplace = validate_bool_kwarg(inplace, "inplace") 

288 

289 if isinstance(expr, str): 

290 _check_expression(expr) 

291 exprs = [e.strip() for e in expr.splitlines() if e.strip() != ""] 

292 else: 

293 exprs = [expr] 

294 multi_line = len(exprs) > 1 

295 

296 if multi_line and target is None: 

297 raise ValueError( 

298 "multi-line expressions are only valid in the " 

299 "context of data, use DataFrame.eval" 

300 ) 

301 

302 ret = None 

303 first_expr = True 

304 target_modified = False 

305 

306 for expr in exprs: 

307 expr = _convert_expression(expr) 

308 engine = _check_engine(engine) 

309 _check_parser(parser) 

310 _check_resolvers(resolvers) 

311 _check_for_locals(expr, level, parser) 

312 

313 # get our (possibly passed-in) scope

316 global_dict=global_dict,

318 resolvers=resolvers,

325 eng = _engines[engine] 

326 eng_inst = eng(parsed_expr)

330 if multi_line:

336 raise ValueError("Cannot operate inplace " "if there is no assignment")

347 except AttributeError:

359 target[assigner] = ret 

360 except (TypeError, IndexError): 

361 raise ValueError("Cannot assign expression output to target") 

362 

363 if not resolvers: 

364 resolvers = ({assigner: ret},) 

365 else: 

366 # existing resolver needs updated to handle 

367 # case of mutating existing column in copy 

368 for resolver in resolvers: 

369 if assigner in resolver: 

370 resolver[assigner] = ret 

371 break 

372 else: 

373 resolvers += ({assigner: ret},) 

374 

375 ret = None 

376 first_expr = False 

377 

378 # We want to exclude `inplace=None` as being False. 

379 if inplace is False: 

380 return target if target_modified else ret
