740def cycler(*args, **kwargs): 

741 """ 

742 Creates a `~cycler.Cycler` object much like :func:`cycler.cycler`, 

743 but includes input validation. 

744 

745 Call signatures:: 

746 

747 cycler(cycler) 

748 cycler(label=values[, label2=values2[, ...]]) 

749 cycler(label, values) 

750 

751 Form 1 copies a given `~cycler.Cycler` object. 

752 

753 Form 2 creates a `~cycler.Cycler` which cycles over one or more 

754 properties simultaneously. If multiple properties are given, their 

755 value lists must have the same length. 

756 

757 Form 3 creates a `~cycler.Cycler` for a single property. This form 

758 exists for compatibility with the original cycler. Its use is 

759 discouraged in favor of the kwarg form, i.e. ``cycler(label=values)``. 

760 

761 Parameters 

762 ---------- 

763 cycler : Cycler 

764 Copy constructor for Cycler. 

765 

766 label : str 

767 The property key. Must be a valid `.Artist` property. 

768 For example, 'color' or 'linestyle'. Aliases are allowed, 

769 such as 'c' for 'color' and 'lw' for 'linewidth'. 

770 

771 values : iterable 

772 Finite-length iterable of the property values. These values 

773 are validated and will raise a ValueError if invalid. 

774 

775 Returns 

776 ------- 

777 cycler : Cycler 

778 A new :class:`~cycler.Cycler` for the given properties. 

779 

780 Examples 

781 -------- 

782 Creating a cycler for a single property: 

783 

784 >>> c = cycler(color=['red', 'green', 'blue']) 

785 

786 Creating a cycler for simultaneously cycling over multiple properties 

787 (e.g. red circle, green plus, blue cross): 

788 

789 >>> c = cycler(color=['red', 'green', 'blue'], 

790 ... marker=['o', '+', 'x']) 

791 

792 """ 

793 if args and kwargs: 

794 raise TypeError("cycler() can only accept positional OR keyword " 

795 "arguments -- not both.") 

796 elif not args and not kwargs: 

797 raise TypeError("cycler() must have positional OR keyword arguments") 

798 

799 if len(args) == 1: 

800 if not isinstance(args[0], Cycler): 

801 raise TypeError("If only one positional argument given, it must " 

802 " be a Cycler instance.") 

803 return validate_cycler(args[0]) 

804 elif len(args) == 2: 

805 pairs = [(args[0], args[1])] 

806 elif len(args) > 2: 

807 raise TypeError("No more than 2 positional arguments allowed") 

808 else: 

809 pairs = kwargs.items() 

810 

811 validated = [] 

812 for prop, vals in pairs: 

813 norm_prop = _prop_aliases.get(prop, prop) 

814 validator = _prop_validators.get(norm_prop, None) 

815 if validator is None: 

816 raise TypeError("Unknown artist property: %s" % prop) 

817 vals = validator(vals) 

818 # We will normalize the property names as well to reduce 

819 # the amount of alias handling code elsewhere. 

820 validated.append((norm_prop, vals)) 

821 

822 return reduce(operator.add, (ccycler(k, v) for k, v in validated))

825def validate_cycler(s): 

826 'return a Cycler object from a string repr or the object itself' 

827 if isinstance(s, str): 

828 try: 

829 # TODO: We might want to rethink this... 

830 # While I think I have it quite locked down, 

831 # it is execution of arbitrary code without 

832 # sanitation. 

833 # Combine this with the possibility that rcparams 

834 # might come from the internet (future plans), this 

835 # could be downright dangerous. 

836 # I locked it down by only having the 'cycler()' function 

837 # available. 

838 # UPDATE: Partly plugging a security hole. 

839 # I really should have read this: 

840 # http://nedbatchelder.com/blog/201206/eval_really_is_dangerous.html 

841 # We should replace this eval with a combo of PyParsing and 

842 # ast.literal_eval() 

843 if '.__' in s.replace(' ', ''): 

844 raise ValueError("'%s' seems to have dunder methods. Raising" 

845 " an exception for your safety") 

846 s = eval(s, {'cycler': cycler, '__builtins__': {}}) 

847 except BaseException as e: 

848 raise ValueError("'%s' is not a valid cycler construction: %s" % 

849 (s, e)) 

850 # Should make sure what comes from the above eval() 

851 # is a Cycler object. 

852 if isinstance(s, Cycler): 

853 cycler_inst = s 

854 else: 

855 raise ValueError("object was not a string or Cycler instance: %s" % s) 

856 

857 unknowns = cycler_inst.keys - (set(_prop_validators) | set(_prop_aliases)) 

858 if unknowns: 

859 raise ValueError("Unknown artist properties: %s" % unknowns) 

860 

861 # Not a full validation, but it'll at least normalize property names 

862 # A fuller validation would require v0.10 of cycler. 

863 checker = set() 

864 for prop in cycler_inst.keys: 

865 norm_prop = _prop_aliases.get(prop, prop) 

866 if norm_prop != prop and norm_prop in cycler_inst.keys: 

867 raise ValueError("Cannot specify both '{0}' and alias '{1}'" 

868 " in the same prop_cycle".format(norm_prop, prop)) 

869 if norm_prop in checker: 

870 raise ValueError("Another property was already aliased to '{0}'." 

871 " Collision normalizing '{1}'.".format(norm_prop, 

872 prop)) 

873 checker.update([norm_prop]) 

874 

875 # This is just an extra-careful check, just in case there is some 

876 # edge-case I haven't thought of. 

877 assert len(checker) == len(cycler_inst.keys) 

878 

879 # Now, it should be safe to mutate this cycler 

880 for prop in cycler_inst.keys: 

881 norm_prop = _prop_aliases.get(prop, prop) 

882 cycler_inst.change_key(prop, norm_prop) 

883 

884 for key, vals in cycler_inst.by_key().items(): 

885 _prop_validators[key](vals) 

886 

887 return cycler_inst
