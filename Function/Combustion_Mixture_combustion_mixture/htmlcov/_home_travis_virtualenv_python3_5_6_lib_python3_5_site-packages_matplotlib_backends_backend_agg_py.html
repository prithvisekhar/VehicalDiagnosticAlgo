68 # we want to cache the fonts at the class level so that when 

69 # multiple figures are created we can reuse them. This helps with 

70 # a bug on windows where the creation of too many figures leads to 

71 # too many open file handles. However, storing them at the class 

72 # level is not thread safe. The solution here is to let the 

73 # FigureCanvas acquire a lock on the fontd at the start of the 

74 # draw, and release it when it is done. This allows multiple 

75 # renderers to share the cached fonts, but only one figure can 

76 # draw at time and so the font cache is used by only one 

77 # renderer at a time.

336 def stop_filter(self, post_processing): 

337 """ 

338 Save the plot in the current canvas as a image and apply 

339 the *post_processing* function. 

340 

341 def post_processing(image, dpi): 

342 # ny, nx, depth = image.shape 

343 # image (numpy array) has RGBA channels and has a depth of 4. 

344 ... 

345 # create a new_image (numpy array of 4 channels, size can be 

346 # different). The resulting image may have offsets from 

347 # lower-left corner of the original image 

348 return new_image, offset_x, offset_y 

349 

350 The saved renderer is restored and the returned image from 

351 post_processing is plotted (using draw_image) on it. 

352 """ 

353 

354 width, height = int(self.width), int(self.height) 

355 

356 buffer, (l, b, w, h) = self.tostring_rgba_minimized() 

357 

358 self._renderer = self._filter_renderers.pop() 

359 self._update_methods() 

360 

361 if w > 0 and h > 0: 

362 img = np.fromstring(buffer, np.uint8) 

363 img, ox, oy = post_processing(img.reshape((h, w, 4)) / 255., 

364 self.dpi) 

365 gc = self.new_gc() 

366 if img.dtype.kind == 'f': 

367 img = np.asarray(img * 255., np.uint8) 

368 img = img[::-1] 

369 self._renderer.draw_image(gc, l + ox, height - b - h + oy, img)

409 def get_renderer(self, cleared=False): 

410 l, b, w, h = self.figure.bbox.bounds 

411 key = w, h, self.figure.dpi 

412 try: self._lastKey, self.renderer 

413 except AttributeError: need_new_renderer = True 

414 else: need_new_renderer = (self._lastKey != key) 

415 

416 if need_new_renderer: 

417 self.renderer = RendererAgg(w, h, self.figure.dpi) 

418 self._lastKey = key 

419 elif cleared: 

420 self.renderer.clear()
