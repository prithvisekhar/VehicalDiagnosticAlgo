12def _check_is_partition(parts, whole):

14 parts = [set(x) for x in parts]

89def _sparse_series_to_coo(ss, row_levels=(0,), column_levels=(1,), sort_labels=False): 

90 """ 

91 Convert a SparseSeries to a scipy.sparse.coo_matrix using index 

92 levels row_levels, column_levels as the row and column 

93 labels respectively. Returns the sparse_matrix, row and column labels. 

94 """ 

95 

96 import scipy.sparse 

97 

98 if ss.index.nlevels < 2: 

99 raise ValueError("to_coo requires MultiIndex with nlevels > 2") 

100 if not ss.index.is_unique: 

101 raise ValueError( 

102 "Duplicate index entries are not allowed in to_coo " "transformation." 

103 ) 

104 

105 # to keep things simple, only rely on integer indexing (not labels) 

106 row_levels = [ss.index._get_level_number(x) for x in row_levels] 

107 column_levels = [ss.index._get_level_number(x) for x in column_levels] 

108 

109 v, i, j, rows, columns = _to_ijv( 

110 ss, row_levels=row_levels, column_levels=column_levels, sort_labels=sort_labels 

111 ) 

112 sparse_matrix = scipy.sparse.coo_matrix( 

113 (v, (i, j)), shape=(len(rows), len(columns)) 

114 ) 

115 return sparse_matrix, rows, columns
