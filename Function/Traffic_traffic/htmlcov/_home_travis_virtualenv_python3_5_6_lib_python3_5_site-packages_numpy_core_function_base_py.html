25def linspace(start, stop, num=50, endpoint=True, retstep=False, dtype=None): 

26 """ 

27 Return evenly spaced numbers over a specified interval. 

28 

29 Returns `num` evenly spaced samples, calculated over the 

30 interval [`start`, `stop`]. 

31 

32 The endpoint of the interval can optionally be excluded. 

33 

34 Parameters 

35 ---------- 

36 start : scalar 

37 The starting value of the sequence. 

38 stop : scalar 

39 The end value of the sequence, unless `endpoint` is set to False. 

40 In that case, the sequence consists of all but the last of ``num + 1`` 

41 evenly spaced samples, so that `stop` is excluded. Note that the step 

42 size changes when `endpoint` is False. 

43 num : int, optional 

44 Number of samples to generate. Default is 50. Must be non-negative. 

45 endpoint : bool, optional 

46 If True, `stop` is the last sample. Otherwise, it is not included. 

47 Default is True. 

48 retstep : bool, optional 

49 If True, return (`samples`, `step`), where `step` is the spacing 

50 between samples. 

51 dtype : dtype, optional 

52 The type of the output array. If `dtype` is not given, infer the data 

53 type from the other input arguments. 

54 

55 .. versionadded:: 1.9.0 

56 

57 Returns 

58 ------- 

59 samples : ndarray 

60 There are `num` equally spaced samples in the closed interval 

61 ``[start, stop]`` or the half-open interval ``[start, stop)`` 

62 (depending on whether `endpoint` is True or False). 

63 step : float, optional 

64 Only returned if `retstep` is True 

65 

66 Size of spacing between samples. 

67 

68 

69 See Also 

70 -------- 

71 arange : Similar to `linspace`, but uses a step size (instead of the 

72 number of samples). 

73 logspace : Samples uniformly distributed in log space. 

74 

75 Examples 

76 -------- 

77 >>> np.linspace(2.0, 3.0, num=5) 

78 array([ 2. , 2.25, 2.5 , 2.75, 3. ]) 

79 >>> np.linspace(2.0, 3.0, num=5, endpoint=False) 

80 array([ 2. , 2.2, 2.4, 2.6, 2.8]) 

81 >>> np.linspace(2.0, 3.0, num=5, retstep=True) 

82 (array([ 2. , 2.25, 2.5 , 2.75, 3. ]), 0.25) 

83 

84 Graphical illustration: 

85 

86 >>> import matplotlib.pyplot as plt 

87 >>> N = 8 

88 >>> y = np.zeros(N) 

89 >>> x1 = np.linspace(0, 10, N, endpoint=True) 

90 >>> x2 = np.linspace(0, 10, N, endpoint=False) 

91 >>> plt.plot(x1, y, 'o') 

92 [<matplotlib.lines.Line2D object at 0x...>] 

93 >>> plt.plot(x2, y + 0.5, 'o') 

94 [<matplotlib.lines.Line2D object at 0x...>] 

95 >>> plt.ylim([-0.5, 1]) 

96 (-0.5, 1) 

97 >>> plt.show() 

98 

99 """ 

100 # 2016-02-25, 1.12 

101 num = _index_deprecate(num) 

102 if num < 0: 

103 raise ValueError("Number of samples, %s, must be non-negative." % num) 

104 div = (num - 1) if endpoint else num 

105 

106 # Convert float/complex array scalars to float, gh-3504 

107 # and make sure one can use variables that have an __array_interface__, gh-6634 

108 start = asanyarray(start) * 1.0 

109 stop = asanyarray(stop) * 1.0 

110 

111 dt = result_type(start, stop, float(num)) 

112 if dtype is None: 

113 dtype = dt 

114 

115 y = _nx.arange(0, num, dtype=dt) 

116 

117 delta = stop - start 

118 # In-place multiplication y *= delta/div is faster, but prevents the multiplicant 

119 # from overriding what class is produced, and thus prevents, e.g. use of Quantities, 

120 # see gh-7142. Hence, we multiply in place only for standard scalar types. 

121 _mult_inplace = _nx.isscalar(delta) 

122 if num > 1: 

123 step = delta / div 

124 if step == 0: 

125 # Special handling for denormal numbers, gh-5437 

126 y /= div 

127 if _mult_inplace: 

128 y *= delta 

129 else: 

130 y = y * delta 

131 else: 

132 if _mult_inplace: 

133 y *= step 

134 else: 

135 y = y * step 

136 else: 

137 # 0 and 1 item long sequences have an undefined step 

138 step = NaN 

139 # Multiply with delta to allow possible override of output class. 

140 y = y * delta 

141 

142 y += start 

143 

144 if endpoint and num > 1: 

145 y[-1] = stop 

146 

147 if retstep: 

148 return y.astype(dtype, copy=False), step 

149 else: 

150 return y.astype(dtype, copy=False)

153def logspace(start, stop, num=50, endpoint=True, base=10.0, dtype=None): 

154 """ 

155 Return numbers spaced evenly on a log scale. 

156 

157 In linear space, the sequence starts at ``base ** start`` 

158 (`base` to the power of `start`) and ends with ``base ** stop`` 

159 (see `endpoint` below). 

160 

161 Parameters 

162 ---------- 

163 start : float 

164 ``base ** start`` is the starting value of the sequence. 

165 stop : float 

166 ``base ** stop`` is the final value of the sequence, unless `endpoint` 

167 is False. In that case, ``num + 1`` values are spaced over the 

168 interval in log-space, of which all but the last (a sequence of 

169 length `num`) are returned. 

170 num : integer, optional 

171 Number of samples to generate. Default is 50. 

172 endpoint : boolean, optional 

173 If true, `stop` is the last sample. Otherwise, it is not included. 

174 Default is True. 

175 base : float, optional 

176 The base of the log space. The step size between the elements in 

177 ``ln(samples) / ln(base)`` (or ``log_base(samples)``) is uniform. 

178 Default is 10.0. 

179 dtype : dtype 

180 The type of the output array. If `dtype` is not given, infer the data 

181 type from the other input arguments. 

182 

183 Returns 

184 ------- 

185 samples : ndarray 

186 `num` samples, equally spaced on a log scale. 

187 

188 See Also 

189 -------- 

190 arange : Similar to linspace, with the step size specified instead of the 

191 number of samples. Note that, when used with a float endpoint, the 

192 endpoint may or may not be included. 

193 linspace : Similar to logspace, but with the samples uniformly distributed 

194 in linear space, instead of log space. 

195 geomspace : Similar to logspace, but with endpoints specified directly. 

196 

197 Notes 

198 ----- 

199 Logspace is equivalent to the code 

200 

201 >>> y = np.linspace(start, stop, num=num, endpoint=endpoint) 

202 ... # doctest: +SKIP 

203 >>> power(base, y).astype(dtype) 

204 ... # doctest: +SKIP 

205 

206 Examples 

207 -------- 

208 >>> np.logspace(2.0, 3.0, num=4) 

209 array([ 100. , 215.443469 , 464.15888336, 1000. ]) 

210 >>> np.logspace(2.0, 3.0, num=4, endpoint=False) 

211 array([ 100. , 177.827941 , 316.22776602, 562.34132519]) 

212 >>> np.logspace(2.0, 3.0, num=4, base=2.0) 

213 array([ 4. , 5.0396842 , 6.34960421, 8. ]) 

214 

215 Graphical illustration: 

216 

217 >>> import matplotlib.pyplot as plt 

218 >>> N = 10 

219 >>> x1 = np.logspace(0.1, 1, N, endpoint=True) 

220 >>> x2 = np.logspace(0.1, 1, N, endpoint=False) 

221 >>> y = np.zeros(N) 

222 >>> plt.plot(x1, y, 'o') 

223 [<matplotlib.lines.Line2D object at 0x...>] 

224 >>> plt.plot(x2, y + 0.5, 'o') 

225 [<matplotlib.lines.Line2D object at 0x...>] 

226 >>> plt.ylim([-0.5, 1]) 

227 (-0.5, 1) 

228 >>> plt.show() 

229 

230 """ 

231 y = linspace(start, stop, num=num, endpoint=endpoint) 

232 if dtype is None: 

233 return _nx.power(base, y) 

234 return _nx.power(base, y).astype(dtype)

237def geomspace(start, stop, num=50, endpoint=True, dtype=None): 

238 """ 

239 Return numbers spaced evenly on a log scale (a geometric progression). 

240 

241 This is similar to `logspace`, but with endpoints specified directly. 

242 Each output sample is a constant multiple of the previous. 

243 

244 Parameters 

245 ---------- 

246 start : scalar 

247 The starting value of the sequence. 

248 stop : scalar 

249 The final value of the sequence, unless `endpoint` is False. 

250 In that case, ``num + 1`` values are spaced over the 

251 interval in log-space, of which all but the last (a sequence of 

252 length `num`) are returned. 

253 num : integer, optional 

254 Number of samples to generate. Default is 50. 

255 endpoint : boolean, optional 

256 If true, `stop` is the last sample. Otherwise, it is not included. 

257 Default is True. 

258 dtype : dtype 

259 The type of the output array. If `dtype` is not given, infer the data 

260 type from the other input arguments. 

261 

262 Returns 

263 ------- 

264 samples : ndarray 

265 `num` samples, equally spaced on a log scale. 

266 

267 See Also 

268 -------- 

269 logspace : Similar to geomspace, but with endpoints specified using log 

270 and base. 

271 linspace : Similar to geomspace, but with arithmetic instead of geometric 

272 progression. 

273 arange : Similar to linspace, with the step size specified instead of the 

274 number of samples. 

275 

276 Notes 

277 ----- 

278 If the inputs or dtype are complex, the output will follow a logarithmic 

279 spiral in the complex plane. (There are an infinite number of spirals 

280 passing through two points; the output will follow the shortest such path.) 

281 

282 Examples 

283 -------- 

284 >>> np.geomspace(1, 1000, num=4) 

285 array([ 1., 10., 100., 1000.]) 

286 >>> np.geomspace(1, 1000, num=3, endpoint=False) 

287 array([ 1., 10., 100.]) 

288 >>> np.geomspace(1, 1000, num=4, endpoint=False) 

289 array([ 1. , 5.62341325, 31.6227766 , 177.827941 ]) 

290 >>> np.geomspace(1, 256, num=9) 

291 array([ 1., 2., 4., 8., 16., 32., 64., 128., 256.]) 

292 

293 Note that the above may not produce exact integers: 

294 

295 >>> np.geomspace(1, 256, num=9, dtype=int) 

296 array([ 1, 2, 4, 7, 16, 32, 63, 127, 256]) 

297 >>> np.around(np.geomspace(1, 256, num=9)).astype(int) 

298 array([ 1, 2, 4, 8, 16, 32, 64, 128, 256]) 

299 

300 Negative, decreasing, and complex inputs are allowed: 

301 

302 >>> np.geomspace(1000, 1, num=4) 

303 array([ 1000., 100., 10., 1.]) 

304 >>> np.geomspace(-1000, -1, num=4) 

305 array([-1000., -100., -10., -1.]) 

306 >>> np.geomspace(1j, 1000j, num=4) # Straight line 

307 array([ 0. +1.j, 0. +10.j, 0. +100.j, 0.+1000.j]) 

308 >>> np.geomspace(-1+0j, 1+0j, num=5) # Circle 

309 array([-1.00000000+0.j , -0.70710678+0.70710678j, 

310 0.00000000+1.j , 0.70710678+0.70710678j, 

311 1.00000000+0.j ]) 

312 

313 Graphical illustration of ``endpoint`` parameter: 

314 

315 >>> import matplotlib.pyplot as plt 

316 >>> N = 10 

317 >>> y = np.zeros(N) 

318 >>> plt.semilogx(np.geomspace(1, 1000, N, endpoint=True), y + 1, 'o') 

319 >>> plt.semilogx(np.geomspace(1, 1000, N, endpoint=False), y + 2, 'o') 

320 >>> plt.axis([0.5, 2000, 0, 3]) 

321 >>> plt.grid(True, color='0.7', linestyle='-', which='both', axis='both') 

322 >>> plt.show() 

323 

324 """ 

325 if start == 0 or stop == 0: 

326 raise ValueError('Geometric sequence cannot include zero') 

327 

328 dt = result_type(start, stop, float(num)) 

329 if dtype is None: 

330 dtype = dt 

331 else: 

332 # complex to dtype('complex128'), for instance 

333 dtype = _nx.dtype(dtype) 

334 

335 # Avoid negligible real or imaginary parts in output by rotating to 

336 # positive real, calculating, then undoing rotation 

337 out_sign = 1 

338 if start.real == stop.real == 0: 

339 start, stop = start.imag, stop.imag 

340 out_sign = 1j * out_sign 

341 if _nx.sign(start) == _nx.sign(stop) == -1: 

342 start, stop = -start, -stop 

343 out_sign = -out_sign 

344 

345 # Promote both arguments to the same dtype in case, for instance, one is 

346 # complex and another is negative and log would produce NaN otherwise 

347 start = start + (stop - stop) 

348 stop = stop + (start - start) 

349 if _nx.issubdtype(dtype, _nx.complexfloating): 

350 start = start + 0j 

351 stop = stop + 0j 

352 

353 log_start = _nx.log10(start) 

354 log_stop = _nx.log10(stop) 

355 result = out_sign * logspace(log_start, log_stop, num=num, 

356 endpoint=endpoint, base=10.0, dtype=dtype) 

357 

358 return result.astype(dtype)
