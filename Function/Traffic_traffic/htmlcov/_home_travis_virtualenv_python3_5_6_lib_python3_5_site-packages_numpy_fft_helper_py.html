22def fftshift(x, axes=None): 

23 """ 

24 Shift the zero-frequency component to the center of the spectrum. 

25 

26 This function swaps half-spaces for all axes listed (defaults to all). 

27 Note that ``y[0]`` is the Nyquist component only if ``len(x)`` is even. 

28 

29 Parameters 

30 ---------- 

31 x : array_like 

32 Input array. 

33 axes : int or shape tuple, optional 

34 Axes over which to shift. Default is None, which shifts all axes. 

35 

36 Returns 

37 ------- 

38 y : ndarray 

39 The shifted array. 

40 

41 See Also 

42 -------- 

43 ifftshift : The inverse of `fftshift`. 

44 

45 Examples 

46 -------- 

47 >>> freqs = np.fft.fftfreq(10, 0.1) 

48 >>> freqs 

49 array([ 0., 1., 2., 3., 4., -5., -4., -3., -2., -1.]) 

50 >>> np.fft.fftshift(freqs) 

51 array([-5., -4., -3., -2., -1., 0., 1., 2., 3., 4.]) 

52 

53 Shift the zero-frequency component only along the second axis: 

54 

55 >>> freqs = np.fft.fftfreq(9, d=1./9).reshape(3, 3) 

56 >>> freqs 

57 array([[ 0., 1., 2.], 

58 [ 3., 4., -4.], 

59 [-3., -2., -1.]]) 

60 >>> np.fft.fftshift(freqs, axes=(1,)) 

61 array([[ 2., 0., 1.], 

62 [-4., 3., 4.], 

63 [-1., -3., -2.]]) 

64 

65 """ 

66 x = asarray(x) 

67 if axes is None: 

68 axes = tuple(range(x.ndim)) 

69 shift = [dim // 2 for dim in x.shape] 

70 elif isinstance(axes, integer_types): 

71 shift = x.shape[axes] // 2 

72 else: 

73 shift = [x.shape[ax] // 2 for ax in axes] 

74 

75 return roll(x, shift, axes)

78def ifftshift(x, axes=None): 

79 """ 

80 The inverse of `fftshift`. Although identical for even-length `x`, the 

81 functions differ by one sample for odd-length `x`. 

82 

83 Parameters 

84 ---------- 

85 x : array_like 

86 Input array. 

87 axes : int or shape tuple, optional 

88 Axes over which to calculate. Defaults to None, which shifts all axes. 

89 

90 Returns 

91 ------- 

92 y : ndarray 

93 The shifted array. 

94 

95 See Also 

96 -------- 

97 fftshift : Shift zero-frequency component to the center of the spectrum. 

98 

99 Examples 

100 -------- 

101 >>> freqs = np.fft.fftfreq(9, d=1./9).reshape(3, 3) 

102 >>> freqs 

103 array([[ 0., 1., 2.], 

104 [ 3., 4., -4.], 

105 [-3., -2., -1.]]) 

106 >>> np.fft.ifftshift(np.fft.fftshift(freqs)) 

107 array([[ 0., 1., 2.], 

108 [ 3., 4., -4.], 

109 [-3., -2., -1.]]) 

110 

111 """ 

112 x = asarray(x) 

113 if axes is None: 

114 axes = tuple(range(x.ndim)) 

115 shift = [-(dim // 2) for dim in x.shape] 

116 elif isinstance(axes, integer_types): 

117 shift = -(x.shape[axes] // 2) 

118 else: 

119 shift = [-(x.shape[ax] // 2) for ax in axes] 

120 

121 return roll(x, shift, axes)

124def fftfreq(n, d=1.0): 

125 """ 

126 Return the Discrete Fourier Transform sample frequencies. 

127 

128 The returned float array `f` contains the frequency bin centers in cycles 

129 per unit of the sample spacing (with zero at the start). For instance, if 

130 the sample spacing is in seconds, then the frequency unit is cycles/second. 

131 

132 Given a window length `n` and a sample spacing `d`:: 

133 

134 f = [0, 1, ..., n/2-1, -n/2, ..., -1] / (d*n) if n is even 

135 f = [0, 1, ..., (n-1)/2, -(n-1)/2, ..., -1] / (d*n) if n is odd 

136 

137 Parameters 

138 ---------- 

139 n : int 

140 Window length. 

141 d : scalar, optional 

142 Sample spacing (inverse of the sampling rate). Defaults to 1. 

143 

144 Returns 

145 ------- 

146 f : ndarray 

147 Array of length `n` containing the sample frequencies. 

148 

149 Examples 

150 -------- 

151 >>> signal = np.array([-2, 8, 6, 4, 1, 0, 3, 5], dtype=float) 

152 >>> fourier = np.fft.fft(signal) 

153 >>> n = signal.size 

154 >>> timestep = 0.1 

155 >>> freq = np.fft.fftfreq(n, d=timestep) 

156 >>> freq 

157 array([ 0. , 1.25, 2.5 , 3.75, -5. , -3.75, -2.5 , -1.25]) 

158 

159 """ 

160 if not isinstance(n, integer_types): 

161 raise ValueError("n should be an integer") 

162 val = 1.0 / (n * d) 

163 results = empty(n, int) 

164 N = (n-1)//2 + 1 

165 p1 = arange(0, N, dtype=int) 

166 results[:N] = p1 

167 p2 = arange(-(n//2), 0, dtype=int) 

168 results[N:] = p2 

169 return results * val

173def rfftfreq(n, d=1.0): 

174 """ 

175 Return the Discrete Fourier Transform sample frequencies 

176 (for usage with rfft, irfft). 

177 

178 The returned float array `f` contains the frequency bin centers in cycles 

179 per unit of the sample spacing (with zero at the start). For instance, if 

180 the sample spacing is in seconds, then the frequency unit is cycles/second. 

181 

182 Given a window length `n` and a sample spacing `d`:: 

183 

184 f = [0, 1, ..., n/2-1, n/2] / (d*n) if n is even 

185 f = [0, 1, ..., (n-1)/2-1, (n-1)/2] / (d*n) if n is odd 

186 

187 Unlike `fftfreq` (but like `scipy.fftpack.rfftfreq`) 

188 the Nyquist frequency component is considered to be positive. 

189 

190 Parameters 

191 ---------- 

192 n : int 

193 Window length. 

194 d : scalar, optional 

195 Sample spacing (inverse of the sampling rate). Defaults to 1. 

196 

197 Returns 

198 ------- 

199 f : ndarray 

200 Array of length ``n//2 + 1`` containing the sample frequencies. 

201 

202 Examples 

203 -------- 

204 >>> signal = np.array([-2, 8, 6, 4, 1, 0, 3, 5, -3, 4], dtype=float) 

205 >>> fourier = np.fft.rfft(signal) 

206 >>> n = signal.size 

207 >>> sample_rate = 100 

208 >>> freq = np.fft.fftfreq(n, d=1./sample_rate) 

209 >>> freq 

210 array([ 0., 10., 20., 30., 40., -50., -40., -30., -20., -10.]) 

211 >>> freq = np.fft.rfftfreq(n, d=1./sample_rate) 

212 >>> freq 

213 array([ 0., 10., 20., 30., 40., 50.]) 

214 

215 """ 

216 if not isinstance(n, integer_types): 

217 raise ValueError("n should be an integer") 

218 val = 1.0/(n*d) 

219 N = n//2 + 1 

220 results = arange(0, N, dtype=int) 

221 return results * val

224class _FFTCache(object): 

225 """ 

226 Cache for the FFT twiddle factors as an LRU (least recently used) cache. 

227 

228 Parameters 

229 ---------- 

230 max_size_in_mb : int 

231 Maximum memory usage of the cache before items are being evicted. 

232 max_item_count : int 

233 Maximum item count of the cache before items are being evicted. 

234 

235 Notes 

236 ----- 

237 Items will be evicted if either limit has been reached upon getting and 

238 setting. The maximum memory usages is not strictly the given 

239 ``max_size_in_mb`` but rather 

240 ``max(max_size_in_mb, 1.5 * size_of_largest_item)``. Thus the cache will 

241 never be completely cleared - at least one item will remain and a single 

242 large item can cause the cache to retain several smaller items even if the 

243 given maximum cache size has been exceeded. 

244 """ 

245 def __init__(self, max_size_in_mb, max_item_count): 

246 self._max_size_in_bytes = max_size_in_mb * 1024 ** 2 

247 self._max_item_count = max_item_count 

248 self._dict = collections.OrderedDict() 

249 self._lock = threading.Lock() 

250 

251 def put_twiddle_factors(self, n, factors): 

252 """ 

253 Store twiddle factors for an FFT of length n in the cache. 

254 

255 Putting multiple twiddle factors for a certain n will store it multiple 

256 times. 

257 

258 Parameters 

259 ---------- 

260 n : int 

261 Data length for the FFT. 

262 factors : ndarray 

263 The actual twiddle values. 

264 """ 

265 with self._lock: 

266 # Pop + later add to move it to the end for LRU behavior. 

267 # Internally everything is stored in a dictionary whose values are 

268 # lists. 

269 try: 

270 value = self._dict.pop(n) 

271 except KeyError: 

272 value = [] 

273 value.append(factors) 

274 self._dict[n] = value 

275 self._prune_cache() 

276 

277 def pop_twiddle_factors(self, n): 

278 """ 

279 Pop twiddle factors for an FFT of length n from the cache. 

280 

281 Will return None if the requested twiddle factors are not available in 

282 the cache. 

283 

284 Parameters 

285 ---------- 

286 n : int 

287 Data length for the FFT. 

288 

289 Returns 

290 ------- 

291 out : ndarray or None 

292 The retrieved twiddle factors if available, else None. 

293 """ 

294 with self._lock: 

295 if n not in self._dict or not self._dict[n]: 

296 return None 

297 # Pop + later add to move it to the end for LRU behavior. 

298 all_values = self._dict.pop(n) 

299 value = all_values.pop() 

300 # Only put pack if there are still some arrays left in the list. 

301 if all_values: 

302 self._dict[n] = all_values 

303 return value 

304 

305 def _prune_cache(self): 

306 # Always keep at least one item. 

307 while len(self._dict) > 1 and ( 

308 len(self._dict) > self._max_item_count or self._check_size()): 

309 self._dict.popitem(last=False) 

310 

311 def _check_size(self): 

312 item_sizes = [sum(_j.nbytes for _j in _i) 

313 for _i in self._dict.values() if _i] 

314 if not item_sizes: 

315 return False 

316 max_size = max(self._max_size_in_bytes, 1.5 * max(item_sizes)) 

317 return sum(item_sizes) > max_size
