1""" 

2The legend module defines the Legend class, which is responsible for 

3drawing legends associated with axes and/or figures. 

4 

5.. important:: 

6 

7 It is unlikely that you would ever create a Legend instance 

8 manually. Most users would normally create a legend via the 

9 :meth:`~matplotlib.axes.Axes.legend` function. For more details on legends 

10 there is also a :doc:`legend guide </tutorials/intermediate/legend_guide>`. 

11 

12The Legend class can be considered as a container of legend handles and 

13legend texts. Creation of corresponding legend handles from the plot elements 

14in the axes or figures (e.g., lines, patches, etc.) are specified by the 

15handler map, which defines the mapping between the plot elements and the 

16legend handlers to be used (the default legend handlers are defined in the 

17:mod:`~matplotlib.legend_handler` module). Note that not all kinds of 

18artist are supported by the legend yet by default but it is possible to 

19extend the legend handler's capabilities to support arbitrary objects. See 

20the :doc:`legend guide </tutorials/intermediate/legend_guide>` for more 

21information.

602 "Helper function to locate the legend."

604 if self._loc == 0: # "best". 

605 x, y = self._find_best_position(width, height, renderer) 

606 elif self._loc in Legend.codes.values(): # Fixed location. 

607 bbox = Bbox.from_bounds(0, 0, width, height) 

608 x, y = self._get_anchored_bbox(self._loc, bbox, 

609 self.get_bbox_to_anchor(), 

610 renderer) 

611 else: # Axes or figure coordinates. 

612 fx, fy = self._loc 

613 bbox = self.get_bbox_to_anchor() 

614 x, y = bbox.x0 + bbox.width * fx, bbox.y0 + bbox.height * fy 

615 

616 return x + xdescent, y + ydescent

748 def _init_legend_box(self, handles, labels, markerfirst=True): 

749 """ 

750 Initialize the legend_box. The legend_box is an instance of 

751 the OffsetBox, which is packed with legend handles and 

752 texts. Once packed, their location is calculated during the 

753 drawing time. 

754 """ 

755 

756 fontsize = self._fontsize 

757 

758 # legend_box is a HPacker, horizontally packed with 

759 # columns. Each column is a VPacker, vertically packed with 

760 # legend items. Each legend item is HPacker packed with 

761 # legend handleBox and labelBox. handleBox is an instance of 

762 # offsetbox.DrawingArea which contains legend handle. labelBox 

763 # is an instance of offsetbox.TextArea which contains legend 

764 # text. 

765 

766 text_list = [] # the list of text instances 

767 handle_list = [] # the list of text instances 

768 handles_and_labels = [] 

769 

770 label_prop = dict(verticalalignment='baseline', 

771 horizontalalignment='left', 

772 fontproperties=self.prop, 

773 ) 

774 

775 # The approximate height and descent of text. These values are 

776 # only used for plotting the legend handle. 

777 descent = 0.35 * self._approx_text_height() * (self.handleheight - 0.7) 

778 # 0.35 and 0.7 are just heuristic numbers and may need to be improved. 

779 height = self._approx_text_height() * self.handleheight - descent 

780 # each handle needs to be drawn inside a box of (x, y, w, h) = 

781 # (0, -descent, width, height). And their coordinates should 

782 # be given in the display coordinates. 

783 

784 # The transformation of each handle will be automatically set 

785 # to self.get_trasnform(). If the artist does not use its 

786 # default transform (e.g., Collections), you need to 

787 # manually set their transform to the self.get_transform(). 

788 legend_handler_map = self.get_legend_handler_map() 

789 

790 for orig_handle, lab in zip(handles, labels): 

791 handler = self.get_legend_handler(legend_handler_map, orig_handle) 

792 if handler is None: 

793 warnings.warn( 

794 "Legend does not support {!r} instances.\nA proxy artist " 

795 "may be used instead.\nSee: " 

796 "http://matplotlib.org/users/legend_guide.html" 

797 "#creating-artists-specifically-for-adding-to-the-legend-" 

798 "aka-proxy-artists".format(orig_handle) 

799 ) 

800 # We don't have a handle for this artist, so we just defer 

801 # to None. 

802 handle_list.append(None) 

803 else: 

804 textbox = TextArea(lab, textprops=label_prop, 

805 multilinebaseline=True, 

806 minimumdescent=True) 

807 handlebox = DrawingArea(width=self.handlelength * fontsize, 

808 height=height, 

809 xdescent=0., ydescent=descent) 

810 

811 text_list.append(textbox._text) 

812 # Create the artist for the legend which represents the 

813 # original artist/handle. 

814 handle_list.append(handler.legend_artist(self, orig_handle, 

815 fontsize, handlebox)) 

816 handles_and_labels.append((handlebox, textbox)) 

817 

818 if handles_and_labels: 

819 # We calculate number of rows in each column. The first 

820 # (num_largecol) columns will have (nrows+1) rows, and remaining 

821 # (num_smallcol) columns will have (nrows) rows. 

822 ncol = min(self._ncol, len(handles_and_labels)) 

823 nrows, num_largecol = divmod(len(handles_and_labels), ncol) 

824 num_smallcol = ncol - num_largecol 

825 # starting index of each column and number of rows in it. 

826 rows_per_col = [nrows + 1] * num_largecol + [nrows] * num_smallcol 

827 start_idxs = np.concatenate([[0], np.cumsum(rows_per_col)[:-1]]) 

828 cols = zip(start_idxs, rows_per_col) 

829 else: 

830 cols = [] 

831 

832 columnbox = [] 

833 for i0, di in cols: 

834 # pack handleBox and labelBox into itemBox 

835 itemBoxes = [HPacker(pad=0, 

836 sep=self.handletextpad * fontsize, 

837 children=[h, t] if markerfirst else [t, h], 

838 align="baseline") 

839 for h, t in handles_and_labels[i0:i0 + di]] 

840 # minimumdescent=False for the text of the last row of the column 

841 if markerfirst: 

842 itemBoxes[-1].get_children()[1].set_minimumdescent(False) 

843 else: 

844 itemBoxes[-1].get_children()[0].set_minimumdescent(False) 

845 

846 # pack columnBox 

847 alignment = "baseline" if markerfirst else "right" 

848 columnbox.append(VPacker(pad=0, 

849 sep=self.labelspacing * fontsize, 

850 align=alignment, 

851 children=itemBoxes)) 

852 

853 mode = "expand" if self._mode == "expand" else "fixed" 

854 sep = self.columnspacing * fontsize 

855 self._legend_handle_box = HPacker(pad=0, 

856 sep=sep, align="baseline", 

857 mode=mode, 

858 children=columnbox) 

859 self._legend_title_box = TextArea("") 

860 self._legend_box = VPacker(pad=self.borderpad * fontsize, 

861 sep=self.labelspacing * fontsize, 

862 align="center", 

863 children=[self._legend_title_box, 

864 self._legend_handle_box]) 

865 self._legend_box.set_figure(self.figure) 

866 self.texts = text_list 

867 self.legendHandles = handle_list

869 def _auto_legend_data(self): 

870 """ 

871 Returns list of vertices and extents covered by the plot. 

872 

873 Returns a two long list. 

874 

875 First element is a list of (x, y) vertices (in 

876 display-coordinates) covered by all the lines and line 

877 collections, in the legend's handles. 

878 

879 Second element is a list of bounding boxes for all the patches in 

880 the legend's handles. 

881 """ 

882 # should always hold because function is only called internally 

883 assert self.isaxes 

884 

885 ax = self.parent 

886 bboxes = [] 

887 lines = [] 

888 offsets = [] 

889 

890 for handle in ax.lines: 

891 assert isinstance(handle, Line2D) 

892 path = handle.get_path() 

893 trans = handle.get_transform() 

894 tpath = trans.transform_path(path) 

895 lines.append(tpath) 

896 

897 for handle in ax.patches: 

898 assert isinstance(handle, Patch) 

899 

900 if isinstance(handle, Rectangle): 

901 transform = handle.get_data_transform() 

902 bboxes.append(handle.get_bbox().transformed(transform)) 

903 else: 

904 transform = handle.get_transform() 

905 bboxes.append(handle.get_path().get_extents(transform)) 

906 

907 for handle in ax.collections: 

908 transform, transOffset, hoffsets, paths = handle._prepare_points() 

909 

910 if len(hoffsets): 

911 for offset in transOffset.transform(hoffsets): 

912 offsets.append(offset) 

913 

914 try: 

915 vertices = np.concatenate([l.vertices for l in lines]) 

916 except ValueError: 

917 vertices = np.array([]) 

918 

919 return [vertices, bboxes, lines, offsets]

1063 def _get_anchored_bbox(self, loc, bbox, parentbbox, renderer): 

1064 """

1066 location code. Return the (x,y) coordinate of the bbox. 

1067

1070 "best". 

1071 

1072 - bbox: bbox to be placed, display coordinate units. 

1073 - parentbbox: a parent box which will contain the bbox. In 

1074 display coordinates. 

1075 """

1077 

1078 BEST, UR, UL, LL, LR, R, CL, CR, LC, UC, C = range(11) 

1079 

1080 anchor_coefs = {UR: "NE", 

1081 UL: "NW", 

1082 LL: "SW",

1085 CL: "W", 

1086 CR: "E", 

1087 LC: "S", 

1088 UC: "N",

1098 def _find_best_position(self, width, height, renderer, consider=None): 

1099 """ 

1100 Determine the best location to place the legend. 

1101 

1102 *consider* is a list of ``(x, y)`` pairs to consider as a potential 

1103 lower-left corner of the legend. All are display coords. 

1104 """ 

1105 # should always hold because function is only called internally 

1106 assert self.isaxes 

1107 

1108 verts, bboxes, lines, offsets = self._auto_legend_data() 

1109 

1110 bbox = Bbox.from_bounds(0, 0, width, height) 

1111 if consider is None: 

1112 consider = [self._get_anchored_bbox(x, bbox, 

1113 self.get_bbox_to_anchor(), 

1114 renderer) 

1115 for x in range(1, len(self.codes))] 

1116 

1117 candidates = [] 

1118 for idx, (l, b) in enumerate(consider): 

1119 legendBox = Bbox.from_bounds(l, b, width, height) 

1120 badness = 0 

1121 # XXX TODO: If markers are present, it would be good to 

1122 # take them into account when checking vertex overlaps in 

1123 # the next line. 

1124 badness = (legendBox.count_contains(verts) 

1125 + legendBox.count_contains(offsets) 

1126 + legendBox.count_overlaps(bboxes) 

1127 + sum(line.intersects_bbox(legendBox, filled=False) 

1128 for line in lines)) 

1129 if badness == 0: 

1130 return l, b 

1131 # Include the index to favor lower codes in case of a tie. 

1132 candidates.append((badness, idx, (l, b))) 

1133 

1134 _, _, (l, b) = min(candidates) 

1135 return l, b

1257def _parse_legend_args(axs, *args, handles=None, labels=None, **kwargs): 

1258 """ 

1259 Get the handles and labels from the calls to either ``figure.legend`` 

1260 or ``axes.legend``. 

1261 

1262 ``axs`` is a list of axes (to get legend artists from) 

1263 """ 

1264 log = logging.getLogger(__name__) 

1265 

1266 handlers = kwargs.get('handler_map', {}) or {} 

1267 extra_args = () 

1268 

1269 if (handles is not None or labels is not None) and args: 

1270 warnings.warn("You have mixed positional and keyword arguments, some " 

1271 "input may be discarded.") 

1272 

1273 # if got both handles and labels as kwargs, make same length 

1274 if handles and labels: 

1275 handles, labels = zip(*zip(handles, labels)) 

1276 

1277 elif handles is not None and labels is None: 

1278 labels = [handle.get_label() for handle in handles] 

1279 

1280 elif labels is not None and handles is None: 

1281 # Get as many handles as there are labels. 

1282 handles = [handle for handle, label 

1283 in zip(_get_legend_handles(axs, handlers), labels)] 

1284 

1285 # No arguments - automatically detect labels and handles. 

1286 elif len(args) == 0: 

1287 handles, labels = _get_legend_handles_labels(axs, handlers) 

1288 if not handles: 

1289 log.warning('No handles with labels found to put in legend.') 

1290 

1291 # One argument. User defined labels - automatic handle detection. 

1292 elif len(args) == 1: 

1293 labels, = args 

1294 # Get as many handles as there are labels. 

1295 handles = [handle for handle, label 

1296 in zip(_get_legend_handles(axs, handlers), labels)] 

1297 

1298 # Two arguments: 

1299 # * user defined handles and labels 

1300 elif len(args) >= 2: 

1301 handles, labels = args[:2] 

1302 extra_args = args[2:] 

1303 

1304 else: 

1305 raise TypeError('Invalid arguments to legend.') 

1306 

1307 return handles, labels, extra_args, kwargs
