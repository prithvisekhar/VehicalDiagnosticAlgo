29def _combine_hash_arrays(arrays, num_items): 

30 """ 

31 Parameters 

32 ---------- 

33 arrays : generator 

34 num_items : int 

35 

36 Should be the same as CPython's tupleobject.c 

37 """ 

38 try: 

39 first = next(arrays) 

40 except StopIteration: 

41 return np.array([], dtype=np.uint64) 

42 

43 arrays = itertools.chain([first], arrays) 

44 

45 mult = np.uint64(1000003) 

46 out = np.zeros_like(first) + np.uint64(0x345678) 

47 for i, a in enumerate(arrays): 

48 inverse_i = num_items - i 

49 out ^= a 

50 out *= mult 

51 mult += np.uint64(82520 + inverse_i + inverse_i) 

52 assert i + 1 == num_items, "Fed in wrong num_items" 

53 out += np.uint64(97531) 

54 return out

139def hash_tuples(vals, encoding="utf8", hash_key=None): 

140 """ 

141 Hash an MultiIndex / list-of-tuples efficiently 

142 

143 .. versionadded:: 0.20.0 

144 

145 Parameters 

146 ---------- 

147 vals : MultiIndex, list-of-tuples, or single tuple 

148 encoding : string, default 'utf8' 

149 hash_key : string key to encode, default to _default_hash_key 

150 

151 Returns 

152 ------- 

153 ndarray of hashed values array 

154 """ 

155 is_tuple = False 

156 if isinstance(vals, tuple): 

157 vals = [vals] 

158 is_tuple = True 

159 elif not is_list_like(vals): 

160 raise TypeError("must be convertible to a list-of-tuples") 

161 

162 from pandas import Categorical, MultiIndex 

163 

164 if not isinstance(vals, ABCMultiIndex): 

165 vals = MultiIndex.from_tuples(vals) 

166 

167 # create a list-of-Categoricals 

168 vals = [ 

169 Categorical(vals.codes[level], vals.levels[level], ordered=False, fastpath=True) 

170 for level in range(vals.nlevels) 

171 ] 

172 

173 # hash the list-of-ndarrays 

174 hashes = ( 

175 _hash_categorical(cat, encoding=encoding, hash_key=hash_key) for cat in vals 

176 ) 

177 h = _combine_hash_arrays(hashes, len(vals)) 

178 if is_tuple: 

179 h = h[0] 

180 

181 return h

206def _hash_categorical(c, encoding, hash_key): 

207 """ 

208 Hash a Categorical by hashing its categories, and then mapping the codes 

209 to the hashes 

210 

211 Parameters 

212 ---------- 

213 c : Categorical 

214 encoding : string, default 'utf8' 

215 hash_key : string key to encode, default to _default_hash_key 

216 

217 Returns 

218 ------- 

219 ndarray of hashed values array, same size as len(c) 

220 """ 

221 # Convert ExtensionArrays to ndarrays 

222 values = np.asarray(c.categories.values) 

223 hashed = hash_array(values, encoding, hash_key, categorize=False) 

224 

225 # we have uint64, as we don't directly support missing values 

226 # we don't want to use take_nd which will coerce to float 

227 # instead, directly construct the result with a 

228 # max(np.uint64) as the missing value indicator 

229 # 

230 # TODO: GH 15362 

231 

232 mask = c.isna() 

233 if len(hashed): 

234 result = hashed.take(c.codes) 

235 else: 

236 result = np.zeros(len(mask), dtype="uint64") 

237 

238 if mask.any(): 

239 result[mask] = np.iinfo(np.uint64).max 

240 

241 return result

244def hash_array(vals, encoding="utf8", hash_key=None, categorize=True): 

245 """ 

246 Given a 1d array, return an array of deterministic integers. 

247 

248 .. versionadded:: 0.19.2 

249 

250 Parameters 

251 ---------- 

252 vals : ndarray, Categorical 

253 encoding : string, default 'utf8' 

254 encoding for data & key when strings 

255 hash_key : string key to encode, default to _default_hash_key 

256 categorize : bool, default True 

257 Whether to first categorize object arrays before hashing. This is more 

258 efficient when the array contains duplicate values. 

259 

260 .. versionadded:: 0.20.0 

261 

262 Returns 

263 ------- 

264 1d uint64 numpy array of hash values, same length as the vals 

265 """ 

266 

267 if not hasattr(vals, "dtype"): 

268 raise TypeError("must pass a ndarray-like") 

269 dtype = vals.dtype 

270 

271 if hash_key is None: 

272 hash_key = _default_hash_key 

273 

274 # For categoricals, we hash the categories, then remap the codes to the 

275 # hash values. (This check is above the complex check so that we don't ask 

276 # numpy if categorical is a subdtype of complex, as it will choke). 

277 if is_categorical_dtype(dtype): 

278 return _hash_categorical(vals, encoding, hash_key) 

279 elif is_extension_array_dtype(dtype): 

280 vals, _ = vals._values_for_factorize() 

281 dtype = vals.dtype 

282 

283 # we'll be working with everything as 64-bit values, so handle this 

284 # 128-bit value early 

285 if np.issubdtype(dtype, np.complex128): 

286 return hash_array(np.real(vals)) + 23 * hash_array(np.imag(vals)) 

287 

288 # First, turn whatever array this is into unsigned 64-bit ints, if we can 

289 # manage it. 

290 elif isinstance(dtype, np.bool): 

291 vals = vals.astype("u8") 

292 elif issubclass(dtype.type, (np.datetime64, np.timedelta64)): 

293 vals = vals.view("i8").astype("u8", copy=False) 

294 elif issubclass(dtype.type, np.number) and dtype.itemsize <= 8: 

295 vals = vals.view("u{}".format(vals.dtype.itemsize)).astype("u8") 

296 else: 

297 # With repeated values, its MUCH faster to categorize object dtypes, 

298 # then hash and rename categories. We allow skipping the categorization 

299 # when the values are known/likely to be unique. 

300 if categorize: 

301 from pandas import factorize, Categorical, Index 

302 

303 codes, categories = factorize(vals, sort=False) 

304 cat = Categorical(codes, Index(categories), ordered=False, fastpath=True) 

305 return _hash_categorical(cat, encoding, hash_key) 

306 

307 try: 

308 vals = hashing.hash_object_array(vals, hash_key, encoding) 

309 except TypeError: 

310 # we have mixed types 

311 vals = hashing.hash_object_array( 

312 vals.astype(str).astype(object), hash_key, encoding 

313 ) 

314 

315 # Then, redistribute these 64-bit ints within the space of 64-bit ints 

316 vals ^= vals >> 30 

317 vals *= np.uint64(0xBF58476D1CE4E5B9) 

318 vals ^= vals >> 27 

319 vals *= np.uint64(0x94D049BB133111EB) 

320 vals ^= vals >> 31 

321 return vals

324def _hash_scalar(val, encoding="utf8", hash_key=None): 

325 """ 

326 Hash scalar value 

327 

328 Returns 

329 ------- 

330 1d uint64 numpy array of hash value, of length 1 

331 """ 

332 

333 if isna(val): 

334 # this is to be consistent with the _hash_categorical implementation 

335 return np.array([np.iinfo(np.uint64).max], dtype="u8") 

336 

337 if getattr(val, "tzinfo", None) is not None: 

338 # for tz-aware datetimes, we need the underlying naive UTC value and 

339 # not the tz aware object or pd extension type (as 

340 # infer_dtype_from_scalar would do) 

341 if not isinstance(val, tslibs.Timestamp): 

342 val = tslibs.Timestamp(val) 

343 val = val.tz_convert(None) 

344 

345 dtype, val = infer_dtype_from_scalar(val) 

346 vals = np.array([val], dtype=dtype) 

347 

348 return hash_array(vals, hash_key=hash_key, encoding=encoding, categorize=False)
