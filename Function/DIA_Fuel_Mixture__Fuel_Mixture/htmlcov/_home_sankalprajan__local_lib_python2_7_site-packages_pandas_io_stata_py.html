509def _cast_to_stata_types(data): 

510 """Checks the dtypes of the columns of a pandas DataFrame for 

511 compatibility with the data types and ranges supported by Stata, and 

512 converts if necessary. 

513 

514 Parameters 

515 ---------- 

516 data : DataFrame 

517 The DataFrame to check and convert 

518 

519 Notes 

520 ----- 

521 Numeric columns in Stata must be one of int8, int16, int32, float32 or 

522 float64, with some additional value restrictions. int8 and int16 columns 

523 are checked for violations of the value restrictions and upcast if needed. 

524 int64 data is not usable in Stata, and so it is downcast to int32 whenever 

525 the value are in the int32 range, and sidecast to float64 when larger than 

526 this range. If the int64 values are outside of the range of those 

527 perfectly representable as float64 values, a warning is raised. 

528 

529 bool columns are cast to int8. uint columns are converted to int of the 

530 same size if there is no loss in precision, otherwise are upcast to a 

531 larger type. uint64 is currently not supported since it is concerted to 

532 object in a DataFrame. 

533 """ 

534 ws = '' 

535 # original, if small, if large 

536 conversion_data = ((np.bool, np.int8, np.int8), 

537 (np.uint8, np.int8, np.int16), 

538 (np.uint16, np.int16, np.int32), 

539 (np.uint32, np.int32, np.int64)) 

540 

541 float32_max = struct.unpack('<f', b'\xff\xff\xff\x7e')[0] 

542 float64_max = struct.unpack('<d', b'\xff\xff\xff\xff\xff\xff\xdf\x7f')[0] 

543 

544 for col in data: 

545 dtype = data[col].dtype 

546 # Cast from unsupported types to supported types 

547 for c_data in conversion_data: 

548 if dtype == c_data[0]: 

549 if data[col].max() <= np.iinfo(c_data[1]).max: 

550 dtype = c_data[1] 

551 else: 

552 dtype = c_data[2] 

553 if c_data[2] == np.float64: # Warn if necessary 

554 if data[col].max() >= 2 ** 53: 

555 ws = precision_loss_doc % ('uint64', 'float64') 

556 

557 data[col] = data[col].astype(dtype) 

558 

559 # Check values and upcast if necessary 

560 if dtype == np.int8: 

561 if data[col].max() > 100 or data[col].min() < -127: 

562 data[col] = data[col].astype(np.int16) 

563 elif dtype == np.int16: 

564 if data[col].max() > 32740 or data[col].min() < -32767: 

565 data[col] = data[col].astype(np.int32) 

566 elif dtype == np.int64: 

567 if (data[col].max() <= 2147483620 and 

568 data[col].min() >= -2147483647): 

569 data[col] = data[col].astype(np.int32) 

570 else: 

571 data[col] = data[col].astype(np.float64) 

572 if data[col].max() >= 2 ** 53 or data[col].min() <= -2 ** 53: 

573 ws = precision_loss_doc % ('int64', 'float64') 

574 elif dtype in (np.float32, np.float64): 

575 value = data[col].max() 

576 if np.isinf(value): 

577 raise ValueError('Column {col} has a maximum value of ' 

578 'infinity which is outside the range ' 

579 'supported by Stata.'.format(col=col)) 

580 if dtype == np.float32 and value > float32_max: 

581 data[col] = data[col].astype(np.float64) 

582 elif dtype == np.float64: 

583 if value > float64_max: 

584 raise ValueError('Column {col} has a maximum value ' 

585 '({val}) outside the range supported by ' 

586 'Stata ({float64_max})' 

587 .format(col=col, val=value, 

588 float64_max=float64_max)) 

589 

590 if ws: 

591 warnings.warn(ws, PossiblePrecisionLoss) 

592 

593 return data

720class StataMissingValue(StringMixin): 

721 """ 

722 An observation's missing value. 

723 

724 Parameters 

725 ----------- 

726 value : int8, int16, int32, float32 or float64 

727 The Stata missing value code 

728 

729 Attributes 

730 ---------- 

731 string : string 

732 String representation of the Stata missing value 

733 value : int8, int16, int32, float32 or float64 

734 The original encoded missing value 

735 

736 Notes 

737 ----- 

738 More information: <http://www.stata.com/help.cgi?missing> 

739 

740 Integer missing values make the code '.', '.a', ..., '.z' to the ranges 

741 101 ... 127 (for int8), 32741 ... 32767 (for int16) and 2147483621 ... 

742 2147483647 (for int32). Missing values for floating point data types are 

743 more complex but the pattern is simple to discern from the following table. 

744 

745 np.float32 missing values (float in Stata) 

746 0000007f . 

747 0008007f .a 

748 0010007f .b 

749 ... 

750 00c0007f .x 

751 00c8007f .y 

752 00d0007f .z 

753 

754 np.float64 missing values (double in Stata) 

755 000000000000e07f . 

756 000000000001e07f .a 

757 000000000002e07f .b 

758 ... 

759 000000000018e07f .x 

760 000000000019e07f .y 

761 00000000001ae07f .z 

762 """ 

763 

764 # Construct a dictionary of missing values 

765 MISSING_VALUES = {} 

766 bases = (101, 32741, 2147483621) 

767 for b in bases: 

768 # Conversion to long to avoid hash issues on 32 bit platforms #8968 

769 MISSING_VALUES[compat.long(b)] = '.' 

770 for i in range(1, 27): 

771 MISSING_VALUES[compat.long(i + b)] = '.' + chr(96 + i) 

772 

773 float32_base = b'\x00\x00\x00\x7f' 

774 increment = struct.unpack('<i', b'\x00\x08\x00\x00')[0] 

775 for i in range(27): 

776 value = struct.unpack('<f', float32_base)[0] 

777 MISSING_VALUES[value] = '.' 

778 if i > 0: 

779 MISSING_VALUES[value] += chr(96 + i) 

780 int_value = struct.unpack('<i', struct.pack('<f', value))[ 

781 0] + increment 

782 float32_base = struct.pack('<i', int_value) 

783 

784 float64_base = b'\x00\x00\x00\x00\x00\x00\xe0\x7f' 

785 increment = struct.unpack('q', b'\x00\x00\x00\x00\x00\x01\x00\x00')[0] 

786 for i in range(27): 

787 value = struct.unpack('<d', float64_base)[0] 

788 MISSING_VALUES[value] = '.' 

789 if i > 0: 

790 MISSING_VALUES[value] += chr(96 + i) 

791 int_value = struct.unpack('q', struct.pack('<d', value))[0] + increment 

792 float64_base = struct.pack('q', int_value) 

793 

794 BASE_MISSING_VALUES = {'int8': 101, 

795 'int16': 32741, 

796 'int32': 2147483621, 

797 'float32': struct.unpack('<f', float32_base)[0], 

798 'float64': struct.unpack('<d', float64_base)[0]}
