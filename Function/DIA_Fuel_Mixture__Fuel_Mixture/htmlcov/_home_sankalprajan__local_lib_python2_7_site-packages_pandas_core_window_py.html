2534def _get_center_of_mass(comass, span, halflife, alpha): 

2535 valid_count = com.count_not_none(comass, span, halflife, alpha) 

2536 if valid_count > 1: 

2537 raise ValueError("comass, span, halflife, and alpha " 

2538 "are mutually exclusive") 

2539 

2540 # Convert to center of mass; domain checks ensure 0 < alpha <= 1 

2541 if comass is not None: 

2542 if comass < 0: 

2543 raise ValueError("comass must satisfy: comass >= 0") 

2544 elif span is not None: 

2545 if span < 1: 

2546 raise ValueError("span must satisfy: span >= 1") 

2547 comass = (span - 1) / 2. 

2548 elif halflife is not None: 

2549 if halflife <= 0: 

2550 raise ValueError("halflife must satisfy: halflife > 0") 

2551 decay = 1 - np.exp(np.log(0.5) / halflife) 

2552 comass = 1 / decay - 1 

2553 elif alpha is not None: 

2554 if alpha <= 0 or alpha > 1: 

2555 raise ValueError("alpha must satisfy: 0 < alpha <= 1") 

2556 comass = (1.0 - alpha) / alpha 

2557 else: 

2558 raise ValueError("Must pass one of comass, span, halflife, or alpha")
