23def get_package_name(filepath): 

24 """ 

25 Given a path where a package is installed, determine its name. 

26 

27 Parameters 

28 ---------- 

29 filepath : str 

30 Path to a file. If the determination fails, "numpy" is returned. 

31 

32 Examples 

33 -------- 

34 >>> np.testing.nosetester.get_package_name('nonsense') 

35 'numpy' 

36 

37 """ 

38 

39 fullpath = filepath[:] 

40 pkg_name = [] 

41 while 'site-packages' in filepath or 'dist-packages' in filepath: 

42 filepath, p2 = os.path.split(filepath) 

43 if p2 in ('site-packages', 'dist-packages'): 

44 break 

45 pkg_name.append(p2) 

46 

47 # if package name determination failed, just default to numpy/scipy 

48 if not pkg_name: 

49 if 'scipy' in fullpath: 

50 return 'scipy' 

51 else: 

52 return 'numpy' 

53 

54 # otherwise, reverse to get correct order and return 

55 pkg_name.reverse() 

56 

57 # don't include the outer egg directory 

58 if pkg_name[0].endswith('.egg'): 

59 pkg_name.pop(0) 

60 

61 return '.'.join(pkg_name)

64def run_module_suite(file_to_run=None, argv=None): 

65 """ 

66 Run a test module. 

67 

68 Equivalent to calling ``$ nosetests <argv> <file_to_run>`` from 

69 the command line 

70 

71 Parameters 

72 ---------- 

73 file_to_run : str, optional 

74 Path to test module, or None. 

75 By default, run the module from which this function is called. 

76 argv : list of strings 

77 Arguments to be passed to the nose test runner. ``argv[0]`` is 

78 ignored. All command line arguments accepted by ``nosetests`` 

79 will work. If it is the default value None, sys.argv is used. 

80 

81 .. versionadded:: 1.9.0 

82 

83 Examples 

84 -------- 

85 Adding the following:: 

86 

87 if __name__ == "__main__" : 

88 run_module_suite(argv=sys.argv) 

89 

90 at the end of a test module will run the tests when that module is 

91 called in the python interpreter. 

92 

93 Alternatively, calling:: 

94 

95 >>> run_module_suite(file_to_run="numpy/tests/test_matlib.py") # doctest: +SKIP 

96 

97 from an interpreter will run all the test routine in 'test_matlib.py'. 

98 """ 

99 if file_to_run is None: 

100 f = sys._getframe(1) 

101 file_to_run = f.f_locals.get('__file__', None) 

102 if file_to_run is None: 

103 raise AssertionError 

104 

105 if argv is None: 

106 argv = sys.argv + [file_to_run] 

107 else: 

108 argv = argv + [file_to_run] 

109 

110 nose = import_nose() 

111 from .noseclasses import KnownFailurePlugin 

112 nose.run(argv=argv, addplugins=[KnownFailurePlugin()])

115class NoseTester(object): 

116 """ 

117 Nose test runner. 

118 

119 This class is made available as numpy.testing.Tester, and a test function 

120 is typically added to a package's __init__.py like so:: 

121 

122 from numpy.testing import Tester 

123 test = Tester().test 

124 

125 Calling this test function finds and runs all tests associated with the 

126 package and all its sub-packages. 

127 

128 Attributes 

129 ---------- 

130 package_path : str 

131 Full path to the package to test. 

132 package_name : str 

133 Name of the package to test. 

134 

135 Parameters 

136 ---------- 

137 package : module, str or None, optional 

138 The package to test. If a string, this should be the full path to 

139 the package. If None (default), `package` is set to the module from 

140 which `NoseTester` is initialized. 

141 raise_warnings : None, str or sequence of warnings, optional 

142 This specifies which warnings to configure as 'raise' instead 

143 of being shown once during the test execution. Valid strings are: 

144 

145 - "develop" : equals ``(Warning,)`` 

146 - "release" : equals ``()``, don't raise on any warnings. 

147 

148 Default is "release". 

149 depth : int, optional 

150 If `package` is None, then this can be used to initialize from the 

151 module of the caller of (the caller of (...)) the code that 

152 initializes `NoseTester`. Default of 0 means the module of the 

153 immediate caller; higher values are useful for utility routines that 

154 want to initialize `NoseTester` objects on behalf of other code. 

155 

156 """ 

157 def __init__(self, package=None, raise_warnings="release", depth=0, 

158 check_fpu_mode=False): 

159 # Back-compat: 'None' used to mean either "release" or "develop" 

160 # depending on whether this was a release or develop version of 

161 # numpy. Those semantics were fine for testing numpy, but not so 

162 # helpful for downstream projects like scipy that use 

163 # numpy.testing. (They want to set this based on whether *they* are a 

164 # release or develop version, not whether numpy is.) So we continue to 

165 # accept 'None' for back-compat, but it's now just an alias for the 

166 # default "release". 

167 if raise_warnings is None: 

168 raise_warnings = "release" 

169 

170 package_name = None 

171 if package is None: 

172 f = sys._getframe(1 + depth) 

173 package_path = f.f_locals.get('__file__', None) 

174 if package_path is None: 

175 raise AssertionError 

176 package_path = os.path.dirname(package_path) 

177 package_name = f.f_locals.get('__name__', None) 

178 elif isinstance(package, type(os)): 

179 package_path = os.path.dirname(package.__file__) 

180 package_name = getattr(package, '__name__', None) 

181 else: 

182 package_path = str(package) 

183 

184 self.package_path = package_path 

185 

186 # Find the package name under test; this name is used to limit coverage 

187 # reporting (if enabled). 

188 if package_name is None: 

189 package_name = get_package_name(package_path) 

190 self.package_name = package_name 

191 

192 # Set to "release" in constructor in maintenance branches. 

193 self.raise_warnings = raise_warnings 

194 

195 # Whether to check for FPU mode changes 

196 self.check_fpu_mode = check_fpu_mode

198 def _test_argv(self, label, verbose, extra_argv): 

199 ''' Generate argv for nosetest command 

200 

201 Parameters 

202 ---------- 

203 label : {'fast', 'full', '', attribute identifier}, optional 

204 see ``test`` docstring 

205 verbose : int, optional 

206 Verbosity value for test outputs, in the range 1-10. Default is 1. 

207 extra_argv : list, optional 

208 List with any extra arguments to pass to nosetests. 

209 

210 Returns 

211 ------- 

212 argv : list 

213 command line arguments that will be passed to nose 

214 ''' 

215 argv = [__file__, self.package_path, '-s'] 

216 if label and label != 'full': 

217 if not isinstance(label, basestring): 

218 raise TypeError('Selection label should be a string') 

219 if label == 'fast': 

220 label = 'not slow' 

221 argv += ['-A', label] 

222 argv += ['--verbosity', str(verbose)] 

223 

224 # When installing with setuptools, and also in some other cases, the 

225 # test_*.py files end up marked +x executable. Nose, by default, does 

226 # not run files marked with +x as they might be scripts. However, in 

227 # our case nose only looks for test_*.py files under the package 

228 # directory, which should be safe. 

229 argv += ['--exe'] 

230 

231 if extra_argv: 

232 argv += extra_argv 

233 return argv

265 def prepare_test_args(self, label='fast', verbose=1, extra_argv=None, 

266 doctests=False, coverage=False, timer=False): 

267 """ 

268 Run tests for module using nose. 

269 

270 This method does the heavy lifting for the `test` method. It takes all 

271 the same arguments, for details see `test`. 

272 

273 See Also 

274 -------- 

275 test 

276 

277 """ 

278 # fail with nice error message if nose is not present 

279 import_nose() 

280 # compile argv 

281 argv = self._test_argv(label, verbose, extra_argv) 

282 # our way of doing coverage 

283 if coverage: 

284 argv += ['--cover-package=%s' % self.package_name, '--with-coverage', 

285 '--cover-tests', '--cover-erase'] 

286 

287 if timer: 

288 if timer is True: 

289 argv += ['--with-timer'] 

290 elif isinstance(timer, int): 

291 argv += ['--with-timer', '--timer-top-n', str(timer)] 

292 

293 # construct list of plugins 

294 import nose.plugins.builtin 

295 from nose.plugins import EntryPointPluginManager 

296 from .noseclasses import (KnownFailurePlugin, Unplugger, 

297 FPUModeCheckPlugin) 

298 plugins = [KnownFailurePlugin()] 

299 plugins += [p() for p in nose.plugins.builtin.plugins] 

300 if self.check_fpu_mode: 

301 plugins += [FPUModeCheckPlugin()] 

302 argv += ["--with-fpumodecheckplugin"] 

303 try: 

304 # External plugins (like nose-timer) 

305 entrypoint_manager = EntryPointPluginManager() 

306 entrypoint_manager.loadPlugins() 

307 plugins += [p for p in entrypoint_manager.plugins] 

308 except ImportError: 

309 # Relies on pkg_resources, not a hard dependency 

310 pass 

311 

312 # add doctesting if required 

313 doctest_argv = '--with-doctest' in argv 

314 if doctests == False and doctest_argv: 

315 doctests = True 

316 plug = self._get_custom_doctester() 

317 if plug is None: 

318 # use standard doctesting 

319 if doctests and not doctest_argv: 

320 argv += ['--with-doctest'] 

321 else: # custom doctesting 

322 if doctest_argv: # in fact the unplugger would take care of this 

323 argv.remove('--with-doctest') 

324 plugins += [Unplugger('doctest'), plug] 

325 if doctests: 

326 argv += ['--with-' + plug.name] 

327 return argv, plugins

329 def test(self, label='fast', verbose=1, extra_argv=None, 

330 doctests=False, coverage=False, raise_warnings=None, 

331 timer=False): 

332 """ 

333 Run tests for module using nose. 

334 

335 Parameters 

336 ---------- 

337 label : {'fast', 'full', '', attribute identifier}, optional 

338 Identifies the tests to run. This can be a string to pass to 

339 the nosetests executable with the '-A' option, or one of several 

340 special values. Special values are: 

341 

342 * 'fast' - the default - which corresponds to the ``nosetests -A`` 

343 option of 'not slow'. 

344 * 'full' - fast (as above) and slow tests as in the 

345 'no -A' option to nosetests - this is the same as ''. 

346 * None or '' - run all tests. 

347 * attribute_identifier - string passed directly to nosetests as '-A'. 

348 

349 verbose : int, optional 

350 Verbosity value for test outputs, in the range 1-10. Default is 1. 

351 extra_argv : list, optional 

352 List with any extra arguments to pass to nosetests. 

353 doctests : bool, optional 

354 If True, run doctests in module. Default is False. 

355 coverage : bool, optional 

356 If True, report coverage of NumPy code. Default is False. 

357 (This requires the 

358 `coverage module <https://nedbatchelder.com/code/modules/coveragehtml>`_). 

359 raise_warnings : None, str or sequence of warnings, optional 

360 This specifies which warnings to configure as 'raise' instead 

361 of being shown once during the test execution. Valid strings are: 

362 

363 * "develop" : equals ``(Warning,)`` 

364 * "release" : equals ``()``, do not raise on any warnings. 

365 timer : bool or int, optional 

366 Timing of individual tests with ``nose-timer`` (which needs to be 

367 installed). If True, time tests and report on all of them. 

368 If an integer (say ``N``), report timing results for ``N`` slowest 

369 tests. 

370 

371 Returns 

372 ------- 

373 result : object 

374 Returns the result of running the tests as a 

375 ``nose.result.TextTestResult`` object. 

376 

377 Notes 

378 ----- 

379 Each NumPy module exposes `test` in its namespace to run all tests for it. 

380 For example, to run all tests for numpy.lib: 

381 

382 >>> np.lib.test() #doctest: +SKIP 

383 

384 Examples 

385 -------- 

386 >>> result = np.lib.test() #doctest: +SKIP 

387 Running unit tests for numpy.lib 

388 ... 

389 Ran 976 tests in 3.933s 

390 

391 OK 

392 

393 >>> result.errors #doctest: +SKIP 

394 [] 

395 >>> result.knownfail #doctest: +SKIP 

396 [] 

397 """ 

398 

399 # cap verbosity at 3 because nose becomes *very* verbose beyond that 

400 verbose = min(verbose, 3) 

401 

402 from . import utils 

403 utils.verbose = verbose 

404 

405 argv, plugins = self.prepare_test_args( 

406 label, verbose, extra_argv, doctests, coverage, timer) 

407 

408 if doctests: 

409 print("Running unit tests and doctests for %s" % self.package_name) 

410 else: 

411 print("Running unit tests for %s" % self.package_name) 

412 

413 self._show_system_info() 

414 

415 # reset doctest state on every run 

416 import doctest 

417 doctest.master = None 

418 

419 if raise_warnings is None: 

420 raise_warnings = self.raise_warnings 

421 

422 _warn_opts = dict(develop=(Warning,), 

423 release=()) 

424 if isinstance(raise_warnings, basestring): 

425 raise_warnings = _warn_opts[raise_warnings] 

426 

427 with suppress_warnings("location") as sup: 

428 # Reset the warning filters to the default state, 

429 # so that running the tests is more repeatable. 

430 warnings.resetwarnings() 

431 # Set all warnings to 'warn', this is because the default 'once' 

432 # has the bad property of possibly shadowing later warnings. 

433 warnings.filterwarnings('always') 

434 # Force the requested warnings to raise 

435 for warningtype in raise_warnings: 

436 warnings.filterwarnings('error', category=warningtype) 

437 # Filter out annoying import messages. 

438 sup.filter(message='Not importing directory') 

439 sup.filter(message="numpy.dtype size changed") 

440 sup.filter(message="numpy.ufunc size changed") 

441 sup.filter(category=np.ModuleDeprecationWarning) 

442 # Filter out boolean '-' deprecation messages. This allows 

443 # older versions of scipy to test without a flood of messages. 

444 sup.filter(message=".*boolean negative.*") 

445 sup.filter(message=".*boolean subtract.*") 

446 # Filter out distutils cpu warnings (could be localized to 

447 # distutils tests). ASV has problems with top level import, 

448 # so fetch module for suppression here. 

449 with warnings.catch_warnings(): 

450 warnings.simplefilter("always") 

451 from ...distutils import cpuinfo 

452 sup.filter(category=UserWarning, module=cpuinfo) 

453 # See #7949: Filter out deprecation warnings due to the -3 flag to 

454 # python 2 

455 if sys.version_info.major == 2 and sys.py3kwarning: 

456 # This is very specific, so using the fragile module filter 

457 # is fine 

458 import threading 

459 sup.filter(DeprecationWarning, 

460 r"sys\.exc_clear\(\) not supported in 3\.x", 

461 module=threading) 

462 sup.filter(DeprecationWarning, message=r"in 3\.x, __setslice__") 

463 sup.filter(DeprecationWarning, message=r"in 3\.x, __getslice__") 

464 sup.filter(DeprecationWarning, message=r"buffer\(\) not supported in 3\.x") 

465 sup.filter(DeprecationWarning, message=r"CObject type is not supported in 3\.x") 

466 sup.filter(DeprecationWarning, message=r"comparing unequal types not supported in 3\.x") 

467 # Filter out some deprecation warnings inside nose 1.3.7 when run 

468 # on python 3.5b2. See 

469 # https://github.com/nose-devs/nose/issues/929 

470 # Note: it is hard to filter based on module for sup (lineno could 

471 # be implemented). 

472 warnings.filterwarnings("ignore", message=".*getargspec.*", 

473 category=DeprecationWarning, 

474 module=r"nose\.") 

475 

476 from .noseclasses import NumpyTestProgram 

477 

478 t = NumpyTestProgram(argv=argv, exit=False, plugins=plugins) 

479 

480 return t.result

482 def bench(self, label='fast', verbose=1, extra_argv=None): 

483 """ 

484 Run benchmarks for module using nose. 

485 

486 Parameters 

487 ---------- 

488 label : {'fast', 'full', '', attribute identifier}, optional 

489 Identifies the benchmarks to run. This can be a string to pass to 

490 the nosetests executable with the '-A' option, or one of several 

491 special values. Special values are: 

492 

493 * 'fast' - the default - which corresponds to the ``nosetests -A`` 

494 option of 'not slow'. 

495 * 'full' - fast (as above) and slow benchmarks as in the 

496 'no -A' option to nosetests - this is the same as ''. 

497 * None or '' - run all tests. 

498 * attribute_identifier - string passed directly to nosetests as '-A'. 

499 

500 verbose : int, optional 

501 Verbosity value for benchmark outputs, in the range 1-10. Default is 1. 

502 extra_argv : list, optional 

503 List with any extra arguments to pass to nosetests. 

504 

505 Returns 

506 ------- 

507 success : bool 

508 Returns True if running the benchmarks works, False if an error 

509 occurred. 

510 

511 Notes 

512 ----- 

513 Benchmarks are like tests, but have names starting with "bench" instead 

514 of "test", and can be found under the "benchmarks" sub-directory of the 

515 module. 

516 

517 Each NumPy module exposes `bench` in its namespace to run all benchmarks 

518 for it. 

519 

520 Examples 

521 -------- 

522 >>> success = np.lib.bench() #doctest: +SKIP 

523 Running benchmarks for numpy.lib 

524 ... 

525 using 562341 items: 

526 unique: 

527 0.11 

528 unique1d: 

529 0.11 

530 ratio: 1.0 

531 nUnique: 56230 == 56230 

532 ... 

533 OK 

534 

535 >>> success #doctest: +SKIP 

536 True 

537 

538 """ 

539 

540 print("Running benchmarks for %s" % self.package_name) 

541 self._show_system_info() 

542 

543 argv = self._test_argv(label, verbose, extra_argv) 

544 argv += ['--match', r'(?:^|[\\b_\\.%s-])[Bb]ench' % os.sep] 

545 

546 # import nose or make informative error 

547 nose = import_nose() 

548 

549 # get plugin to disable doctests 

550 from .noseclasses import Unplugger 

551 add_plugins = [Unplugger('doctest')] 

552 

553 return nose.run(argv=argv, addplugins=add_plugins)
