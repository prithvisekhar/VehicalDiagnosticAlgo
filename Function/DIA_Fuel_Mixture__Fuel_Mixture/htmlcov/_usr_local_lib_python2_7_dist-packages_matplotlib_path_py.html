26class Path(object): 

27 """ 

28 :class:`Path` represents a series of possibly disconnected, 

29 possibly closed, line and curve segments. 

30 

31 The underlying storage is made up of two parallel numpy arrays: 

32 - *vertices*: an Nx2 float array of vertices 

33 - *codes*: an N-length uint8 array of vertex types 

34 

35 These two arrays always have the same length in the first 

36 dimension. For example, to represent a cubic curve, you must 

37 provide three vertices as well as three codes ``CURVE3``. 

38 

39 The code types are: 

40 

41 - ``STOP`` : 1 vertex (ignored) 

42 A marker for the end of the entire path (currently not 

43 required and ignored) 

44 

45 - ``MOVETO`` : 1 vertex 

46 Pick up the pen and move to the given vertex. 

47 

48 - ``LINETO`` : 1 vertex 

49 Draw a line from the current position to the given vertex. 

50 

51 - ``CURVE3`` : 1 control point, 1 endpoint 

52 Draw a quadratic Bezier curve from the current position, 

53 with the given control point, to the given end point. 

54 

55 - ``CURVE4`` : 2 control points, 1 endpoint 

56 Draw a cubic Bezier curve from the current position, with 

57 the given control points, to the given end point. 

58 

59 - ``CLOSEPOLY`` : 1 vertex (ignored) 

60 Draw a line segment to the start point of the current 

61 polyline. 

62 

63 Users of Path objects should not access the vertices and codes 

64 arrays directly. Instead, they should use :meth:`iter_segments` 

65 or :meth:`cleaned` to get the vertex/code pairs. This is important, 

66 since many :class:`Path` objects, as an optimization, do not store a 

67 *codes* at all, but have a default one provided for them by 

68 :meth:`iter_segments`. 

69 

70 Some behavior of Path objects can be controlled by rcParams. See 

71 the rcParams whose keys contain 'path.'. 

72 

73 .. note:: 

74 

75 The vertices and codes arrays should be treated as 

76 immutable -- there are a number of optimizations and assumptions 

77 made up front in the constructor that will not change when the 

78 data changes. 

79 

80 """

101 def __init__(self, vertices, codes=None, _interpolation_steps=1, 

102 closed=False, readonly=False): 

103 """ 

104 Create a new path with the given vertices and codes. 

105 

106 Parameters 

107 ---------- 

108 vertices : array_like 

109 The ``(n, 2)`` float array, masked array or sequence of pairs 

110 representing the vertices of the path. 

111 

112 If *vertices* contains masked values, they will be converted 

113 to NaNs which are then handled correctly by the Agg 

114 PathIterator and other consumers of path data, such as 

115 :meth:`iter_segments`. 

116 codes : {None, array_like}, optional 

117 n-length array integers representing the codes of the path. 

118 If not None, codes must be the same length as vertices. 

119 If None, *vertices* will be treated as a series of line segments. 

120 _interpolation_steps : int, optional 

121 Used as a hint to certain projections, such as Polar, that this 

122 path should be linearly interpolated immediately before drawing. 

123 This attribute is primarily an implementation detail and is not 

124 intended for public use. 

125 closed : bool, optional 

126 If *codes* is None and closed is True, vertices will be treated as 

127 line segments of a closed polygon. 

128 readonly : bool, optional 

129 Makes the path behave in an immutable way and sets the vertices 

130 and codes as read-only arrays. 

131 """ 

132 vertices = _to_unmasked_float_array(vertices) 

133 if (vertices.ndim != 2) or (vertices.shape[1] != 2): 

134 raise ValueError( 

135 "'vertices' must be a 2D list or array with shape Nx2") 

136 

137 if codes is not None: 

138 codes = np.asarray(codes, self.code_type) 

139 if (codes.ndim != 1) or len(codes) != len(vertices): 

140 raise ValueError("'codes' must be a 1D list or array with the " 

141 "same length of 'vertices'") 

142 if len(codes) and codes[0] != self.MOVETO: 

143 raise ValueError("The first element of 'code' must be equal " 

144 "to 'MOVETO' ({})".format(self.MOVETO)) 

145 elif closed: 

146 codes = np.empty(len(vertices), dtype=self.code_type) 

147 codes[0] = self.MOVETO 

148 codes[1:-1] = self.LINETO 

149 codes[-1] = self.CLOSEPOLY 

150 

151 self._vertices = vertices 

152 self._codes = codes 

153 self._interpolation_steps = _interpolation_steps 

154 self._update_values() 

155 

156 if readonly: 

157 self._vertices.flags.writeable = False 

158 if self._codes is not None: 

159 self._codes.flags.writeable = False 

160 self._readonly = True 

161 else: 

162 self._readonly = False

303 @classmethod 

304 def make_compound_path_from_polys(cls, XY): 

305 """ 

306 Make a compound path object to draw a number 

307 of polygons with equal numbers of sides XY is a (numpolys x 

308 numsides x 2) numpy array of vertices. Return object is a 

309 :class:`Path` 

310 

311 .. plot:: gallery/api/histogram_path.py 

312 

313 """ 

314 

315 # for each poly: 1 for the MOVETO, (numsides-1) for the LINETO, 1 for 

316 # the CLOSEPOLY; the vert for the closepoly is ignored but we still 

317 # need it to keep the codes aligned with the vertices 

318 numpolys, numsides, two = XY.shape 

319 if two != 2: 

320 raise ValueError("The third dimension of 'XY' must be 2") 

321 stride = numsides + 1 

322 nverts = numpolys * stride 

323 verts = np.zeros((nverts, 2)) 

324 codes = np.ones(nverts, int) * cls.LINETO 

325 codes[0::stride] = cls.MOVETO 

326 codes[numsides::stride] = cls.CLOSEPOLY 

327 for i in range(numsides): 

328 verts[i::stride] = XY[:, i] 

329 

330 return cls(verts, codes)

332 @classmethod 

333 def make_compound_path(cls, *args): 

334 """Make a compound path from a list of Path objects.""" 

335 # Handle an empty list in args (i.e. no args). 

336 if not args: 

337 return Path(np.empty([0, 2], dtype=np.float32)) 

338 

339 lengths = [len(x) for x in args] 

340 total_length = sum(lengths) 

341 

342 vertices = np.vstack([x.vertices for x in args]) 

343 vertices.reshape((total_length, 2)) 

344 

345 codes = np.empty(total_length, dtype=cls.code_type) 

346 i = 0 

347 for path in args: 

348 if path.codes is None: 

349 codes[i] = cls.MOVETO 

350 codes[i + 1:i + len(path.vertices)] = cls.LINETO 

351 else: 

352 codes[i:i + len(path.codes)] = path.codes 

353 i += len(path.vertices) 

354 

355 return cls(vertices, codes)

363 def iter_segments(self, transform=None, remove_nans=True, clip=None, 

364 snap=False, stroke_width=1.0, simplify=None, 

365 curves=True, sketch=None): 

366 """ 

367 Iterates over all of the curve segments in the path. Each 

368 iteration returns a 2-tuple (*vertices*, *code*), where 

369 *vertices* is a sequence of 1 - 3 coordinate pairs, and *code* is 

370 one of the :class:`Path` codes. 

371 

372 Additionally, this method can provide a number of standard 

373 cleanups and conversions to the path. 

374 

375 Parameters 

376 ---------- 

377 transform : None or :class:`~matplotlib.transforms.Transform` instance 

378 If not None, the given affine transformation will 

379 be applied to the path. 

380 remove_nans : {False, True}, optional 

381 If True, will remove all NaNs from the path and 

382 insert MOVETO commands to skip over them. 

383 clip : None or sequence, optional 

384 If not None, must be a four-tuple (x1, y1, x2, y2) 

385 defining a rectangle in which to clip the path. 

386 snap : None or bool, optional 

387 If None, auto-snap to pixels, to reduce 

388 fuzziness of rectilinear lines. If True, force snapping, and 

389 if False, don't snap. 

390 stroke_width : float, optional 

391 The width of the stroke being drawn. Needed 

392 as a hint for the snapping algorithm. 

393 simplify : None or bool, optional 

394 If True, perform simplification, to remove 

395 vertices that do not affect the appearance of the path. If 

396 False, perform no simplification. If None, use the 

397 should_simplify member variable. See also the rcParams 

398 path.simplify and path.simplify_threshold. 

399 curves : {True, False}, optional 

400 If True, curve segments will be returned as curve 

401 segments. If False, all curves will be converted to line 

402 segments. 

403 sketch : None or sequence, optional 

404 If not None, must be a 3-tuple of the form 

405 (scale, length, randomness), representing the sketch 

406 parameters. 

407 """ 

408 if not len(self): 

409 return 

410 

411 cleaned = self.cleaned(transform=transform, 

412 remove_nans=remove_nans, clip=clip, 

413 snap=snap, stroke_width=stroke_width, 

414 simplify=simplify, curves=curves, 

415 sketch=sketch) 

416 vertices = cleaned.vertices 

417 codes = cleaned.codes 

418 len_vertices = vertices.shape[0] 

419 

420 # Cache these object lookups for performance in the loop. 

421 NUM_VERTICES_FOR_CODE = self.NUM_VERTICES_FOR_CODE 

422 STOP = self.STOP 

423 

424 i = 0 

425 while i < len_vertices: 

426 code = codes[i] 

427 if code == STOP: 

428 return 

429 else: 

430 num_vertices = NUM_VERTICES_FOR_CODE[code] 

431 curr_vertices = vertices[i:i+num_vertices].flatten() 

432 yield curr_vertices, code 

433 i += num_vertices

580 def to_polygons(self, transform=None, width=0, height=0, closed_only=True): 

581 """ 

582 Convert this path to a list of polygons or polylines. Each 

583 polygon/polyline is an Nx2 array of vertices. In other words, 

584 each polygon has no ``MOVETO`` instructions or curves. This 

585 is useful for displaying in backends that do not support 

586 compound paths or Bezier curves, such as GDK. 

587 

588 If *width* and *height* are both non-zero then the lines will 

589 be simplified so that vertices outside of (0, 0), (width, 

590 height) will be clipped. 

591 

592 If *closed_only* is `True` (default), only closed polygons, 

593 with the last point being the same as the first point, will be 

594 returned. Any unclosed polylines in the path will be 

595 explicitly closed. If *closed_only* is `False`, any unclosed 

596 polygons in the path will be returned as unclosed polygons, 

597 and the closed polygons will be returned explicitly closed by 

598 setting the last point to the same as the first point. 

599 """ 

600 if len(self.vertices) == 0: 

601 return [] 

602 

603 if transform is not None: 

604 transform = transform.frozen() 

605 

606 if self.codes is None and (width == 0 or height == 0): 

607 vertices = self.vertices 

608 if closed_only: 

609 if len(vertices) < 3: 

610 return [] 

611 elif np.any(vertices[0] != vertices[-1]): 

612 vertices = list(vertices) + [vertices[0]] 

613 

614 if transform is None: 

615 return [vertices] 

616 else: 

617 return [transform.transform(vertices)] 

618 

619 # Deal with the case where there are curves and/or multiple 

620 # subpaths (using extension code) 

621 return _path.convert_path_to_polygons( 

622 self, transform, width, height, closed_only)

645 """

673 def unit_regular_star(cls, numVertices, innerCircle=0.5): 

674 """ 

675 Return a :class:`Path` for a unit regular star 

676 with the given numVertices and radius of 1.0, centered at (0, 

677 0). 

678 """ 

679 if numVertices <= 16: 

680 path = cls._unit_regular_stars.get((numVertices, innerCircle)) 

681 else: 

682 path = None 

683 if path is None: 

684 ns2 = numVertices * 2 

685 theta = (2*np.pi/ns2 * np.arange(ns2 + 1)) 

686 # This initial rotation is to make sure the polygon always 

687 # "points-up" 

688 theta += np.pi / 2.0 

689 r = np.ones(ns2 + 1) 

690 r[1::2] = innerCircle 

691 verts = np.vstack((r*np.cos(theta), r*np.sin(theta))).transpose() 

692 codes = np.empty((ns2 + 1,)) 

693 codes[0] = cls.MOVETO 

694 codes[1:-1] = cls.LINETO 

695 codes[-1] = cls.CLOSEPOLY 

696 path = cls(verts, codes, readonly=True) 

697 if numVertices <= 16: 

698 cls._unit_regular_stars[(numVertices, innerCircle)] = path 

699 return path

758 [SQRTHALF, -SQRTHALF], 

759

761 [1.0, -MAGIC], 

762 [1.0, 0.0], 

763

844 @classmethod 

845 def arc(cls, theta1, theta2, n=None, is_wedge=False): 

846 """ 

847 Return an arc on the unit circle from angle 

848 *theta1* to angle *theta2* (in degrees). 

849 

850 *theta2* is unwrapped to produce the shortest arc within 360 degrees. 

851 That is, if *theta2* > *theta1* + 360, the arc will be from *theta1* to 

852 *theta2* - 360 and not a full circle plus some extra overlap. 

853 

854 If *n* is provided, it is the number of spline segments to make. 

855 If *n* is not provided, the number of spline segments is 

856 determined based on the delta between *theta1* and *theta2*. 

857 

858 Masionobe, L. 2003. `Drawing an elliptical arc using 

859 polylines, quadratic or cubic Bezier curves 

860 <http://www.spaceroots.org/documents/ellipse/index.html>`_. 

861 """ 

862 halfpi = np.pi * 0.5 

863 

864 eta1 = theta1 

865 eta2 = theta2 - 360 * np.floor((theta2 - theta1) / 360) 

866 # Ensure 2pi range is not flattened to 0 due to floating-point errors, 

867 # but don't try to expand existing 0 range. 

868 if theta2 != theta1 and eta2 <= eta1: 

869 eta2 += 360 

870 eta1, eta2 = np.deg2rad([eta1, eta2]) 

871 

872 # number of curve segments to make 

873 if n is None: 

874 n = int(2 ** np.ceil((eta2 - eta1) / halfpi)) 

875 if n < 1: 

876 raise ValueError("n must be >= 1 or None") 

877 

878 deta = (eta2 - eta1) / n 

879 t = np.tan(0.5 * deta) 

880 alpha = np.sin(deta) * (np.sqrt(4.0 + 3.0 * t * t) - 1) / 3.0 

881 

882 steps = np.linspace(eta1, eta2, n + 1, True) 

883 cos_eta = np.cos(steps) 

884 sin_eta = np.sin(steps) 

885 

886 xA = cos_eta[:-1] 

887 yA = sin_eta[:-1] 

888 xA_dot = -yA 

889 yA_dot = xA 

890 

891 xB = cos_eta[1:] 

892 yB = sin_eta[1:] 

893 xB_dot = -yB 

894 yB_dot = xB 

895 

896 if is_wedge: 

897 length = n * 3 + 4 

898 vertices = np.zeros((length, 2), float) 

899 codes = cls.CURVE4 * np.ones((length, ), cls.code_type) 

900 vertices[1] = [xA[0], yA[0]] 

901 codes[0:2] = [cls.MOVETO, cls.LINETO] 

902 codes[-2:] = [cls.LINETO, cls.CLOSEPOLY] 

903 vertex_offset = 2 

904 end = length - 2 

905 else: 

906 length = n * 3 + 1 

907 vertices = np.empty((length, 2), float) 

908 codes = cls.CURVE4 * np.ones((length, ), cls.code_type) 

909 vertices[0] = [xA[0], yA[0]] 

910 codes[0] = cls.MOVETO 

911 vertex_offset = 1 

912 end = length 

913 

914 vertices[vertex_offset:end:3, 0] = xA + alpha * xA_dot 

915 vertices[vertex_offset:end:3, 1] = yA + alpha * yA_dot 

916 vertices[vertex_offset+1:end:3, 0] = xB - alpha * xB_dot 

917 vertices[vertex_offset+1:end:3, 1] = yB - alpha * yB_dot 

918 vertices[vertex_offset+2:end:3, 0] = xB 

919 vertices[vertex_offset+2:end:3, 1] = yB 

920 

921 return cls(vertices, codes, readonly=True)
