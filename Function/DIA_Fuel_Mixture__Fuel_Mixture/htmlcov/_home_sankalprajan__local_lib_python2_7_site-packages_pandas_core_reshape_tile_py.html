23def cut(x, bins, right=True, labels=None, retbins=False, precision=3, 

24 include_lowest=False, duplicates='raise'): 

25 """ 

26 Bin values into discrete intervals. 

27 

28 Use `cut` when you need to segment and sort data values into bins. This 

29 function is also useful for going from a continuous variable to a 

30 categorical variable. For example, `cut` could convert ages to groups of 

31 age ranges. Supports binning into an equal number of bins, or a 

32 pre-specified array of bins. 

33 

34 Parameters 

35 ---------- 

36 x : array-like 

37 The input array to be binned. Must be 1-dimensional. 

38 bins : int, sequence of scalars, or pandas.IntervalIndex 

39 The criteria to bin by. 

40 

41 * int : Defines the number of equal-width bins in the range of `x`. The 

42 range of `x` is extended by .1% on each side to include the minimum 

43 and maximum values of `x`. 

44 * sequence of scalars : Defines the bin edges allowing for non-uniform 

45 width. No extension of the range of `x` is done. 

46 * IntervalIndex : Defines the exact bins to be used. Note that 

47 IntervalIndex for `bins` must be non-overlapping. 

48 

49 right : bool, default True 

50 Indicates whether `bins` includes the rightmost edge or not. If 

51 ``right == True`` (the default), then the `bins` ``[1, 2, 3, 4]`` 

52 indicate (1,2], (2,3], (3,4]. This argument is ignored when 

53 `bins` is an IntervalIndex. 

54 labels : array or bool, optional 

55 Specifies the labels for the returned bins. Must be the same length as 

56 the resulting bins. If False, returns only integer indicators of the 

57 bins. This affects the type of the output container (see below). 

58 This argument is ignored when `bins` is an IntervalIndex. 

59 retbins : bool, default False 

60 Whether to return the bins or not. Useful when bins is provided 

61 as a scalar. 

62 precision : int, default 3 

63 The precision at which to store and display the bins labels. 

64 include_lowest : bool, default False 

65 Whether the first interval should be left-inclusive or not. 

66 duplicates : {default 'raise', 'drop'}, optional 

67 If bin edges are not unique, raise ValueError or drop non-uniques. 

68 

69 .. versionadded:: 0.23.0 

70 

71 Returns 

72 ------- 

73 out : pandas.Categorical, Series, or ndarray 

74 An array-like object representing the respective bin for each value 

75 of `x`. The type depends on the value of `labels`. 

76 

77 * True (default) : returns a Series for Series `x` or a 

78 pandas.Categorical for all other inputs. The values stored within 

79 are Interval dtype. 

80 

81 * sequence of scalars : returns a Series for Series `x` or a 

82 pandas.Categorical for all other inputs. The values stored within 

83 are whatever the type in the sequence is. 

84 

85 * False : returns an ndarray of integers. 

86 

87 bins : numpy.ndarray or IntervalIndex. 

88 The computed or specified bins. Only returned when `retbins=True`. 

89 For scalar or sequence `bins`, this is an ndarray with the computed 

90 bins. If set `duplicates=drop`, `bins` will drop non-unique bin. For 

91 an IntervalIndex `bins`, this is equal to `bins`. 

92 

93 See Also 

94 -------- 

95 qcut : Discretize variable into equal-sized buckets based on rank 

96 or based on sample quantiles. 

97 pandas.Categorical : Array type for storing data that come from a 

98 fixed set of values. 

99 Series : One-dimensional array with axis labels (including time series). 

100 pandas.IntervalIndex : Immutable Index implementing an ordered, 

101 sliceable set. 

102 

103 Notes 

104 ----- 

105 Any NA values will be NA in the result. Out of bounds values will be NA in 

106 the resulting Series or pandas.Categorical object. 

107 

108 Examples 

109 -------- 

110 Discretize into three equal-sized bins. 

111 

112 >>> pd.cut(np.array([1, 7, 5, 4, 6, 3]), 3) 

113 ... # doctest: +ELLIPSIS 

114 [(0.994, 3.0], (5.0, 7.0], (3.0, 5.0], (3.0, 5.0], (5.0, 7.0], ... 

115 Categories (3, interval[float64]): [(0.994, 3.0] < (3.0, 5.0] ... 

116 

117 >>> pd.cut(np.array([1, 7, 5, 4, 6, 3]), 3, retbins=True) 

118 ... # doctest: +ELLIPSIS 

119 ([(0.994, 3.0], (5.0, 7.0], (3.0, 5.0], (3.0, 5.0], (5.0, 7.0], ... 

120 Categories (3, interval[float64]): [(0.994, 3.0] < (3.0, 5.0] ... 

121 array([0.994, 3. , 5. , 7. ])) 

122 

123 Discovers the same bins, but assign them specific labels. Notice that 

124 the returned Categorical's categories are `labels` and is ordered. 

125 

126 >>> pd.cut(np.array([1, 7, 5, 4, 6, 3]), 

127 ... 3, labels=["bad", "medium", "good"]) 

128 [bad, good, medium, medium, good, bad] 

129 Categories (3, object): [bad < medium < good] 

130 

131 ``labels=False`` implies you just want the bins back. 

132 

133 >>> pd.cut([0, 1, 1, 2], bins=4, labels=False) 

134 array([0, 1, 1, 3]) 

135 

136 Passing a Series as an input returns a Series with categorical dtype: 

137 

138 >>> s = pd.Series(np.array([2, 4, 6, 8, 10]), 

139 ... index=['a', 'b', 'c', 'd', 'e']) 

140 >>> pd.cut(s, 3) 

141 ... # doctest: +ELLIPSIS 

142 a (1.992, 4.667] 

143 b (1.992, 4.667] 

144 c (4.667, 7.333] 

145 d (7.333, 10.0] 

146 e (7.333, 10.0] 

147 dtype: category 

148 Categories (3, interval[float64]): [(1.992, 4.667] < (4.667, ... 

149 

150 Passing a Series as an input returns a Series with mapping value. 

151 It is used to map numerically to intervals based on bins. 

152 

153 >>> s = pd.Series(np.array([2, 4, 6, 8, 10]), 

154 ... index=['a', 'b', 'c', 'd', 'e']) 

155 >>> pd.cut(s, [0, 2, 4, 6, 8, 10], labels=False, retbins=True, right=False) 

156 ... # doctest: +ELLIPSIS 

157 (a 0.0 

158 b 1.0 

159 c 2.0 

160 d 3.0 

161 e 4.0 

162 dtype: float64, array([0, 2, 4, 6, 8])) 

163 

164 Use `drop` optional when bins is not unique 

165 

166 >>> pd.cut(s, [0, 2, 4, 6, 10, 10], labels=False, retbins=True, 

167 ... right=False, duplicates='drop') 

168 ... # doctest: +ELLIPSIS 

169 (a 0.0 

170 b 1.0 

171 c 2.0 

172 d 3.0 

173 e 3.0 

174 dtype: float64, array([0, 2, 4, 6, 8])) 

175 

176 Passing an IntervalIndex for `bins` results in those categories exactly. 

177 Notice that values not covered by the IntervalIndex are set to NaN. 0 

178 is to the left of the first bin (which is closed on the right), and 1.5 

179 falls between two bins. 

180 

181 >>> bins = pd.IntervalIndex.from_tuples([(0, 1), (2, 3), (4, 5)]) 

182 >>> pd.cut([0, 0.5, 1.5, 2.5, 4.5], bins) 

183 [NaN, (0, 1], NaN, (2, 3], (4, 5]] 

184 Categories (3, interval[int64]): [(0, 1] < (2, 3] < (4, 5]] 

185 """ 

186 # NOTE: this binning code is changed a bit from histogram for var(x) == 0 

187 

188 # for handling the cut for datetime and timedelta objects 

189 x_is_series, series_index, name, x = _preprocess_for_cut(x) 

190 x, dtype = _coerce_to_type(x) 

191 

192 if not np.iterable(bins): 

193 if is_scalar(bins) and bins < 1: 

194 raise ValueError("`bins` should be a positive integer.") 

195 

196 try: # for array-like 

197 sz = x.size 

198 except AttributeError: 

199 x = np.asarray(x) 

200 sz = x.size 

201 

202 if sz == 0: 

203 raise ValueError('Cannot cut empty array') 

204 

205 rng = (nanops.nanmin(x), nanops.nanmax(x)) 

206 mn, mx = [mi + 0.0 for mi in rng] 

207 

208 if np.isinf(mn) or np.isinf(mx): 

209 # GH 24314 

210 raise ValueError('cannot specify integer `bins` when input data ' 

211 'contains infinity') 

212 elif mn == mx: # adjust end points before binning 

213 mn -= .001 * abs(mn) if mn != 0 else .001 

214 mx += .001 * abs(mx) if mx != 0 else .001 

215 bins = np.linspace(mn, mx, bins + 1, endpoint=True) 

216 else: # adjust end points after binning 

217 bins = np.linspace(mn, mx, bins + 1, endpoint=True) 

218 adj = (mx - mn) * 0.001 # 0.1% of the range 

219 if right: 

220 bins[0] -= adj 

221 else: 

222 bins[-1] += adj 

223 

224 elif isinstance(bins, IntervalIndex): 

225 if bins.is_overlapping: 

226 raise ValueError('Overlapping IntervalIndex is not accepted.') 

227 

228 else: 

229 if is_datetime64tz_dtype(bins): 

230 bins = np.asarray(bins, dtype=_NS_DTYPE) 

231 else: 

232 bins = np.asarray(bins) 

233 bins = _convert_bin_to_numeric_type(bins, dtype) 

234 if (np.diff(bins) < 0).any(): 

235 raise ValueError('bins must increase monotonically.') 

236 

237 fac, bins = _bins_to_cuts(x, bins, right=right, labels=labels, 

238 precision=precision, 

239 include_lowest=include_lowest, 

240 dtype=dtype, 

241 duplicates=duplicates) 

242 

243 return _postprocess_for_cut(fac, bins, retbins, x_is_series, 

244 series_index, name, dtype)

247def qcut(x, q, labels=None, retbins=False, precision=3, duplicates='raise'): 

248 """ 

249 Quantile-based discretization function. Discretize variable into 

250 equal-sized buckets based on rank or based on sample quantiles. For example 

251 1000 values for 10 quantiles would produce a Categorical object indicating 

252 quantile membership for each data point. 

253 

254 Parameters 

255 ---------- 

256 x : 1d ndarray or Series 

257 q : integer or array of quantiles 

258 Number of quantiles. 10 for deciles, 4 for quartiles, etc. Alternately 

259 array of quantiles, e.g. [0, .25, .5, .75, 1.] for quartiles 

260 labels : array or boolean, default None 

261 Used as labels for the resulting bins. Must be of the same length as 

262 the resulting bins. If False, return only integer indicators of the 

263 bins. 

264 retbins : bool, optional 

265 Whether to return the (bins, labels) or not. Can be useful if bins 

266 is given as a scalar. 

267 precision : int, optional 

268 The precision at which to store and display the bins labels 

269 duplicates : {default 'raise', 'drop'}, optional 

270 If bin edges are not unique, raise ValueError or drop non-uniques. 

271 

272 .. versionadded:: 0.20.0 

273 

274 Returns 

275 ------- 

276 out : Categorical or Series or array of integers if labels is False 

277 The return type (Categorical or Series) depends on the input: a Series 

278 of type category if input is a Series else Categorical. Bins are 

279 represented as categories when categorical data is returned. 

280 bins : ndarray of floats 

281 Returned only if `retbins` is True. 

282 

283 Notes 

284 ----- 

285 Out of bounds values will be NA in the resulting Categorical object 

286 

287 Examples 

288 -------- 

289 >>> pd.qcut(range(5), 4) 

290 ... # doctest: +ELLIPSIS 

291 [(-0.001, 1.0], (-0.001, 1.0], (1.0, 2.0], (2.0, 3.0], (3.0, 4.0]] 

292 Categories (4, interval[float64]): [(-0.001, 1.0] < (1.0, 2.0] ... 

293 

294 >>> pd.qcut(range(5), 3, labels=["good", "medium", "bad"]) 

295 ... # doctest: +SKIP 

296 [good, good, medium, bad, bad] 

297 Categories (3, object): [good < medium < bad] 

298 

299 >>> pd.qcut(range(5), 4, labels=False) 

300 array([0, 0, 1, 2, 3]) 

301 """ 

302 x_is_series, series_index, name, x = _preprocess_for_cut(x) 

303 

304 x, dtype = _coerce_to_type(x) 

305 

306 if is_integer(q): 

307 quantiles = np.linspace(0, 1, q + 1) 

308 else: 

309 quantiles = q 

310 bins = algos.quantile(x, quantiles) 

311 fac, bins = _bins_to_cuts(x, bins, labels=labels, 

312 precision=precision, include_lowest=True, 

313 dtype=dtype, duplicates=duplicates) 

314 

315 return _postprocess_for_cut(fac, bins, retbins, x_is_series, 

316 series_index, name, dtype)

319def _bins_to_cuts(x, bins, right=True, labels=None, 

320 precision=3, include_lowest=False, 

321 dtype=None, duplicates='raise'): 

322 

323 if duplicates not in ['raise', 'drop']: 

324 raise ValueError("invalid value for 'duplicates' parameter, " 

325 "valid options are: raise, drop") 

326 

327 if isinstance(bins, IntervalIndex): 

328 # we have a fast-path here 

329 ids = bins.get_indexer(x) 

330 result = algos.take_nd(bins, ids) 

331 result = Categorical(result, categories=bins, ordered=True) 

332 return result, bins 

333 

334 unique_bins = algos.unique(bins) 

335 if len(unique_bins) < len(bins) and len(bins) != 2: 

336 if duplicates == 'raise': 

337 raise ValueError("Bin edges must be unique: {bins!r}.\nYou " 

338 "can drop duplicate edges by setting " 

339 "the 'duplicates' kwarg".format(bins=bins)) 

340 else: 

341 bins = unique_bins 

342 

343 side = 'left' if right else 'right' 

344 ids = ensure_int64(bins.searchsorted(x, side=side)) 

345 

346 if include_lowest: 

347 ids[x == bins[0]] = 1 

348 

349 na_mask = isna(x) | (ids == len(bins)) | (ids == 0) 

350 has_nas = na_mask.any() 

351 

352 if labels is not False: 

353 if labels is None: 

354 labels = _format_labels(bins, precision, right=right, 

355 include_lowest=include_lowest, 

356 dtype=dtype) 

357 else: 

358 if len(labels) != len(bins) - 1: 

359 raise ValueError('Bin labels must be one fewer than ' 

360 'the number of bin edges') 

361 if not is_categorical_dtype(labels): 

362 labels = Categorical(labels, categories=labels, ordered=True) 

363 

364 np.putmask(ids, na_mask, 0) 

365 result = algos.take_nd(labels, ids - 1) 

366 

367 else: 

368 result = ids - 1 

369 if has_nas: 

370 result = result.astype(np.float64) 

371 np.putmask(result, na_mask, np.nan) 

372 

373 return result, bins

384def _coerce_to_type(x): 

385 """ 

386 if the passed data is of datetime/timedelta type, 

387 this method converts it to numeric so that cut method can 

388 handle it 

389 """ 

390 dtype = None

408def _convert_bin_to_numeric_type(bins, dtype): 

409 """ 

410 if the passed bin is of datetime/timedelta type, 

411 this method converts it to integer 

412 

413 Parameters 

414 ---------- 

415 bins : list-like of bins 

416 dtype : dtype of data 

417 

418 Raises 

419 ------ 

420 ValueError if bins are not of a compat dtype to dtype 

421 """ 

422 bins_dtype = infer_dtype(bins, skipna=False) 

423 if is_timedelta64_dtype(dtype): 

424 if bins_dtype in ['timedelta', 'timedelta64']: 

425 bins = to_timedelta(bins).view(np.int64) 

426 else: 

427 raise ValueError("bins must be of timedelta64 dtype") 

428 elif is_datetime64_dtype(dtype) or is_datetime64tz_dtype(dtype): 

429 if bins_dtype in ['datetime', 'datetime64']: 

430 bins = to_datetime(bins).view(np.int64) 

431 else: 

432 raise ValueError("bins must be of datetime64 dtype") 

433 

434 return bins

437def _convert_bin_to_datelike_type(bins, dtype): 

438 """ 

439 Convert bins to a DatetimeIndex or TimedeltaIndex if the orginal dtype is

460def _format_labels(bins, precision, right=True, 

461 include_lowest=False, dtype=None): 

462 """ based on the dtype, return our labels """ 

463 

464 closed = 'right' if right else 'left' 

465 

466 if is_datetime64tz_dtype(dtype): 

467 formatter = partial(Timestamp, tz=dtype.tz) 

468 adjust = lambda x: x - Timedelta('1ns') 

469 elif is_datetime64_dtype(dtype): 

470 formatter = Timestamp 

471 adjust = lambda x: x - Timedelta('1ns') 

472 elif is_timedelta64_dtype(dtype): 

473 formatter = Timedelta 

474 adjust = lambda x: x - Timedelta('1ns') 

475 else: 

476 precision = _infer_precision(precision, bins) 

477 formatter = lambda x: _round_frac(x, precision) 

478 adjust = lambda x: x - 10 ** (-precision) 

479 

480 breaks = [formatter(b) for b in bins] 

481 labels = IntervalIndex.from_breaks(breaks, closed=closed) 

482 

483 if right and include_lowest: 

484 # we will adjust the left hand side by precision to 

485 # account that we are all right closed 

486 v = adjust(labels[0].left) 

487 

488 i = IntervalIndex([Interval(v, labels[0].right, closed='right')]) 

489 labels = i.append(labels[1:]) 

490 

491 return labels

495 """ 

496 handles preprocessing for cut where we convert passed

537def _round_frac(x, precision): 

538 """ 

539 Round the fractional part of the given number 

540 """ 

541 if not np.isfinite(x) or x == 0: 

542 return x 

543 else: 

544 frac, whole = np.modf(x) 

545 if whole == 0: 

546 digits = -int(np.floor(np.log10(abs(frac)))) - 1 + precision 

547 else: 

548 digits = precision 

549 return np.around(x, digits)

552def _infer_precision(base_precision, bins): 

553 """Infer an appropriate precision for _round_frac 

554 """ 

555 for precision in range(base_precision, 20): 

556 levels = [_round_frac(b, precision) for b in bins] 

557 if algos.unique(levels).size == bins.size: 

558 return precision 

559 return base_precision # default
