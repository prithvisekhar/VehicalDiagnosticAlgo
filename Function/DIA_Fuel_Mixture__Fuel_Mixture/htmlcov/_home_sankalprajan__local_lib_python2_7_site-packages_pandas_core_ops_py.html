66def _maybe_match_name(a, b): 

67 """ 

68 Try to find a name to attach to the result of an operation between 

69 a and b. If only one of these has a `name` attribute, return that 

70 name. Otherwise return a consensus name if they match of None if 

71 they have different names. 

72 

73 Parameters 

74 ---------- 

75 a : object 

76 b : object 

77 

78 Returns 

79 ------- 

80 name : str or None 

81 

82 See Also 

83 -------- 

84 pandas.core.common.consensus_name_attr 

85 """ 

86 a_has = hasattr(a, 'name') 

87 b_has = hasattr(b, 'name') 

88 if a_has and b_has: 

89 if a.name == b.name: 

90 return a.name 

91 else: 

92 # TODO: what if they both have np.nan for their names? 

93 return None 

94 elif a_has: 

95 return a.name 

96 elif b_has: 

97 return b.name 

98 return None

101def maybe_upcast_for_op(obj): 

102 """ 

103 Cast non-pandas objects to pandas types to unify behavior of arithmetic 

104 and comparison operations. 

105 

106 Parameters 

107 ---------- 

108 obj: object 

109 

110 Returns 

111 ------- 

112 out : object 

113 

114 Notes 

115 ----- 

116 Be careful to call this *after* determining the `name` attribute to be 

117 attached to the result of the arithmetic operation. 

118 """ 

119 if type(obj) is datetime.timedelta: 

120 # GH#22390 cast up to Timedelta to rely on Timedelta 

121 # implementation; otherwise operation against numeric-dtype 

122 # raises TypeError 

123 return pd.Timedelta(obj) 

124 elif isinstance(obj, np.timedelta64) and not isna(obj): 

125 # In particular non-nanosecond timedelta64 needs to be cast to 

126 # nanoseconds, or else we get undesired behavior like 

127 # np.timedelta64(3, 'D') / 2 == np.timedelta64(1, 'D') 

128 # The isna check is to avoid casting timedelta64("NaT"), which would 

129 # return NaT and incorrectly be treated as a datetime-NaT. 

130 return pd.Timedelta(obj) 

131 elif isinstance(obj, np.ndarray) and is_timedelta64_dtype(obj): 

132 # GH#22390 Unfortunately we need to special-case right-hand 

133 # timedelta64 dtypes because numpy casts integer dtypes to 

134 # timedelta64 when operating with timedelta64 

135 return pd.TimedeltaIndex(obj) 

136 return obj

905def fill_binop(left, right, fill_value): 

906 """ 

907 If a non-None fill_value is given, replace null entries in left and right

1102def dispatch_to_series(left, right, func, str_rep=None, axis=None): 

1103 """ 

1104 Evaluate the frame operation func(left, right) by evaluating 

1105 column-by-column, dispatching to the Series implementation. 

1106 

1107 Parameters 

1108 ---------- 

1109 left : DataFrame 

1110 right : scalar or DataFrame 

1111 func : arithmetic or comparison operator 

1112 str_rep : str or None, default None 

1113 axis : {None, 0, 1, "index", "columns"} 

1114 

1115 Returns 

1116 ------- 

1117 DataFrame 

1118 """ 

1119 # Note: we use iloc to access columns for compat with cases 

1120 # with non-unique columns. 

1121 import pandas.core.computation.expressions as expressions 

1122 

1123 right = lib.item_from_zerodim(right) 

1124 if lib.is_scalar(right) or np.ndim(right) == 0: 

1125 

1126 def column_op(a, b): 

1127 return {i: func(a.iloc[:, i], b) 

1128 for i in range(len(a.columns))} 

1129 

1130 elif isinstance(right, ABCDataFrame): 

1131 assert right._indexed_same(left) 

1132 

1133 def column_op(a, b): 

1134 return {i: func(a.iloc[:, i], b.iloc[:, i]) 

1135 for i in range(len(a.columns))} 

1136 

1137 elif isinstance(right, ABCSeries) and axis == "columns": 

1138 # We only get here if called via left._combine_match_columns, 

1139 # in which case we specifically want to operate row-by-row 

1140 assert right.index.equals(left.columns) 

1141 

1142 def column_op(a, b): 

1143 return {i: func(a.iloc[:, i], b.iloc[i]) 

1144 for i in range(len(a.columns))} 

1145 

1146 elif isinstance(right, ABCSeries): 

1147 assert right.index.equals(left.index) # Handle other cases later 

1148 

1149 def column_op(a, b): 

1150 return {i: func(a.iloc[:, i], b) 

1151 for i in range(len(a.columns))} 

1152 

1153 else: 

1154 # Remaining cases have less-obvious dispatch rules 

1155 raise NotImplementedError(right) 

1156 

1157 new_data = expressions.evaluate(column_op, str_rep, left, right) 

1158 

1159 result = left._constructor(new_data, index=left.index, copy=False) 

1160 # Pin columns instead of passing to constructor for compat with 

1161 # non-unique columns case 

1162 result.columns = left.columns 

1163 return result

1167 """

1169 to the index implementation. DatetimeIndex and TimedeltaIndex perform 

1170 type checking, timezone handling, overflow checks, etc. 

1171 

1172 Parameters 

1173 ---------- 

1174 op : binary operator (operator.add, operator.sub, ...)

1176 right : object

1196 'operation [{name}]'.format(name=op.__name__))
