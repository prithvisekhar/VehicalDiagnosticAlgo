34def _get_dash_pattern(style): 

35 """Convert linestyle -> dash pattern 

36 """ 

37 # go from short hand -> full strings 

38 if isinstance(style, six.string_types): 

39 style = ls_mapper.get(style, style) 

40 # un-dashed styles 

41 if style in ['solid', 'None']: 

42 offset, dashes = None, None 

43 # dashed styles 

44 elif style in ['dashed', 'dashdot', 'dotted']: 

45 offset = 0 

46 dashes = tuple(rcParams['lines.{}_pattern'.format(style)]) 

47 # 

48 elif isinstance(style, tuple): 

49 offset, dashes = style 

50 else: 

51 raise ValueError('Unrecognized linestyle: %s' % str(style))

53 # normalize offset to be positive and shorter than the dash cycle 

54 if dashes is not None and offset is not None: 

55 dsum = sum(dashes)

57 offset %= dsum

76def segment_hits(cx, cy, x, y, radius): 

77 """ 

78 Determine if any line segments are within radius of a 

79 point. Returns the list of line segments that are within that 

80 radius. 

81 """ 

82 # Process single points specially 

83 if len(x) < 2: 

84 res, = np.nonzero((cx - x) ** 2 + (cy - y) ** 2 <= radius ** 2) 

85 return res 

86 

87 # We need to lop the last element off a lot. 

88 xr, yr = x[:-1], y[:-1] 

89 

90 # Only look at line segments whose nearest point to C on the line 

91 # lies within the segment. 

92 dx, dy = x[1:] - xr, y[1:] - yr 

93 Lnorm_sq = dx ** 2 + dy ** 2 # Possibly want to eliminate Lnorm==0 

94 u = ((cx - xr) * dx + (cy - yr) * dy) / Lnorm_sq 

95 candidates = (u >= 0) & (u <= 1) 

96 

97 # Note that there is a little area near one side of each point 

98 # which will be near neither segment, and another which will 

99 # be near both, depending on the angle of the lines. The 

100 # following radius test eliminates these ambiguities. 

101 point_hits = (cx - x) ** 2 + (cy - y) ** 2 <= radius ** 2 

102 candidates = candidates & ~(point_hits[:-1] | point_hits[1:]) 

103 

104 # For those candidates which remain, determine how far they lie away 

105 # from the line. 

106 px, py = xr + u * dx, yr + u * dy 

107 line_hits = (cx - px) ** 2 + (cy - py) ** 2 <= radius ** 2 

108 line_hits = line_hits & candidates 

109 points, = point_hits.ravel().nonzero() 

110 lines, = line_hits.ravel().nonzero() 

111 return np.concatenate((points, lines))

114def _mark_every_path(markevery, tpath, affine, ax_transform): 

115 """ 

116 Helper function that sorts out how to deal the input 

117 `markevery` and returns the points where markers should be drawn. 

118 

119 Takes in the `markevery` value and the line path and returns the 

120 sub-sampled path. 

121 """ 

122 # pull out the two bits of data we want from the path 

123 codes, verts = tpath.codes, tpath.vertices 

124 

125 def _slice_or_none(in_v, slc): 

126 ''' 

127 Helper function to cope with `codes` being an 

128 ndarray or `None` 

129 ''' 

130 if in_v is None: 

131 return None 

132 return in_v[slc] 

133 

134 # if just a float, assume starting at 0.0 and make a tuple 

135 if isinstance(markevery, float): 

136 markevery = (0.0, markevery) 

137 # if just an int, assume starting at 0 and make a tuple 

138 elif isinstance(markevery, int): 

139 markevery = (0, markevery) 

140 # if just an numpy int, assume starting at 0 and make a tuple 

141 elif isinstance(markevery, np.integer): 

142 markevery = (0, markevery.item()) 

143 

144 if isinstance(markevery, tuple): 

145 if len(markevery) != 2: 

146 raise ValueError('`markevery` is a tuple but its ' 

147 'len is not 2; ' 

148 'markevery=%s' % (markevery,)) 

149 start, step = markevery 

150 # if step is an int, old behavior 

151 if isinstance(step, int): 

152 #tuple of 2 int is for backwards compatibility, 

153 if not(isinstance(start, int)): 

154 raise ValueError('`markevery` is a tuple with ' 

155 'len 2 and second element is an int, but ' 

156 'the first element is not an int; ' 

157 'markevery=%s' % (markevery,)) 

158 # just return, we are done here 

159 

160 return Path(verts[slice(start, None, step)], 

161 _slice_or_none(codes, slice(start, None, step))) 

162 

163 elif isinstance(step, float): 

164 if not (isinstance(start, int) or 

165 isinstance(start, float)): 

166 raise ValueError('`markevery` is a tuple with ' 

167 'len 2 and second element is a float, but ' 

168 'the first element is not a float or an ' 

169 'int; ' 

170 'markevery=%s' % (markevery,)) 

171 #calc cumulative distance along path (in display 

172 # coords): 

173 disp_coords = affine.transform(tpath.vertices) 

174 delta = np.empty((len(disp_coords), 2), 

175 dtype=float) 

176 delta[0, :] = 0.0 

177 delta[1:, :] = (disp_coords[1:, :] - 

178 disp_coords[:-1, :]) 

179 delta = np.sum(delta**2, axis=1) 

180 delta = np.sqrt(delta) 

181 delta = np.cumsum(delta) 

182 #calc distance between markers along path based on 

183 # the axes bounding box diagonal being a distance 

184 # of unity: 

185 scale = ax_transform.transform( 

186 np.array([[0, 0], [1, 1]])) 

187 scale = np.diff(scale, axis=0) 

188 scale = np.sum(scale**2) 

189 scale = np.sqrt(scale) 

190 marker_delta = np.arange(start * scale, 

191 delta[-1], 

192 step * scale) 

193 #find closest actual data point that is closest to 

194 # the theoretical distance along the path: 

195 inds = np.abs(delta[np.newaxis, :] - 

196 marker_delta[:, np.newaxis]) 

197 inds = inds.argmin(axis=1) 

198 inds = np.unique(inds) 

199 # return, we are done here 

200 return Path(verts[inds], 

201 _slice_or_none(codes, inds)) 

202 else: 

203 raise ValueError('`markevery` is a tuple with ' 

204 'len 2, but its second element is not an int ' 

205 'or a float; ' 

206 'markevery=%s' % (markevery,)) 

207 

208 elif isinstance(markevery, slice): 

209 # mazol tov, it's already a slice, just return 

210 return Path(verts[markevery], 

211 _slice_or_none(codes, markevery)) 

212 

213 elif iterable(markevery): 

214 #fancy indexing 

215 try: 

216 return Path(verts[markevery], 

217 _slice_or_none(codes, markevery)) 

218 

219 except (ValueError, IndexError): 

220 raise ValueError('`markevery` is iterable but ' 

221 'not a valid form of numpy fancy indexing; ' 

222 'markevery=%s' % (markevery,)) 

223 else: 

224 raise ValueError('Value of `markevery` is not ' 

225 'recognized; ' 

226 'markevery=%s' % (markevery,))

441 def contains(self, mouseevent): 

442 """ 

443 Test whether the mouse event occurred on the line. The pick 

444 radius determines the precision of the location test (usually 

445 within five points of the value). Use 

446 :meth:`~matplotlib.lines.Line2D.get_pickradius` or 

447 :meth:`~matplotlib.lines.Line2D.set_pickradius` to view or 

448 modify it. 

449 

450 Returns *True* if any values are within the radius along with 

451 ``{'ind': pointlist}``, where *pointlist* is the set of points 

452 within the radius. 

453 

454 TODO: sort returned indices by distance 

455 """ 

456 if callable(self._contains): 

457 return self._contains(self, mouseevent) 

458 

459 if not is_numlike(self.pickradius): 

460 raise ValueError("pick radius should be a distance") 

461 

462 # Make sure we have data to plot 

463 if self._invalidy or self._invalidx: 

464 self.recache() 

465 if len(self._xy) == 0: 

466 return False, {} 

467 

468 # Convert points to pixels 

469 transformed_path = self._get_transformed_path() 

470 path, affine = transformed_path.get_transformed_path_and_affine() 

471 path = affine.transform_path(path) 

472 xy = path.vertices 

473 xt = xy[:, 0] 

474 yt = xy[:, 1] 

475 

476 # Convert pick radius from points to pixels 

477 if self.figure is None: 

478 warnings.warn('no figure set when check if mouse is on line') 

479 pixels = self.pickradius 

480 else: 

481 pixels = self.figure.dpi / 72. * self.pickradius 

482 

483 # the math involved in checking for containment (here and inside of 

484 # segment_hits) assumes that it is OK to overflow. In case the 

485 # application has set the error flags such that an exception is raised 

486 # on overflow, we temporarily set the appropriate error flags here and 

487 # set them back when we are finished. 

488 with np.errstate(all='ignore'): 

489 # Check for collision 

490 if self._linestyle in ['None', None]: 

491 # If no line, return the nearby point(s) 

492 d = (xt - mouseevent.x) ** 2 + (yt - mouseevent.y) ** 2 

493 ind, = np.nonzero(np.less_equal(d, pixels ** 2)) 

494 else: 

495 # If line, return the nearby segment(s) 

496 ind = segment_hits(mouseevent.x, mouseevent.y, xt, yt, pixels) 

497 if self._drawstyle.startswith("steps"): 

498 ind //= 2 

499 

500 ind += self.ind_offset 

501 

502 # Return the point(s) within radius 

503 return len(ind) > 0, dict(ind=ind)

537 def set_markevery(self, every): 

538 """Set the markevery property to subsample the plot when using markers. 

539 

540 e.g., if `every=5`, every 5-th marker will be plotted. 

541 

542 ACCEPTS: [None | int | length-2 tuple of int | slice | 

543 list/array of int | float | length-2 tuple of float] 

544 

545 Parameters 

546 ---------- 

547 every: None | int | length-2 tuple of int | slice | list/array of int \ 

548| float | length-2 tuple of float 

549 Which markers to plot. 

550 

551 - every=None, every point will be plotted. 

552 - every=N, every N-th marker will be plotted starting with 

553 marker 0. 

554 - every=(start, N), every N-th marker, starting at point 

555 start, will be plotted. 

556 - every=slice(start, end, N), every N-th marker, starting at 

557 point start, upto but not including point end, will be plotted. 

558 - every=[i, j, m, n], only markers at points i, j, m, and n 

559 will be plotted. 

560 - every=0.1, (i.e. a float) then markers will be spaced at 

561 approximately equal distances along the line; the distance 

562 along the line between markers is determined by multiplying the 

563 display-coordinate distance of the axes bounding-box diagonal 

564 by the value of every. 

565 - every=(0.5, 0.1) (i.e. a length-2 tuple of float), the 

566 same functionality as every=0.1 is exhibited but the first 

567 marker will be 0.5 multiplied by the 

568 display-cordinate-diagonal-distance along the line. 

569 

570 Notes 

571 ----- 

572 Setting the markevery property will only show markers at actual data 

573 points. When using float arguments to set the markevery property 

574 on irregularly spaced data, the markers will likely not appear evenly 

575 spaced because the actual data points do not coincide with the 

576 theoretical spacing between markers. 

577 

578 When using a start offset to specify the first marker, the offset will 

579 be from the first data point which may be different from the first 

580 the visible data point if the plot is zoomed in. 

581 

582 If zooming in on a plot when using float arguments then the actual 

583 data points that have markers will change because the distance between 

584 markers is always determined from the display-coordinates 

585 axes-bounding-box-diagonal regardless of the actual axes data limits. 

586 

587 """ 

588 if self._markevery != every: 

589 self.stale = True 

590 self._markevery = every

662 self._x, self._y = self._xy.T # views 

663 

664 self._subslice = False 

665 if (self.axes and len(x) > 1000 and self._is_sorted(x) and 

666 self.axes.name == 'rectilinear' and 

667 self.axes.get_xscale() == 'linear' and 

668 self._markevery is None and 

669 self.get_clip_on() is True): 

670 self._subslice = True 

671 nanmask = np.isnan(x) 

672 if nanmask.any(): 

673 self._x_filled = self._x.copy() 

674 indices = np.arange(len(x)) 

675 self._x_filled[nanmask] = np.interp(indices[nanmask], 

676 indices[~nanmask], self._x[~nanmask]) 

677 else: 

678 self._x_filled = self._x 

679 

680 if self._path is not None: 

681 interpolation_steps = self._path._interpolation_steps 

682 else: 

683 interpolation_steps = 1 

684 xy = STEP_LOOKUP_MAP[self._drawstyle](*self._xy.T) 

685 self._path = Path(np.asarray(xy).T, 

686 _interpolation_steps=interpolation_steps) 

687 self._transformed_path = None 

688 self._invalidx = False 

689 self._invalidy = False

698 if subslice is not None:

700 _path = Path(np.asarray(xy).T, 

701 _interpolation_steps=self._path._interpolation_steps) 

702 else: 

703 _path = self._path

731 @allow_rasterization 

732 def draw(self, renderer): 

733 """draw the Line with `renderer` unless visibility is False""" 

734 if not self.get_visible(): 

735 return 

736 

737 if self._invalidy or self._invalidx: 

738 self.recache() 

739 self.ind_offset = 0 # Needed for contains() method. 

740 if self._subslice and self.axes: 

741 x0, x1 = self.axes.get_xbound() 

742 i0, = self._x_filled.searchsorted([x0], 'left') 

743 i1, = self._x_filled.searchsorted([x1], 'right') 

744 subslice = slice(max(i0 - 1, 0), i1 + 1) 

745 self.ind_offset = subslice.start 

746 self._transform_path(subslice) 

747 else: 

748 subslice = None 

749 

750 if self.get_path_effects(): 

751 from matplotlib.patheffects import PathEffectRenderer 

752 renderer = PathEffectRenderer(self.get_path_effects(), renderer) 

753 

754 renderer.open_group('line2d', self.get_gid()) 

755 if self._lineStyles[self._linestyle] != '_draw_nothing': 

756 tpath, affine = (self._get_transformed_path() 

757 .get_transformed_path_and_affine()) 

758 if len(tpath.vertices): 

759 gc = renderer.new_gc() 

760 self._set_gc_clip(gc) 

761 

762 lc_rgba = mcolors.to_rgba(self._color, self._alpha) 

763 gc.set_foreground(lc_rgba, isRGBA=True) 

764 

765 gc.set_antialiased(self._antialiased) 

766 gc.set_linewidth(self._linewidth) 

767 

768 if self.is_dashed(): 

769 cap = self._dashcapstyle 

770 join = self._dashjoinstyle 

771 else: 

772 cap = self._solidcapstyle 

773 join = self._solidjoinstyle 

774 gc.set_joinstyle(join) 

775 gc.set_capstyle(cap) 

776 gc.set_snap(self.get_snap()) 

777 if self.get_sketch_params() is not None: 

778 gc.set_sketch_params(*self.get_sketch_params()) 

779 

780 gc.set_dashes(self._dashOffset, self._dashSeq) 

781 renderer.draw_path(gc, tpath, affine.frozen()) 

782 gc.restore() 

783 

784 if self._marker and self._markersize > 0: 

785 gc = renderer.new_gc() 

786 self._set_gc_clip(gc) 

787 gc.set_linewidth(self._markeredgewidth) 

788 gc.set_antialiased(self._antialiased) 

789 

790 ec_rgba = mcolors.to_rgba( 

791 self.get_markeredgecolor(), self._alpha) 

792 fc_rgba = mcolors.to_rgba( 

793 self._get_markerfacecolor(), self._alpha) 

794 fcalt_rgba = mcolors.to_rgba( 

795 self._get_markerfacecolor(alt=True), self._alpha) 

796 # If the edgecolor is "auto", it is set according to the *line* 

797 # color but inherits the alpha value of the *face* color, if any. 

798 if (cbook._str_equal(self._markeredgecolor, "auto") 

799 and not cbook._str_lower_equal( 

800 self.get_markerfacecolor(), "none")): 

801 ec_rgba = ec_rgba[:3] + (fc_rgba[3],) 

802 gc.set_foreground(ec_rgba, isRGBA=True) 

803 

804 marker = self._marker 

805 

806 # Markers *must* be drawn ignoring the drawstyle (but don't pay the 

807 # recaching if drawstyle is already "default"). 

808 if self.get_drawstyle() != "default": 

809 with cbook._setattr_cm( 

810 self, _drawstyle="default", _transformed_path=None): 

811 self.recache() 

812 self._transform_path(subslice) 

813 tpath, affine = (self._get_transformed_path() 

814 .get_transformed_path_and_affine()) 

815 else: 

816 tpath, affine = (self._get_transformed_path() 

817 .get_transformed_path_and_affine()) 

818 

819 if len(tpath.vertices): 

820 # subsample the markers if markevery is not None 

821 markevery = self.get_markevery() 

822 if markevery is not None: 

823 subsampled = _mark_every_path(markevery, tpath, 

824 affine, self.axes.transAxes) 

825 else: 

826 subsampled = tpath 

827 

828 snap = marker.get_snap_threshold() 

829 if type(snap) == float: 

830 snap = renderer.points_to_pixels(self._markersize) >= snap 

831 gc.set_snap(snap) 

832 gc.set_joinstyle(marker.get_joinstyle()) 

833 gc.set_capstyle(marker.get_capstyle()) 

834 marker_path = marker.get_path() 

835 marker_trans = marker.get_transform() 

836 w = renderer.points_to_pixels(self._markersize) 

837 

838 if cbook._str_equal(marker.get_marker(), ","): 

839 gc.set_linewidth(0) 

840 else: 

841 # Don't scale for pixels, and don't stroke them 

842 marker_trans = marker_trans.scale(w) 

843 

844 renderer.draw_markers(gc, marker_path, marker_trans, 

845 subsampled, affine.frozen(), 

846 fc_rgba) 

847 

848 alt_marker_path = marker.get_alt_path() 

849 if alt_marker_path: 

850 alt_marker_trans = marker.get_alt_transform() 

851 alt_marker_trans = alt_marker_trans.scale(w) 

852 renderer.draw_markers( 

853 gc, alt_marker_path, alt_marker_trans, subsampled, 

854 affine.frozen(), fcalt_rgba) 

855 

856 gc.restore() 

857 

858 renderer.close_group('line2d') 

859 self.stale = False

1025 def _split_drawstyle_linestyle(self, ls): 

1026 '''Split drawstyle from linestyle string 

1027 

1028 If `ls` is only a drawstyle default to returning a linestyle 

1029 of '-'. 

1030 

1031 Parameters 

1032 ---------- 

1033 ls : str 

1034 The linestyle to be processed 

1035 

1036 Returns 

1037 ------- 

1038 ret_ds : str or None 

1039 If the linestyle string does not contain a drawstyle prefix 

1040 return None, otherwise return it. 

1041 

1042 ls : str 

1043 The linestyle with the drawstyle (if any) stripped. 

1044 ''' 

1045 ret_ds = None 

1046 for ds in self.drawStyleKeys: # long names are first in the list 

1047 if ls.startswith(ds): 

1048 ret_ds = ds 

1049 if len(ls) > len(ds): 

1050 ls = ls[len(ds):] 

1051 else: 

1052 ls = '-' 

1053 break 

1054 

1055 return ret_ds, ls

1057 def set_linestyle(self, ls): 

1058 """ 

1059 Set the linestyle of the line (also accepts drawstyles, 

1060 e.g., ``'steps--'``) 

1061 

1062 

1063 =========================== ================= 

1064 linestyle description 

1065 =========================== ================= 

1066 ``'-'`` or ``'solid'`` solid line 

1067 ``'--'`` or ``'dashed'`` dashed line 

1068 ``'-.'`` or ``'dashdot'`` dash-dotted line 

1069 ``':'`` or ``'dotted'`` dotted line 

1070 ``'None'`` draw nothing 

1071 ``' '`` draw nothing 

1072 ``''`` draw nothing 

1073 =========================== ================= 

1074 

1075 'steps' is equivalent to 'steps-pre' and is maintained for 

1076 backward-compatibility. 

1077 

1078 Alternatively a dash tuple of the following form can be provided:: 

1079 

1080 (offset, onoffseq), 

1081 

1082 where ``onoffseq`` is an even length tuple of on and off ink 

1083 in points. 

1084 

1085 

1086 ACCEPTS: ['solid' | 'dashed', 'dashdot', 'dotted' | 

1087 (offset, on-off-dash-seq) | 

1088 ``'-'`` | ``'--'`` | ``'-.'`` | ``':'`` | ``'None'`` | 

1089 ``' '`` | ``''``] 

1090 

1091 .. seealso:: 

1092 

1093 :meth:`set_drawstyle` 

1094 To set the drawing style (stepping) of the plot. 

1095 

1096 Parameters 

1097 ---------- 

1098 ls : { ``'-'``, ``'--'``, ``'-.'``, ``':'``} and more see description 

1099 The line style. 

1100 """ 

1101 if isinstance(ls, six.string_types): 

1102 ds, ls = self._split_drawstyle_linestyle(ls) 

1103 if ds is not None: 

1104 self.set_drawstyle(ds) 

1105 

1106 if ls in [' ', '', 'none']: 

1107 ls = 'None' 

1108 

1109 if ls not in self._lineStyles: 

1110 try: 

1111 ls = ls_mapper_r[ls] 

1112 except KeyError: 

1113 raise ValueError(("You passed in an invalid linestyle, " 

1114 "`{0}`. See " 

1115 "docs of Line2D.set_linestyle for " 

1116 "valid values.").format(ls)) 

1117 self._linestyle = ls 

1118 else: 

1119 self._linestyle = '--' 

1120 

1121 # get the unscaled dashes 

1122 self._us_dashOffset, self._us_dashSeq = _get_dash_pattern(ls) 

1123 # compute the linewidth scaled dashes 

1124 self._dashOffset, self._dashSeq = _scale_dashes( 

1125 self._us_dashOffset, self._us_dashSeq, self._linewidth)
