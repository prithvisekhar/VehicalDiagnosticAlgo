88def _pad_simple(array, pad_width, fill_value=None): 

89 """ 

90 Pad array on all sides with either a single value or undefined values. 

91 

92 Parameters 

93 ---------- 

94 array : ndarray 

95 Array to grow. 

96 pad_width : sequence of tuple[int, int] 

97 Pad width on both sides for each dimension in `arr`. 

98 fill_value : scalar, optional 

99 If provided the padded area is filled with this value, otherwise 

100 the pad area left undefined. 

101 

102 Returns 

103 ------- 

104 padded : ndarray 

105 The padded array with the same dtype as`array`. Its order will default 

106 to C-style if `array` is not F-contiguous. 

107 original_area_slice : tuple 

108 A tuple of slices pointing to the area of the original array. 

109 """ 

110 # Allocate grown array 

111 new_shape = tuple( 

112 left + size + right 

113 for size, (left, right) in zip(array.shape, pad_width) 

114 ) 

115 order = 'F' if array.flags.fnc else 'C' # Fortran and not also C-order 

116 padded = np.empty(new_shape, dtype=array.dtype, order=order) 

117 

118 if fill_value is not None: 

119 padded.fill(fill_value) 

120 

121 # Copy old array into correct space 

122 original_area_slice = tuple( 

123 slice(left, left + size) 

124 for size, (left, right) in zip(array.shape, pad_width) 

125 ) 

126 padded[original_area_slice] = array 

127 

128 return padded, original_area_slice

157 """ 

158 Retrieve edge values from empty-padded array in given dimension.

163 Empty-padded array.

189 """ 

190 Construct linear ramps for empty-padded array in given dimension. 

191 

192 Parameters 

193 ---------- 

194 padded : ndarray 

195 Empty-padded array. 

196 axis : int 

197 Dimension in which the ramps are constructed. 

198 width_pair : (int, int) 

199 Pair of widths that mark the pad area on both sides in the given 

200 dimension. 

201 end_value_pair : (scalar, scalar) 

202 End values for the linear ramps which form the edge of the fully padded 

203 array. These values are included in the linear ramps. 

204 

205 Returns 

206 -------

225 endpoint=False, 

226 dtype=padded.dtype, 

227 axis=axis, 

228 )

230 right_ramp = right_ramp[_slice_at_axis(slice(None, None, -1), axis)]

235def _get_stats(padded, axis, width_pair, length_pair, stat_func): 

236 """ 

237 Calculate statistic for the empty-padded array in given dimnsion. 

238 

239 Parameters 

240 ---------- 

241 padded : ndarray 

242 Empty-padded array. 

243 axis : int 

244 Dimension in which the statistic is calculated. 

245 width_pair : (int, int) 

246 Pair of widths that mark the pad area on both sides in the given 

247 dimension. 

248 length_pair : 2-element sequence of None or int 

249 Gives the number of values in valid area from each side that is 

250 taken into account when calculating the statistic. If None the entire 

251 valid area in `padded` is considered. 

252 stat_func : function 

253 Function to compute statistic. The expected signature is 

254 ``stat_func(x: ndarray, axis: int, keepdims: bool) -> ndarray``. 

255 

256 Returns 

257 ------- 

258 left_stat, right_stat : ndarray 

259 Calculated statistic for both sides of `padded`. 

260 """ 

261 # Calculate indices of the edges of the area with original values 

262 left_index = width_pair[0] 

263 right_index = padded.shape[axis] - width_pair[1] 

264 # as well as its length 

265 max_length = right_index - left_index 

266 

267 # Limit stat_lengths to max_length 

268 left_length, right_length = length_pair 

269 if left_length is None or max_length < left_length: 

270 left_length = max_length 

271 if right_length is None or max_length < right_length: 

272 right_length = max_length 

273 

274 if (left_length == 0 or right_length == 0) \ 

275 and stat_func in {np.amax, np.amin}: 

276 # amax and amin can't operate on an emtpy array, 

277 # raise a more descriptive warning here instead of the default one 

278 raise ValueError("stat_length of 0 yields no value for padding") 

279 

280 # Calculate statistic for the left side 

281 left_slice = _slice_at_axis( 

282 slice(left_index, left_index + left_length), axis) 

283 left_chunk = padded[left_slice] 

284 left_stat = stat_func(left_chunk, axis=axis, keepdims=True) 

285 _round_if_needed(left_stat, padded.dtype) 

286 

287 if left_length == right_length == max_length: 

288 # return early as right_stat must be identical to left_stat 

289 return left_stat, left_stat 

290 

291 # Calculate statistic for the right side 

292 right_slice = _slice_at_axis( 

293 slice(right_index - right_length, right_index), axis) 

294 right_chunk = padded[right_slice] 

295 right_stat = stat_func(right_chunk, axis=axis, keepdims=True) 

296 _round_if_needed(right_stat, padded.dtype) 

297 

298 return left_stat, right_stat

301def _set_reflect_both(padded, axis, width_pair, method, include_edge=False): 

302 """ 

303 Pad `axis` of `arr` with reflection. 

304 

305 Parameters 

306 ---------- 

307 padded : ndarray 

308 Input array of arbitrary shape. 

309 axis : int 

310 Axis along which to pad `arr`. 

311 width_pair : (int, int) 

312 Pair of widths that mark the pad area on both sides in the given 

313 dimension. 

314 method : str 

315 Controls method of reflection; options are 'even' or 'odd'. 

316 include_edge : bool 

317 If true, edge value is included in reflection, otherwise the edge 

318 value forms the symmetric axis to the reflection. 

319 

320 Returns 

321 ------- 

322 pad_amt : tuple of ints, length 2 

323 New index positions of padding to do along the `axis`. If these are 

324 both 0, padding is done in this dimension. 

325 """ 

326 left_pad, right_pad = width_pair 

327 old_length = padded.shape[axis] - right_pad - left_pad 

328 

329 if include_edge: 

330 # Edge is included, we need to offset the pad amount by 1 

331 edge_offset = 1 

332 else: 

333 edge_offset = 0 # Edge is not included, no need to offset pad amount 

334 old_length -= 1 # but must be omitted from the chunk 

335 

336 if left_pad > 0: 

337 # Pad with reflected values on left side: 

338 # First limit chunk size which can't be larger than pad area 

339 chunk_length = min(old_length, left_pad) 

340 # Slice right to left, stop on or next to edge, start relative to stop 

341 stop = left_pad - edge_offset 

342 start = stop + chunk_length 

343 left_slice = _slice_at_axis(slice(start, stop, -1), axis) 

344 left_chunk = padded[left_slice] 

345 

346 if method == "odd": 

347 # Negate chunk and align with edge 

348 edge_slice = _slice_at_axis(slice(left_pad, left_pad + 1), axis) 

349 left_chunk = 2 * padded[edge_slice] - left_chunk 

350 

351 # Insert chunk into padded area 

352 start = left_pad - chunk_length 

353 stop = left_pad 

354 pad_area = _slice_at_axis(slice(start, stop), axis) 

355 padded[pad_area] = left_chunk 

356 # Adjust pointer to left edge for next iteration 

357 left_pad -= chunk_length 

358 

359 if right_pad > 0: 

360 # Pad with reflected values on right side: 

361 # First limit chunk size which can't be larger than pad area 

362 chunk_length = min(old_length, right_pad) 

363 # Slice right to left, start on or next to edge, stop relative to start 

364 start = -right_pad + edge_offset - 2 

365 stop = start - chunk_length 

366 right_slice = _slice_at_axis(slice(start, stop, -1), axis) 

367 right_chunk = padded[right_slice] 

368 

369 if method == "odd": 

370 # Negate chunk and align with edge 

371 edge_slice = _slice_at_axis( 

372 slice(-right_pad - 1, -right_pad), axis) 

373 right_chunk = 2 * padded[edge_slice] - right_chunk 

374 

375 # Insert chunk into padded area 

376 start = padded.shape[axis] - right_pad 

377 stop = start + chunk_length 

378 pad_area = _slice_at_axis(slice(start, stop), axis) 

379 padded[pad_area] = right_chunk 

380 # Adjust pointer to right edge for next iteration 

381 right_pad -= chunk_length 

382 

383 return left_pad, right_pad

386def _set_wrap_both(padded, axis, width_pair): 

387 """ 

388 Pad `axis` of `arr` with wrapped values. 

389 

390 Parameters 

391 ---------- 

392 padded : ndarray 

393 Input array of arbitrary shape. 

394 axis : int 

395 Axis along which to pad `arr`. 

396 width_pair : (int, int) 

397 Pair of widths that mark the pad area on both sides in the given 

398 dimension. 

399 

400 Returns 

401 ------- 

402 pad_amt : tuple of ints, length 2 

403 New index positions of padding to do along the `axis`. If these are 

404 both 0, padding is done in this dimension. 

405 """ 

406 left_pad, right_pad = width_pair 

407 period = padded.shape[axis] - right_pad - left_pad 

408 

409 # If the current dimension of `arr` doesn't contain enough valid values 

410 # (not part of the undefined pad area) we need to pad multiple times. 

411 # Each time the pad area shrinks on both sides which is communicated with 

412 # these variables. 

413 new_left_pad = 0 

414 new_right_pad = 0 

415 

416 if left_pad > 0: 

417 # Pad with wrapped values on left side 

418 # First slice chunk from right side of the non-pad area. 

419 # Use min(period, left_pad) to ensure that chunk is not larger than 

420 # pad area 

421 right_slice = _slice_at_axis( 

422 slice(-right_pad - min(period, left_pad), 

423 -right_pad if right_pad != 0 else None), 

424 axis 

425 ) 

426 right_chunk = padded[right_slice] 

427 

428 if left_pad > period: 

429 # Chunk is smaller than pad area 

430 pad_area = _slice_at_axis(slice(left_pad - period, left_pad), axis) 

431 new_left_pad = left_pad - period 

432 else: 

433 # Chunk matches pad area 

434 pad_area = _slice_at_axis(slice(None, left_pad), axis) 

435 padded[pad_area] = right_chunk 

436 

437 if right_pad > 0: 

438 # Pad with wrapped values on right side 

439 # First slice chunk from left side of the non-pad area. 

440 # Use min(period, right_pad) to ensure that chunk is not larger than 

441 # pad area 

442 left_slice = _slice_at_axis( 

443 slice(left_pad, left_pad + min(period, right_pad),), axis) 

444 left_chunk = padded[left_slice] 

445 

446 if right_pad > period: 

447 # Chunk is smaller than pad area 

448 pad_area = _slice_at_axis( 

449 slice(-right_pad, -right_pad + period), axis) 

450 new_right_pad = right_pad - period 

451 else: 

452 # Chunk matches pad area 

453 pad_area = _slice_at_axis(slice(-right_pad, None), axis) 

454 padded[pad_area] = left_chunk 

455 

456 return new_left_pad, new_right_pad

459def _as_pairs(x, ndim, as_index=False): 

460 """ 

461 Broadcast `x` to an array with the shape (`ndim`, 2). 

462 

463 A helper function for `pad` that prepares and validates arguments like 

464 `pad_width` for iteration in pairs. 

465 

466 Parameters 

467 ---------- 

468 x : {None, scalar, array-like} 

469 The object to broadcast to the shape (`ndim`, 2). 

470 ndim : int 

471 Number of pairs the broadcasted `x` will have. 

472 as_index : bool, optional 

473 If `x` is not None, try to round each element of `x` to an integer 

474 (dtype `np.intp`) and ensure every element is positive. 

475 

476 Returns 

477 ------- 

478 pairs : nested iterables, shape (`ndim`, 2) 

479 The broadcasted version of `x`. 

480 

481 Raises 

482 ------ 

483 ValueError 

484 If `as_index` is True and `x` contains negative elements. 

485 Or if `x` is not broadcastable to the shape (`ndim`, 2). 

486 """ 

487 if x is None: 

488 # Pass through None as a special case, otherwise np.round(x) fails 

489 # with an AttributeError 

490 return ((None, None),) * ndim 

491 

492 x = np.array(x) 

493 if as_index: 

494 x = np.round(x).astype(np.intp, copy=False) 

495 

496 if x.ndim < 3: 

497 # Optimization: Possibly use faster paths for cases where `x` has 

498 # only 1 or 2 elements. `np.broadcast_to` could handle these as well 

499 # but is currently slower 

500 

501 if x.size == 1: 

502 # x was supplied as a single value 

503 x = x.ravel() # Ensure x[0] works for x.ndim == 0, 1, 2 

504 if as_index and x < 0: 

505 raise ValueError("index can't contain negative values") 

506 return ((x[0], x[0]),) * ndim 

507 

508 if x.size == 2 and x.shape != (2, 1): 

509 # x was supplied with a single value for each side 

510 # but except case when each dimension has a single value 

511 # which should be broadcasted to a pair, 

512 # e.g. [[1], [2]] -> [[1, 1], [2, 2]] not [[1, 2], [1, 2]] 

513 x = x.ravel() # Ensure x[0], x[1] works 

514 if as_index and (x[0] < 0 or x[1] < 0): 

515 raise ValueError("index can't contain negative values") 

516 return ((x[0], x[1]),) * ndim 

517 

518 if as_index and x.min() < 0: 

519 raise ValueError("index can't contain negative values") 

520 

521 # Converting the array with `tolist` seems to improve performance 

522 # when iterating and indexing the result (see usage in `pad`) 

523 return np.broadcast_to(x, (ndim, 2)).tolist()

534@array_function_dispatch(_pad_dispatcher, module='numpy') 

535def pad(array, pad_width, mode='constant', **kwargs): 

536 """ 

537 Pad an array. 

538 

539 Parameters 

540 ---------- 

541 array : array_like of rank N 

542 The array to pad. 

543 pad_width : {sequence, array_like, int} 

544 Number of values padded to the edges of each axis. 

545 ((before_1, after_1), ... (before_N, after_N)) unique pad widths 

546 for each axis. 

547 ((before, after),) yields same before and after pad for each axis. 

548 (pad,) or int is a shortcut for before = after = pad width for all 

549 axes. 

550 mode : str or function, optional 

551 One of the following string values or a user supplied function. 

552 

553 'constant' (default) 

554 Pads with a constant value. 

555 'edge' 

556 Pads with the edge values of array. 

557 'linear_ramp' 

558 Pads with the linear ramp between end_value and the 

559 array edge value. 

560 'maximum' 

561 Pads with the maximum value of all or part of the 

562 vector along each axis. 

563 'mean' 

564 Pads with the mean value of all or part of the 

565 vector along each axis. 

566 'median' 

567 Pads with the median value of all or part of the 

568 vector along each axis. 

569 'minimum' 

570 Pads with the minimum value of all or part of the 

571 vector along each axis. 

572 'reflect' 

573 Pads with the reflection of the vector mirrored on 

574 the first and last values of the vector along each 

575 axis. 

576 'symmetric' 

577 Pads with the reflection of the vector mirrored 

578 along the edge of the array. 

579 'wrap' 

580 Pads with the wrap of the vector along the axis. 

581 The first values are used to pad the end and the 

582 end values are used to pad the beginning. 

583 'empty' 

584 Pads with undefined values. 

585 

586 .. versionadded:: 1.17 

587 

588 <function> 

589 Padding function, see Notes. 

590 stat_length : sequence or int, optional 

591 Used in 'maximum', 'mean', 'median', and 'minimum'. Number of 

592 values at edge of each axis used to calculate the statistic value. 

593 

594 ((before_1, after_1), ... (before_N, after_N)) unique statistic 

595 lengths for each axis. 

596 

597 ((before, after),) yields same before and after statistic lengths 

598 for each axis. 

599 

600 (stat_length,) or int is a shortcut for before = after = statistic 

601 length for all axes. 

602 

603 Default is ``None``, to use the entire axis. 

604 constant_values : sequence or scalar, optional 

605 Used in 'constant'. The values to set the padded values for each 

606 axis. 

607 

608 ``((before_1, after_1), ... (before_N, after_N))`` unique pad constants 

609 for each axis. 

610 

611 ``((before, after),)`` yields same before and after constants for each 

612 axis. 

613 

614 ``(constant,)`` or ``constant`` is a shortcut for ``before = after = constant`` for 

615 all axes. 

616 

617 Default is 0. 

618 end_values : sequence or scalar, optional 

619 Used in 'linear_ramp'. The values used for the ending value of the 

620 linear_ramp and that will form the edge of the padded array. 

621 

622 ``((before_1, after_1), ... (before_N, after_N))`` unique end values 

623 for each axis. 

624 

625 ``((before, after),)`` yields same before and after end values for each 

626 axis. 

627 

628 ``(constant,)`` or ``constant`` is a shortcut for ``before = after = constant`` for 

629 all axes. 

630 

631 Default is 0. 

632 reflect_type : {'even', 'odd'}, optional 

633 Used in 'reflect', and 'symmetric'. The 'even' style is the 

634 default with an unaltered reflection around the edge value. For 

635 the 'odd' style, the extended part of the array is created by 

636 subtracting the reflected values from two times the edge value. 

637 

638 Returns 

639 ------- 

640 pad : ndarray 

641 Padded array of rank equal to `array` with shape increased 

642 according to `pad_width`. 

643 

644 Notes 

645 ----- 

646 .. versionadded:: 1.7.0 

647 

648 For an array with rank greater than 1, some of the padding of later 

649 axes is calculated from padding of previous axes. This is easiest to 

650 think about with a rank 2 array where the corners of the padded array 

651 are calculated by using padded values from the first axis. 

652 

653 The padding function, if used, should modify a rank 1 array in-place. It 

654 has the following signature:: 

655 

656 padding_func(vector, iaxis_pad_width, iaxis, kwargs) 

657 

658 where 

659 

660 vector : ndarray 

661 A rank 1 array already padded with zeros. Padded values are 

662 vector[:iaxis_pad_width[0]] and vector[-iaxis_pad_width[1]:]. 

663 iaxis_pad_width : tuple 

664 A 2-tuple of ints, iaxis_pad_width[0] represents the number of 

665 values padded at the beginning of vector where 

666 iaxis_pad_width[1] represents the number of values padded at 

667 the end of vector. 

668 iaxis : int 

669 The axis currently being calculated. 

670 kwargs : dict 

671 Any keyword arguments the function requires. 

672 

673 Examples 

674 -------- 

675 >>> a = [1, 2, 3, 4, 5] 

676 >>> np.pad(a, (2, 3), 'constant', constant_values=(4, 6)) 

677 array([4, 4, 1, ..., 6, 6, 6]) 

678 

679 >>> np.pad(a, (2, 3), 'edge') 

680 array([1, 1, 1, ..., 5, 5, 5])

742 pad_width = np.asarray(pad_width) 

743 

744 if not pad_width.dtype.kind == 'i': 

745 raise TypeError('`pad_width` must be of integral type.') 

746 

747 # Broadcast to shape (array.ndim, 2) 

748 pad_width = _as_pairs(pad_width, array.ndim, as_index=True) 

749 

750 if callable(mode): 

751 # Old behavior: Use user-supplied function with np.apply_along_axis 

752 function = mode 

753 # Create a new zero padded array 

754 padded, _ = _pad_simple(array, pad_width, fill_value=0) 

755 # And apply along each axis 

756 

757 for axis in range(padded.ndim): 

758 # Iterate using ndindex as in apply_along_axis, but assuming that 

759 # function operates inplace on the padded array. 

760 

761 # view with the iteration axis at the end 

762 view = np.moveaxis(padded, axis, -1) 

763 

764 # compute indices for the iteration axes, and append a trailing 

765 # ellipsis to prevent 0d arrays decaying to scalars (gh-8642) 

766 inds = ndindex(view.shape[:-1]) 

767 inds = (ind + (Ellipsis,) for ind in inds) 

768 for ind in inds: 

769 function(view[ind], pad_width[axis], axis, kwargs) 

770 

771 return padded

773 # Make sure that no unsupported keywords were passed for the current mode 

774 allowed_kwargs = { 

775 'empty': [], 'edge': [], 'wrap': [], 

776 'constant': ['constant_values'], 

777 'linear_ramp': ['end_values'], 

778 'maximum': ['stat_length'], 

779 'mean': ['stat_length'], 

780 'median': ['stat_length'], 

781 'minimum': ['stat_length'], 

782 'reflect': ['reflect_type'], 

783 'symmetric': ['reflect_type'], 

784 }

786 unsupported_kwargs = set(kwargs) - set(allowed_kwargs[mode]) 

787 except KeyError: 

788 raise ValueError("mode '{}' is not supported".format(mode)) 

789 if unsupported_kwargs: 

790 raise ValueError("unsupported keyword arguments for mode '{}': {}" 

791 .format(mode, unsupported_kwargs)) 

792 

793 stat_functions = {"maximum": np.amax, "minimum": np.amin, 

794 "mean": np.mean, "median": np.median}

797 # (padded area is undefined) 

798 padded, original_area_slice = _pad_simple(array, pad_width)

800 # (zipping may be more readable than using enumerate)

808 _set_pad_area(roi, axis, width_pair, value_pair) 

809 

810 elif mode == "empty": 

811 pass # Do nothing as _pad_simple already returned the correct result 

812 

813 elif array.size == 0: 

814 # Only modes "constant" and "empty" can extend empty axes, all other 

815 # modes depend on `array` not being empty 

816 # -> ensure every empty axis is only "padded with 0" 

817 for axis, width_pair in zip(axes, pad_width): 

818 if array.shape[axis] == 0 and any(width_pair): 

819 raise ValueError( 

820 "can't extend empty axis {} using modes other than " 

821 "'constant' or 'empty'".format(axis) 

822 ) 

823 # passed, don't need to do anything more as _pad_simple already 

824 # returned the correct result 

825

827 for axis, width_pair in zip(axes, pad_width): 

828 roi = _view_roi(padded, original_area_slice, axis) 

829 edge_pair = _get_edges(roi, axis, width_pair) 

830 _set_pad_area(roi, axis, width_pair, edge_pair)

832 elif mode == "linear_ramp":

836 roi = _view_roi(padded, original_area_slice, axis) 

837 ramp_pair = _get_linear_ramps(roi, axis, width_pair, value_pair) 

838 _set_pad_area(roi, axis, width_pair, ramp_pair) 

839

841 func = stat_functions[mode] 

842 length = kwargs.get("stat_length", None) 

843 length = _as_pairs(length, padded.ndim, as_index=True) 

844 for axis, width_pair, length_pair in zip(axes, pad_width, length): 

845 roi = _view_roi(padded, original_area_slice, axis) 

846 stat_pair = _get_stats(roi, axis, width_pair, length_pair, func) 

847 _set_pad_area(roi, axis, width_pair, stat_pair) 

848

850 method = kwargs.get("reflect_type", "even") 

851 include_edge = True if mode == "symmetric" else False 

852 for axis, (left_index, right_index) in zip(axes, pad_width): 

853 if array.shape[axis] == 1 and (left_index > 0 or right_index > 0): 

854 # Extending singleton dimension for 'reflect' is legacy 

855 # behavior; it really should raise an error. 

856 edge_pair = _get_edges(padded, axis, (left_index, right_index)) 

857 _set_pad_area( 

858 padded, axis, (left_index, right_index), edge_pair) 

859 continue 

860

862 while left_index > 0 or right_index > 0: 

863 # Iteratively pad until dimension is filled with reflected 

864 # values. This is necessary if the pad area is larger than 

865 # the length of the original values in the current dimension.

867 roi, axis, (left_index, right_index), 

868 method, include_edge 

869 )

873 roi = _view_roi(padded, original_area_slice, axis) 

874 while left_index > 0 or right_index > 0: 

875 # Iteratively pad until dimension is filled with wrapped 

876 # values. This is necessary if the pad area is larger than 

877 # the length of the original values in the current dimension. 

878 left_index, right_index = _set_wrap_both( 

879 roi, axis, (left_index, right_index))
