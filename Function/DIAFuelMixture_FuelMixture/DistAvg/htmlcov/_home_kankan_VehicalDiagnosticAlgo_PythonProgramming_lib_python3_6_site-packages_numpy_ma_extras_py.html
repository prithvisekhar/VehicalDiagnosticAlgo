303class _fromnxfunction_args(_fromnxfunction): 

304 """ 

305 A version of `_fromnxfunction` that is called with multiple array 

306 arguments. The first non-array-like input marks the beginning of the 

307 arguments that are passed verbatim for both the data and mask calls. 

308 Array arguments are processed independently and the results are 

309 returned in a list. If only one array is found, the return value is 

310 just the processed array instead of a list. 

311 """ 

312 def __call__(self, *args, **params): 

313 func = getattr(np, self.__name__) 

314 arrays = [] 

315 args = list(args) 

316 while len(args) > 0 and issequence(args[0]): 

317 arrays.append(args.pop(0)) 

318 res = [] 

319 for x in arrays: 

320 _d = func(np.asarray(x), *args, **params) 

321 _m = func(getmaskarray(x), *args, **params) 

322 res.append(masked_array(_d, mask=_m)) 

323 if len(arrays) == 1: 

324 return res[0] 

325 return res

535def average(a, axis=None, weights=None, returned=False): 

536 """ 

537 Return the weighted average of array over the given axis. 

538 

539 Parameters 

540 ---------- 

541 a : array_like 

542 Data to be averaged. 

543 Masked entries are not taken into account in the computation. 

544 axis : int, optional 

545 Axis along which to average `a`. If None, averaging is done over 

546 the flattened array. 

547 weights : array_like, optional 

548 The importance that each element has in the computation of the average. 

549 The weights array can either be 1-D (in which case its length must be 

550 the size of `a` along the given axis) or of the same shape as `a`. 

551 If ``weights=None``, then all data in `a` are assumed to have a 

552 weight equal to one. The 1-D calculation is:: 

553 

554 avg = sum(a * weights) / sum(weights) 

555 

556 The only constraint on `weights` is that `sum(weights)` must not be 0. 

557 returned : bool, optional 

558 Flag indicating whether a tuple ``(result, sum of weights)`` 

559 should be returned as output (True), or just the result (False). 

560 Default is False. 

561 

562 Returns 

563 ------- 

564 average, [sum_of_weights] : (tuple of) scalar or MaskedArray 

565 The average along the specified axis. When returned is `True`, 

566 return a tuple with the average as the first element and the sum 

567 of the weights as the second element. The return type is `np.float64` 

568 if `a` is of integer type and floats smaller than `float64`, or the 

569 input data-type, otherwise. If returned, `sum_of_weights` is always 

570 `float64`. 

571 

572 Examples 

573 -------- 

574 >>> a = np.ma.array([1., 2., 3., 4.], mask=[False, False, True, True]) 

575 >>> np.ma.average(a, weights=[3, 1, 0, 0]) 

576 1.25 

577 

578 >>> x = np.ma.arange(6.).reshape(3, 2) 

579 >>> x 

580 masked_array( 

581 data=[[0., 1.], 

582 [2., 3.], 

583 [4., 5.]], 

584 mask=False, 

585 fill_value=1e+20) 

586 >>> avg, sumweights = np.ma.average(x, axis=0, weights=[1, 2, 3], 

587 ... returned=True) 

588 >>> avg 

589 masked_array(data=[2.6666666666666665, 3.6666666666666665], 

590 mask=[False, False], 

591 fill_value=1e+20) 

592 

593 """ 

594 a = asarray(a) 

595 m = getmask(a) 

596 

597 # inspired by 'average' in numpy/lib/function_base.py 

598 

599 if weights is None: 

600 avg = a.mean(axis) 

601 scl = avg.dtype.type(a.count(axis)) 

602 else: 

603 wgt = np.asanyarray(weights) 

604 

605 if issubclass(a.dtype.type, (np.integer, np.bool_)): 

606 result_dtype = np.result_type(a.dtype, wgt.dtype, 'f8') 

607 else: 

608 result_dtype = np.result_type(a.dtype, wgt.dtype) 

609 

610 # Sanity checks 

611 if a.shape != wgt.shape: 

612 if axis is None: 

613 raise TypeError( 

614 "Axis must be specified when shapes of a and weights " 

615 "differ.") 

616 if wgt.ndim != 1: 

617 raise TypeError( 

618 "1D weights expected when shapes of a and weights differ.") 

619 if wgt.shape[0] != a.shape[axis]: 

620 raise ValueError( 

621 "Length of weights not compatible with specified axis.") 

622 

623 # setup wgt to broadcast along axis 

624 wgt = np.broadcast_to(wgt, (a.ndim-1)*(1,) + wgt.shape) 

625 wgt = wgt.swapaxes(-1, axis) 

626 

627 if m is not nomask: 

628 wgt = wgt*(~a.mask) 

629 

630 scl = wgt.sum(axis=axis, dtype=result_dtype) 

631 avg = np.multiply(a, wgt, dtype=result_dtype).sum(axis)/scl 

632 

633 if returned: 

634 if scl.shape != avg.shape: 

635 scl = np.broadcast_to(scl, avg.shape).copy() 

636 return avg, scl 

637 else: 

638 return avg

723def _median(a, axis=None, out=None, overwrite_input=False): 

724 # when an unmasked NaN is present return it, so we need to sort the NaN 

725 # values behind the mask 

726 if np.issubdtype(a.dtype, np.inexact): 

727 fill_value = np.inf 

728 else: 

729 fill_value = None 

730 if overwrite_input: 

731 if axis is None: 

732 asorted = a.ravel() 

733 asorted.sort(fill_value=fill_value) 

734 else: 

735 a.sort(axis=axis, fill_value=fill_value) 

736 asorted = a 

737 else: 

738 asorted = sort(a, axis=axis, fill_value=fill_value) 

739 

740 if axis is None: 

741 axis = 0 

742 else: 

743 axis = normalize_axis_index(axis, asorted.ndim) 

744 

745 if asorted.shape[axis] == 0: 

746 # for empty axis integer indices fail so use slicing to get same result 

747 # as median (which is mean of empty slice = nan) 

748 indexer = [slice(None)] * asorted.ndim 

749 indexer[axis] = slice(0, 0) 

750 indexer = tuple(indexer) 

751 return np.ma.mean(asorted[indexer], axis=axis, out=out) 

752 

753 if asorted.ndim == 1: 

754 counts = count(asorted) 

755 idx, odd = divmod(count(asorted), 2) 

756 mid = asorted[idx + odd - 1:idx + 1] 

757 if np.issubdtype(asorted.dtype, np.inexact) and asorted.size > 0: 

758 # avoid inf / x = masked 

759 s = mid.sum(out=out) 

760 if not odd: 

761 s = np.true_divide(s, 2., casting='safe', out=out) 

762 s = np.lib.utils._median_nancheck(asorted, s, axis, out) 

763 else: 

764 s = mid.mean(out=out) 

765 

766 # if result is masked either the input contained enough 

767 # minimum_fill_value so that it would be the median or all values 

768 # masked 

769 if np.ma.is_masked(s) and not np.all(asorted.mask): 

770 return np.ma.minimum_fill_value(asorted) 

771 return s 

772 

773 counts = count(asorted, axis=axis, keepdims=True) 

774 h = counts // 2 

775 

776 # duplicate high if odd number of elements so mean does nothing 

777 odd = counts % 2 == 1 

778 l = np.where(odd, h, h-1) 

779 

780 lh = np.concatenate([l,h], axis=axis) 

781 

782 # get low and high median 

783 low_high = np.take_along_axis(asorted, lh, axis=axis)

786 # Replace masked entries with minimum_full_value unless it all values 

787 # are masked. This is required as the sort order of values equal or 

788 # larger than the fill value is undefined and a valid value placed 

789 # elsewhere, e.g. [4, --, inf]. 

790 if np.ma.is_masked(s): 

791 rep = (~np.all(asorted.mask, axis=axis, keepdims=True)) & s.mask 

792 s.data[rep] = np.ma.minimum_fill_value(asorted) 

793 s.mask[rep] = False

795 replace_masked(low_high) 

796 

797 if np.issubdtype(asorted.dtype, np.inexact): 

798 # avoid inf / x = masked 

799 s = np.ma.sum(low_high, axis=axis, out=out) 

800 np.true_divide(s.data, 2., casting='unsafe', out=s.data) 

801 

802 s = np.lib.utils._median_nancheck(asorted, s, axis, out) 

803 else: 

804 s = np.ma.mean(low_high, axis=axis, out=out)

1159 array.

1269 """ 

1270 Private function for the computation of covariance and correlation 

1271 coefficients.

1273 """
