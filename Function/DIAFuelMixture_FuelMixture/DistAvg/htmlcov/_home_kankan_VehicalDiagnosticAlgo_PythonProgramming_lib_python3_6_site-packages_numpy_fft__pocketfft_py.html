47# `inv_norm` is a float by which the result of the transform needs to be 

48# divided. This replaces the original, more intuitive 'fct` parameter to avoid 

49# divisions by zero (or alternatively additional checks) in the case of 

50# zero-length axes during its computation. 

51def _raw_fft(a, n, axis, is_real, is_forward, inv_norm): 

52 axis = normalize_axis_index(axis, a.ndim) 

53 if n is None: 

54 n = a.shape[axis] 

55 

56 if n < 1: 

57 raise ValueError("Invalid number of FFT data points (%d) specified." 

58 % n) 

59 

60 fct = 1/inv_norm 

61 

62 if a.shape[axis] != n: 

63 s = list(a.shape) 

64 if s[axis] > n: 

65 index = [slice(None)]*len(s) 

66 index[axis] = slice(0, n) 

67 a = a[tuple(index)] 

68 else: 

69 index = [slice(None)]*len(s) 

70 index[axis] = slice(0, s[axis]) 

71 s[axis] = n 

72 z = zeros(s, a.dtype.char) 

73 z[tuple(index)] = a 

74 a = z 

75 

76 if axis == a.ndim-1: 

77 r = pfi.execute(a, is_real, is_forward, fct) 

78 else: 

79 a = swapaxes(a, axis, -1) 

80 r = pfi.execute(a, is_real, is_forward, fct) 

81 r = swapaxes(r, axis, -1) 

82 return r

98@array_function_dispatch(_fft_dispatcher) 

99def fft(a, n=None, axis=-1, norm=None): 

100 """ 

101 Compute the one-dimensional discrete Fourier Transform. 

102 

103 This function computes the one-dimensional *n*-point discrete Fourier 

104 Transform (DFT) with the efficient Fast Fourier Transform (FFT) 

105 algorithm [CT]. 

106 

107 Parameters 

108 ---------- 

109 a : array_like 

110 Input array, can be complex. 

111 n : int, optional 

112 Length of the transformed axis of the output. 

113 If `n` is smaller than the length of the input, the input is cropped. 

114 If it is larger, the input is padded with zeros. If `n` is not given, 

115 the length of the input along the axis specified by `axis` is used. 

116 axis : int, optional 

117 Axis over which to compute the FFT. If not given, the last axis is 

118 used. 

119 norm : {None, "ortho"}, optional 

120 .. versionadded:: 1.10.0 

121 

122 Normalization mode (see `numpy.fft`). Default is None. 

123 

124 Returns 

125 ------- 

126 out : complex ndarray 

127 The truncated or zero-padded input, transformed along the axis 

128 indicated by `axis`, or the last one if `axis` is not specified. 

129 

130 Raises 

131 ------ 

132 IndexError 

133 if `axes` is larger than the last axis of `a`. 

134 

135 See Also 

136 -------- 

137 numpy.fft : for definition of the DFT and conventions used. 

138 ifft : The inverse of `fft`. 

139 fft2 : The two-dimensional FFT. 

140 fftn : The *n*-dimensional FFT. 

141 rfftn : The *n*-dimensional FFT of real input. 

142 fftfreq : Frequency bins for given FFT parameters. 

143 

144 Notes 

145 ----- 

146 FFT (Fast Fourier Transform) refers to a way the discrete Fourier 

147 Transform (DFT) can be calculated efficiently, by using symmetries in the 

148 calculated terms. The symmetry is highest when `n` is a power of 2, and 

149 the transform is therefore most efficient for these sizes. 

150 

151 The DFT is defined, with the conventions used in this implementation, in 

152 the documentation for the `numpy.fft` module. 

153 

154 References 

155 ---------- 

156 .. [CT] Cooley, James W., and John W. Tukey, 1965, "An algorithm for the 

157 machine calculation of complex Fourier series," *Math. Comput.* 

158 19: 297-301. 

159 

160 Examples 

161 -------- 

162 >>> np.fft.fft(np.exp(2j * np.pi * np.arange(8) / 8)) 

163 array([-2.33486982e-16+1.14423775e-17j, 8.00000000e+00-1.25557246e-15j, 

164 2.33486982e-16+2.33486982e-16j, 0.00000000e+00+1.22464680e-16j, 

165 -1.14423775e-17+2.33486982e-16j, 0.00000000e+00+5.20784380e-16j, 

166 1.14423775e-17+1.14423775e-17j, 0.00000000e+00+1.22464680e-16j]) 

167 

168 In this example, real input has an FFT which is Hermitian, i.e., symmetric 

169 in the real part and anti-symmetric in the imaginary part, as described in 

170 the `numpy.fft` documentation: 

171 

172 >>> import matplotlib.pyplot as plt 

173 >>> t = np.arange(256) 

174 >>> sp = np.fft.fft(np.sin(t)) 

175 >>> freq = np.fft.fftfreq(t.shape[-1]) 

176 >>> plt.plot(freq, sp.real, freq, sp.imag) 

177 [<matplotlib.lines.Line2D object at 0x...>, <matplotlib.lines.Line2D object at 0x...>] 

178 >>> plt.show() 

179 

180 """ 

181 

182 a = asarray(a) 

183 if n is None: 

184 n = a.shape[axis] 

185 inv_norm = 1 

186 if norm is not None and _unitary(norm): 

187 inv_norm = sqrt(n) 

188 output = _raw_fft(a, n, axis, False, True, inv_norm) 

189 return output

192@array_function_dispatch(_fft_dispatcher) 

193def ifft(a, n=None, axis=-1, norm=None): 

194 """ 

195 Compute the one-dimensional inverse discrete Fourier Transform. 

196 

197 This function computes the inverse of the one-dimensional *n*-point 

198 discrete Fourier transform computed by `fft`. In other words, 

199 ``ifft(fft(a)) == a`` to within numerical accuracy. 

200 For a general description of the algorithm and definitions, 

201 see `numpy.fft`. 

202 

203 The input should be ordered in the same way as is returned by `fft`, 

204 i.e., 

205 

206 * ``a[0]`` should contain the zero frequency term, 

207 * ``a[1:n//2]`` should contain the positive-frequency terms, 

208 * ``a[n//2 + 1:]`` should contain the negative-frequency terms, in 

209 increasing order starting from the most negative frequency. 

210 

211 For an even number of input points, ``A[n//2]`` represents the sum of 

212 the values at the positive and negative Nyquist frequencies, as the two 

213 are aliased together. See `numpy.fft` for details. 

214 

215 Parameters 

216 ---------- 

217 a : array_like 

218 Input array, can be complex. 

219 n : int, optional 

220 Length of the transformed axis of the output. 

221 If `n` is smaller than the length of the input, the input is cropped. 

222 If it is larger, the input is padded with zeros. If `n` is not given, 

223 the length of the input along the axis specified by `axis` is used. 

224 See notes about padding issues. 

225 axis : int, optional 

226 Axis over which to compute the inverse DFT. If not given, the last 

227 axis is used. 

228 norm : {None, "ortho"}, optional 

229 .. versionadded:: 1.10.0 

230 

231 Normalization mode (see `numpy.fft`). Default is None. 

232 

233 Returns 

234 ------- 

235 out : complex ndarray 

236 The truncated or zero-padded input, transformed along the axis 

237 indicated by `axis`, or the last one if `axis` is not specified. 

238 

239 Raises 

240 ------ 

241 IndexError 

242 If `axes` is larger than the last axis of `a`. 

243 

244 See Also 

245 -------- 

246 numpy.fft : An introduction, with definitions and general explanations. 

247 fft : The one-dimensional (forward) FFT, of which `ifft` is the inverse 

248 ifft2 : The two-dimensional inverse FFT. 

249 ifftn : The n-dimensional inverse FFT. 

250 

251 Notes 

252 ----- 

253 If the input parameter `n` is larger than the size of the input, the input 

254 is padded by appending zeros at the end. Even though this is the common 

255 approach, it might lead to surprising results. If a different padding is 

256 desired, it must be performed before calling `ifft`. 

257 

258 Examples 

259 -------- 

260 >>> np.fft.ifft([0, 4, 0, 0]) 

261 array([ 1.+0.j, 0.+1.j, -1.+0.j, 0.-1.j]) # may vary 

262 

263 Create and plot a band-limited signal with random phases: 

264 

265 >>> import matplotlib.pyplot as plt 

266 >>> t = np.arange(400) 

267 >>> n = np.zeros((400,), dtype=complex) 

268 >>> n[40:60] = np.exp(1j*np.random.uniform(0, 2*np.pi, (20,))) 

269 >>> s = np.fft.ifft(n) 

270 >>> plt.plot(t, s.real, 'b-', t, s.imag, 'r--') 

271 [<matplotlib.lines.Line2D object at ...>, <matplotlib.lines.Line2D object at ...>] 

272 >>> plt.legend(('real', 'imaginary')) 

273 <matplotlib.legend.Legend object at ...> 

274 >>> plt.show() 

275 

276 """ 

277 a = asarray(a) 

278 if n is None: 

279 n = a.shape[axis] 

280 if norm is not None and _unitary(norm): 

281 inv_norm = sqrt(max(n, 1)) 

282 else: 

283 inv_norm = n 

284 output = _raw_fft(a, n, axis, False, False, inv_norm) 

285 return output

289@array_function_dispatch(_fft_dispatcher) 

290def rfft(a, n=None, axis=-1, norm=None): 

291 """ 

292 Compute the one-dimensional discrete Fourier Transform for real input. 

293 

294 This function computes the one-dimensional *n*-point discrete Fourier 

295 Transform (DFT) of a real-valued array by means of an efficient algorithm 

296 called the Fast Fourier Transform (FFT). 

297 

298 Parameters 

299 ---------- 

300 a : array_like 

301 Input array 

302 n : int, optional 

303 Number of points along transformation axis in the input to use. 

304 If `n` is smaller than the length of the input, the input is cropped. 

305 If it is larger, the input is padded with zeros. If `n` is not given, 

306 the length of the input along the axis specified by `axis` is used. 

307 axis : int, optional 

308 Axis over which to compute the FFT. If not given, the last axis is 

309 used. 

310 norm : {None, "ortho"}, optional 

311 .. versionadded:: 1.10.0 

312 

313 Normalization mode (see `numpy.fft`). Default is None. 

314 

315 Returns 

316 ------- 

317 out : complex ndarray 

318 The truncated or zero-padded input, transformed along the axis 

319 indicated by `axis`, or the last one if `axis` is not specified. 

320 If `n` is even, the length of the transformed axis is ``(n/2)+1``. 

321 If `n` is odd, the length is ``(n+1)/2``. 

322 

323 Raises 

324 ------ 

325 IndexError 

326 If `axis` is larger than the last axis of `a`. 

327 

328 See Also 

329 -------- 

330 numpy.fft : For definition of the DFT and conventions used. 

331 irfft : The inverse of `rfft`. 

332 fft : The one-dimensional FFT of general (complex) input. 

333 fftn : The *n*-dimensional FFT. 

334 rfftn : The *n*-dimensional FFT of real input. 

335 

336 Notes 

337 ----- 

338 When the DFT is computed for purely real input, the output is 

339 Hermitian-symmetric, i.e. the negative frequency terms are just the complex 

340 conjugates of the corresponding positive-frequency terms, and the 

341 negative-frequency terms are therefore redundant. This function does not 

342 compute the negative frequency terms, and the length of the transformed 

343 axis of the output is therefore ``n//2 + 1``. 

344 

345 When ``A = rfft(a)`` and fs is the sampling frequency, ``A[0]`` contains 

346 the zero-frequency term 0*fs, which is real due to Hermitian symmetry. 

347 

348 If `n` is even, ``A[-1]`` contains the term representing both positive 

349 and negative Nyquist frequency (+fs/2 and -fs/2), and must also be purely 

350 real. If `n` is odd, there is no term at fs/2; ``A[-1]`` contains 

351 the largest positive frequency (fs/2*(n-1)/n), and is complex in the 

352 general case. 

353 

354 If the input `a` contains an imaginary part, it is silently discarded. 

355 

356 Examples 

357 -------- 

358 >>> np.fft.fft([0, 1, 0, 0]) 

359 array([ 1.+0.j, 0.-1.j, -1.+0.j, 0.+1.j]) # may vary 

360 >>> np.fft.rfft([0, 1, 0, 0]) 

361 array([ 1.+0.j, 0.-1.j, -1.+0.j]) # may vary 

362 

363 Notice how the final element of the `fft` output is the complex conjugate 

364 of the second element, for real input. For `rfft`, this symmetry is 

365 exploited to compute only the non-negative frequency terms. 

366 

367 """ 

368 a = asarray(a) 

369 inv_norm = 1 

370 if norm is not None and _unitary(norm): 

371 if n is None: 

372 n = a.shape[axis] 

373 inv_norm = sqrt(n) 

374 output = _raw_fft(a, n, axis, True, True, inv_norm) 

375 return output

378@array_function_dispatch(_fft_dispatcher) 

379def irfft(a, n=None, axis=-1, norm=None): 

380 """ 

381 Compute the inverse of the n-point DFT for real input. 

382 

383 This function computes the inverse of the one-dimensional *n*-point 

384 discrete Fourier Transform of real input computed by `rfft`. 

385 In other words, ``irfft(rfft(a), len(a)) == a`` to within numerical 

386 accuracy. (See Notes below for why ``len(a)`` is necessary here.) 

387 

388 The input is expected to be in the form returned by `rfft`, i.e. the 

389 real zero-frequency term followed by the complex positive frequency terms 

390 in order of increasing frequency. Since the discrete Fourier Transform of 

391 real input is Hermitian-symmetric, the negative frequency terms are taken 

392 to be the complex conjugates of the corresponding positive frequency terms. 

393 

394 Parameters 

395 ---------- 

396 a : array_like 

397 The input array. 

398 n : int, optional 

399 Length of the transformed axis of the output. 

400 For `n` output points, ``n//2+1`` input points are necessary. If the 

401 input is longer than this, it is cropped. If it is shorter than this, 

402 it is padded with zeros. If `n` is not given, it is taken to be 

403 ``2*(m-1)`` where ``m`` is the length of the input along the axis 

404 specified by `axis`. 

405 axis : int, optional 

406 Axis over which to compute the inverse FFT. If not given, the last 

407 axis is used. 

408 norm : {None, "ortho"}, optional 

409 .. versionadded:: 1.10.0 

410 

411 Normalization mode (see `numpy.fft`). Default is None. 

412 

413 Returns 

414 ------- 

415 out : ndarray 

416 The truncated or zero-padded input, transformed along the axis 

417 indicated by `axis`, or the last one if `axis` is not specified. 

418 The length of the transformed axis is `n`, or, if `n` is not given, 

419 ``2*(m-1)`` where ``m`` is the length of the transformed axis of the 

420 input. To get an odd number of output points, `n` must be specified. 

421 

422 Raises 

423 ------ 

424 IndexError 

425 If `axis` is larger than the last axis of `a`. 

426 

427 See Also 

428 -------- 

429 numpy.fft : For definition of the DFT and conventions used. 

430 rfft : The one-dimensional FFT of real input, of which `irfft` is inverse. 

431 fft : The one-dimensional FFT. 

432 irfft2 : The inverse of the two-dimensional FFT of real input. 

433 irfftn : The inverse of the *n*-dimensional FFT of real input. 

434 

435 Notes 

436 ----- 

437 Returns the real valued `n`-point inverse discrete Fourier transform 

438 of `a`, where `a` contains the non-negative frequency terms of a 

439 Hermitian-symmetric sequence. `n` is the length of the result, not the 

440 input. 

441 

442 If you specify an `n` such that `a` must be zero-padded or truncated, the 

443 extra/removed values will be added/removed at high frequencies. One can 

444 thus resample a series to `m` points via Fourier interpolation by: 

445 ``a_resamp = irfft(rfft(a), m)``. 

446 

447 The correct interpretation of the hermitian input depends on the length of 

448 the original data, as given by `n`. This is because each input shape could 

449 correspond to either an odd or even length signal. By default, `irfft` 

450 assumes an even output length which puts the last entry at the Nyquist 

451 frequency; aliasing with its symmetric counterpart. By Hermitian symmetry, 

452 the value is thus treated as purely real. To avoid losing information, the 

453 correct length of the real input **must** be given. 

454 

455 Examples 

456 -------- 

457 >>> np.fft.ifft([1, -1j, -1, 1j]) 

458 array([0.+0.j, 1.+0.j, 0.+0.j, 0.+0.j]) # may vary 

459 >>> np.fft.irfft([1, -1j, -1]) 

460 array([0., 1., 0., 0.]) 

461 

462 Notice how the last term in the input to the ordinary `ifft` is the 

463 complex conjugate of the second term, and the output has zero imaginary 

464 part everywhere. When calling `irfft`, the negative frequencies are not 

465 specified, and the output array is purely real. 

466 

467 """ 

468 a = asarray(a) 

469 if n is None: 

470 n = (a.shape[axis] - 1) * 2 

471 inv_norm = n 

472 if norm is not None and _unitary(norm): 

473 inv_norm = sqrt(n) 

474 output = _raw_fft(a, n, axis, True, False, inv_norm) 

475 return output

478@array_function_dispatch(_fft_dispatcher) 

479def hfft(a, n=None, axis=-1, norm=None): 

480 """ 

481 Compute the FFT of a signal that has Hermitian symmetry, i.e., a real 

482 spectrum. 

483 

484 Parameters 

485 ---------- 

486 a : array_like 

487 The input array. 

488 n : int, optional 

489 Length of the transformed axis of the output. For `n` output 

490 points, ``n//2 + 1`` input points are necessary. If the input is 

491 longer than this, it is cropped. If it is shorter than this, it is 

492 padded with zeros. If `n` is not given, it is taken to be ``2*(m-1)`` 

493 where ``m`` is the length of the input along the axis specified by 

494 `axis`. 

495 axis : int, optional 

496 Axis over which to compute the FFT. If not given, the last 

497 axis is used. 

498 norm : {None, "ortho"}, optional 

499 Normalization mode (see `numpy.fft`). Default is None. 

500 

501 .. versionadded:: 1.10.0 

502 

503 Returns 

504 ------- 

505 out : ndarray 

506 The truncated or zero-padded input, transformed along the axis 

507 indicated by `axis`, or the last one if `axis` is not specified. 

508 The length of the transformed axis is `n`, or, if `n` is not given, 

509 ``2*m - 2`` where ``m`` is the length of the transformed axis of 

510 the input. To get an odd number of output points, `n` must be 

511 specified, for instance as ``2*m - 1`` in the typical case, 

512 

513 Raises 

514 ------ 

515 IndexError 

516 If `axis` is larger than the last axis of `a`. 

517 

518 See also 

519 -------- 

520 rfft : Compute the one-dimensional FFT for real input. 

521 ihfft : The inverse of `hfft`. 

522 

523 Notes 

524 ----- 

525 `hfft`/`ihfft` are a pair analogous to `rfft`/`irfft`, but for the 

526 opposite case: here the signal has Hermitian symmetry in the time 

527 domain and is real in the frequency domain. So here it's `hfft` for 

528 which you must supply the length of the result if it is to be odd. 

529 

530 * even: ``ihfft(hfft(a, 2*len(a) - 2) == a``, within roundoff error, 

531 * odd: ``ihfft(hfft(a, 2*len(a) - 1) == a``, within roundoff error. 

532 

533 The correct interpretation of the hermitian input depends on the length of 

534 the original data, as given by `n`. This is because each input shape could 

535 correspond to either an odd or even length signal. By default, `hfft` 

536 assumes an even output length which puts the last entry at the Nyquist 

537 frequency; aliasing with its symmetric counterpart. By Hermitian symmetry, 

538 the value is thus treated as purely real. To avoid losing information, the 

539 shape of the full signal **must** be given. 

540 

541 Examples 

542 -------- 

543 >>> signal = np.array([1, 2, 3, 4, 3, 2]) 

544 >>> np.fft.fft(signal) 

545 array([15.+0.j, -4.+0.j, 0.+0.j, -1.-0.j, 0.+0.j, -4.+0.j]) # may vary 

546 >>> np.fft.hfft(signal[:4]) # Input first half of signal 

547 array([15., -4., 0., -1., 0., -4.]) 

548 >>> np.fft.hfft(signal, 6) # Input entire signal and truncate 

549 array([15., -4., 0., -1., 0., -4.]) 

550 

551 

552 >>> signal = np.array([[1, 1.j], [-1.j, 2]]) 

553 >>> np.conj(signal.T) - signal # check Hermitian symmetry 

554 array([[ 0.-0.j, -0.+0.j], # may vary 

555 [ 0.+0.j, 0.-0.j]]) 

556 >>> freq_spectrum = np.fft.hfft(signal) 

557 >>> freq_spectrum 

558 array([[ 1., 1.], 

559 [ 2., -2.]]) 

560 

561 """ 

562 a = asarray(a) 

563 if n is None: 

564 n = (a.shape[axis] - 1) * 2 

565 unitary = _unitary(norm) 

566 return irfft(conjugate(a), n, axis) * (sqrt(n) if unitary else n)

572 Compute the inverse FFT of a signal that has Hermitian symmetry. 

573 

574 Parameters

578 n : int, optional

582 the input is padded with zeros. If `n` is not given, the length of 

583 the input along the axis specified by `axis` is used. 

584 axis : int, optional 

585 Axis over which to compute the inverse FFT. If not given, the last 

586 axis is used. 

587 norm : {None, "ortho"}, optional 

588 Normalization mode (see `numpy.fft`). Default is None.

590 .. versionadded:: 1.10.0 

591

593 ------- 

594 out : complex ndarray 

595 The truncated or zero-padded input, transformed along the axis

597 The length of the transformed axis is ``n//2 + 1``. 

598 

599 See also 

600 -------- 

601 hfft, irfft 

602 

603 Notes 

604 ----- 

605 `hfft`/`ihfft` are a pair analogous to `rfft`/`irfft`, but for the 

606 opposite case: here the signal has Hermitian symmetry in the time 

607 domain and is real in the frequency domain. So here it's `hfft` for 

608 which you must supply the length of the result if it is to be odd: 

609 

610 * even: ``ihfft(hfft(a, 2*len(a) - 2) == a``, within roundoff error, 

611 * odd: ``ihfft(hfft(a, 2*len(a) - 1) == a``, within roundoff error. 

612 

613 Examples 

614 -------- 

615 >>> spectrum = np.array([ 15, -4, 0, -1, 0, -4]) 

616 >>> np.fft.ifft(spectrum) 

617 array([1.+0.j, 2.+0.j, 3.+0.j, 4.+0.j, 3.+0.j, 2.+0.j]) # may vary 

618 >>> np.fft.ihfft(spectrum) 

619 array([ 1.-0.j, 2.-0.j, 3.-0.j, 4.-0.j]) # may vary 

620 

621 """ 

622 a = asarray(a) 

623 if n is None: 

624 n = a.shape[axis] 

625 unitary = _unitary(norm) 

626 output = conjugate(rfft(a, n, axis)) 

627 return output * (1 / (sqrt(n) if unitary else n)) 

628 

629 

630def _cook_nd_args(a, s=None, axes=None, invreal=0): 

631 if s is None: 

632 shapeless = 1 

633 if axes is None: 

634 s = list(a.shape) 

635 else: 

636 s = take(a.shape, axes) 

637 else: 

638 shapeless = 0 

639 s = list(s) 

640 if axes is None: 

641 axes = list(range(-len(s), 0)) 

642 if len(s) != len(axes): 

643 raise ValueError("Shape and axes have different lengths.") 

644 if invreal and shapeless: 

645 s[-1] = (a.shape[axes[-1]] - 1) * 2 

646 return s, axes

649def _raw_fftnd(a, s=None, axes=None, function=fft, norm=None): 

650 a = asarray(a) 

651 s, axes = _cook_nd_args(a, s, axes) 

652 itl = list(range(len(axes))) 

653 itl.reverse() 

654 for ii in itl: 

655 a = function(a, n=s[ii], axis=axes[ii], norm=norm) 

656 return a

1038@array_function_dispatch(_fftn_dispatcher) 

1039def rfftn(a, s=None, axes=None, norm=None): 

1040 """ 

1041 Compute the N-dimensional discrete Fourier Transform for real input. 

1042 

1043 This function computes the N-dimensional discrete Fourier Transform over 

1044 any number of axes in an M-dimensional real array by means of the Fast

1049 Parameters

1051 a : array_like 

1052 Input array, taken to be real. 

1053 s : sequence of ints, optional 

1054 Shape (length along each transformed axis) to use from the input. 

1055 (``s[0]`` refers to axis 0, ``s[1]`` to axis 1, etc.). 

1056 The final element of `s` corresponds to `n` for ``rfft(x, n)``, while 

1057 for the remaining axes, it corresponds to `n` for ``fft(x, n)``. 

1058 Along any axis, if the given shape is smaller than that of the input, 

1059 the input is cropped. If it is larger, the input is padded with zeros. 

1060 if `s` is not given, the shape of the input along the axes specified 

1061 by `axes` is used. 

1062 axes : sequence of ints, optional 

1063 Axes over which to compute the FFT. If not given, the last ``len(s)`` 

1064 axes are used, or all axes if `s` is also not specified. 

1065 norm : {None, "ortho"}, optional 

1066 .. versionadded:: 1.10.0 

1067 

1068 Normalization mode (see `numpy.fft`). Default is None. 

1069 

1070 Returns 

1071 ------- 

1072 out : complex ndarray 

1073 The truncated or zero-padded input, transformed along the axes 

1074 indicated by `axes`, or by a combination of `s` and `a`, 

1075 as explained in the parameters section above. 

1076 The length of the last axis transformed will be ``s[-1]//2+1``, 

1077 while the remaining transformed axes will have lengths according to 

1078 `s`, or unchanged from the input. 

1079 

1080 Raises 

1081 ------ 

1082 ValueError 

1083 If `s` and `axes` have different length. 

1084 IndexError 

1085 If an element of `axes` is larger than than the number of axes of `a`. 

1086 

1087 See Also 

1088 -------- 

1089 irfftn : The inverse of `rfftn`, i.e. the inverse of the n-dimensional FFT 

1090 of real input. 

1091 fft : The one-dimensional FFT, with definitions and conventions used. 

1092 rfft : The one-dimensional FFT of real input. 

1093 fftn : The n-dimensional FFT. 

1094 rfft2 : The two-dimensional FFT of real input. 

1095 

1096 Notes 

1097 ----- 

1098 The transform for real input is performed over the last transformation 

1099 axis, as by `rfft`, then the transform over the remaining axes is 

1100 performed as by `fftn`. The order of the output is as for `rfft` for the 

1101 final transformation axis, and as for `fftn` for the remaining 

1102 transformation axes. 

1103 

1104 See `fft` for details, definitions and conventions used. 

1105 

1106 Examples 

1107 -------- 

1108 >>> a = np.ones((2, 2, 2))

1118 [[0.+0.j, 0.+0.j],

1124 a = rfft(a, s[-1], axes[-1], norm) 

1125 for ii in range(len(axes)-1): 

1126 a = fft(a, s[ii], axes[ii], norm) 

1127 return a

1174 Fourier Transform for real input over any number of axes in an 

1175 M-dimensional array by means of the Fast Fourier Transform (FFT). In 

1176 other words, ``irfftn(rfftn(a), a.shape) == a`` to within numerical 

1177 accuracy. (The ``a.shape`` is necessary like ``len(a)`` is for `irfft`, 

1178 and for the same reason.) 

1179 

1180 The input should be ordered in the same way as is returned by `rfftn`, 

1181 i.e. as for `irfft` for the final transformation axis, and as for `ifftn` 

1182 along all the other axes. 

1183 

1184 Parameters 

1185 ---------- 

1186 a : array_like 

1187 Input array. 

1188 s : sequence of ints, optional 

1189 Shape (length of each transformed axis) of the output 

1190 (``s[0]`` refers to axis 0, ``s[1]`` to axis 1, etc.). `s` is also the 

1191 number of input points used along this axis, except for the last axis, 

1192 where ``s[-1]//2+1`` points of the input are used. 

1193 Along any axis, if the shape indicated by `s` is smaller than that of 

1194 the input, the input is cropped. If it is larger, the input is padded 

1195 with zeros. If `s` is not given, the shape of the input along the axes 

1196 specified by axes is used. Except for the last axis which is taken to be 

1197 ``2*(m-1)`` where ``m`` is the length of the input along that axis. 

1198 axes : sequence of ints, optional 

1199 Axes over which to compute the inverse FFT. If not given, the last 

1200 `len(s)` axes are used, or all axes if `s` is also not specified. 

1201 Repeated indices in `axes` means that the inverse transform over that 

1202 axis is performed multiple times. 

1203 norm : {None, "ortho"}, optional 

1204 .. versionadded:: 1.10.0 

1205 

1206 Normalization mode (see `numpy.fft`). Default is None. 

1207 

1208 Returns 

1209 ------- 

1210 out : ndarray 

1211 The truncated or zero-padded input, transformed along the axes 

1212 indicated by `axes`, or by a combination of `s` or `a`, 

1213 as explained in the parameters section above. 

1214 The length of each transformed axis is as given by the corresponding 

1215 element of `s`, or the length of the input in every axis except for the 

1216 last one if `s` is not given. In the final transformed axis the length 

1217 of the output when `s` is not given is ``2*(m-1)`` where ``m`` is the 

1218 length of the final transformed axis of the input. To get an odd 

1219 number of output points in the final axis, `s` must be specified. 

1220 

1221 Raises 

1222 ------ 

1223 ValueError 

1224 If `s` and `axes` have different length. 

1225 IndexError 

1226 If an element of `axes` is larger than than the number of axes of `a`. 

1227 

1228 See Also 

1229 -------- 

1230 rfftn : The forward n-dimensional FFT of real input, 

1231 of which `ifftn` is the inverse. 

1232 fft : The one-dimensional FFT, with definitions and conventions used. 

1233 irfft : The inverse of the one-dimensional FFT of real input. 

1234 irfft2 : The inverse of the two-dimensional FFT of real input. 

1235 

1236 Notes 

1237 ----- 

1238 See `fft` for definitions and conventions used. 

1239 

1240 See `rfft` for definitions and conventions used for real input. 

1241 

1242 The correct interpretation of the hermitian input depends on the shape of 

1243 the original data, as given by `s`. This is because each input shape could 

1244 correspond to either an odd or even length signal. By default, `irfftn` 

1245 assumes an even output length which puts the last entry at the Nyquist 

1246 frequency; aliasing with its symmetric counterpart. When performing the 

1247 final complex to real transform, the last value is thus treated as purely 

1248 real. To avoid losing information, the correct shape of the real input 

1249 **must** be given. 

1250 

1251 Examples 

1252 -------- 

1253 >>> a = np.zeros((3, 2, 2)) 

1254 >>> a[0, 0, 0] = 3 * 2 * 2 

1255 >>> np.fft.irfftn(a) 

1256 array([[[1., 1.], 

1257 [1., 1.]], 

1258 [[1., 1.], 

1259 [1., 1.]], 

1260 [[1., 1.], 

1261 [1., 1.]]]) 

1262 

1263 """ 

1264 a = asarray(a) 

1265 s, axes = _cook_nd_args(a, s, axes, invreal=1) 

1266 for ii in range(len(axes)-1): 

1267 a = ifft(a, s[ii], axes[ii], norm) 

1268 a = irfft(a, s[-1], axes[-1], norm) 

1269 return a
