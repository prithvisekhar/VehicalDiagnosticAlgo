1225 def _map_values(self, mapper, na_action=None): 

1226 """ 

1227 An internal function that maps values using the input 

1228 correspondence (which can be a dict, Series, or function). 

1229 

1230 Parameters 

1231 ---------- 

1232 mapper : function, dict, or Series 

1233 The input correspondence object 

1234 na_action : {None, 'ignore'} 

1235 If 'ignore', propagate NA values, without passing them to the 

1236 mapping function 

1237 

1238 Returns 

1239 ------- 

1240 Union[Index, MultiIndex], inferred 

1241 The output of the mapping function applied to the index. 

1242 If the function returns a tuple with more than one element 

1243 a MultiIndex will be returned. 

1244 

1245 """ 

1246 

1247 # we can fastpath dict/Series to an efficient map 

1248 # as we know that we are not going to have to yield 

1249 # python types 

1250 if isinstance(mapper, dict): 

1251 if hasattr(mapper, "__missing__"): 

1252 # If a dictionary subclass defines a default value method, 

1253 # convert mapper to a lookup function (GH #15999). 

1254 dict_with_default = mapper 

1255 mapper = lambda x: dict_with_default[x] 

1256 else: 

1257 # Dictionary does not have a default. Thus it's safe to 

1258 # convert to an Series for efficiency. 

1259 # we specify the keys here to handle the 

1260 # possibility that they are tuples 

1261 from pandas import Series 

1262 

1263 mapper = Series(mapper) 

1264 

1265 if isinstance(mapper, ABCSeries): 

1266 # Since values were input this means we came from either 

1267 # a dict or a series and mapper should be an index 

1268 if is_categorical_dtype(self._values): 

1269 # use the built in categorical series mapper which saves 

1270 # time by mapping the categories instead of all values 

1271 return self._values.map(mapper) 

1272 if is_extension_type(self.dtype): 

1273 values = self._values 

1274 else: 

1275 values = self.values 

1276 

1277 indexer = mapper.index.get_indexer(values) 

1278 new_values = algorithms.take_1d(mapper._values, indexer) 

1279 

1280 return new_values 

1281 

1282 # we must convert to python types 

1283 if is_extension_type(self.dtype): 

1284 values = self._values 

1285 if na_action is not None: 

1286 raise NotImplementedError 

1287 map_f = lambda values, f: values.map(f) 

1288 else: 

1289 values = self.astype(object) 

1290 values = getattr(values, "values", values) 

1291 if na_action == "ignore": 

1292 

1293 def map_f(values, f): 

1294 return lib.map_infer_mask(values, f, isna(values).view(np.uint8)) 

1295 

1296 else: 

1297 map_f = lib.map_infer 

1298 

1299 # mapper is a function 

1300 new_values = map_f(values, mapper) 

1301 

1302 return new_values
