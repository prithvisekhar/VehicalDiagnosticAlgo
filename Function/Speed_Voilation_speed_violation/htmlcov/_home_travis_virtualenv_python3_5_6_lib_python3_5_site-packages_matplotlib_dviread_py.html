36# Many dvi related files are looked for by external processes, require 

37# additional parsing, and are used many times per rendering, which is why they 

38# are cached using lru_cache(). 

39 

40# Dvi is a bytecode format documented in 

41# http://mirrors.ctan.org/systems/knuth/dist/texware/dvitype.web 

42# http://texdoc.net/texmf-dist/doc/generic/knuth/texware/dvitype.pdf 

43# 

44# The file consists of a preamble, some number of pages, a postamble, 

45# and a finale. Different opcodes are allowed in different contexts, 

46# so the Dvi object has a parser state: 

47# 

48# pre: expecting the preamble 

49# outer: between pages (followed by a page or the postamble, 

50# also e.g. font definitions are allowed) 

51# page: processing a page 

52# post_post: state after the postamble (our current implementation 

53# just stops reading) 

54# finale: the finale (unimplemented in our current implementation) 

55 

56_dvistate = enum.Enum('DviState', 'pre outer inpage post_post finale') 

57 

58# The marks on a page consist of text and boxes. A page also has dimensions. 

59Page = namedtuple('Page', 'text boxes height width descent') 

60Text = namedtuple('Text', 'x y font glyph width') 

61Box = namedtuple('Box', 'x y height width')

64# Opcode argument parsing 

65# 

66# Each of the following functions takes a Dvi object and delta, 

67# which is the difference between the opcode and the minimum opcode 

68# with the same meaning. Dvi opcodes often encode the number of 

69# argument bytes in this delta. 

70 

71def _arg_raw(dvi, delta): 

72 """Return *delta* without reading anything more from the dvi file""" 

73 return delta

76def _arg(bytes, signed, dvi, _): 

77 """Read *bytes* bytes, returning the bytes interpreted as a 

78 signed integer if *signed* is true, unsigned otherwise.""" 

79 return dvi._arg(bytes, signed)

82def _arg_slen(dvi, delta): 

83 """Signed, length *delta*

85 Read *delta* bytes, returning None if *delta* is zero, and

88 return None

124def _dispatch(table, min, max=None, state=None, args=('raw',)): 

125 """Decorator for dispatch by opcode. Sets the values in *table* 

126 from *min* to *max* to this method, adds a check that the Dvi state 

127 matches *state* if not None, reads arguments from the file according 

128 to *args*. 

129 

130 *table* 

131 the dispatch table to be filled in 

132 

133 *min* 

134 minimum opcode for calling this function 

135 

136 *max* 

137 maximum opcode for calling this function, None if only *min* is allowed 

138 

139 *state* 

140 state of the Dvi object in which these opcodes are allowed 

141 

142 *args* 

143 sequence of argument specifications: 

144 

145 ``'raw'``: opcode minus minimum 

146 ``'u1'``: read one unsigned byte 

147 ``'u4'``: read four bytes, treat as an unsigned number 

148 ``'s4'``: read four bytes, treat as a signed number 

149 ``'slen'``: read (opcode - minimum) bytes, treat as signed 

150 ``'slen1'``: read (opcode - minimum + 1) bytes, treat as signed 

151 ``'ulen1'``: read (opcode - minimum + 1) bytes, treat as unsigned 

152 ``'olen1'``: read (opcode - minimum + 1) bytes, treat as unsigned 

153 if under four bytes, signed if four bytes 

154 """ 

155 def decorate(method): 

156 get_args = [_arg_mapping[x] for x in args] 

157 

158 @wraps(method) 

159 def wrapper(self, byte): 

160 if state is not None and self.state != state: 

161 raise ValueError("state precondition failed") 

162 return method(self, *[f(self, byte-min) for f in get_args]) 

163 if max is None: 

164 table[min] = wrapper 

165 else: 

166 for i in range(min, max+1): 

167 assert table[i] is None 

168 table[i] = wrapper 

169 return wrapper 

170 return decorate

173class Dvi(object): 

174 """ 

175 A reader for a dvi ("device-independent") file, as produced by TeX. 

176 The current implementation can only iterate through pages in order, 

177 and does not even attempt to verify the postamble. 

178 

179 This class can be used as a context manager to close the underlying 

180 file upon exit. Pages can be read via iteration. Here is an overly 

181 simple way to extract text without trying to detect whitespace:: 

182 

183 >>> with matplotlib.dviread.Dvi('input.dvi', 72) as dvi: 

184 ... for page in dvi: 

185 ... print(''.join(chr(t.glyph) for t in page.text)) 

186 """ 

187 # dispatch table 

188 _dtable = [None] * 256 

189 _dispatch = partial(_dispatch, _dtable) 

190 

191 def __init__(self, filename, dpi): 

192 """ 

193 Read the data from the file named *filename* and convert 

194 TeX's internal units to units of *dpi* per inch. 

195 *dpi* only sets the units and does not limit the resolution. 

196 Use None to return TeX's internal units. 

197 """ 

198 _log.debug('Dvi: %s', filename) 

199 self.file = open(filename, 'rb') 

200 self.dpi = dpi 

201 self.fonts = {} 

202 self.state = _dvistate.pre 

203 self.baseline = self._get_baseline(filename) 

204 

205 def _get_baseline(self, filename): 

206 if rcParams['text.latex.preview']: 

207 base, ext = os.path.splitext(filename) 

208 baseline_filename = base + ".baseline" 

209 if os.path.exists(baseline_filename): 

210 with open(baseline_filename, 'rb') as fd: 

211 l = fd.read().split() 

212 height, depth, width = l 

213 return float(depth) 

214 return None 

215 

216 def __enter__(self): 

217 """ 

218 Context manager enter method, does nothing. 

219 """ 

220 return self 

221 

222 def __exit__(self, etype, evalue, etrace): 

223 """ 

224 Context manager exit method, closes the underlying file if it is open. 

225 """ 

226 self.close() 

227 

228 def __iter__(self): 

229 """ 

230 Iterate through the pages of the file. 

231 

232 Yields 

233 ------ 

234 Page 

235 Details of all the text and box objects on the page. 

236 The Page tuple contains lists of Text and Box tuples and 

237 the page dimensions, and the Text and Box tuples contain 

238 coordinates transformed into a standard Cartesian 

239 coordinate system at the dpi value given when initializing. 

240 The coordinates are floating point numbers, but otherwise 

241 precision is not lost and coordinate values are not clipped to 

242 integers. 

243 """ 

244 while self._read(): 

245 yield self._output() 

246 

247 def close(self): 

248 """ 

249 Close the underlying file if it is open. 

250 """ 

251 if not self.file.closed: 

252 self.file.close()

254 def _output(self): 

255 """ 

256 Output the text and boxes belonging to the most recent page. 

257 page = dvi._output() 

258 """ 

259 minx, miny, maxx, maxy = np.inf, np.inf, -np.inf, -np.inf 

260 maxy_pure = -np.inf 

261 for elt in self.text + self.boxes: 

262 if isinstance(elt, Box): 

263 x, y, h, w = elt 

264 e = 0 # zero depth 

265 else: # glyph 

266 x, y, font, g, w = elt 

267 h, e = font._height_depth_of(g) 

268 minx = min(minx, x) 

269 miny = min(miny, y - h) 

270 maxx = max(maxx, x + w) 

271 maxy = max(maxy, y + e) 

272 maxy_pure = max(maxy_pure, y) 

273 

274 if self.dpi is None: 

275 # special case for ease of debugging: output raw dvi coordinates 

276 return Page(text=self.text, boxes=self.boxes, 

277 width=maxx-minx, height=maxy_pure-miny, 

278 descent=maxy-maxy_pure) 

279 

280 # convert from TeX's "scaled points" to dpi units 

281 d = self.dpi / (72.27 * 2**16) 

282 if self.baseline is None: 

283 descent = (maxy - maxy_pure) * d 

284 else: 

285 descent = self.baseline 

286 

287 text = [Text((x-minx)*d, (maxy-y)*d - descent, f, g, w*d) 

288 for (x, y, f, g, w) in self.text] 

289 boxes = [Box((x-minx)*d, (maxy-y)*d - descent, h*d, w*d) 

290 for (x, y, h, w) in self.boxes] 

291 

292 return Page(text=text, boxes=boxes, width=(maxx-minx)*d, 

293 height=(maxy_pure-miny)*d, descent=descent)

295 def _read(self): 

296 """ 

297 Read one page from the file. Return True if successful, 

298 False if there were no more pages. 

299 """ 

300 while True: 

301 byte = self.file.read(1)[0] 

302 self._dtable[byte](self, byte) 

303 if byte == 140: # end of page 

304 return True 

305 if self.state is _dvistate.post_post: # end of file 

306 self.close() 

307 return False

309 def _arg(self, nbytes, signed=False): 

310 """ 

311 Read and return an integer argument *nbytes* long. 

312 Signedness is determined by the *signed* keyword. 

313 """ 

314 str = self.file.read(nbytes) 

315 value = str[0] 

316 if signed and value >= 0x80: 

317 value = value - 0x100 

318 for i in range(1, nbytes): 

319 value = 0x100*value + str[i] 

320 return value

490class DviFont(object): 

491 """ 

492 Encapsulation of a font that a DVI file can refer to. 

493 

494 This class holds a font's texname and size, supports comparison, 

495 and knows the widths of glyphs in the same units as the AFM file. 

496 There are also internal attributes (for use by dviread.py) that 

497 are *not* used for comparison. 

498 

499 The size is in Adobe points (converted from TeX points). 

500 

501 Parameters 

502 ---------- 

503 

504 scale : float 

505 Factor by which the font is scaled from its natural size. 

506 tfm : Tfm 

507 TeX font metrics for this font 

508 texname : bytes 

509 Name of the font as used internally by TeX and friends, as an 

510 ASCII bytestring. This is usually very different from any external 

511 font names, and :class:`dviread.PsfontsMap` can be used to find 

512 the external name of the font. 

513 vf : Vf 

514 A TeX "virtual font" file, or None if this font is not virtual. 

515 

516 Attributes 

517 ---------- 

518 

519 texname : bytes 

520 size : float 

521 Size of the font in Adobe points, converted from the slightly 

522 smaller TeX points. 

523 widths : list 

524 Widths of glyphs in glyph-space units, typically 1/1000ths of 

525 the point size. 

526 

527 """ 

528 __slots__ = ('texname', 'size', 'widths', '_scale', '_vf', '_tfm') 

529 

530 def __init__(self, scale, tfm, texname, vf): 

531 if not isinstance(texname, bytes): 

532 raise ValueError("texname must be a bytestring, got %s" 

533 % type(texname)) 

534 self._scale, self._tfm, self.texname, self._vf = \ 

535 scale, tfm, texname, vf 

536 self.size = scale * (72.0 / (72.27 * 2**16)) 

537 try: 

538 nchars = max(tfm.width) + 1 

539 except ValueError: 

540 nchars = 0 

541 self.widths = [(1000*tfm.width.get(char, 0)) >> 20 

542 for char in range(nchars)] 

543 

544 def __eq__(self, other): 

545 return self.__class__ == other.__class__ and \ 

546 self.texname == other.texname and self.size == other.size 

547 

548 def __ne__(self, other): 

549 return not self.__eq__(other) 

550 

551 def _width_of(self, char): 

552 """ 

553 Width of char in dvi units. For internal use by dviread.py. 

554 """ 

555 

556 width = self._tfm.width.get(char, None) 

557 if width is not None: 

558 return _mul2012(width, self._scale) 

559 _log.debug('No width for char %d in font %s.', char, self.texname) 

560 return 0 

561 

562 def _height_depth_of(self, char): 

563 """ 

564 Height and depth of char in dvi units. For internal use by dviread.py. 

565 """ 

566 

567 result = [] 

568 for metric, name in ((self._tfm.height, "height"), 

569 (self._tfm.depth, "depth")): 

570 value = metric.get(char, None) 

571 if value is None: 

572 _log.debug('No %s for char %d in font %s', 

573 name, char, self.texname) 

574 result.append(0) 

575 else: 

576 result.append(_mul2012(value, self._scale)) 

577 return result

580class Vf(Dvi): 

581 """ 

582 A virtual font (\\*.vf file) containing subroutines for dvi files. 

583 

584 Usage:: 

585 

586 vf = Vf(filename) 

587 glyph = vf[code] 

588 glyph.text, glyph.boxes, glyph.width 

589 

590 Parameters 

591 ---------- 

592 

593 filename : string or bytestring 

594 

595 Notes 

596 ----- 

597 

598 The virtual font format is a derivative of dvi: 

599 http://mirrors.ctan.org/info/knuth/virtual-fonts 

600 This class reuses some of the machinery of `Dvi` 

601 but replaces the `_read` loop and dispatch mechanism. 

602 """ 

603 

604 def __init__(self, filename): 

605 Dvi.__init__(self, filename, 0) 

606 try: 

607 self._first_font = None 

608 self._chars = {} 

609 self._read() 

610 finally: 

611 self.close() 

612 

613 def __getitem__(self, code): 

614 return self._chars[code] 

615 

616 def _read(self): 

617 """ 

618 Read one page from the file. Return True if successful, 

619 False if there were no more pages. 

620 """ 

621 packet_char, packet_ends = None, None 

622 packet_len, packet_width = None, None 

623 while True: 

624 byte = self.file.read(1)[0] 

625 # If we are in a packet, execute the dvi instructions 

626 if self.state is _dvistate.inpage: 

627 byte_at = self.file.tell()-1 

628 if byte_at == packet_ends: 

629 self._finalize_packet(packet_char, packet_width) 

630 packet_len, packet_char, packet_width = None, None, None 

631 # fall through to out-of-packet code 

632 elif byte_at > packet_ends: 

633 raise ValueError("Packet length mismatch in vf file") 

634 else: 

635 if byte in (139, 140) or byte >= 243: 

636 raise ValueError( 

637 "Inappropriate opcode %d in vf file" % byte) 

638 Dvi._dtable[byte](self, byte) 

639 continue 

640 

641 # We are outside a packet 

642 if byte < 242: # a short packet (length given by byte) 

643 packet_len = byte 

644 packet_char, packet_width = self._arg(1), self._arg(3) 

645 packet_ends = self._init_packet(byte) 

646 self.state = _dvistate.inpage 

647 elif byte == 242: # a long packet 

648 packet_len, packet_char, packet_width = \ 

649 [self._arg(x) for x in (4, 4, 4)] 

650 self._init_packet(packet_len) 

651 elif 243 <= byte <= 246: 

652 k = self._arg(byte - 242, byte == 246) 

653 c, s, d, a, l = [self._arg(x) for x in (4, 4, 4, 1, 1)] 

654 self._fnt_def_real(k, c, s, d, a, l) 

655 if self._first_font is None: 

656 self._first_font = k 

657 elif byte == 247: # preamble 

658 i, k = self._arg(1), self._arg(1) 

659 x = self.file.read(k) 

660 cs, ds = self._arg(4), self._arg(4) 

661 self._pre(i, x, cs, ds) 

662 elif byte == 248: # postamble (just some number of 248s) 

663 break 

664 else: 

665 raise ValueError("unknown vf opcode %d" % byte) 

666 

667 def _init_packet(self, pl): 

668 if self.state != _dvistate.outer: 

669 raise ValueError("Misplaced packet in vf file") 

670 self.h, self.v, self.w, self.x, self.y, self.z = 0, 0, 0, 0, 0, 0 

671 self.stack, self.text, self.boxes = [], [], [] 

672 self.f = self._first_font 

673 return self.file.tell() + pl 

674 

675 def _finalize_packet(self, packet_char, packet_width): 

676 self._chars[packet_char] = Page( 

677 text=self.text, boxes=self.boxes, width=packet_width, 

678 height=None, descent=None) 

679 self.state = _dvistate.outer 

680 

681 def _pre(self, i, x, cs, ds): 

682 if self.state is not _dvistate.pre: 

683 raise ValueError("pre command in middle of vf file") 

684 if i != 202: 

685 raise ValueError("Unknown vf format %d" % i) 

686 if len(x): 

687 _log.debug('vf file comment: %s', x) 

688 self.state = _dvistate.outer 

689 # cs = checksum, ds = design size

692def _fix2comp(num): 

693 """ 

694 Convert from two's complement to negative. 

695 """ 

696 assert 0 <= num < 2**32 

697 if num & 2**31: 

698 return num - 2**32 

699 else: 

700 return num

704 """ 

705 Multiply two numbers in 20.12 fixed point format.

708 return (num1*num2) >> 20

711class Tfm(object): 

712 """ 

713 A TeX Font Metric file. 

714 

715 This implementation covers only the bare minimum needed by the Dvi class. 

716 

717 Parameters 

718 ---------- 

719 filename : string or bytestring 

720 

721 Attributes 

722 ---------- 

723 checksum : int 

724 Used for verifying against the dvi file. 

725 design_size : int 

726 Design size of the font (unknown units) 

727 width, height, depth : dict 

728 Dimensions of each character, need to be scaled by the factor 

729 specified in the dvi file. These are dicts because indexing may 

730 not start from 0. 

731 """ 

732 __slots__ = ('checksum', 'design_size', 'width', 'height', 'depth') 

733 

734 def __init__(self, filename): 

735 _log.debug('opening tfm file %s', filename) 

736 with open(filename, 'rb') as file: 

737 header1 = file.read(24) 

738 lh, bc, ec, nw, nh, nd = \ 

739 struct.unpack('!6H', header1[2:14]) 

740 _log.debug('lh=%d, bc=%d, ec=%d, nw=%d, nh=%d, nd=%d', 

741 lh, bc, ec, nw, nh, nd) 

742 header2 = file.read(4*lh) 

743 self.checksum, self.design_size = \ 

744 struct.unpack('!2I', header2[:8]) 

745 # there is also encoding information etc. 

746 char_info = file.read(4*(ec-bc+1)) 

747 widths = file.read(4*nw) 

748 heights = file.read(4*nh) 

749 depths = file.read(4*nd) 

750 

751 self.width, self.height, self.depth = {}, {}, {} 

752 widths, heights, depths = \ 

753 [struct.unpack('!%dI' % (len(x)/4), x) 

754 for x in (widths, heights, depths)] 

755 for idx, char in enumerate(range(bc, ec+1)): 

756 byte0 = char_info[4*idx] 

757 byte1 = char_info[4*idx+1] 

758 self.width[char] = _fix2comp(widths[byte0]) 

759 self.height[char] = _fix2comp(heights[byte1 >> 4]) 

760 self.depth[char] = _fix2comp(depths[byte1 & 0xf])

766class PsfontsMap(object): 

767 """ 

768 A psfonts.map formatted file, mapping TeX fonts to PS fonts. 

769 

770 Usage:: 

771 

772 >>> map = PsfontsMap(find_tex_file('pdftex.map')) 

773 >>> entry = map[b'ptmbo8r'] 

774 >>> entry.texname 

775 b'ptmbo8r' 

776 >>> entry.psname 

777 b'Times-Bold' 

778 >>> entry.encoding 

779 '/usr/local/texlive/2008/texmf-dist/fonts/enc/dvips/base/8r.enc' 

780 >>> entry.effects 

781 {'slant': 0.16700000000000001} 

782 >>> entry.filename 

783 

784 Parameters 

785 ---------- 

786 

787 filename : string or bytestring 

788 

789 Notes 

790 ----- 

791 

792 For historical reasons, TeX knows many Type-1 fonts by different 

793 names than the outside world. (For one thing, the names have to 

794 fit in eight characters.) Also, TeX's native fonts are not Type-1 

795 but Metafont, which is nontrivial to convert to PostScript except 

796 as a bitmap. While high-quality conversions to Type-1 format exist 

797 and are shipped with modern TeX distributions, we need to know 

798 which Type-1 fonts are the counterparts of which native fonts. For 

799 these reasons a mapping is needed from internal font names to font 

800 file names. 

801 

802 A texmf tree typically includes mapping files called e.g. 

803 :file:`psfonts.map`, :file:`pdftex.map`, or :file:`dvipdfm.map`. 

804 The file :file:`psfonts.map` is used by :program:`dvips`, 

805 :file:`pdftex.map` by :program:`pdfTeX`, and :file:`dvipdfm.map` 

806 by :program:`dvipdfm`. :file:`psfonts.map` might avoid embedding 

807 the 35 PostScript fonts (i.e., have no filename for them, as in 

808 the Times-Bold example above), while the pdf-related files perhaps 

809 only avoid the "Base 14" pdf fonts. But the user may have 

810 configured these files differently. 

811 """ 

812 __slots__ = ('_font', '_filename') 

813 

814 # Create a filename -> PsfontsMap cache, so that calling 

815 # `PsfontsMap(filename)` with the same filename a second time immediately 

816 # returns the same object. 

817 @lru_cache() 

818 def __new__(cls, filename): 

819 self = object.__new__(cls) 

820 self._font = {} 

821 self._filename = os.fsdecode(filename) 

822 with open(filename, 'rb') as file: 

823 self._parse(file) 

824 return self 

825 

826 def __getitem__(self, texname): 

827 assert isinstance(texname, bytes) 

828 try: 

829 result = self._font[texname] 

830 except KeyError: 

831 fmt = ('A PostScript file for the font whose TeX name is "{0}" ' 

832 'could not be found in the file "{1}". The dviread module ' 

833 'can only handle fonts that have an associated PostScript ' 

834 'font file. ' 

835 'This problem can often be solved by installing ' 

836 'a suitable PostScript font package in your (TeX) ' 

837 'package manager.') 

838 msg = fmt.format(texname.decode('ascii'), self._filename) 

839 msg = textwrap.fill(msg, break_on_hyphens=False, 

840 break_long_words=False) 

841 _log.info(msg) 

842 raise 

843 fn, enc = result.filename, result.encoding 

844 if fn is not None and not fn.startswith(b'/'): 

845 fn = find_tex_file(fn) 

846 if enc is not None and not enc.startswith(b'/'): 

847 enc = find_tex_file(result.encoding) 

848 return result._replace(filename=fn, encoding=enc)

850 def _parse(self, file): 

851 """ 

852 Parse the font mapping file. 

853 

854 The format is, AFAIK: texname fontname [effects and filenames] 

855 Effects are PostScript snippets like ".177 SlantFont", 

856 filenames begin with one or two less-than signs. A filename 

857 ending in enc is an encoding file, other filenames are font 

858 files. This can be overridden with a left bracket: <[foobar 

859 indicates an encoding file named foobar. 

860 

861 There is some difference between <foo.pfb and <<bar.pfb in 

862 subsetting, but I have no example of << in my TeX installation. 

863 """ 

864 # If the map file specifies multiple encodings for a font, we 

865 # follow pdfTeX in choosing the last one specified. Such 

866 # entries are probably mistakes but they have occurred. 

867 # http://tex.stackexchange.com/questions/10826/ 

868 # http://article.gmane.org/gmane.comp.tex.pdftex/4914 

869 

870 empty_re = re.compile(br'%|\s*$') 

871 word_re = re.compile( 

872 br'''(?x) (?: 

873 "<\[ (?P<enc1> [^"]+ )" | # quoted encoding marked by [ 

874 "< (?P<enc2> [^"]+.enc)" | # quoted encoding, ends in .enc 

875 "<<? (?P<file1> [^"]+ )" | # quoted font file name 

876 " (?P<eff1> [^"]+ )" | # quoted effects or font name 

877 <\[ (?P<enc3> \S+ ) | # encoding marked by [ 

878 < (?P<enc4> \S+ .enc) | # encoding, ends in .enc 

879 <<? (?P<file2> \S+ ) | # font file name 

880 (?P<eff2> \S+ ) # effects or font name 

881 )''') 

882 effects_re = re.compile( 

883 br'''(?x) (?P<slant> -?[0-9]*(?:\.[0-9]+)) \s* SlantFont 

884 | (?P<extend>-?[0-9]*(?:\.[0-9]+)) \s* ExtendFont''') 

885 

886 lines = (line.strip() 

887 for line in file 

888 if not empty_re.match(line)) 

889 for line in lines: 

890 effects, encoding, filename = b'', None, None 

891 words = word_re.finditer(line) 

892 

893 # The named groups are mutually exclusive and are 

894 # referenced below at an estimated order of probability of 

895 # occurrence based on looking at my copy of pdftex.map. 

896 # The font names are probably unquoted: 

897 w = next(words) 

898 texname = w.group('eff2') or w.group('eff1') 

899 w = next(words) 

900 psname = w.group('eff2') or w.group('eff1') 

901 

902 for w in words: 

903 # Any effects are almost always quoted: 

904 eff = w.group('eff1') or w.group('eff2') 

905 if eff: 

906 effects = eff 

907 continue 

908 # Encoding files usually have the .enc suffix 

909 # and almost never need quoting: 

910 enc = (w.group('enc4') or w.group('enc3') or 

911 w.group('enc2') or w.group('enc1')) 

912 if enc: 

913 if encoding is not None: 

914 _log.debug('Multiple encodings for %s = %s', 

915 texname, psname) 

916 encoding = enc 

917 continue 

918 # File names are probably unquoted: 

919 filename = w.group('file2') or w.group('file1') 

920 

921 effects_dict = {} 

922 for match in effects_re.finditer(effects): 

923 slant = match.group('slant') 

924 if slant: 

925 effects_dict['slant'] = float(slant) 

926 else: 

927 effects_dict['extend'] = float(match.group('extend')) 

928 

929 self._font[texname] = PsFont( 

930 texname=texname, psname=psname, effects=effects_dict, 

931 encoding=encoding, filename=filename)

934class Encoding(object): 

935 """ 

936 Parses a \\*.enc file referenced from a psfonts.map style file. 

937 The format this class understands is a very limited subset of 

938 PostScript. 

939 

940 Usage (subject to change):: 

941 

942 for name in Encoding(filename): 

943 whatever(name) 

944 

945 Parameters 

946 ---------- 

947 filename : string or bytestring 

948 

949 Attributes 

950 ---------- 

951 encoding : list 

952 List of character names 

953 """

955 

956 def __init__(self, filename): 

957 with open(filename, 'rb') as file: 

958 _log.debug('Parsing TeX encoding %s', filename)

965 @staticmethod 

966 def _parse(file): 

967 result = [] 

968

971 beginning = data.find(b'[') 

972 if beginning < 0: 

973 raise ValueError("Cannot locate beginning of encoding in {}" 

974 .format(file))

976 end = data.find(b']') 

977 if end < 0: 

978 raise ValueError("Cannot locate end of encoding in {}" 

979 .format(file))

981 

982 return re.findall(br'/([^][{}<>\s]+)', data)

985@lru_cache() 

986def find_tex_file(filename, format=None): 

987 """ 

988 Find a file in the texmf tree. 

989 

990 Calls :program:`kpsewhich` which is an interface to the kpathsea 

991 library [1]_. Most existing TeX distributions on Unix-like systems use 

992 kpathsea. It is also available as part of MikTeX, a popular 

993 distribution on Windows. 

994 

995 Parameters 

996 ---------- 

997 filename : string or bytestring 

998 format : string or bytestring 

999 Used as the value of the `--format` option to :program:`kpsewhich`. 

1000 Could be e.g. 'tfm' or 'vf' to limit the search to that type of files. 

1001 

1002 References 

1003 ---------- 

1004 

1005 .. [1] `Kpathsea documentation <http://www.tug.org/kpathsea/>`_ 

1006 The library that :program:`kpsewhich` is part of. 

1007 """ 

1008 

1009 # we expect these to always be ascii encoded, but use utf-8 

1010 # out of caution 

1011 if isinstance(filename, bytes): 

1012 filename = filename.decode('utf-8', errors='replace') 

1013 if isinstance(format, bytes): 

1014 format = format.decode('utf-8', errors='replace') 

1015 

1016 cmd = ['kpsewhich'] 

1017 if format is not None: 

1018 cmd += ['--format=' + format] 

1019 cmd += [filename] 

1020 _log.debug('find_tex_file(%s): %s', filename, cmd) 

1021 pipe = subprocess.Popen(cmd, stdout=subprocess.PIPE) 

1022 result = pipe.communicate()[0].rstrip() 

1023 _log.debug('find_tex_file result: %s', result) 

1024 return result.decode('ascii')
