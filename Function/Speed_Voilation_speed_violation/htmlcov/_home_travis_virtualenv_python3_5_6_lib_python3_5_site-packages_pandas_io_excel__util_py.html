56def _excel2num(x): 

57 """ 

58 Convert Excel column name like 'AB' to 0-based column index. 

59 

60 Parameters 

61 ---------- 

62 x : str 

63 The Excel column name to convert to a 0-based column index. 

64 

65 Returns 

66 ------- 

67 num : int 

68 The column index corresponding to the name. 

69 

70 Raises 

71 ------ 

72 ValueError 

73 Part of the Excel column name was invalid. 

74 """ 

75 index = 0 

76 

77 for c in x.upper().strip(): 

78 cp = ord(c) 

79 

80 if cp < ord("A") or cp > ord("Z"): 

81 raise ValueError("Invalid column name: {x}".format(x=x)) 

82 

83 index = index * 26 + cp - ord("A") + 1 

84 

85 return index - 1

89 """

91 

92 Parameters

94 areas : str 

95 A string containing a sequence of column ranges (or areas).

106 >>> _range2cols('A,C,Z:AB') 

107 [0, 2, 25, 26, 27] 

108 """ 

109 cols = [] 

110 

111 for rng in areas.split(","): 

112 if ":" in rng:
