754 def _cumcount_array(self, ascending=True): 

755 """ 

756 Parameters 

757 ---------- 

758 ascending : bool, default True 

759 If False, number in reverse, from length of group - 1 to 0. 

760 

761 Notes

770 if count == 0:

1680

1682 ---------- 

1683 n : int or list of ints

1685 dropna : None or str, optional

1695 --------

1718 1 NaN

1746 

1747 valid_containers = (set, list, tuple) 

1748 if not isinstance(n, (valid_containers, int)): 

1749 raise TypeError("n needs to be an int or a list/set/tuple of ints") 

1750 

1751 if not dropna: 

1752 

1753 if isinstance(n, int): 

1754 nth_values = [n] 

1755 elif isinstance(n, valid_containers): 

1756 nth_values = list(set(n)) 

1757 

1758 nth_array = np.array(nth_values, dtype=np.intp) 

1759 self._set_group_selection() 

1760 

1761 mask_left = np.in1d(self._cumcount_array(), nth_array) 

1762 mask_right = np.in1d(self._cumcount_array(ascending=False) + 1, -nth_array) 

1763 mask = mask_left | mask_right 

1764 

1765 ids, _, _ = self.grouper.group_info 

1766 

1767 # Drop NA values in grouping 

1768 mask = mask & (ids != -1) 

1769 

1770 out = self._selected_obj[mask] 

1771 if not self.as_index: 

1772 return out 

1773 

1774 result_index = self.grouper.result_index 

1775 out.index = result_index[ids[mask]] 

1776 

1777 if not self.observed and isinstance(result_index, CategoricalIndex): 

1778 out = out.reindex(result_index) 

1779 

1780 return out.sort_index() if self.sort else out 

1781 

1782 # dropna is truthy 

1783 if isinstance(n, valid_containers): 

1784 raise ValueError("dropna option with a list of nth values is not supported") 

1785 

1786 if dropna not in ["any", "all"]: 

1787 # Note: when agg-ing picker doesn't raise this, just returns NaN 

1788 raise ValueError( 

1789 "For a DataFrame groupby, dropna must be " 

1790 "either None, 'any' or 'all', " 

1791 "(was passed {dropna}).".format(dropna=dropna) 

1792 ) 

1793 

1794 # old behaviour, but with all and any support for DataFrames. 

1795 # modified in GH 7559 to have better perf 

1796 max_len = n if n >= 0 else -1 - n 

1797 dropped = self.obj.dropna(how=dropna, axis=self.axis) 

1798 

1799 # get a new grouper for our dropped obj 

1800 if self.keys is None and self.level is None: 

1801 

1802 # we don't have the grouper info available 

1803 # (e.g. we have selected out 

1804 # a column that is not in the current object) 

1805 axis = self.grouper.axis 

1806 grouper = axis[axis.isin(dropped.index)] 

1807 

1808 else: 

1809 

1810 # create a grouper with the original parameters, but on dropped 

1811 # object 

1812 from pandas.core.groupby.grouper import _get_grouper 

1813 

1814 grouper, _, _ = _get_grouper( 

1815 dropped, 

1816 key=self.keys, 

1817 axis=self.axis, 

1818 level=self.level, 

1819 sort=self.sort, 

1820 mutated=self.mutated, 

1821 ) 

1822 

1823 grb = dropped.groupby(grouper, as_index=self.as_index, sort=self.sort) 

1824 sizes, result = grb.size(), grb.nth(n) 

1825 mask = (sizes < max_len).values 

1826 

1827 # set the results which don't meet the criteria 

1828 if len(result) and mask.any(): 

1829 result.loc[mask] = np.nan 

1830 

1831 # reset/reindex to the original groups 

1832 if len(self.obj) == len(dropped) or len(result) == len( 

1833 self.grouper.result_index 

1834 ): 

1835 result.index = self.grouper.result_index 

1836 else: 

1837 result = result.reindex(self.grouper.result_index) 

1838 

1839 return result
