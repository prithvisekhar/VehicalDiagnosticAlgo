706 nanos = delta_to_nanoseconds(other) 

707 

708 elif isinstance(other, np.ndarray): 

709 # numpy timedelta64 array; all entries must be compatible 

710 assert other.dtype.kind == "m" 

711 if other.dtype != _TD_DTYPE: 

712 # i.e. non-nano unit 

713 # TODO: disallow unit-less timedelta64 

714 other = other.astype(_TD_DTYPE) 

715 nanos = other.view("i8") 

716 else: 

717 # TimedeltaArray/Index 

718 nanos = other.asi8 

719 

720 if np.all(nanos % base_nanos == 0): 

721 # nanos being added is an integer multiple of the 

722 # base-frequency to self.freq 

723 delta = nanos // base_nanos 

724 # delta is the integer (or integer-array) number of periods 

725 # by which will be added to self. 

726 return delta

769def period_array( 

770 data: Sequence[Optional[Period]], freq: Optional[Tick] = None, copy: bool = False 

771) -> PeriodArray: 

772 """ 

773 Construct a new PeriodArray from a sequence of Period scalars. 

774 

775 Parameters 

776 ---------- 

777 data : Sequence of Period objects 

778 A sequence of Period objects. These are required to all have 

779 the same ``freq.`` Missing values can be indicated by ``None`` 

780 or ``pandas.NaT``. 

781 freq : str, Tick, or Offset 

782 The frequency of every element of the array. This can be specified 

783 to avoid inferring the `freq` from `data`. 

784 copy : bool, default False 

785 Whether to ensure a copy of the data is made. 

786 

787 Returns 

788 ------- 

789 PeriodArray 

790 

791 See Also 

792 -------- 

793 PeriodArray 

794 pandas.PeriodIndex 

795 

796 Examples 

797 -------- 

798 >>> period_array([pd.Period('2017', freq='A'), 

799 ... pd.Period('2018', freq='A')]) 

800 <PeriodArray> 

801 ['2017', '2018'] 

802 Length: 2, dtype: period[A-DEC] 

803 

804 >>> period_array([pd.Period('2017', freq='A'), 

805 ... pd.Period('2018', freq='A'), 

806 ... pd.NaT]) 

807 <PeriodArray> 

808 ['2017', '2018', 'NaT'] 

809 Length: 3, dtype: period[A-DEC] 

810 

811 Integers that look like years are handled 

812 

813 >>> period_array([2000, 2001, 2002], freq='D') 

814 ['2000-01-01', '2001-01-01', '2002-01-01'] 

815 Length: 3, dtype: period[D] 

816 

817 Datetime-like strings may also be passed 

818 

819 >>> period_array(['2000-Q1', '2000-Q2', '2000-Q3', '2000-Q4'], freq='Q') 

820 <PeriodArray> 

821 ['2000Q1', '2000Q2', '2000Q3', '2000Q4'] 

822 Length: 4, dtype: period[Q-DEC] 

823 """ 

824 if is_datetime64_dtype(data): 

825 return PeriodArray._from_datetime64(data, freq) 

826 if isinstance(data, (ABCPeriodIndex, ABCSeries, PeriodArray)): 

827 return PeriodArray(data, freq) 

828 

829 # other iterable of some kind 

830 if not isinstance(data, (np.ndarray, list, tuple)): 

831 data = list(data) 

832 

833 data = np.asarray(data) 

834 

835 if freq: 

836 # typed Optional here because the else block below assigns None 

837 dtype = PeriodDtype(freq) # type: Optional[PeriodDtype] 

838 else: 

839 dtype = None 

840 

841 if is_float_dtype(data) and len(data) > 0: 

842 raise TypeError("PeriodIndex does not allow " "floating point in construction") 

843 

844 data = ensure_object(data) 

845 

846 return PeriodArray._from_sequence(data, dtype=dtype)

849def validate_dtype_freq(dtype, freq): 

850 """ 

851 If both a dtype and a freq are available, ensure they match. If only 

852 dtype is available, extract the implied freq. 

853 

854 Parameters 

855 ---------- 

856 dtype : dtype 

857 freq : DateOffset or None 

858 

859 Returns 

860 ------- 

861 freq : DateOffset 

862 

863 Raises 

864 ------ 

865 ValueError : non-period dtype 

866 IncompatibleFrequency : mismatch between dtype and freq 

867 """ 

868 if freq is not None: 

869 freq = frequencies.to_offset(freq) 

870 

871 if dtype is not None: 

872 dtype = pandas_dtype(dtype) 

873 if not is_period_dtype(dtype): 

874 raise ValueError("dtype must be PeriodDtype") 

875 if freq is None: 

876 freq = dtype.freq 

877 elif freq != dtype.freq: 

878 raise IncompatibleFrequency("specified freq and dtype " "are different") 

879 return freq

920def _get_ordinal_range(start, end, periods, freq, mult=1): 

921 if com.count_not_none(start, end, periods) != 2: 

922 raise ValueError( 

923 "Of the three parameters: start, end, and periods, " 

924 "exactly two must be specified" 

925 ) 

926 

927 if freq is not None: 

928 _, mult = libfrequencies.get_freq_code(freq) 

929 

930 if start is not None: 

931 start = Period(start, freq) 

932 if end is not None: 

933 end = Period(end, freq) 

934 

935 is_start_per = isinstance(start, Period) 

936 is_end_per = isinstance(end, Period) 

937 

938 if is_start_per and is_end_per and start.freq != end.freq: 

939 raise ValueError("start and end must have same freq") 

940 if start is NaT or end is NaT: 

941 raise ValueError("start and end must not be NaT") 

942 

943 if freq is None: 

944 if is_start_per: 

945 freq = start.freq 

946 elif is_end_per: 

947 freq = end.freq 

948 else: # pragma: no cover 

949 raise ValueError("Could not infer freq from start/end") 

950 

951 if periods is not None: 

952 periods = periods * mult 

953 if start is None: 

954 data = np.arange( 

955 end.ordinal - periods + mult, end.ordinal + 1, mult, dtype=np.int64 

956 ) 

957 else: 

958 data = np.arange( 

959 start.ordinal, start.ordinal + periods, mult, dtype=np.int64 

960 ) 

961 else: 

962 data = np.arange(start.ordinal, end.ordinal + 1, mult, dtype=np.int64) 

963 

964 return data, freq
