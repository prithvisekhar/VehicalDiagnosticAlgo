13class DirNamesMixin:

19 def _dir_deletions(self): 

20 """ 

21 Delete unwanted __dir__ for this object. 

22 """ 

23 return self._accessors | self._deprecations

25 def _dir_additions(self): 

26 """ 

27 Add additional __dir__ for this object. 

28 """ 

29 rv = set() 

30 for accessor in self._accessors: 

31 try:

33 rv.add(accessor)

48class PandasDelegate: 

49 """ 

50 An abstract base class for delegating methods/properties. 

51 """ 

52 

53 def _delegate_property_get(self, name, *args, **kwargs): 

54 raise TypeError("You cannot access the " "property {name}".format(name=name)) 

55 

56 def _delegate_property_set(self, name, value, *args, **kwargs): 

57 raise TypeError("The property {name} cannot be set".format(name=name)) 

58 

59 def _delegate_method(self, name, *args, **kwargs): 

60 raise TypeError("You cannot call method {name}".format(name=name))

62 @classmethod 

63 def _add_delegate_accessors(cls, delegate, accessors, typ, overwrite=False): 

64 """ 

65 Add accessors to cls from the delegate class. 

66 

67 Parameters 

68 ---------- 

69 cls : the class to add the methods/properties to 

70 delegate : the class to get methods/properties & doc-strings 

71 accessors : string list of accessors to add 

72 typ : 'property' or 'method' 

73 overwrite : boolean, default False 

74 overwrite the method/property in the target class if it exists. 

75 """ 

76 

77 def _create_delegator_property(name): 

78 def _getter(self): 

79 return self._delegate_property_get(name) 

80 

81 def _setter(self, new_values): 

82 return self._delegate_property_set(name, new_values) 

83 

84 _getter.__name__ = name 

85 _setter.__name__ = name 

86 

87 return property( 

88 fget=_getter, fset=_setter, doc=getattr(delegate, name).__doc__ 

89 ) 

90 

91 def _create_delegator_method(name): 

92 def f(self, *args, **kwargs): 

93 return self._delegate_method(name, *args, **kwargs) 

94 

95 f.__name__ = name 

96 f.__doc__ = getattr(delegate, name).__doc__ 

97 

98 return f 

99 

100 for name in accessors: 

101 

102 if typ == "property": 

103 f = _create_delegator_property(name) 

104 else: 

105 f = _create_delegator_method(name) 

106 

107 # don't overwrite existing methods/properties 

108 if overwrite or not hasattr(cls, name): 

109 setattr(cls, name, f)

153class CachedAccessor: 

154 """ 

155 Custom property-like object (descriptor) for caching accessors. 

156 

157 Parameters 

158 ---------- 

159 name : str 

160 The namespace this will be accessed under, e.g. ``df.foo`` 

161 accessor : cls 

162 The class with the extension methods. The class' __init__ method 

163 should expect one of a ``Series``, ``DataFrame`` or ``Index`` as 

164 the single argument ``data`` 

165 """ 

166 

167 def __init__(self, name, accessor): 

168 self._name = name 

169 self._accessor = accessor 

170 

171 def __get__(self, obj, cls): 

172 if obj is None: 

173 # we're accessing the attribute of the class, i.e., Dataset.geo 

174 return self._accessor 

175 accessor_obj = self._accessor(obj) 

176 # Replace the property with the accessor object. Inspired by: 

177 # http://www.pydanny.com/cached-property.html 

178 # We need to use object.__setattr__ because we overwrite __setattr__ on 

179 # NDFrame 

180 object.__setattr__(obj, self._name, accessor_obj) 

181 return accessor_obj

184def _register_accessor(name, cls): 

185 def decorator(accessor): 

186 if hasattr(cls, name): 

187 warnings.warn( 

188 "registration of accessor {!r} under name {!r} for type " 

189 "{!r} is overriding a preexisting attribute with the same " 

190 "name.".format(accessor, name, cls), 

191 UserWarning, 

192 stacklevel=2, 

193 ) 

194 setattr(cls, name, CachedAccessor(name, accessor)) 

195 cls._accessors.add(name) 

196 return accessor 

197 

198 return decorator

219Notes 

220----- 

221When accessed, your accessor will be initialized with the pandas object 

222the user is interacting with. So the signature must be

224.. code-block:: python

232>>> pd.Series(['a', 'b']).dt
