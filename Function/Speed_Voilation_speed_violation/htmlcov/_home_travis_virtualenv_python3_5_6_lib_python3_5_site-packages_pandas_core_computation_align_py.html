59def _align_core(terms): 

60 term_index = [i for i, term in enumerate(terms) if hasattr(term.value, "axes")] 

61 term_dims = [terms[i].value.ndim for i in term_index] 

62 ndims = pd.Series(dict(zip(term_index, term_dims))) 

63 

64 # initial axes are the axes of the largest-axis'd term 

65 biggest = terms[ndims.idxmax()].value 

66 typ = biggest._constructor 

67 axes = biggest.axes 

68 naxes = len(axes) 

69 gt_than_one_axis = naxes > 1 

70 

71 for value in (terms[i].value for i in term_index): 

72 is_series = isinstance(value, pd.Series) 

73 is_series_and_gt_one_axis = is_series and gt_than_one_axis 

74 

75 for axis, items in enumerate(value.axes): 

76 if is_series_and_gt_one_axis: 

77 ax, itm = naxes - 1, value.index 

78 else: 

79 ax, itm = axis, items 

80 

81 if not axes[ax].is_(itm): 

82 axes[ax] = axes[ax].join(itm, how="outer") 

83 

84 for i, ndim in ndims.items(): 

85 for axis, items in zip(range(ndim), axes): 

86 ti = terms[i].value 

87 

88 if hasattr(ti, "reindex"): 

89 transpose = isinstance(ti, pd.Series) and naxes > 1 

90 reindexer = axes[naxes - 1] if transpose else items 

91 

92 term_axis_size = len(ti.axes[axis]) 

93 reindexer_size = len(reindexer) 

94 

95 ordm = np.log10(max(1, abs(reindexer_size - term_axis_size))) 

96 if ordm >= 1 and reindexer_size >= 10000: 

97 w = ( 

98 "Alignment difference on axis {axis} is larger " 

99 "than an order of magnitude on term {term!r}, by " 

100 "more than {ordm:.4g}; performance may suffer" 

101 ).format(axis=axis, term=terms[i].name, ordm=ordm) 

102 warnings.warn(w, category=PerformanceWarning, stacklevel=6) 

103 

104 f = partial(ti.reindex, reindexer, axis=axis, copy=False) 

105 

106 terms[i].update(f()) 

107 

108 terms[i].update(terms[i].value.values) 

109 

110 return typ, _zip_axes_from_type(typ, axes)

113def _align(terms): 

114 """Align a set of terms""" 

115 try: 

116 # flatten the parse tree (a nested list, really) 

117 terms = list(com.flatten(terms)) 

118 except TypeError: 

119 # can't iterate so it must just be a constant or single variable 

120 if isinstance(terms.value, pd.core.generic.NDFrame): 

121 typ = type(terms.value) 

122 return typ, _zip_axes_from_type(typ, terms.value.axes) 

123 return np.result_type(terms.type), None 

124 

125 # if all resolved variables are numeric scalars 

126 if all(term.is_scalar for term in terms): 

127 return _result_type_many(*(term.value for term in terms)).type, None 

128 

129 # perform the main alignment 

130 typ, axes = _align_core(terms) 

131 return typ, axes

134def _reconstruct_object(typ, obj, axes, dtype): 

135 """Reconstruct an object given its type, raw value, and possibly empty 

136 (None) axes. 

137 

138 Parameters 

139 ---------- 

140 typ : object 

141 A type 

142 obj : object 

143 The value to use in the type constructor 

144 axes : dict 

145 The axes to use to construct the resulting pandas object 

146 

147 Returns 

148 ------- 

149 ret : typ 

150 An object of type ``typ`` with the value `obj` and possible axes 

151 `axes`. 

152 """ 

153 try: 

154 typ = typ.type 

155 except AttributeError: 

156 pass 

157 

158 res_t = np.result_type(obj.dtype, dtype) 

159 

160 if not isinstance(typ, partial) and issubclass(typ, pd.core.generic.PandasObject): 

161 return typ(obj, dtype=res_t, **axes) 

162 

163 # special case for pathological things like ~True/~False 

164 if hasattr(res_t, "type") and typ == np.bool_ and res_t != np.bool_: 

165 ret_value = res_t.type(obj) 

166 else: 

167 ret_value = typ(obj).astype(res_t) 

168 # The condition is to distinguish 0-dim array (returned in case of 

169 # scalar) and 1 element array 

170 # e.g. np.array(0) and np.array([0]) 

171 if len(obj.shape) == 1 and len(obj) == 1: 

172 if not isinstance(ret_value, np.ndarray): 

173 ret_value = np.array([ret_value]).astype(res_t) 

174 

175 return ret_value
