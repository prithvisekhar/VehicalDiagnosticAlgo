390def _logged_cached(fmt, func=None): 

391 """ 

392 Decorator that logs a function's return value, and memoizes that value. 

393 

394 After :: 

395 

396 @_logged_cached(fmt) 

397 def func(): ... 

398 

399 the first call to *func* will log its return value at the DEBUG level using 

400 %-format string *fmt*, and memoize it; later calls to *func* will directly 

401 return that value. 

402 """ 

403 if func is None: # Return the actual decorator. 

404 return functools.partial(_logged_cached, fmt) 

405 

406 called = False 

407 ret = None 

408 

409 @functools.wraps(func) 

410 def wrapper(): 

411 nonlocal called, ret 

412 if not called: 

413 ret = func() 

414 called = True 

415 _log.debug(fmt, ret) 

416 return ret 

417 

418 return wrapper

703def matplotlib_fname(): 

704 """ 

705 Get the location of the config file. 

706 

707 The file location is determined in the following order 

708 

709 - `$PWD/matplotlibrc` 

710 

711 - `$MATPLOTLIBRC` if it is a file (or a named pipe, which can be created 

712 e.g. by process substitution) 

713 

714 - `$MATPLOTLIBRC/matplotlibrc` 

715 

716 - `$MPLCONFIGDIR/matplotlibrc` 

717 

718 - On Linux, 

719 

720 - `$XDG_CONFIG_HOME/matplotlib/matplotlibrc` (if 

721 $XDG_CONFIG_HOME is defined) 

722 

723 - or `$HOME/.config/matplotlib/matplotlibrc` (if 

724 $XDG_CONFIG_HOME is not defined) 

725 

726 - On other platforms, 

727 

728 - `$HOME/.matplotlib/matplotlibrc` if `$HOME` is defined. 

729 

730 - Lastly, it looks in `$MATPLOTLIBDATA/matplotlibrc` for a 

731 system-defined copy. 

732 """

735 yield os.path.join(os.getcwd(), 'matplotlibrc')

738 except KeyError:

744 yield os.path.join(get_data_path(), 'matplotlibrc')

746 for fname in gen_candidates():

748 st_mode = os.stat(fname).st_mode

750 break

752 # valid (in that case, a warning is raised at startup by `rc_params`).

978def _rc_params_in_file(fname, fail_on_error=False): 

979 """Return :class:`matplotlib.RcParams` from the contents of the given file. 

980 

981 Unlike `rc_params_from_file`, the configuration class only contains the 

982 parameters specified in the file (i.e. default values are not filled in). 

983 """ 

984 cnt = 0 

985 rc_temp = {} 

986 with _open_file_or_url(fname) as fd: 

987 try: 

988 for line in fd: 

989 cnt += 1 

990 strippedline = line.split('#', 1)[0].strip() 

991 if not strippedline: 

992 continue 

993 tup = strippedline.split(':', 1) 

994 if len(tup) != 2: 

995 error_details = _error_details_fmt % (cnt, line, fname) 

996 warnings.warn('Illegal %s' % error_details) 

997 continue 

998 key, val = tup 

999 key = key.strip() 

1000 val = val.strip() 

1001 if key in rc_temp: 

1002 warnings.warn('Duplicate key in file "%s", line #%d' % 

1003 (fname, cnt))

1005 except UnicodeDecodeError: 

1006 warnings.warn( 

1007 ('Cannot decode configuration file %s with ' 

1008 'encoding %s, check LANG and LC_* variables') 

1009 % (fname, locale.getpreferredencoding(do_setlocale=False) or 

1010 'utf-8 (default)')) 

1011 raise 

1012 

1013 config = RcParams() 

1014 

1015 for key in ('verbose.level', 'verbose.fileo'): 

1016 if key in rc_temp: 

1017 val, line, cnt = rc_temp.pop(key) 

1018 if fail_on_error: 

1019 config[key] = val # try to convert to proper type or raise 

1020 else: 

1021 try: 

1022 config[key] = val # try to convert to proper type or skip 

1023 except Exception as msg: 

1024 error_details = _error_details_fmt % (cnt, line, fname) 

1025 warnings.warn('Bad val "%s" on %s\n\t%s' % 

1026 (val, error_details, msg)) 

1027 

1028 for key, (val, line, cnt) in rc_temp.items(): 

1029 if key in defaultParams: 

1030 if fail_on_error: 

1031 config[key] = val # try to convert to proper type or raise 

1032 else:

1034 config[key] = val # try to convert to proper type or skip 

1035 except Exception as msg: 

1036 error_details = _error_details_fmt % (cnt, line, fname) 

1037 warnings.warn('Bad val "%s" on %s\n\t%s' % 

1038 (val, error_details, msg)) 

1039 elif key in _deprecated_ignore_map: 

1040 version, alt_key = _deprecated_ignore_map[key] 

1041 cbook.warn_deprecated( 

1042 version, key, alternative=alt_key, 

1043 addendum="Please update your matplotlibrc.") 

1044 else: 

1045 print(""" 

1046Bad key "%s" on line %d in 

1047%s. 

1048You probably need to get an updated matplotlibrc file from 

1049http://github.com/matplotlib/matplotlib/blob/master/matplotlibrc.template

1133def rc(group, **kwargs): 

1134 """ 

1135 Set the current rc params. *group* is the grouping for the rc, e.g., 

1136 for ``lines.linewidth`` the group is ``lines``, for 

1137 ``axes.facecolor``, the group is ``axes``, and so on. Group may 

1138 also be a list or tuple of group names, e.g., (*xtick*, *ytick*). 

1139 *kwargs* is a dictionary attribute name/value pairs, e.g.,:: 

1140 

1141 rc('lines', linewidth=2, color='r') 

1142 

1143 sets the current rc params and is equivalent to:: 

1144 

1145 rcParams['lines.linewidth'] = 2 

1146 rcParams['lines.color'] = 'r' 

1147 

1148 The following aliases are available to save typing for interactive 

1149 users: 

1150 

1151 ===== ================= 

1152 Alias Property 

1153 ===== ================= 

1154 'lw' 'linewidth' 

1155 'ls' 'linestyle' 

1156 'c' 'color' 

1157 'fc' 'facecolor' 

1158 'ec' 'edgecolor' 

1159 'mew' 'markeredgewidth' 

1160 'aa' 'antialiased' 

1161 ===== ================= 

1162 

1163 Thus you could abbreviate the above rc command as:: 

1164 

1165 rc('lines', lw=2, c='r') 

1166 

1167 

1168 Note you can use python's kwargs dictionary facility to store 

1169 dictionaries of default parameters. e.g., you can customize the 

1170 font rc as follows:: 

1171 

1172 font = {'family' : 'monospace', 

1173 'weight' : 'bold', 

1174 'size' : 'larger'} 

1175 

1176 rc('font', **font) # pass in the font dict as kwargs 

1177 

1178 This enables you to easily switch between several configurations. Use 

1179 ``matplotlib.style.use('default')`` or :func:`~matplotlib.rcdefaults` to 

1180 restore the default rc params after changes. 

1181 """ 

1182 

1183 aliases = { 

1184 'lw': 'linewidth', 

1185 'ls': 'linestyle', 

1186 'c': 'color', 

1187 'fc': 'facecolor', 

1188 'ec': 'edgecolor', 

1189 'mew': 'markeredgewidth', 

1190 'aa': 'antialiased', 

1191 } 

1192 

1193 if isinstance(group, str): 

1194 group = (group,) 

1195 for g in group: 

1196 for k, v in kwargs.items(): 

1197 name = aliases.get(k) or k 

1198 key = '%s.%s' % (g, name) 

1199 try: 

1200 rcParams[key] = v 

1201 except KeyError: 

1202 raise KeyError(('Unrecognized key "%s" for group "%s" and ' 

1203 'name "%s"') % (key, g, name))

1264class rc_context: 

1265 """ 

1266 Return a context manager for managing rc settings. 

1267 

1268 This allows one to do:: 

1269 

1270 with mpl.rc_context(fname='screen.rc'): 

1271 plt.plot(x, a) 

1272 with mpl.rc_context(fname='print.rc'): 

1273 plt.plot(x, b) 

1274 plt.plot(x, c) 

1275 

1276 The 'a' vs 'x' and 'c' vs 'x' plots would have settings from 

1277 'screen.rc', while the 'b' vs 'x' plot would have settings from 

1278 'print.rc'. 

1279 

1280 A dictionary can also be passed to the context manager:: 

1281 

1282 with mpl.rc_context(rc={'text.usetex': True}, fname='screen.rc'): 

1283 plt.plot(x, a) 

1284 

1285 The 'rc' dictionary takes precedence over the settings loaded from 

1286 'fname'. Passing a dictionary only is also valid. For example a 

1287 common usage is:: 

1288 

1289 with mpl.rc_context(rc={'interactive': False}): 

1290 fig, ax = plt.subplots() 

1291 ax.plot(range(3), range(3)) 

1292 fig.savefig('A.png', format='png') 

1293 plt.close(fig) 

1294 """ 

1295 # While it may seem natural to implement rc_context using 

1296 # contextlib.contextmanager, that would entail always calling the finally: 

1297 # clause of the contextmanager (which restores the original rcs) including 

1298 # during garbage collection; as a result, something like `plt.xkcd(); 

1299 # gc.collect()` would result in the style being lost (as `xkcd()` is 

1300 # implemented on top of rc_context, and nothing is holding onto context 

1301 # manager except possibly circular references. 

1302 

1303 def __init__(self, rc=None, fname=None): 

1304 self._orig = rcParams.copy() 

1305 try: 

1306 if fname:

1599 label_namer=None, positional_parameter_names=None): 

1600 """ 

1601 A decorator to add a 'data' kwarg to any a function. The signature 

1602 of the input function must include the ax argument at the first position :: 

1603 

1604 def foo(ax, *args, **kwargs) 

1605 

1606 so this is suitable for use with Axes methods. 

1607 

1608 Parameters 

1609 ---------- 

1610 replace_names : list of strings, optional, default: None 

1611 The list of parameter names which arguments should be replaced by 

1612 `data[name]`. If None, all arguments are replaced if they are 

1613 included in `data`. 

1614 replace_all_args : bool, default: False 

1615 If True, all arguments in *args get replaced, even if they are not 

1616 in replace_names. 

1617 label_namer : string, optional, default: None 

1618 The name of the parameter which argument should be used as label, if 

1619 label is not set. If None, the label keyword argument is not set. 

1620 positional_parameter_names : list of strings or callable, optional 

1621 The full list of positional parameter names (excluding an explicit 

1622 `ax`/'self' argument at the first place and including all possible 

1623 positional parameter in `*args`), in the right order. Can also include 

1624 all other keyword parameter. Only needed if the wrapped function does 

1625 contain `*args` and (replace_names is not None or replace_all_args is 

1626 False). If it is a callable, it will be called with the actual 

1627 tuple of *args and the data and should return a list like 

1628 above. 

1629 NOTE: callables should only be used when the names and order of *args 

1630 can only be determined at runtime. Please use list of names 

1631 when the order and names of *args is clear before runtime! 

1632 

1633 .. note:: decorator also converts MappingView input data to list. 

1634 """ 

1635 if replace_names is not None: 

1636 replace_names = set(replace_names)

1644 for p in params: 

1645 if p.kind is Parameter.VAR_POSITIONAL: 

1646 _has_varargs = True 

1647 elif p.kind is Parameter.VAR_KEYWORD: 

1648 _has_varkwargs = True 

1649 else: 

1650 _arg_names.append(p.name) 

1651 data_param = Parameter('data', Parameter.KEYWORD_ONLY, default=None) 

1652 if _has_varkwargs: 

1653 params.insert(-1, data_param) 

1654 else: 

1655 params.append(data_param)

1660 # positional args can end up in **kwargs, so only *varargs make 

1661 # problems.

1672 arg_names = None 

1673 elif len(replace_names) == 0:

1675 arg_names = []

1677 # we got no manual parameter names but more than an 'ax' ...

1680 arg_names = _arg_names[1:]
