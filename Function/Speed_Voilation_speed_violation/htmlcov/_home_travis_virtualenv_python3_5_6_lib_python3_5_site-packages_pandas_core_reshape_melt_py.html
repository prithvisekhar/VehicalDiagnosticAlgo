154 keys = list(groups.keys()) 

155 values = list(groups.values()) 

156 else: 

157 keys, values = zip(*groups)

165 if len(seq) != K:

192def wide_to_long(df, stubnames, i, j, sep="", suffix=r"\d+"): 

193 r""" 

194 Wide panel to long format. Less flexible but more user-friendly than melt. 

195 

196 With stubnames ['A', 'B'], this function expects to find one or more 

197 group of columns with format 

198 A-suffix1, A-suffix2,..., B-suffix1, B-suffix2,... 

199 You specify what you want to call this suffix in the resulting long format 

200 with `j` (for example `j='year'`) 

201 

202 Each row of these wide variables are assumed to be uniquely identified by 

203 `i` (can be a single column name or a list of column names) 

204 

205 All remaining variables in the data frame are left intact. 

206 

207 Parameters 

208 ---------- 

209 df : DataFrame 

210 The wide-format DataFrame 

211 stubnames : str or list-like 

212 The stub name(s). The wide format variables are assumed to 

213 start with the stub names. 

214 i : str or list-like 

215 Column(s) to use as id variable(s) 

216 j : str 

217 The name of the sub-observation variable. What you wish to name your 

218 suffix in the long format. 

219 sep : str, default "" 

220 A character indicating the separation of the variable names 

221 in the wide format, to be stripped from the names in the long format. 

222 For example, if your column names are A-suffix1, A-suffix2, you 

223 can strip the hyphen by specifying `sep='-'` 

224 

225 .. versionadded:: 0.20.0 

226 

227 suffix : str, default '\\d+' 

228 A regular expression capturing the wanted suffixes. '\\d+' captures 

229 numeric suffixes. Suffixes with no numbers could be specified with the 

230 negated character class '\\D+'. You can also further disambiguate 

231 suffixes, for example, if your wide variables are of the form 

232 A-one, B-two,.., and you have an unrelated column A-rating, you can 

233 ignore the last one by specifying `suffix='(!?one|two)'` 

234 

235 .. versionadded:: 0.20.0 

236 

237 .. versionchanged:: 0.23.0

421 def get_var_names(df, stub, sep, suffix): 

422 regex = r"^{stub}{sep}{suffix}$".format( 

423 stub=re.escape(stub), sep=re.escape(sep), suffix=suffix 

424 ) 

425 pattern = re.compile(regex) 

426 return [col for col in df.columns if pattern.match(col)]
