590 def _bcast_lwls(linewidths, dashes): 

591 '''Internal helper function to broadcast + scale ls/lw 

592 

593 In the collection drawing code the linewidth and linestyle are 

594 cycled through as circular buffers (via v[i % len(v)]). Thus, 

595 if we are going to scale the dash pattern at set time (not 

596 draw time) we need to do the broadcasting now and expand both 

597 lists to be the same length. 

598 

599 Parameters 

600 ---------- 

601 linewidths : list 

602 line widths of collection 

603 

604 dashes : list 

605 dash specification (offset, (dash pattern tuple)) 

606 

607 Returns 

608 ------- 

609 linewidths, dashes : list 

610 Will be the same length, dashes are scaled by paired linewidth 

611 

612 ''' 

613 if mpl.rcParams['_internal.classic_mode']: 

614 return linewidths, dashes 

615 # make sure they are the same length so we can zip them 

616 if len(dashes) != len(linewidths): 

617 l_dashes = len(dashes) 

618 l_lw = len(linewidths) 

619 gcd = math.gcd(l_dashes, l_lw) 

620 dashes = list(dashes) * (l_lw // gcd) 

621 linewidths = list(linewidths) * (l_dashes // gcd) 

622 

623 # scale the dash patters 

624 dashes = [mlines._scale_dashes(o, d, lw) 

625 for (o, d), lw in zip(dashes, linewidths)] 

626 

627 return linewidths, dashes

1790 (*m*, *n*), (*m*, *n* + 1), (*m* + 1, *n* + 1), and (*m* + 1, *n*)
