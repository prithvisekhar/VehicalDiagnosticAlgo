273def get_empty_dtype_and_na(join_units): 

274 """ 

275 Return dtype and N/A values to use when concatenating specified units. 

276 

277 Returned N/A value may be None which means there was no casting involved. 

278 

279 Returns 

280 ------- 

281 dtype 

282 na 

283 """

285 blk = join_units[0].block 

286 if blk is None: 

287 return np.float64, np.nan 

288 

289 if is_uniform_reindex(join_units): 

290 # XXX: integrate property 

291 empty_dtype = join_units[0].block.dtype 

292 upcasted_na = join_units[0].block.fill_value 

293 return empty_dtype, upcasted_na

298 if unit.block is None:

301 dtypes[i] = unit.dtype 

302

305 for dtype, unit in zip(dtypes, join_units): 

306 if dtype is None: 

307 continue 

308 

309 if is_categorical_dtype(dtype): 

310 upcast_cls = "category" 

311 elif is_datetime64tz_dtype(dtype): 

312 upcast_cls = "datetimetz" 

313 elif issubclass(dtype.type, np.bool_): 

314 upcast_cls = "bool" 

315 elif issubclass(dtype.type, np.object_): 

316 upcast_cls = "object" 

317 elif is_datetime64_dtype(dtype): 

318 upcast_cls = "datetime" 

319 elif is_timedelta64_dtype(dtype): 

320 upcast_cls = "timedelta" 

321 elif is_sparse(dtype): 

322 upcast_cls = dtype.subtype.name 

323 elif is_extension_array_dtype(dtype): 

324 upcast_cls = "object" 

325 elif is_float_dtype(dtype) or is_numeric_dtype(dtype): 

326 upcast_cls = dtype.name 

327 else: 

328 upcast_cls = "float" 

329 

330 # Null blocks should not influence upcast class selection, unless there 

331 # are only null blocks, when same upcasting rules must be applied to 

332 # null upcast classes. 

333 if unit.is_na: 

334 null_upcast_classes[upcast_cls].append(dtype) 

335 else: 

336 upcast_classes[upcast_cls].append(dtype) 

337 

338 if not upcast_classes: 

339 upcast_classes = null_upcast_classes 

340 

341 # create the result 

342 if "object" in upcast_classes:

356 elif "datetime" in upcast_classes: 

357 return np.dtype("M8[ns]"), tslibs.iNaT 

358 elif "timedelta" in upcast_classes: 

359 return np.dtype("m8[ns]"), tslibs.iNaT

361 try: 

362 g = np.find_common_type(upcast_classes, []) 

363 except TypeError: 

364 # At least one is an ExtensionArray 

365 return np.dtype(np.object_), np.nan 

366 else: 

367 if is_float_dtype(g): 

368 return g, g.type(np.nan) 

369 elif is_numeric_dtype(g): 

370 if has_none_blocks:

380 """

384

386 return (

438def combine_concat_plans(plans, concat_axis): 

439 """ 

440 Combine multiple concatenation plans into one. 

441 

442 existing_plan is updated in-place. 

443 """ 

444 if len(plans) == 1: 

445 for p in plans[0]: 

446 yield p[0], [p[1]] 

447 

448 elif concat_axis == 0: 

449 offset = 0 

450 for plan in plans: 

451 last_plc = None 

452 

453 for plc, unit in plan: 

454 yield plc.add(offset), [unit] 

455 last_plc = plc 

456 

457 if last_plc is not None: 

458 offset += last_plc.as_slice.stop 

459 

460 else: 

461 num_ended = [0] 

462 

463 def _next_or_none(seq): 

464 retval = next(seq, None) 

465 if retval is None: 

466 num_ended[0] += 1 

467 return retval 

468 

469 plans = list(map(iter, plans)) 

470 next_items = list(map(_next_or_none, plans)) 

471 

472 while num_ended[0] != len(next_items): 

473 if num_ended[0] > 0: 

474 raise ValueError("Plan shapes are not aligned") 

475 

476 placements, units = zip(*next_items) 

477 

478 lengths = list(map(len, placements)) 

479 min_len, max_len = min(lengths), max(lengths) 

480 

481 if min_len == max_len: 

482 yield placements[0], units 

483 next_items[:] = map(_next_or_none, plans) 

484 else: 

485 yielded_placement = None 

486 yielded_units = [None] * len(next_items) 

487 for i, (plc, unit) in enumerate(next_items): 

488 yielded_units[i] = unit 

489 if len(plc) > min_len: 

490 # trim_join_unit updates unit in place, so only 

491 # placement needs to be sliced to skip min_len. 

492 next_items[i] = (plc[min_len:], trim_join_unit(unit, min_len)) 

493 else: 

494 yielded_placement = plc 

495 next_items[i] = _next_or_none(plans[i]) 

496 

497 yield yielded_placement, yielded_units
