149 def get_basefile(self, tex, fontsize, dpi=None): 

150 """ 

151 Return a filename based on a hash of the string, fontsize, and dpi. 

152 """ 

153 s = ''.join([tex, self.get_font_config(), '%f' % fontsize, 

154 self.get_custom_preamble(), str(dpi or '')]) 

155 return os.path.join( 

156 self.texcache, hashlib.md5(s.encode('utf-8')).hexdigest())

158 def get_font_config(self): 

159 """Reinitializes self if relevant rcParams on have changed.""" 

160 if self._rc_cache is None: 

161 self._rc_cache = dict.fromkeys(self._rc_cache_keys) 

162 changed = [par for par in self._rc_cache_keys 

163 if rcParams[par] != self._rc_cache[par]] 

164 if changed: 

165 _log.debug('following keys changed: %s', changed) 

166 for k in changed: 

167 _log.debug('%-20s: %-10s -> %-10s', 

168 k, self._rc_cache[k], rcParams[k]) 

169 # deepcopy may not be necessary, but feels more future-proof 

170 self._rc_cache[k] = copy.deepcopy(rcParams[k]) 

171 _log.debug('RE-INIT\nold fontconfig: %s', self._fontconfig) 

172 self.__init__() 

173 _log.debug('fontconfig: %s', self._fontconfig) 

174 return self._fontconfig

187 """ 

188 Generate a tex file to render the tex string at a specific font size. 

189

195 fontcmd = {'sans-serif': r'{\sffamily %s}',

197 r'{\rmfamily %s}')

199 

200 if rcParams['text.latex.unicode']: 

201 unicode_preamble = r""" 

202\usepackage[utf8]{inputenc}"""

208%s 

209%s 

210%s 

211\usepackage[papersize={72in,72in},body={70in,70in},margin={1in,1in}]{geometry} 

212\pagestyle{empty} 

213\begin{document} 

214\fontsize{%f}{%f}%s 

215\end{document} 

216""" % (self._font_preamble, unicode_preamble, custom_preamble,

219 if rcParams['text.latex.unicode']:

293 def _run_checked_subprocess(self, command, tex): 

294 _log.debug(command) 

295 try: 

296 report = subprocess.check_output(command, 

297 cwd=self.texcache, 

298 stderr=subprocess.STDOUT) 

299 except subprocess.CalledProcessError as exc: 

300 raise RuntimeError( 

301 '{prog} was not able to process the following string:\n' 

302 '{tex!r}\n\n' 

303 'Here is the full report generated by {prog}:\n' 

304 '{exc}\n\n'.format( 

305 prog=command[0], 

306 tex=tex.encode('unicode_escape'), 

307 exc=exc.output.decode('utf-8'))) 

308 _log.debug(report) 

309 return report

338 def make_dvi_preview(self, tex, fontsize): 

339 """ 

340 Generate a dvi file containing latex's layout of tex string. 

341 

342 It calls make_tex_preview() method and store the size information 

343 (width, height, descent) in a separate file. 

344 

345 Return the file name. 

346 """ 

347 basefile = self.get_basefile(tex, fontsize) 

348 dvifile = '%s.dvi' % basefile 

349 baselinefile = '%s.baseline' % basefile

351 if not os.path.exists(dvifile) or not os.path.exists(baselinefile): 

352 texfile = self.make_tex_preview(tex, fontsize) 

353 report = self._run_checked_subprocess( 

354 ["latex", "-interaction=nonstopmode", "--halt-on-error", 

355 texfile], tex) 

356 

357 # find the box extent information in the latex output 

358 # file and store them in ".baseline" file 

359 m = TexManager._re_vbox.search(report.decode("utf-8")) 

360 with open(basefile + '.baseline', "w") as fh: 

361 fh.write(" ".join(m.groups())) 

362 

363 for fname in glob.glob(basefile + '*'): 

364 if not fname.endswith(('dvi', 'tex', 'baseline')): 

365 try: 

366 os.remove(fname) 

367 except OSError:

415 def get_grey(self, tex, fontsize=None, dpi=None): 

416 """Return the alpha channel.""" 

417 key = tex, self.get_font_config(), fontsize, dpi 

418 alpha = self.grey_arrayd.get(key) 

419 if alpha is None: 

420 pngfile = self.make_png(tex, fontsize, dpi) 

421 X = _png.read_png(os.path.join(self.texcache, pngfile)) 

422 self.grey_arrayd[key] = alpha = X[:, :, -1] 

423 return alpha

425 def get_rgba(self, tex, fontsize=None, dpi=None, rgb=(0, 0, 0)): 

426 """Return latex's rendering of the tex string as an rgba array.""" 

427 if not fontsize: 

428 fontsize = rcParams['font.size'] 

429 if not dpi: 

430 dpi = rcParams['savefig.dpi'] 

431 r, g, b = rgb 

432 key = tex, self.get_font_config(), fontsize, dpi, tuple(rgb) 

433 Z = self.rgba_arrayd.get(key) 

434 

435 if Z is None: 

436 alpha = self.get_grey(tex, fontsize, dpi) 

437 Z = np.dstack([r, g, b, alpha]) 

438 self.rgba_arrayd[key] = Z 

439 

440 return Z

442 def get_text_width_height_descent(self, tex, fontsize, renderer=None): 

443 """Return width, height and descent of the text.""" 

444 if tex.strip() == '': 

445 return 0, 0, 0 

446 

447 dpi_fraction = renderer.points_to_pixels(1.) if renderer else 1 

448 

449 if rcParams['text.latex.preview']: 

450 # use preview.sty 

451 basefile = self.get_basefile(tex, fontsize) 

452 baselinefile = '%s.baseline' % basefile 

453 

454 if not os.path.exists(baselinefile): 

455 dvifile = self.make_dvi_preview(tex, fontsize) 

456 

457 with open(baselinefile) as fh: 

458 l = fh.read().split() 

459 height, depth, width = [float(l1) * dpi_fraction for l1 in l] 

460 return width, height + depth, depth 

461 

462 else: 

463 # use dviread. It sometimes returns a wrong descent. 

464 dvifile = self.make_dvi(tex, fontsize) 

465 with dviread.Dvi(dvifile, 72 * dpi_fraction) as dvi: 

466 page = next(iter(dvi)) 

467 # A total height (including the descent) needs to be returned. 

468 return page.width, page.height + page.descent, page.descent
