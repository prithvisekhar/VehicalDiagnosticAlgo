21 Computes the number of FLOPS in the contraction.

24 ----------

26 The indices involved in the contraction

28 Does this contraction require an inner product?

86def _find_contraction(positions, input_sets, output_set): 

87 """ 

88 Finds the contraction for a given set of input and output sets. 

89 

90 Parameters 

91 ---------- 

92 positions : iterable 

93 Integer positions of terms used in the contraction. 

94 input_sets : list 

95 List of sets that represent the lhs side of the einsum subscript 

96 output_set : set 

97 Set that represents the rhs side of the overall einsum subscript 

98 

99 Returns 

100 ------- 

101 new_result : set 

102 The indices of the resulting contraction 

103 remaining : list 

104 List of sets that have not been contracted, the new set is appended to 

105 the end of this list 

106 idx_removed : set 

107 Indices removed from the entire contraction 

108 idx_contraction : set 

109 The indices used in the current contraction 

110 

111 Examples 

112 -------- 

113 

114 # A simple dot product test case 

115 >>> pos = (0, 1) 

116 >>> isets = [set('ab'), set('bc')] 

117 >>> oset = set('ac') 

118 >>> _find_contraction(pos, isets, oset) 

119 ({'a', 'c'}, [{'a', 'c'}], {'b'}, {'a', 'b', 'c'}) 

120 

121 # A more complex case with additional terms in the contraction 

122 >>> pos = (0, 2) 

123 >>> isets = [set('abd'), set('ac'), set('bdc')] 

124 >>> oset = set('ac') 

125 >>> _find_contraction(pos, isets, oset) 

126 ({'a', 'c'}, [{'a', 'c'}, {'a', 'c'}], {'b', 'd'}, {'a', 'b', 'c', 'd'}) 

127 """ 

128 

129 idx_contract = set() 

130 idx_remain = output_set.copy() 

131 remaining = [] 

132 for ind, value in enumerate(input_sets): 

133 if ind in positions: 

134 idx_contract |= value 

135 else: 

136 remaining.append(value) 

137 idx_remain |= value 

138 

139 new_result = idx_remain & idx_contract 

140 idx_removed = (idx_contract - new_result) 

141 remaining.append(new_result) 

142 

143 return (new_result, remaining, idx_removed, idx_contract)

146def _optimal_path(input_sets, output_set, idx_dict, memory_limit): 

147 """ 

148 Computes all possible pair contractions, sieves the results based 

149 on ``memory_limit`` and returns the lowest cost path. This algorithm 

150 scales factorial with respect to the elements in the list ``input_sets``. 

151 

152 Parameters 

153 ---------- 

154 input_sets : list 

155 List of sets that represent the lhs side of the einsum subscript 

156 output_set : set 

157 Set that represents the rhs side of the overall einsum subscript 

158 idx_dict : dictionary 

159 Dictionary of index sizes 

160 memory_limit : int 

161 The maximum number of elements in a temporary array 

162 

163 Returns 

164 ------- 

165 path : list 

166 The optimal contraction order within the memory limit constraint. 

167 

168 Examples 

169 -------- 

170 >>> isets = [set('abd'), set('ac'), set('bdc')] 

171 >>> oset = set('') 

172 >>> idx_sizes = {'a': 1, 'b':2, 'c':3, 'd':4} 

173 >>> _path__optimal_path(isets, oset, idx_sizes, 5000) 

174 [(0, 2), (0, 1)] 

175 """ 

176 

177 full_results = [(0, [], input_sets)] 

178 for iteration in range(len(input_sets) - 1): 

179 iter_results = [] 

180 

181 # Compute all unique pairs 

182 for curr in full_results: 

183 cost, positions, remaining = curr 

184 for con in itertools.combinations(range(len(input_sets) - iteration), 2): 

185 

186 # Find the contraction 

187 cont = _find_contraction(con, remaining, output_set) 

188 new_result, new_input_sets, idx_removed, idx_contract = cont 

189 

190 # Sieve the results based on memory_limit 

191 new_size = _compute_size_by_dict(new_result, idx_dict) 

192 if new_size > memory_limit: 

193 continue 

194 

195 # Build (total_cost, positions, indices_remaining) 

196 total_cost = cost + _flop_count(idx_contract, idx_removed, len(con), idx_dict) 

197 new_pos = positions + [con] 

198 iter_results.append((total_cost, new_pos, new_input_sets)) 

199 

200 # Update combinatorial list, if we did not find anything return best 

201 # path + remaining contractions 

202 if iter_results: 

203 full_results = iter_results 

204 else: 

205 path = min(full_results, key=lambda x: x[0])[1] 

206 path += [tuple(range(len(input_sets) - iteration))] 

207 return path 

208 

209 # If we have not found anything return single einsum contraction 

210 if len(full_results) == 0: 

211 return [tuple(range(len(input_sets)))] 

212 

213 path = min(full_results, key=lambda x: x[0])[1] 

214 return path

216def _parse_possible_contraction(positions, input_sets, output_set, idx_dict, memory_limit, path_cost, naive_cost): 

217 """Compute the cost (removed size + flops) and resultant indices for 

218 performing the contraction specified by ``positions``. 

219 

220 Parameters 

221 ---------- 

222 positions : tuple of int 

223 The locations of the proposed tensors to contract. 

224 input_sets : list of sets 

225 The indices found on each tensors. 

226 output_set : set 

227 The output indices of the expression. 

228 idx_dict : dict 

229 Mapping of each index to its size. 

230 memory_limit : int 

231 The total allowed size for an intermediary tensor. 

232 path_cost : int 

233 The contraction cost so far. 

234 naive_cost : int 

235 The cost of the unoptimized expression. 

236 

237 Returns 

238 ------- 

239 cost : (int, int) 

240 A tuple containing the size of any indices removed, and the flop cost. 

241 positions : tuple of int 

242 The locations of the proposed tensors to contract. 

243 new_input_sets : list of sets 

244 The resulting new list of indices if this proposed contraction is performed. 

245 

246 """ 

247 

248 # Find the contraction 

249 contract = _find_contraction(positions, input_sets, output_set) 

250 idx_result, new_input_sets, idx_removed, idx_contract = contract 

251 

252 # Sieve the results based on memory_limit 

253 new_size = _compute_size_by_dict(idx_result, idx_dict) 

254 if new_size > memory_limit: 

255 return None 

256 

257 # Build sort tuple 

258 old_sizes = (_compute_size_by_dict(input_sets[p], idx_dict) for p in positions) 

259 removed_size = sum(old_sizes) - new_size 

260 

261 # NB: removed_size used to be just the size of any removed indices i.e.: 

262 # helpers.compute_size_by_dict(idx_removed, idx_dict) 

263 cost = _flop_count(idx_contract, idx_removed, len(positions), idx_dict) 

264 sort = (-removed_size, cost) 

265 

266 # Sieve based on total cost as well 

267 if (path_cost + cost) > naive_cost: 

268 return None 

269 

270 # Add contraction to possible choices 

271 return [sort, positions, new_input_sets]

274def _update_other_results(results, best): 

275 """Update the positions and provisional input_sets of ``results`` based on 

276 performing the contraction result ``best``. Remove any involving the tensors 

277 contracted. 

278 

279 Parameters 

280 ---------- 

281 results : list 

282 List of contraction results produced by ``_parse_possible_contraction``. 

283 best : list 

284 The best contraction of ``results`` i.e. the one that will be performed. 

285 

286 Returns 

287 ------- 

288 mod_results : list 

289 The list of modifed results, updated with outcome of ``best`` contraction. 

290 """ 

291 

292 best_con = best[1] 

293 bx, by = best_con 

294 mod_results = [] 

295 

296 for cost, (x, y), con_sets in results: 

297 

298 # Ignore results involving tensors just contracted 

299 if x in best_con or y in best_con: 

300 continue 

301 

302 # Update the input_sets 

303 del con_sets[by - int(by > x) - int(by > y)] 

304 del con_sets[bx - int(bx > x) - int(bx > y)] 

305 con_sets.insert(-1, best[2][-1]) 

306 

307 # Update the position indices 

308 mod_con = x - int(x > bx) - int(x > by), y - int(y > bx) - int(y > by) 

309 mod_results.append((cost, mod_con, con_sets)) 

310 

311 return mod_results

313def _greedy_path(input_sets, output_set, idx_dict, memory_limit): 

314 """ 

315 Finds the path by contracting the best pair until the input list is 

316 exhausted. The best pair is found by minimizing the tuple 

317 ``(-prod(indices_removed), cost)``. What this amounts to is prioritizing 

318 matrix multiplication or inner product operations, then Hadamard like 

319 operations, and finally outer operations. Outer products are limited by 

320 ``memory_limit``. This algorithm scales cubically with respect to the 

321 number of elements in the list ``input_sets``. 

322 

323 Parameters 

324 ---------- 

325 input_sets : list 

326 List of sets that represent the lhs side of the einsum subscript 

327 output_set : set 

328 Set that represents the rhs side of the overall einsum subscript 

329 idx_dict : dictionary 

330 Dictionary of index sizes 

331 memory_limit_limit : int 

332 The maximum number of elements in a temporary array 

333 

334 Returns 

335 ------- 

336 path : list 

337 The greedy contraction order within the memory limit constraint. 

338 

339 Examples 

340 -------- 

341 >>> isets = [set('abd'), set('ac'), set('bdc')] 

342 >>> oset = set('') 

343 >>> idx_sizes = {'a': 1, 'b':2, 'c':3, 'd':4} 

344 >>> _path__greedy_path(isets, oset, idx_sizes, 5000) 

345 [(0, 2), (0, 1)] 

346 """ 

347 

348 # Handle trivial cases that leaked through 

349 if len(input_sets) == 1: 

350 return [(0,)] 

351 elif len(input_sets) == 2: 

352 return [(0, 1)] 

353 

354 # Build up a naive cost 

355 contract = _find_contraction(range(len(input_sets)), input_sets, output_set) 

356 idx_result, new_input_sets, idx_removed, idx_contract = contract 

357 naive_cost = _flop_count(idx_contract, idx_removed, len(input_sets), idx_dict) 

358 

359 # Initially iterate over all pairs 

360 comb_iter = itertools.combinations(range(len(input_sets)), 2) 

361 known_contractions = [] 

362 

363 path_cost = 0 

364 path = [] 

365 

366 for iteration in range(len(input_sets) - 1): 

367 

368 # Iterate over all pairs on first step, only previously found pairs on subsequent steps 

369 for positions in comb_iter: 

370 

371 # Always initially ignore outer products 

372 if input_sets[positions[0]].isdisjoint(input_sets[positions[1]]): 

373 continue 

374 

375 result = _parse_possible_contraction(positions, input_sets, output_set, idx_dict, memory_limit, path_cost, 

376 naive_cost) 

377 if result is not None: 

378 known_contractions.append(result) 

379 

380 # If we do not have a inner contraction, rescan pairs including outer products 

381 if len(known_contractions) == 0: 

382 

383 # Then check the outer products 

384 for positions in itertools.combinations(range(len(input_sets)), 2): 

385 result = _parse_possible_contraction(positions, input_sets, output_set, idx_dict, memory_limit, 

386 path_cost, naive_cost) 

387 if result is not None: 

388 known_contractions.append(result) 

389 

390 # If we still did not find any remaining contractions, default back to einsum like behavior 

391 if len(known_contractions) == 0: 

392 path.append(tuple(range(len(input_sets)))) 

393 break 

394 

395 # Sort based on first index 

396 best = min(known_contractions, key=lambda x: x[0]) 

397 

398 # Now propagate as many unused contractions as possible to next iteration 

399 known_contractions = _update_other_results(known_contractions, best) 

400 

401 # Next iteration only compute contractions with the new tensor 

402 # All other contractions have been accounted for 

403 input_sets = best[2] 

404 new_tensor_pos = len(input_sets) - 1 

405 comb_iter = ((i, new_tensor_pos) for i in range(new_tensor_pos)) 

406 

407 # Update path and total cost 

408 path.append(best[1]) 

409 path_cost += best[0][1] 

410 

411 return path

414def _can_dot(inputs, result, idx_removed): 

415 """ 

416 Checks if we can use BLAS (np.tensordot) call and its beneficial to do so. 

417 

418 Parameters 

419 ---------- 

420 inputs : list of str 

421 Specifies the subscripts for summation. 

422 result : str 

423 Resulting summation. 

424 idx_removed : set 

425 Indices that are removed in the summation 

426 

427 

428 Returns 

429 ------- 

430 type : bool 

431 Returns true if BLAS should and can be used, else False 

432 

433 Notes 

434 ----- 

435 If the operations is BLAS level 1 or 2 and is not already aligned 

436 we default back to einsum as the memory movement to copy is more 

437 costly than the operation itself. 

438 

439 

440 Examples 

441 -------- 

442 

443 # Standard GEMM operation 

444 >>> _can_dot(['ij', 'jk'], 'ik', set('j')) 

445 True 

446 

447 # Can use the standard BLAS, but requires odd data movement 

448 >>> _can_dot(['ijj', 'jk'], 'ik', set('j')) 

449 False 

450 

451 # DDOT where the memory is not aligned 

452 >>> _can_dot(['ijk', 'ikj'], '', set('ijk')) 

453 False 

454 

455 """ 

456 

457 # All `dot` calls remove indices 

458 if len(idx_removed) == 0: 

459 return False 

460 

461 # BLAS can only handle two operands 

462 if len(inputs) != 2: 

463 return False 

464 

465 input_left, input_right = inputs 

466 

467 for c in set(input_left + input_right): 

468 # can't deal with repeated indices on same input or more than 2 total 

469 nl, nr = input_left.count(c), input_right.count(c) 

470 if (nl > 1) or (nr > 1) or (nl + nr > 2): 

471 return False 

472 

473 # can't do implicit summation or dimension collapse e.g. 

474 # "ab,bc->c" (implicitly sum over 'a') 

475 # "ab,ca->ca" (take diagonal of 'a') 

476 if nl + nr - 1 == int(c in result): 

477 return False 

478 

479 # Build a few temporaries 

480 set_left = set(input_left) 

481 set_right = set(input_right) 

482 keep_left = set_left - idx_removed 

483 keep_right = set_right - idx_removed 

484 rs = len(idx_removed) 

485 

486 # At this point we are a DOT, GEMV, or GEMM operation 

487 

488 # Handle inner products 

489 

490 # DDOT with aligned data 

491 if input_left == input_right: 

492 return True 

493 

494 # DDOT without aligned data (better to use einsum) 

495 if set_left == set_right: 

496 return False 

497 

498 # Handle the 4 possible (aligned) GEMV or GEMM cases 

499 

500 # GEMM or GEMV no transpose 

501 if input_left[-rs:] == input_right[:rs]: 

502 return True 

503 

504 # GEMM or GEMV transpose both 

505 if input_left[:rs] == input_right[-rs:]: 

506 return True 

507 

508 # GEMM or GEMV transpose right 

509 if input_left[-rs:] == input_right[-rs:]: 

510 return True 

511 

512 # GEMM or GEMV transpose left 

513 if input_left[:rs] == input_right[:rs]: 

514 return True 

515 

516 # Einsum is faster than GEMV if we have to copy data 

517 if not keep_left or not keep_right: 

518 return False 

519 

520 # We are a matrix-matrix product, but we need to copy data 

521 return True

691def einsum_path(*operands, **kwargs): 

692 """ 

693 einsum_path(subscripts, *operands, optimize='greedy') 

694 

695 Evaluates the lowest cost contraction order for an einsum expression by 

696 considering the creation of intermediate arrays. 

697 

698 Parameters 

699 ---------- 

700 subscripts : str 

701 Specifies the subscripts for summation. 

702 *operands : list of array_like 

703 These are the arrays for the operation. 

704 optimize : {bool, list, tuple, 'greedy', 'optimal'} 

705 Choose the type of path. If a tuple is provided, the second argument is 

706 assumed to be the maximum intermediate size created. If only a single 

707 argument is provided the largest input or output array size is used 

708 as a maximum intermediate size. 

709 

710 * if a list is given that starts with ``einsum_path``, uses this as the 

711 contraction path 

712 * if False no optimization is taken 

713 * if True defaults to the 'greedy' algorithm 

714 * 'optimal' An algorithm that combinatorially explores all possible 

715 ways of contracting the listed tensors and choosest the least costly 

716 path. Scales exponentially with the number of terms in the 

717 contraction. 

718 * 'greedy' An algorithm that chooses the best pair contraction 

719 at each step. Effectively, this algorithm searches the largest inner, 

720 Hadamard, and then outer products at each step. Scales cubically with 

721 the number of terms in the contraction. Equivalent to the 'optimal' 

722 path for most contractions. 

723 

724 Default is 'greedy'. 

725 

726 Returns 

727 ------- 

728 path : list of tuples 

729 A list representation of the einsum path. 

730 string_repr : str 

731 A printable representation of the einsum path. 

732 

733 Notes 

734 ----- 

735 The resulting path indicates which terms of the input contraction should be 

736 contracted first, the result of this contraction is then appended to the 

737 end of the contraction list. This list can then be iterated over until all 

738 intermediate contractions are complete. 

739 

740 See Also 

741 -------- 

742 einsum, linalg.multi_dot 

743 

744 Examples 

745 -------- 

746 

747 We can begin with a chain dot example. In this case, it is optimal to 

748 contract the ``b`` and ``c`` tensors first as represented by the first 

749 element of the path ``(1, 2)``. The resulting tensor is added to the end 

750 of the contraction and the remaining contraction ``(0, 1)`` is then 

751 completed. 

752 

753 >>> a = np.random.rand(2, 2) 

754 >>> b = np.random.rand(2, 5) 

755 >>> c = np.random.rand(5, 2) 

756 >>> path_info = np.einsum_path('ij,jk,kl->il', a, b, c, optimize='greedy') 

757 >>> print(path_info[0]) 

758 ['einsum_path', (1, 2), (0, 1)]

764 Optimized FLOP count: 5.600e+01 

765 Theoretical speedup: 2.857 

766 Largest intermediate: 4.000e+00 elements 

767 ------------------------------------------------------------------------- 

768 scaling current remaining 

769 ------------------------------------------------------------------------- 

770 3 kl,jk->jl ij,jl->il 

771 3 jl,ij->il il->il

777 >>> C = np.random.rand(10, 10)

779 optimize='greedy')

782 ['einsum_path', (0, 2), (0, 3), (0, 2), (0, 1)]

785 Naive scaling: 8 

786 Optimized scaling: 5

788 Optimized FLOP count: 8.000e+05

790 Largest intermediate: 1.000e+04 elements

792 scaling current remaining 

793 --------------------------------------------------------------------------

795 5 bcde,fb->cdef gc,hd,cdef->efgh 

796 5 cdef,gc->defg hd,defg->efgh 

797 5 defg,hd->efgh efgh->efgh

800 # Make sure all keywords are valid 

801 valid_contract_kwargs = ['optimize', 'einsum_call'] 

802 unknown_kwargs = [k for (k, v) in kwargs.items() if k 

803 not in valid_contract_kwargs] 

804 if len(unknown_kwargs): 

805 raise TypeError("Did not understand the following kwargs:" 

806 " %s" % unknown_kwargs) 

807 

808 # Figure out what the path really is 

809 path_type = kwargs.pop('optimize', True) 

810 if path_type is True: 

811 path_type = 'greedy' 

812 if path_type is None: 

813 path_type = False 

814 

815 memory_limit = None 

816 

817 # No optimization or a named path algorithm 

818 if (path_type is False) or isinstance(path_type, basestring): 

819 pass 

820 

821 # Given an explicit path 

822 elif len(path_type) and (path_type[0] == 'einsum_path'): 

823 pass 

824 

825 # Path tuple with memory limit 

826 elif ((len(path_type) == 2) and isinstance(path_type[0], basestring) and 

827 isinstance(path_type[1], (int, float))): 

828 memory_limit = int(path_type[1]) 

829 path_type = path_type[0] 

830 

831 else: 

832 raise TypeError("Did not understand the path: %s" % str(path_type)) 

833 

834 # Hidden option, only einsum should call this 

835 einsum_call_arg = kwargs.pop("einsum_call", False) 

836 

837 # Python side parsing 

838 input_subscripts, output_subscript, operands = _parse_einsum_input(operands) 

839 subscripts = input_subscripts + '->' + output_subscript 

840 

841 # Build a few useful list and sets 

842 input_list = input_subscripts.split(',') 

843 input_sets = [set(x) for x in input_list] 

844 output_set = set(output_subscript) 

845 indices = set(input_subscripts.replace(',', '')) 

846 

847 # Get length of each unique dimension and ensure all dimensions are correct 

848 dimension_dict = {} 

849 broadcast_indices = [[] for x in range(len(input_list))] 

850 for tnum, term in enumerate(input_list): 

851 sh = operands[tnum].shape 

852 if len(sh) != len(term): 

853 raise ValueError("Einstein sum subscript %s does not contain the " 

854 "correct number of indices for operand %d." 

855 % (input_subscripts[tnum], tnum)) 

856 for cnum, char in enumerate(term): 

857 dim = sh[cnum] 

858 

859 # Build out broadcast indices 

860 if dim == 1: 

861 broadcast_indices[tnum].append(char) 

862 

863 if char in dimension_dict.keys(): 

864 # For broadcasting cases we always want the largest dim size 

865 if dimension_dict[char] == 1: 

866 dimension_dict[char] = dim 

867 elif dim not in (1, dimension_dict[char]): 

868 raise ValueError("Size of label '%s' for operand %d (%d) " 

869 "does not match previous terms (%d)." 

870 % (char, tnum, dimension_dict[char], dim)) 

871 else: 

872 dimension_dict[char] = dim 

873 

874 # Convert broadcast inds to sets 

875 broadcast_indices = [set(x) for x in broadcast_indices] 

876 

877 # Compute size of each input array plus the output array 

878 size_list = [] 

879 for term in input_list + [output_subscript]: 

880 size_list.append(_compute_size_by_dict(term, dimension_dict)) 

881 max_size = max(size_list) 

882 

883 if memory_limit is None: 

884 memory_arg = max_size 

885 else: 

886 memory_arg = memory_limit 

887 

888 # Compute naive cost 

889 # This isn't quite right, need to look into exactly how einsum does this 

890 inner_product = (sum(len(x) for x in input_sets) - len(indices)) > 0 

891 naive_cost = _flop_count(indices, inner_product, len(input_list), dimension_dict) 

892 

893 # Compute the path 

894 if (path_type is False) or (len(input_list) in [1, 2]) or (indices == output_set): 

895 # Nothing to be optimized, leave it to einsum 

896 path = [tuple(range(len(input_list)))] 

897 elif path_type == "greedy": 

898 path = _greedy_path(input_sets, output_set, dimension_dict, memory_arg) 

899 elif path_type == "optimal": 

900 path = _optimal_path(input_sets, output_set, dimension_dict, memory_arg) 

901 elif path_type[0] == 'einsum_path': 

902 path = path_type[1:] 

903 else: 

904 raise KeyError("Path name %s not found", path_type) 

905 

906 cost_list, scale_list, size_list, contraction_list = [], [], [], [] 

907 

908 # Build contraction tuple (positions, gemm, einsum_str, remaining) 

909 for cnum, contract_inds in enumerate(path): 

910 # Make sure we remove inds from right to left 

911 contract_inds = tuple(sorted(list(contract_inds), reverse=True)) 

912 

913 contract = _find_contraction(contract_inds, input_sets, output_set) 

914 out_inds, input_sets, idx_removed, idx_contract = contract 

915 

916 cost = _flop_count(idx_contract, idx_removed, len(contract_inds), dimension_dict) 

917 cost_list.append(cost) 

918 scale_list.append(len(idx_contract)) 

919 size_list.append(_compute_size_by_dict(out_inds, dimension_dict)) 

920 

921 bcast = set() 

922 tmp_inputs = [] 

923 for x in contract_inds: 

924 tmp_inputs.append(input_list.pop(x)) 

925 bcast |= broadcast_indices.pop(x) 

926 

927 new_bcast_inds = bcast - idx_removed 

928 

929 # If we're broadcasting, nix blas 

930 if not len(idx_removed & bcast): 

931 do_blas = _can_dot(tmp_inputs, out_inds, idx_removed) 

932 else: 

933 do_blas = False 

934 

935 # Last contraction 

936 if (cnum - len(path)) == -1: 

937 idx_result = output_subscript 

938 else: 

939 sort_result = [(dimension_dict[ind], ind) for ind in out_inds] 

940 idx_result = "".join([x[1] for x in sorted(sort_result)]) 

941 

942 input_list.append(idx_result) 

943 broadcast_indices.append(new_bcast_inds) 

944 einsum_str = ",".join(tmp_inputs) + "->" + idx_result 

945 

946 contraction = (contract_inds, idx_removed, einsum_str, input_list[:], do_blas) 

947 contraction_list.append(contraction) 

948 

949 opt_cost = sum(cost_list) + 1 

950 

951 if einsum_call_arg: 

952 return (operands, contraction_list) 

953 

954 # Return the path along with a nice string representation 

955 overall_contraction = input_subscripts + "->" + output_subscript 

956 header = ("scaling", "current", "remaining") 

957 

958 speedup = naive_cost / opt_cost 

959 max_i = max(size_list) 

960 

961 path_print = " Complete contraction: %s\n" % overall_contraction 

962 path_print += " Naive scaling: %d\n" % len(indices) 

963 path_print += " Optimized scaling: %d\n" % max(scale_list) 

964 path_print += " Naive FLOP count: %.3e\n" % naive_cost 

965 path_print += " Optimized FLOP count: %.3e\n" % opt_cost 

966 path_print += " Theoretical speedup: %3.3f\n" % speedup 

967 path_print += " Largest intermediate: %.3e elements\n" % max_i 

968 path_print += "-" * 74 + "\n" 

969 path_print += "%6s %24s %40s\n" % header 

970 path_print += "-" * 74 

971 

972 for n, contraction in enumerate(contraction_list): 

973 inds, idx_rm, einsum_str, remaining, blas = contraction 

974 remaining_str = ",".join(remaining) + "->" + output_subscript 

975 path_run = (scale_list[n], einsum_str, remaining_str) 

976 path_print += "\n%4d %24s %40s" % path_run 

977 

978 path = ['einsum_path'] + path 

979 return (path, path_print)

984 """ 

985 einsum(subscripts, *operands, out=None, dtype=None, order='K', 

986 casting='safe', optimize=False) 

987 

988 Evaluates the Einstein summation convention on the operands. 

989 

990 Using the Einstein summation convention, many common multi-dimensional 

991 array operations can be represented in a simple fashion. This function 

992 provides a way to compute such summations. The best way to understand this 

993 function is to try the examples below, which show how many common NumPy 

994 functions can be implemented as calls to `einsum`. 

995 

996 Parameters 

997 ---------- 

998 subscripts : str 

999 Specifies the subscripts for summation. 

1000 operands : list of array_like 

1001 These are the arrays for the operation. 

1002 out : {ndarray, None}, optional 

1003 If provided, the calculation is done into this array. 

1004 dtype : {data-type, None}, optional 

1005 If provided, forces the calculation to use the data type specified. 

1006 Note that you may have to also give a more liberal `casting` 

1007 parameter to allow the conversions. Default is None. 

1008 order : {'C', 'F', 'A', 'K'}, optional 

1009 Controls the memory layout of the output. 'C' means it should 

1010 be C contiguous. 'F' means it should be Fortran contiguous, 

1011 'A' means it should be 'F' if the inputs are all 'F', 'C' otherwise. 

1012 'K' means it should be as close to the layout as the inputs as 

1013 is possible, including arbitrarily permuted axes. 

1014 Default is 'K'. 

1015 casting : {'no', 'equiv', 'safe', 'same_kind', 'unsafe'}, optional 

1016 Controls what kind of data casting may occur. Setting this to 

1017 'unsafe' is not recommended, as it can adversely affect accumulations. 

1018 

1019 * 'no' means the data types should not be cast at all. 

1020 * 'equiv' means only byte-order changes are allowed. 

1021 * 'safe' means only casts which can preserve values are allowed. 

1022 * 'same_kind' means only safe casts or casts within a kind, 

1023 like float64 to float32, are allowed. 

1024 * 'unsafe' means any data conversions may be done. 

1025 

1026 Default is 'safe'. 

1027 optimize : {False, True, 'greedy', 'optimal'}, optional 

1028 Controls if intermediate optimization should occur. No optimization 

1029 will occur if False and True will default to the 'greedy' algorithm. 

1030 Also accepts an explicit contraction list from the ``np.einsum_path`` 

1031 function. See ``np.einsum_path`` for more details. Defaults to False. 

1032 

1033 Returns 

1034 ------- 

1035 output : ndarray 

1036 The calculation based on the Einstein summation convention. 

1037 

1038 See Also 

1039 -------- 

1040 einsum_path, dot, inner, outer, tensordot, linalg.multi_dot 

1041 

1042 Notes 

1043 ----- 

1044 .. versionadded:: 1.6.0 

1045 

1046 The subscripts string is a comma-separated list of subscript labels, 

1047 where each label refers to a dimension of the corresponding operand. 

1048 Repeated subscripts labels in one operand take the diagonal. For example, 

1049 ``np.einsum('ii', a)`` is equivalent to ``np.trace(a)``. 

1050 

1051 Whenever a label is repeated, it is summed, so ``np.einsum('i,i', a, b)`` 

1052 is equivalent to ``np.inner(a,b)``. If a label appears only once, 

1053 it is not summed, so ``np.einsum('i', a)`` produces a view of ``a`` 

1054 with no changes. 

1055 

1056 The order of labels in the output is by default alphabetical. This 

1057 means that ``np.einsum('ij', a)`` doesn't affect a 2D array, while 

1058 ``np.einsum('ji', a)`` takes its transpose. 

1059 

1060 The output can be controlled by specifying output subscript labels 

1061 as well. This specifies the label order, and allows summing to 

1062 be disallowed or forced when desired. The call ``np.einsum('i->', a)`` 

1063 is like ``np.sum(a, axis=-1)``, and ``np.einsum('ii->i', a)`` 

1064 is like ``np.diag(a)``. The difference is that `einsum` does not 

1065 allow broadcasting by default. 

1066 

1067 To enable and control broadcasting, use an ellipsis. Default 

1068 NumPy-style broadcasting is done by adding an ellipsis 

1069 to the left of each term, like ``np.einsum('...ii->...i', a)``. 

1070 To take the trace along the first and last axes, 

1071 you can do ``np.einsum('i...i', a)``, or to do a matrix-matrix 

1072 product with the left-most indices instead of rightmost, you can do 

1073 ``np.einsum('ij...,jk...->ik...', a, b)``. 

1074 

1075 When there is only one operand, no axes are summed, and no output 

1076 parameter is provided, a view into the operand is returned instead 

1077 of a new array. Thus, taking the diagonal as ``np.einsum('ii->i', a)`` 

1078 produces a view. 

1079 

1080 An alternative way to provide the subscripts and operands is as 

1081 ``einsum(op0, sublist0, op1, sublist1, ..., [sublistout])``. The examples 

1082 below have corresponding `einsum` calls with the two parameter methods. 

1083 

1084 .. versionadded:: 1.10.0 

1085 

1086 Views returned from einsum are now writeable whenever the input array 

1087 is writeable. For example, ``np.einsum('ijk...->kji...', a)`` will now 

1088 have the same effect as ``np.swapaxes(a, 0, 2)`` and 

1089 ``np.einsum('ii->i', a)`` will return a writeable view of the diagonal 

1090 of a 2D array. 

1091 

1092 .. versionadded:: 1.12.0 

1093 

1094 Added the ``optimize`` argument which will optimize the contraction order 

1095 of an einsum expression. For a contraction with three or more operands this 

1096 can greatly increase the computational efficiency at the cost of a larger 

1097 memory footprint during computation. 

1098 

1099 See ``np.einsum_path`` for more details.
