113 def parse(self, pattern): 

114 """ 

115 Parse the given fontconfig *pattern* and return a dictionary 

116 of key/value pairs useful for initializing a 

117 :class:`font_manager.FontProperties` object. 

118 """ 

119 props = self._properties = {} 

120 try: 

121 self._parser.parseString(pattern) 

122 except self.ParseException as e: 

123 raise ValueError( 

124 "Could not parse font string: '%s'\n%s" % (pattern, e)) 

125 

126 self._properties = None 

127 

128 self._parser.resetCache() 

129 

130 return props 

131 

132 def _family(self, s, loc, tokens): 

133 return [family_unescape(r'\1', str(tokens[0]))] 

134 

135 def _size(self, s, loc, tokens): 

136 return [float(tokens[0])] 

137 

138 def _name(self, s, loc, tokens): 

139 return [str(tokens[0])] 

140 

141 def _value(self, s, loc, tokens): 

142 return [value_unescape(r'\1', str(tokens[0]))] 

143 

144 def _families(self, s, loc, tokens): 

145 self._properties['family'] = [str(x) for x in tokens] 

146 return [] 

147 

148 def _point_sizes(self, s, loc, tokens): 

149 self._properties['size'] = [str(x) for x in tokens] 

150 return [] 

151 

152 def _property(self, s, loc, tokens): 

153 if len(tokens) == 1: 

154 if tokens[0] in self._constants: 

155 key, val = self._constants[tokens[0]] 

156 self._properties.setdefault(key, []).append(val) 

157 else: 

158 key = tokens[0] 

159 val = tokens[1:] 

160 self._properties.setdefault(key, []).extend(val) 

161 return []

168parse_fontconfig_pattern = lru_cache()(FontconfigPatternParser().parse)

171def generate_fontconfig_pattern(d): 

172 """ 

173 Given a dictionary of key/value pairs, generates a fontconfig 

174 pattern string. 

175 """ 

176 props = [] 

177 families = '' 

178 size = '' 

179 for key in 'family style variant weight stretch file size'.split(): 

180 val = getattr(d, 'get_' + key)() 

181 if val is not None and val != []: 

182 if type(val) == list: 

183 val = [value_escape(r'\\\1', str(x)) for x in val 

184 if x is not None] 

185 if val != []: 

186 val = ','.join(val) 

187 props.append(":%s=%s" % (key, val)) 

188 return ''.join(props)
