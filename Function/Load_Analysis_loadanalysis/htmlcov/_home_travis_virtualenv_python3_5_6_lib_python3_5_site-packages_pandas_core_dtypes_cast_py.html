201def maybe_upcast_putmask(result, mask, other): 

202 """ 

203 A safe version of putmask that potentially upcasts the result. 

204 The result is replaced with the first N elements of other, 

205 where N is the number of True values in mask. 

206 If the length of other is shorter than N, other will be repeated. 

207 

208 Parameters 

209 ---------- 

210 result : ndarray 

211 The destination array. This will be mutated in-place if no upcasting is 

212 necessary. 

213 mask : boolean ndarray 

214 other : ndarray or scalar 

215 The source array or value 

216 

217 Returns 

218 ------- 

219 result : ndarray 

220 changed : boolean 

221 Set to true if the result array was upcasted 

222 

223 Examples 

224 -------- 

225 >>> result, _ = maybe_upcast_putmask(np.arange(1,6), 

226 np.array([False, True, False, True, True]), np.arange(21,23)) 

227 >>> result 

228 array([1, 21, 3, 22, 21]) 

229 """

231 if not isinstance(result, np.ndarray): 

232 raise ValueError("The result input must be a ndarray.") 

233 

234 if mask.any(): 

235 # Two conversions for date-like dtypes that can't be done automatically 

236 # in np.place: 

237 # NaN -> NaT 

238 # integer or integer array -> date-like array 

239 if is_datetimelike(result.dtype): 

240 if is_scalar(other): 

241 if isna(other): 

242 other = result.dtype.type("nat") 

243 elif is_integer(other): 

244 other = np.array(other, dtype=result.dtype) 

245 elif is_integer_dtype(other): 

246 other = np.array(other, dtype=result.dtype)

248 def changeit(): 

249 

250 # try to directly set by expanding our array to full 

251 # length of the boolean 

252 try: 

253 om = other[mask] 

254 om_at = om.astype(result.dtype) 

255 if (om == om_at).all(): 

256 new_result = result.values.copy() 

257 new_result[mask] = om_at 

258 result[:] = new_result 

259 return result, False 

260 except Exception: 

261 pass 

262 

263 # we are forced to change the dtype of the result as the input 

264 # isn't compatible 

265 r, _ = maybe_upcast(result, fill_value=other, copy=True)

273 # have values, say integers, in the success portion then it's ok to not 

274 # upcast) 

275 new_dtype, _ = maybe_promote(result.dtype, other) 

276 if new_dtype != result.dtype: 

277 

278 # we have a scalar or len 0 ndarray 

279 # and its nan and we are changing some values 

280 if is_scalar(other) or (isinstance(other, np.ndarray) and other.ndim < 1): 

281 if isna(other): 

282 return changeit() 

283 

284 # we have an ndarray and the masking has nans in it 

285 else: 

286 

287 if isna(other).any(): 

288 return changeit() 

289 

290 try: 

291 np.place(result, mask, other) 

292 except Exception: 

293 return changeit() 

294 

295 return result, False

628def astype_nansafe(arr, dtype, copy=True, skipna=False):

799def soft_convert_objects( 

800 values, datetime=True, numeric=True, timedelta=True, coerce=False, copy=True 

801): 

802 """ if we have an object dtype, try to coerce dates and/or numbers """ 

803 

804 conversion_count = sum((datetime, numeric, timedelta)) 

805 if conversion_count == 0: 

806 raise ValueError( 

807 "At least one of datetime, numeric or timedelta must " "be True." 

808 ) 

809 elif conversion_count > 1 and coerce: 

810 raise ValueError( 

811 "Only one of 'datetime', 'numeric' or " 

812 "'timedelta' can be True when when coerce=True." 

813 ) 

814 

815 if isinstance(values, (list, tuple)): 

816 # List or scalar 

817 values = np.array(values, dtype=np.object_) 

818 elif not hasattr(values, "dtype"): 

819 values = np.array([values], dtype=np.object_) 

820 elif not is_object_dtype(values.dtype): 

821 # If not object, do not attempt conversion 

822 values = values.copy() if copy else values 

823 return values 

824 

825 # If 1 flag is coerce, ensure 2 others are False 

826 if coerce: 

827 # Immediate return if coerce 

828 if datetime: 

829 from pandas import to_datetime 

830 

831 return to_datetime(values, errors="coerce").to_numpy() 

832 elif timedelta: 

833 from pandas import to_timedelta 

834 

835 return to_timedelta(values, errors="coerce").to_numpy() 

836 elif numeric: 

837 from pandas import to_numeric 

838 

839 return to_numeric(values, errors="coerce") 

840 

841 # Soft conversions 

842 if datetime: 

843 # GH 20380, when datetime is beyond year 2262, hence outside 

844 # bound of nanosecond-resolution 64-bit integers. 

845 try: 

846 values = lib.maybe_convert_objects(values, convert_datetime=datetime) 

847 except OutOfBoundsDatetime: 

848 pass 

849 

850 if timedelta and is_object_dtype(values.dtype): 

851 # Object check to ensure only run if previous did not convert 

852 values = lib.maybe_convert_objects(values, convert_timedelta=timedelta) 

853 

854 if numeric and is_object_dtype(values.dtype): 

855 try: 

856 converted = lib.maybe_convert_numeric(values, set(), coerce_numeric=True) 

857 # If all NaNs, then do not-alter 

858 values = converted if not isna(converted).all() else values 

859 values = values.copy() if copy else values 

860 except Exception: 

861 pass 

862 

863 return values

986def maybe_cast_to_datetime(value, dtype, errors="raise"):

1119def find_common_type(types): 

1120 """ 

1121 Find a common data type among the given dtypes. 

1122 

1123 Parameters 

1124 ---------- 

1125 types : list of dtypes 

1126 

1127 Returns 

1128 ------- 

1129 pandas extension or numpy dtype 

1130 

1131 See Also 

1132 -------- 

1133 numpy.find_common_type 

1134 

1135 """

1138 raise ValueError("no types given")

1142 # workaround for find_common_type([np.dtype('datetime64[ns]')] * 2) 

1143 # => object 

1144 if all(is_dtype_equal(first, t) for t in types[1:]):
