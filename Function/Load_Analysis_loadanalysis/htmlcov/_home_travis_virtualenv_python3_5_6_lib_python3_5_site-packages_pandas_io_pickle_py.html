13def to_pickle(obj, path, compression="infer", protocol=pickle.HIGHEST_PROTOCOL): 

14 """ 

15 Pickle (serialize) object to file. 

16 

17 Parameters 

18 ---------- 

19 obj : any object 

20 Any python object. 

21 path : str 

22 File path where the pickled object will be stored. 

23 compression : {'infer', 'gzip', 'bz2', 'zip', 'xz', None}, default 'infer' 

24 A string representing the compression to use in the output file. By 

25 default, infers from the file extension in specified path. 

26 

27 .. versionadded:: 0.20.0 

28 protocol : int 

29 Int which indicates which protocol should be used by the pickler, 

30 default HIGHEST_PROTOCOL (see [1], paragraph 12.1.2). The possible 

31 values for this parameter depend on the version of Python. For Python 

32 2.x, possible values are 0, 1, 2. For Python>=3.0, 3 is a valid value. 

33 For Python >= 3.4, 4 is a valid value. A negative value for the 

34 protocol parameter is equivalent to setting its value to 

35 HIGHEST_PROTOCOL. 

36 

37 .. [1] https://docs.python.org/3/library/pickle.html 

38 .. versionadded:: 0.21.0 

39 

40 See Also 

41 -------- 

42 read_pickle : Load pickled pandas object (or any object) from file. 

43 DataFrame.to_hdf : Write DataFrame to an HDF5 file. 

44 DataFrame.to_sql : Write DataFrame to a SQL database. 

45 DataFrame.to_parquet : Write a DataFrame to the binary parquet format. 

46 

47 Examples 

48 -------- 

49 >>> original_df = pd.DataFrame({"foo": range(5), "bar": range(5, 10)}) 

50 >>> original_df 

51 foo bar 

52 0 0 5 

53 1 1 6 

54 2 2 7 

55 3 3 8 

56 4 4 9 

57 >>> pd.to_pickle(original_df, "./dummy.pkl") 

58 

59 >>> unpickled_df = pd.read_pickle("./dummy.pkl") 

60 >>> unpickled_df 

61 foo bar 

62 0 0 5 

63 1 1 6 

64 2 2 7 

65 3 3 8 

66 4 4 9 

67 

68 >>> import os 

69 >>> os.remove("./dummy.pkl") 

70 """ 

71 path = _stringify_path(path) 

72 f, fh = _get_handle(path, "wb", compression=compression, is_text=False) 

73 if protocol < 0: 

74 protocol = pickle.HIGHEST_PROTOCOL 

75 try: 

76 f.write(pickle.dumps(obj, protocol=protocol)) 

77 finally: 

78 f.close() 

79 for _f in fh: 

80 _f.close()

83def read_pickle(path, compression="infer"): 

84 """ 

85 Load pickled pandas object (or any object) from file. 

86 

87 .. warning:: 

88 

89 Loading pickled data received from untrusted sources can be 

90 unsafe. See `here <https://docs.python.org/3/library/pickle.html>`__. 

91 

92 Parameters 

93 ---------- 

94 path : str 

95 File path where the pickled object will be loaded. 

96 compression : {'infer', 'gzip', 'bz2', 'zip', 'xz', None}, default 'infer' 

97 For on-the-fly decompression of on-disk data. If 'infer', then use 

98 gzip, bz2, xz or zip if path ends in '.gz', '.bz2', '.xz', 

99 or '.zip' respectively, and no decompression otherwise. 

100 Set to None for no decompression. 

101 

102 .. versionadded:: 0.20.0 

103 

104 Returns 

105 ------- 

106 unpickled : same type as object stored in file 

107 

108 See Also 

109 -------- 

110 DataFrame.to_pickle : Pickle (serialize) DataFrame object to file. 

111 Series.to_pickle : Pickle (serialize) Series object to file. 

112 read_hdf : Read HDF5 file into a DataFrame. 

113 read_sql : Read SQL query or database table into a DataFrame. 

114 read_parquet : Load a parquet object, returning a DataFrame. 

115 

116 Notes 

117 ----- 

118 read_pickle is only guaranteed to be backwards compatible to pandas 0.20.3. 

119 

120 Examples 

121 -------- 

122 >>> original_df = pd.DataFrame({"foo": range(5), "bar": range(5, 10)}) 

123 >>> original_df 

124 foo bar 

125 0 0 5 

126 1 1 6 

127 2 2 7 

128 3 3 8 

129 4 4 9 

130 >>> pd.to_pickle(original_df, "./dummy.pkl") 

131 

132 >>> unpickled_df = pd.read_pickle("./dummy.pkl") 

133 >>> unpickled_df 

134 foo bar 

135 0 0 5 

136 1 1 6 

137 2 2 7 

138 3 3 8 

139 4 4 9 

140 

141 >>> import os 

142 >>> os.remove("./dummy.pkl") 

143 """ 

144 path = _stringify_path(path) 

145 f, fh = _get_handle(path, "rb", compression=compression, is_text=False) 

146 

147 # 1) try standard libary Pickle 

148 # 2) try pickle_compat (older pandas version) to handle subclass changes 

149 # 3) try pickle_compat with latin1 encoding 

150 

151 try: 

152 with warnings.catch_warnings(record=True): 

153 # We want to silence any warnings about, e.g. moved modules. 

154 warnings.simplefilter("ignore", Warning) 

155 return pickle.load(f) 

156 except Exception: # noqa: E722 

157 try: 

158 return pc.load(f, encoding=None) 

159 except Exception: # noqa: E722 

160 return pc.load(f, encoding="latin1") 

161 finally: 

162 f.close() 

163 for _f in fh: 

164 _f.close()
