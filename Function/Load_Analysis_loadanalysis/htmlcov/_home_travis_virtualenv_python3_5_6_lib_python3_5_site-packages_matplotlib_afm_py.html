1""" 

2This is a python interface to Adobe Font Metrics Files. Although a 

3number of other python implementations exist, and may be more complete 

4than this, it was decided not to go with them because they were 

5either: 

6 

7 1) copyrighted or used a non-BSD compatible license 

8 

9 2) had too many dependencies and a free standing lib was needed 

10 

11 3) Did more than needed and it was easier to write afresh rather than 

12 figure out how to get just what was needed. 

13 

14It is pretty easy to use, and requires only built-in python libs: 

15 

16 >>> from matplotlib import rcParams 

17 >>> import os.path 

18 >>> afm_fname = os.path.join(rcParams['datapath'], 

19 ... 'fonts', 'afm', 'ptmr8a.afm') 

20 >>> 

21 >>> from matplotlib.afm import AFM 

22 >>> with open(afm_fname, 'rb') as fh: 

23 ... afm = AFM(fh) 

24 >>> afm.string_width_height('What the heck?') 

25 (6220.0, 694) 

26 >>> afm.get_fontname() 

27 'Times-Roman' 

28 >>> afm.get_kern_dist('A', 'f') 

29 0 

30 >>> afm.get_kern_dist('A', 'y') 

31 -92.0 

32 >>> afm.get_bbox_char('!') 

33 [130, -9, 238, 676] 

34 

35As in the Adobe Font Metrics File Format Specification, all dimensions 

36are given in units of 1/1000 of the scale factor (point size) of the font 

37being used. 

38"""

81def _sanity_check(fh): 

82 """ 

83 Check if the file at least looks like AFM. 

84 If not, raise :exc:`RuntimeError`. 

85 """ 

86 

87 # Remember the file position in case the caller wants to 

88 # do something else with the file. 

89 pos = fh.tell() 

90 try: 

91 line = next(fh) 

92 finally: 

93 fh.seek(pos, 0) 

94 

95 # AFM spec, Section 4: The StartFontMetrics keyword [followed by a 

96 # version number] must be the first line in the file, and the 

97 # EndFontMetrics keyword must be the last non-empty line in the 

98 # file. We just check the first line. 

99 if not line.startswith(b'StartFontMetrics'): 

100 raise RuntimeError('Not an AFM file')

103def _parse_header(fh): 

104 """ 

105 Reads the font metrics header (up to the char metrics) and returns 

106 a dictionary mapping *key* to *val*. *val* will be converted to the 

107 appropriate python type as necessary; e.g.: 

108 

109 * 'False'->False 

110 * '0'->0 

111 * '-168 -218 1000 898'-> [-168, -218, 1000, 898] 

112 

113 Dictionary keys are 

114 

115 StartFontMetrics, FontName, FullName, FamilyName, Weight, 

116 ItalicAngle, IsFixedPitch, FontBBox, UnderlinePosition, 

117 UnderlineThickness, Version, Notice, EncodingScheme, CapHeight, 

118 XHeight, Ascender, Descender, StartCharMetrics 

119 

120 """ 

121 headerConverters = { 

122 b'StartFontMetrics': _to_float, 

123 b'FontName': _to_str, 

124 b'FullName': _to_str, 

125 b'FamilyName': _to_str, 

126 b'Weight': _to_str, 

127 b'ItalicAngle': _to_float, 

128 b'IsFixedPitch': _to_bool, 

129 b'FontBBox': _to_list_of_ints, 

130 b'UnderlinePosition': _to_int, 

131 b'UnderlineThickness': _to_int, 

132 b'Version': _to_str, 

133 b'Notice': _to_str, 

134 b'EncodingScheme': _to_str, 

135 b'CapHeight': _to_float, # Is the second version a mistake, or 

136 b'Capheight': _to_float, # do some AFM files contain 'Capheight'? -JKS 

137 b'XHeight': _to_float, 

138 b'Ascender': _to_float, 

139 b'Descender': _to_float, 

140 b'StdHW': _to_float, 

141 b'StdVW': _to_float, 

142 b'StartCharMetrics': _to_int, 

143 b'CharacterSet': _to_str, 

144 b'Characters': _to_int, 

145 } 

146 

147 d = {} 

148 for line in fh: 

149 line = line.rstrip() 

150 if line.startswith(b'Comment'): 

151 continue 

152 lst = line.split(b' ', 1) 

153 

154 key = lst[0] 

155 if len(lst) == 2: 

156 val = lst[1] 

157 else: 

158 val = b'' 

159 

160 try: 

161 d[key] = headerConverters[key](val) 

162 except ValueError: 

163 print('Value error parsing header in AFM:', key, val, 

164 file=sys.stderr) 

165 continue 

166 except KeyError: 

167 print('Found an unknown keyword in AFM header (was %r)' % key, 

168 file=sys.stderr) 

169 continue 

170 if key == b'StartCharMetrics': 

171 return d 

172 raise RuntimeError('Bad parse')

190def _parse_char_metrics(fh): 

191 """ 

192 Parse the given filehandle for character metrics information and return 

193 the information as dicts. 

194 

195 It is assumed that the file cursor is on the line behind 

196 'StartCharMetrics'. 

197 

198 Returns 

199 ------- 

200 ascii_d : dict 

201 A mapping "ASCII num of the character" to `.CharMetrics`. 

202 name_d : dict 

203 A mapping "character name" to `.CharMetrics`. 

204 

205 Notes 

206 ----- 

207 This function is incomplete per the standard, but thus far parses 

208 all the sample afm files tried. 

209 """ 

210 required_keys = {'C', 'WX', 'N', 'B'} 

211 

212 ascii_d = {} 

213 name_d = {} 

214 for line in fh: 

215 # We are defensively letting values be utf8. The spec requires 

216 # ascii, but there are non-compliant fonts in circulation 

217 line = _to_str(line.rstrip()) # Convert from byte-literal 

218 if line.startswith('EndCharMetrics'): 

219 return ascii_d, name_d 

220 # Split the metric line into a dictionary, keyed by metric identifiers 

221 vals = dict(s.strip().split(' ', 1) for s in line.split(';') if s) 

222 # There may be other metrics present, but only these are needed 

223 if not required_keys.issubset(vals): 

224 raise RuntimeError('Bad char metrics line: %s' % line) 

225 num = _to_int(vals['C']) 

226 wx = _to_float(vals['WX']) 

227 name = vals['N'] 

228 bbox = _to_list_of_floats(vals['B']) 

229 bbox = list(map(int, bbox)) 

230 metrics = CharMetrics(wx, name, bbox) 

231 # Workaround: If the character name is 'Euro', give it the 

232 # corresponding character code, according to WinAnsiEncoding (see PDF 

233 # Reference). 

234 if name == 'Euro': 

235 num = 128 

236 if num != -1: 

237 ascii_d[num] = metrics 

238 name_d[name] = metrics 

239 raise RuntimeError('Bad parse')

242def _parse_kern_pairs(fh): 

243 """ 

244 Return a kern pairs dictionary; keys are (*char1*, *char2*) tuples and 

245 values are the kern pair value. For example, a kern pairs line like 

246 ``KPX A y -50`` 

247 

248 will be represented as:: 

249 

250 d[ ('A', 'y') ] = -50 

251 

252 """ 

253 

254 line = next(fh) 

255 if not line.startswith(b'StartKernPairs'): 

256 raise RuntimeError('Bad start of kern pairs data: %s' % line) 

257 

258 d = {} 

259 for line in fh: 

260 line = line.rstrip() 

261 if not line: 

262 continue 

263 if line.startswith(b'EndKernPairs'): 

264 next(fh) # EndKernData 

265 return d 

266 vals = line.split() 

267 if len(vals) != 4 or vals[0] != b'KPX': 

268 raise RuntimeError('Bad kern pairs line: %s' % line) 

269 c1, c2, val = _to_str(vals[1]), _to_str(vals[2]), _to_float(vals[3]) 

270 d[(c1, c2)] = val 

271 raise RuntimeError('Bad kern pairs parse')

282def _parse_composites(fh): 

283 """ 

284 Parse the given filehandle for composites information return them as a 

285 dict. 

286 

287 It is assumed that the file cursor is on the line behind 'StartComposites'. 

288 

289 Returns 

290 ------- 

291 composites : dict 

292 A dict mapping composite character names to a parts list. The parts 

293 list is a list of `.CompositePart` entries describing the parts of 

294 the composite. 

295 

296 Example 

297 ------- 

298 A composite definition line:: 

299 

300 CC Aacute 2 ; PCC A 0 0 ; PCC acute 160 170 ; 

301 

302 will be represented as:: 

303 

304 composites['Aacute'] = [CompositePart(name='A', dx=0, dy=0), 

305 CompositePart(name='acute', dx=160, dy=170)] 

306 

307 """ 

308 composites = {} 

309 for line in fh: 

310 line = line.rstrip() 

311 if not line: 

312 continue 

313 if line.startswith(b'EndComposites'): 

314 return composites 

315 vals = line.split(b';') 

316 cc = vals[0].split() 

317 name, numParts = cc[1], _to_int(cc[2]) 

318 pccParts = [] 

319 for s in vals[1:-1]: 

320 pcc = s.split() 

321 part = CompositePart(pcc[1], _to_float(pcc[2]), _to_float(pcc[3])) 

322 pccParts.append(part) 

323 composites[name] = pccParts 

324 

325 raise RuntimeError('Bad composites parse')

328def _parse_optional(fh): 

329 """ 

330 Parse the optional fields for kern pair data and composites. 

331 

332 Returns 

333 ------- 

334 kern_data : dict 

335 A dict containing kerning information. May be empty. 

336 See `._parse_kern_pairs`. 

337 composites : dict 

338 A dict containing composite information. May be empty. 

339 See `._parse_composites`. 

340 """ 

341 optional = { 

342 b'StartKernData': _parse_kern_pairs, 

343 b'StartComposites': _parse_composites, 

344 } 

345 

346 d = {b'StartKernData': {}, 

347 b'StartComposites': {}} 

348 for line in fh: 

349 line = line.rstrip() 

350 if not line: 

351 continue 

352 key = line.split()[0] 

353 

354 if key in optional: 

355 d[key] = optional[key](fh) 

356 

357 return d[b'StartKernData'], d[b'StartComposites']

365def _parse_afm(fh): 

366 """ 

367 Parse the Adobe Font Metrics file in file handle *fh*. 

368 

369 Returns 

370 ------- 

371 header : dict 

372 A header dict. See :func:`_parse_header`. 

373 cmetrics_by_ascii : dict 

374 From :func:`_parse_char_metrics`. 

375 cmetrics_by_name : dict 

376 From :func:`_parse_char_metrics`.

378 From :func:`_parse_kern_pairs`. 

379 composites : dict

405 def string_width_height(self, s): 

406 """ 

407 Return the string width (including kerning) and string height 

408 as a (*w*, *h*) tuple. 

409 """ 

410 if not len(s): 

411 return 0, 0 

412 total_width = 0 

413 namelast = None 

414 miny = 1e9 

415 maxy = 0 

416 for c in s: 

417 if c == '\n': 

418 continue 

419 wx, name, bbox = self._metrics[ord(c)] 

420 

421 total_width += wx + self._kern.get((namelast, name), 0) 

422 l, b, w, h = bbox 

423 miny = min(miny, b) 

424 maxy = max(maxy, b + h) 

425 

426 namelast = name 

427 

428 return total_width, maxy - miny

430 def get_str_bbox_and_descent(self, s): 

431 """Return the string bounding box and the maximal descent.""" 

432 if not len(s): 

433 return 0, 0, 0, 0, 0 

434 total_width = 0 

435 namelast = None 

436 miny = 1e9 

437 maxy = 0 

438 left = 0 

439 if not isinstance(s, str): 

440 s = _to_str(s) 

441 for c in s: 

442 if c == '\n': 

443 continue 

444 name = uni2type1.get(ord(c), 'question') 

445 try: 

446 wx, _, bbox = self._metrics_by_name[name] 

447 except KeyError: 

448 name = 'question' 

449 wx, _, bbox = self._metrics_by_name[name] 

450 total_width += wx + self._kern.get((namelast, name), 0) 

451 l, b, w, h = bbox 

452 left = min(left, l) 

453 miny = min(miny, b) 

454 maxy = max(maxy, b + h) 

455 

456 namelast = name 

457 

458 return left, miny, total_width, maxy - miny, -miny
