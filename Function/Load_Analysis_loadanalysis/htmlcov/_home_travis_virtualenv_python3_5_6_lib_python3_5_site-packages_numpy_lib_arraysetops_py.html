39def ediff1d(ary, to_end=None, to_begin=None): 

40 """ 

41 The differences between consecutive elements of an array. 

42 

43 Parameters 

44 ---------- 

45 ary : array_like 

46 If necessary, will be flattened before the differences are taken. 

47 to_end : array_like, optional 

48 Number(s) to append at the end of the returned differences. 

49 to_begin : array_like, optional 

50 Number(s) to prepend at the beginning of the returned differences. 

51 

52 Returns 

53 ------- 

54 ediff1d : ndarray 

55 The differences. Loosely, this is ``ary.flat[1:] - ary.flat[:-1]``. 

56 

57 See Also 

58 -------- 

59 diff, gradient 

60 

61 Notes 

62 ----- 

63 When applied to masked arrays, this function drops the mask information 

64 if the `to_begin` and/or `to_end` parameters are used. 

65 

66 Examples 

67 -------- 

68 >>> x = np.array([1, 2, 4, 7, 0]) 

69 >>> np.ediff1d(x) 

70 array([ 1, 2, 3, -7]) 

71 

72 >>> np.ediff1d(x, to_begin=-99, to_end=np.array([88, 99])) 

73 array([-99, 1, 2, 3, -7, 88, 99]) 

74 

75 The returned array is always 1D. 

76 

77 >>> y = [[1, 2, 4], [1, 6, 24]] 

78 >>> np.ediff1d(y) 

79 array([ 1, 2, -3, 5, 18]) 

80 

81 """ 

82 # force a 1d array 

83 ary = np.asanyarray(ary).ravel() 

84 

85 # fast track default case 

86 if to_begin is None and to_end is None: 

87 return ary[1:] - ary[:-1] 

88 

89 if to_begin is None: 

90 l_begin = 0 

91 else: 

92 to_begin = np.asanyarray(to_begin).ravel() 

93 l_begin = len(to_begin) 

94 

95 if to_end is None: 

96 l_end = 0 

97 else: 

98 to_end = np.asanyarray(to_end).ravel() 

99 l_end = len(to_end) 

100 

101 # do the calculation in place and copy to_begin and to_end 

102 l_diff = max(len(ary) - 1, 0) 

103 result = np.empty(l_diff + l_begin + l_end, dtype=ary.dtype) 

104 result = ary.__array_wrap__(result) 

105 if l_begin > 0: 

106 result[:l_begin] = to_begin 

107 if l_end > 0: 

108 result[l_begin + l_diff:] = to_end 

109 np.subtract(ary[1:], ary[:-1], result[l_begin:l_begin + l_diff]) 

110 return result

121def unique(ar, return_index=False, return_inverse=False, 

122 return_counts=False, axis=None): 

123 """ 

124 Find the unique elements of an array. 

125 

126 Returns the sorted unique elements of an array. There are three optional 

127 outputs in addition to the unique elements: 

128 

129 * the indices of the input array that give the unique values 

130 * the indices of the unique array that reconstruct the input array 

131 * the number of times each unique value comes up in the input array 

132 

133 Parameters 

134 ---------- 

135 ar : array_like 

136 Input array. Unless `axis` is specified, this will be flattened if it 

137 is not already 1-D. 

138 return_index : bool, optional 

139 If True, also return the indices of `ar` (along the specified axis, 

140 if provided, or in the flattened array) that result in the unique array. 

141 return_inverse : bool, optional 

142 If True, also return the indices of the unique array (for the specified 

143 axis, if provided) that can be used to reconstruct `ar`. 

144 return_counts : bool, optional 

145 If True, also return the number of times each unique item appears 

146 in `ar`. 

147 

148 .. versionadded:: 1.9.0 

149 

150 axis : int or None, optional 

151 The axis to operate on. If None, `ar` will be flattened. If an integer, 

152 the subarrays indexed by the given axis will be flattened and treated 

153 as the elements of a 1-D array with the dimension of the given axis, 

154 see the notes for more details. Object arrays or structured arrays 

155 that contain objects are not supported if the `axis` kwarg is used. The 

156 default is None. 

157 

158 .. versionadded:: 1.13.0 

159 

160 Returns 

161 ------- 

162 unique : ndarray 

163 The sorted unique values. 

164 unique_indices : ndarray, optional 

165 The indices of the first occurrences of the unique values in the 

166 original array. Only provided if `return_index` is True. 

167 unique_inverse : ndarray, optional 

168 The indices to reconstruct the original array from the 

169 unique array. Only provided if `return_inverse` is True. 

170 unique_counts : ndarray, optional 

171 The number of times each of the unique values comes up in the 

172 original array. Only provided if `return_counts` is True. 

173 

174 .. versionadded:: 1.9.0 

175 

176 See Also 

177 -------- 

178 numpy.lib.arraysetops : Module with a number of other functions for 

179 performing set operations on arrays. 

180 

181 Notes 

182 ----- 

183 When an axis is specified the subarrays indexed by the axis are sorted. 

184 This is done by making the specified axis the first dimension of the array 

185 and then flattening the subarrays in C order. The flattened subarrays are 

186 then viewed as a structured type with each element given a label, with the 

187 effect that we end up with a 1-D array of structured types that can be 

188 treated in the same way as any other 1-D array. The result is that the 

189 flattened subarrays are sorted in lexicographic order starting with the 

190 first element. 

191 

192 Examples 

193 -------- 

194 >>> np.unique([1, 1, 2, 2, 3, 3]) 

195 array([1, 2, 3]) 

196 >>> a = np.array([[1, 1], [2, 3]]) 

197 >>> np.unique(a) 

198 array([1, 2, 3]) 

199 

200 Return the unique rows of a 2D array 

201 

202 >>> a = np.array([[1, 0, 0], [1, 0, 0], [2, 3, 4]]) 

203 >>> np.unique(a, axis=0) 

204 array([[1, 0, 0], [2, 3, 4]]) 

205 

206 Return the indices of the original array that give the unique values: 

207 

208 >>> a = np.array(['a', 'b', 'b', 'c', 'a']) 

209 >>> u, indices = np.unique(a, return_index=True) 

210 >>> u 

211 array(['a', 'b', 'c'], 

212 dtype='|S1') 

213 >>> indices 

214 array([0, 1, 3]) 

215 >>> a[indices] 

216 array(['a', 'b', 'c'], 

217 dtype='|S1') 

218 

219 Reconstruct the input array from the unique values: 

220 

221 >>> a = np.array([1, 2, 6, 4, 2, 3, 2]) 

222 >>> u, indices = np.unique(a, return_inverse=True) 

223 >>> u 

224 array([1, 2, 3, 4, 6]) 

225 >>> indices 

226 array([0, 1, 4, 3, 1, 2, 1]) 

227 >>> u[indices] 

228 array([1, 2, 6, 4, 2, 3, 2]) 

229 

230 """ 

231 ar = np.asanyarray(ar) 

232 if axis is None: 

233 ret = _unique1d(ar, return_index, return_inverse, return_counts) 

234 return _unpack_tuple(ret) 

235 

236 # axis was specified and not None 

237 try: 

238 ar = np.swapaxes(ar, axis, 0) 

239 except np.AxisError: 

240 # this removes the "axis1" or "axis2" prefix from the error message 

241 raise np.AxisError(axis, ar.ndim) 

242 

243 # Must reshape to a contiguous 2D array for this to work... 

244 orig_shape, orig_dtype = ar.shape, ar.dtype 

245 ar = ar.reshape(orig_shape[0], -1) 

246 ar = np.ascontiguousarray(ar) 

247 dtype = [('f{i}'.format(i=i), ar.dtype) for i in range(ar.shape[1])] 

248 

249 try: 

250 consolidated = ar.view(dtype) 

251 except TypeError: 

252 # There's no good way to do this for object arrays, etc... 

253 msg = 'The axis argument to unique is not supported for dtype {dt}' 

254 raise TypeError(msg.format(dt=ar.dtype)) 

255 

256 def reshape_uniq(uniq): 

257 uniq = uniq.view(orig_dtype) 

258 uniq = uniq.reshape(-1, *orig_shape[1:]) 

259 uniq = np.swapaxes(uniq, 0, axis) 

260 return uniq 

261 

262 output = _unique1d(consolidated, return_index, 

263 return_inverse, return_counts) 

264 output = (reshape_uniq(output[0]),) + output[1:] 

265 return _unpack_tuple(output)

268def _unique1d(ar, return_index=False, return_inverse=False, 

269 return_counts=False): 

270 """ 

271 Find the unique elements of an array, ignoring shape. 

272 """ 

273 ar = np.asanyarray(ar).flatten() 

274 

275 optional_indices = return_index or return_inverse 

276 

277 if optional_indices: 

278 perm = ar.argsort(kind='mergesort' if return_index else 'quicksort') 

279 aux = ar[perm] 

280 else: 

281 ar.sort() 

282 aux = ar 

283 mask = np.empty(aux.shape, dtype=np.bool_) 

284 mask[:1] = True 

285 mask[1:] = aux[1:] != aux[:-1] 

286 

287 ret = (aux[mask],) 

288 if return_index: 

289 ret += (perm[mask],) 

290 if return_inverse: 

291 imask = np.cumsum(mask) - 1 

292 inv_idx = np.empty(mask.shape, dtype=np.intp) 

293 inv_idx[perm] = imask 

294 ret += (inv_idx,) 

295 if return_counts: 

296 idx = np.concatenate(np.nonzero(mask) + ([mask.size],)) 

297 ret += (np.diff(idx),) 

298 return ret

301def intersect1d(ar1, ar2, assume_unique=False, return_indices=False): 

302 """ 

303 Find the intersection of two arrays. 

304 

305 Return the sorted, unique values that are in both of the input arrays. 

306 

307 Parameters 

308 ---------- 

309 ar1, ar2 : array_like 

310 Input arrays. Will be flattened if not already 1D. 

311 assume_unique : bool 

312 If True, the input arrays are both assumed to be unique, which 

313 can speed up the calculation. Default is False. 

314 return_indices : bool 

315 If True, the indices which correspond to the intersection of the two 

316 arrays are returned. The first instance of a value is used if there are 

317 multiple. Default is False. 

318 

319 .. versionadded:: 1.15.0 

320 

321 Returns 

322 ------- 

323 intersect1d : ndarray 

324 Sorted 1D array of common and unique elements. 

325 comm1 : ndarray 

326 The indices of the first occurrences of the common values in `ar1`. 

327 Only provided if `return_indices` is True. 

328 comm2 : ndarray 

329 The indices of the first occurrences of the common values in `ar2`. 

330 Only provided if `return_indices` is True. 

331 

332 

333 See Also 

334 -------- 

335 numpy.lib.arraysetops : Module with a number of other functions for 

336 performing set operations on arrays. 

337 

338 Examples 

339 -------- 

340 >>> np.intersect1d([1, 3, 4, 3], [3, 1, 2, 1]) 

341 array([1, 3]) 

342 

343 To intersect more than two arrays, use functools.reduce: 

344 

345 >>> from functools import reduce 

346 >>> reduce(np.intersect1d, ([1, 3, 4, 3], [3, 1, 2, 1], [6, 3, 4, 2])) 

347 array([3]) 

348 

349 To return the indices of the values common to the input arrays 

350 along with the intersected values: 

351 >>> x = np.array([1, 1, 2, 3, 4]) 

352 >>> y = np.array([2, 1, 4, 6]) 

353 >>> xy, x_ind, y_ind = np.intersect1d(x, y, return_indices=True) 

354 >>> x_ind, y_ind 

355 (array([0, 2, 4]), array([1, 0, 2])) 

356 >>> xy, x[x_ind], y[y_ind] 

357 (array([1, 2, 4]), array([1, 2, 4]), array([1, 2, 4])) 

358 

359 """ 

360 ar1 = np.asanyarray(ar1) 

361 ar2 = np.asanyarray(ar2) 

362 

363 if not assume_unique: 

364 if return_indices: 

365 ar1, ind1 = unique(ar1, return_index=True) 

366 ar2, ind2 = unique(ar2, return_index=True) 

367 else: 

368 ar1 = unique(ar1) 

369 ar2 = unique(ar2) 

370 else: 

371 ar1 = ar1.ravel() 

372 ar2 = ar2.ravel() 

373 

374 aux = np.concatenate((ar1, ar2)) 

375 if return_indices: 

376 aux_sort_indices = np.argsort(aux, kind='mergesort') 

377 aux = aux[aux_sort_indices] 

378 else: 

379 aux.sort() 

380 

381 mask = aux[1:] == aux[:-1] 

382 int1d = aux[:-1][mask] 

383 

384 if return_indices: 

385 ar1_indices = aux_sort_indices[:-1][mask] 

386 ar2_indices = aux_sort_indices[1:][mask] - ar1.size 

387 if not assume_unique: 

388 ar1_indices = ind1[ar1_indices] 

389 ar2_indices = ind2[ar2_indices] 

390 

391 return int1d, ar1_indices, ar2_indices 

392 else: 

393 return int1d

397 """ 

398 Find the set exclusive-or of two arrays. 

399 

400 Return the sorted, unique values that are in only one (not both) of the 

401 input arrays. 

402 

403 Parameters 

404 ---------- 

405 ar1, ar2 : array_like 

406 Input arrays. 

407 assume_unique : bool 

408 If True, the input arrays are both assumed to be unique, which 

409 can speed up the calculation. Default is False. 

410 

411 Returns 

412 ------- 

413 setxor1d : ndarray 

414 Sorted 1D array of unique values that are in only one of the input 

415 arrays. 

416 

417 Examples 

418 -------- 

419 >>> a = np.array([1, 2, 3, 2, 4])

421 >>> np.setxor1d(a,b)

423 

424 """ 

425 if not assume_unique:

428

431 return aux

434 flag = np.concatenate(([True], aux[1:] != aux[:-1], [True])) 

435 return aux[flag[1:] & flag[:-1]]

438def in1d(ar1, ar2, assume_unique=False, invert=False): 

439 """ 

440 Test whether each element of a 1-D array is also present in a second array. 

441 

442 Returns a boolean array the same length as `ar1` that is True 

443 where an element of `ar1` is in `ar2` and False otherwise. 

444 

445 We recommend using :func:`isin` instead of `in1d` for new code. 

446 

447 Parameters 

448 ---------- 

449 ar1 : (M,) array_like 

450 Input array. 

451 ar2 : array_like 

452 The values against which to test each value of `ar1`. 

453 assume_unique : bool, optional 

454 If True, the input arrays are both assumed to be unique, which 

455 can speed up the calculation. Default is False. 

456 invert : bool, optional 

457 If True, the values in the returned array are inverted (that is, 

458 False where an element of `ar1` is in `ar2` and True otherwise). 

459 Default is False. ``np.in1d(a, b, invert=True)`` is equivalent 

460 to (but is faster than) ``np.invert(in1d(a, b))``. 

461 

462 .. versionadded:: 1.8.0 

463 

464 Returns 

465 ------- 

466 in1d : (M,) ndarray, bool 

467 The values `ar1[in1d]` are in `ar2`. 

468 

469 See Also 

470 -------- 

471 isin : Version of this function that preserves the 

472 shape of ar1. 

473 numpy.lib.arraysetops : Module with a number of other functions for 

474 performing set operations on arrays. 

475 

476 Notes 

477 ----- 

478 `in1d` can be considered as an element-wise function version of the 

479 python keyword `in`, for 1-D sequences. ``in1d(a, b)`` is roughly 

480 equivalent to ``np.array([item in b for item in a])``. 

481 However, this idea fails if `ar2` is a set, or similar (non-sequence) 

482 container: As ``ar2`` is converted to an array, in those cases 

483 ``asarray(ar2)`` is an object array rather than the expected array of 

484 contained values. 

485 

486 .. versionadded:: 1.4.0 

487 

488 Examples 

489 -------- 

490 >>> test = np.array([0, 1, 2, 5, 0]) 

491 >>> states = [0, 2] 

492 >>> mask = np.in1d(test, states) 

493 >>> mask 

494 array([ True, False, True, False, True]) 

495 >>> test[mask] 

496 array([0, 2, 0]) 

497 >>> mask = np.in1d(test, states, invert=True) 

498 >>> mask 

499 array([False, True, False, True, False]) 

500 >>> test[mask] 

501 array([1, 5]) 

502 """ 

503 # Ravel both arrays, behavior for the first array could be different 

504 ar1 = np.asarray(ar1).ravel() 

505 ar2 = np.asarray(ar2).ravel() 

506 

507 # Check if one of the arrays may contain arbitrary objects 

508 contains_object = ar1.dtype.hasobject or ar2.dtype.hasobject 

509 

510 # This code is run when 

511 # a) the first condition is true, making the code significantly faster 

512 # b) the second condition is true (i.e. `ar1` or `ar2` may contain 

513 # arbitrary objects), since then sorting is not guaranteed to work 

514 if len(ar2) < 10 * len(ar1) ** 0.145 or contains_object: 

515 if invert: 

516 mask = np.ones(len(ar1), dtype=bool) 

517 for a in ar2: 

518 mask &= (ar1 != a) 

519 else: 

520 mask = np.zeros(len(ar1), dtype=bool) 

521 for a in ar2: 

522 mask |= (ar1 == a) 

523 return mask 

524 

525 # Otherwise use sorting 

526 if not assume_unique: 

527 ar1, rev_idx = np.unique(ar1, return_inverse=True) 

528 ar2 = np.unique(ar2) 

529 

530 ar = np.concatenate((ar1, ar2)) 

531 # We need this to be a stable sort, so always use 'mergesort' 

532 # here. The values from the first array should always come before 

533 # the values from the second array. 

534 order = ar.argsort(kind='mergesort') 

535 sar = ar[order] 

536 if invert: 

537 bool_ar = (sar[1:] != sar[:-1]) 

538 else: 

539 bool_ar = (sar[1:] == sar[:-1]) 

540 flag = np.concatenate((bool_ar, [invert])) 

541 ret = np.empty(ar.shape, dtype=bool) 

542 ret[order] = flag 

543 

544 if assume_unique: 

545 return ret[:len(ar1)] 

546 else: 

547 return ret[rev_idx]

550def isin(element, test_elements, assume_unique=False, invert=False): 

551 """ 

552 Calculates `element in test_elements`, broadcasting over `element` only. 

553 Returns a boolean array of the same shape as `element` that is True 

554 where an element of `element` is in `test_elements` and False otherwise. 

555 

556 Parameters 

557 ---------- 

558 element : array_like 

559 Input array. 

560 test_elements : array_like 

561 The values against which to test each value of `element`. 

562 This argument is flattened if it is an array or array_like. 

563 See notes for behavior with non-array-like parameters. 

564 assume_unique : bool, optional 

565 If True, the input arrays are both assumed to be unique, which 

566 can speed up the calculation. Default is False. 

567 invert : bool, optional 

568 If True, the values in the returned array are inverted, as if 

569 calculating `element not in test_elements`. Default is False. 

570 ``np.isin(a, b, invert=True)`` is equivalent to (but faster 

571 than) ``np.invert(np.isin(a, b))``. 

572 

573 Returns 

574 ------- 

575 isin : ndarray, bool 

576 Has the same shape as `element`. The values `element[isin]` 

577 are in `test_elements`. 

578 

579 See Also 

580 -------- 

581 in1d : Flattened version of this function. 

582 numpy.lib.arraysetops : Module with a number of other functions for 

583 performing set operations on arrays. 

584 

585 Notes 

586 ----- 

587 

588 `isin` is an element-wise function version of the python keyword `in`. 

589 ``isin(a, b)`` is roughly equivalent to 

590 ``np.array([item in b for item in a])`` if `a` and `b` are 1-D sequences. 

591 

592 `element` and `test_elements` are converted to arrays if they are not 

593 already. If `test_elements` is a set (or other non-sequence collection) 

594 it will be converted to an object array with one element, rather than an 

595 array of the values contained in `test_elements`. This is a consequence 

596 of the `array` constructor's way of handling non-sequence collections. 

597 Converting the set to a list usually gives the desired behavior. 

598 

599 .. versionadded:: 1.13.0 

600 

601 Examples 

602 -------- 

603 >>> element = 2*np.arange(4).reshape((2, 2)) 

604 >>> element 

605 array([[0, 2], 

606 [4, 6]]) 

607 >>> test_elements = [1, 2, 4, 8] 

608 >>> mask = np.isin(element, test_elements) 

609 >>> mask 

610 array([[ False, True], 

611 [ True, False]]) 

612 >>> element[mask] 

613 array([2, 4]) 

614 >>> mask = np.isin(element, test_elements, invert=True) 

615 >>> mask 

616 array([[ True, False], 

617 [ False, True]]) 

618 >>> element[mask] 

619 array([0, 6]) 

620 

621 Because of how `array` handles sets, the following does not 

622 work as expected: 

623 

624 >>> test_set = {1, 2, 4, 8} 

625 >>> np.isin(element, test_set) 

626 array([[ False, False], 

627 [ False, False]]) 

628 

629 Casting the set to a list gives the expected result: 

630 

631 >>> np.isin(element, list(test_set)) 

632 array([[ False, True], 

633 [ True, False]]) 

634 """ 

635 element = np.asarray(element) 

636 return in1d(element, test_elements, assume_unique=assume_unique, 

637 invert=invert).reshape(element.shape)

675def setdiff1d(ar1, ar2, assume_unique=False): 

676 """
