26def ix_(*args): 

27 """ 

28 Construct an open mesh from multiple sequences. 

29 

30 This function takes N 1-D sequences and returns N outputs with N 

31 dimensions each, such that the shape is 1 in all but one dimension 

32 and the dimension with the non-unit shape value cycles through all 

33 N dimensions. 

34 

35 Using `ix_` one can quickly construct index arrays that will index 

36 the cross product. ``a[np.ix_([1,3],[2,5])]`` returns the array 

37 ``[[a[1,2] a[1,5]], [a[3,2] a[3,5]]]``. 

38 

39 Parameters 

40 ---------- 

41 args : 1-D sequences 

42 Each sequence should be of integer or boolean type. 

43 Boolean sequences will be interpreted as boolean masks for the 

44 corresponding dimension (equivalent to passing in 

45 ``np.nonzero(boolean_sequence)``). 

46 

47 Returns 

48 ------- 

49 out : tuple of ndarrays 

50 N arrays with N dimensions each, with N the number of input 

51 sequences. Together these arrays form an open mesh. 

52 

53 See Also 

54 -------- 

55 ogrid, mgrid, meshgrid 

56 

57 Examples 

58 -------- 

59 >>> a = np.arange(10).reshape(2, 5) 

60 >>> a 

61 array([[0, 1, 2, 3, 4], 

62 [5, 6, 7, 8, 9]]) 

63 >>> ixgrid = np.ix_([0, 1], [2, 4]) 

64 >>> ixgrid 

65 (array([[0], 

66 [1]]), array([[2, 4]])) 

67 >>> ixgrid[0].shape, ixgrid[1].shape 

68 ((2, 1), (1, 2)) 

69 >>> a[ixgrid] 

70 array([[2, 4], 

71 [7, 9]]) 

72 

73 >>> ixgrid = np.ix_([True, True], [2, 4]) 

74 >>> a[ixgrid] 

75 array([[2, 4], 

76 [7, 9]]) 

77 >>> ixgrid = np.ix_([True, True], [False, False, True, False, True]) 

78 >>> a[ixgrid] 

79 array([[2, 4], 

80 [7, 9]]) 

81 

82 """ 

83 out = [] 

84 nd = len(args) 

85 for k, new in enumerate(args): 

86 new = asarray(new) 

87 if new.ndim != 1: 

88 raise ValueError("Cross index must be 1 dimensional") 

89 if new.size == 0: 

90 # Explicitly type empty arrays to avoid float default 

91 new = new.astype(_nx.intp) 

92 if issubdtype(new.dtype, _nx.bool_): 

93 new, = new.nonzero() 

94 new = new.reshape((1,)*k + (new.size,) + (1,)*(nd-k-1)) 

95 out.append(new) 

96 return tuple(out)

258 trans1d = self.trans1d

260 matrix = self.matrix

263 objs = []

270 if isinstance(item, slice): 

271 step = item.step 

272 start = item.start 

273 stop = item.stop

276 if step is None: 

277 step = 1 

278 if isinstance(step, complex): 

279 size = int(abs(step)) 

280 newobj = function_base.linspace(start, stop, num=size) 

281 else: 

282 newobj = _nx.arange(start, stop, step) 

283 if ndmin > 1: 

284 newobj = array(newobj, copy=False, ndmin=ndmin) 

285 if trans1d != -1: 

286 newobj = newobj.swapaxes(-1, trans1d) 

287 elif isinstance(item, str): 

288 if k != 0: 

289 raise ValueError("special directives must be the " 

290 "first entry.") 

291 if item in ('r', 'c'): 

292 matrix = True 

293 col = (item == 'c') 

294 continue 

295 if ',' in item: 

296 vec = item.split(',') 

297 try: 

298 axis, ndmin = [int(x) for x in vec[:2]] 

299 if len(vec) == 3:

301 continue 

302 except Exception: 

303 raise ValueError("unknown special directive")

305 axis = int(item) 

306 continue 

307 except (ValueError, TypeError): 

308 raise ValueError("unknown special directive") 

309 elif type(item) in ScalarType: 

310 newobj = array(item, ndmin=ndmin) 

311 scalars.append(len(objs)) 

312 scalar = True 

313 scalartypes.append(newobj.dtype) 

314 else: 

315 item_ndim = ndim(item) 

316 newobj = array(item, copy=False, subok=True, ndmin=ndmin) 

317 if trans1d != -1 and item_ndim < ndmin: 

318 k2 = ndmin - item_ndim 

319 k1 = trans1d 

320 if k1 < 0: 

321 k1 += k2 + 1 

322 defaxes = list(range(ndmin)) 

323 axes = defaxes[:k1] + defaxes[k2:] + defaxes[k1:k2] 

324 newobj = newobj.transpose(axes) 

325 objs.append(newobj) 

326 if not scalar and isinstance(newobj, _nx.ndarray): 

327 arraytypes.append(newobj.dtype)

329 # Ensure that scalars won't up-cast unless warranted 

330 final_dtype = find_common_type(arraytypes, scalartypes) 

331 if final_dtype is not None: 

332 for k in scalars: 

333 objs[k] = objs[k].astype(final_dtype) 

334 

335 res = self.concatenate(tuple(objs), axis=axis) 

336 

337 if matrix:

339 res = self.makemat(res)

351class RClass(AxisConcatenator): 

352 """ 

353 Translates slice objects to concatenation along the first axis. 

354 

355 This is a simple way to build up arrays quickly. There are two use cases. 

356 

357 1. If the index expression contains comma separated arrays, then stack 

358 them along their first axis. 

359 2. If the index expression contains slice notation or scalars then create 

360 a 1-D array with a range indicated by the slice notation. 

361 

362 If slice notation is used, the syntax ``start:stop:step`` is equivalent 

363 to ``np.arange(start, stop, step)`` inside of the brackets. However, if 

364 ``step`` is an imaginary number (i.e. 100j) then its integer portion is 

365 interpreted as a number-of-points desired and the start and stop are 

366 inclusive. In other words ``start:stop:stepj`` is interpreted as 

367 ``np.linspace(start, stop, step, endpoint=1)`` inside of the brackets. 

368 After expansion of slice notation, all comma separated sequences are 

369 concatenated together. 

370 

371 Optional character strings placed as the first element of the index 

372 expression can be used to change the output. The strings 'r' or 'c' result 

373 in matrix output. If the result is 1-D and 'r' is specified a 1 x N (row) 

374 matrix is produced. If the result is 1-D and 'c' is specified, then a N x 1 

375 (column) matrix is produced. If the result is 2-D then both provide the 

376 same matrix result. 

377 

378 A string integer specifies which axis to stack multiple comma separated 

379 arrays along. A string of two comma-separated integers allows indication 

380 of the minimum number of dimensions to force each entry into as the 

381 second integer (the axis to concatenate along is still the first integer). 

382 

383 A string with three comma-separated integers allows specification of the 

384 axis to concatenate along, the minimum number of dimensions to force the 

385 entries to, and which axis should contain the start of the arrays which 

386 are less than the specified number of dimensions. In other words the third 

387 integer allows you to specify where the 1's should be placed in the shape 

388 of the arrays that have their shapes upgraded. By default, they are placed 

389 in the front of the shape tuple. The third argument allows you to specify 

390 where the start of the array should be instead. Thus, a third argument of 

391 '0' would place the 1's at the end of the array shape. Negative integers 

392 specify where in the new shape tuple the last dimension of upgraded arrays 

393 should be placed, so the default is '-1'. 

394 

395 Parameters 

396 ---------- 

397 Not a function, so takes no parameters 

398 

399 

400 Returns 

401 ------- 

402 A concatenated ndarray or matrix. 

403 

404 See Also 

405 -------- 

406 concatenate : Join a sequence of arrays along an existing axis. 

407 c_ : Translates slice objects to concatenation along the second axis. 

408 

409 Examples 

410 -------- 

411 >>> np.r_[np.array([1,2,3]), 0, 0, np.array([4,5,6])] 

412 array([1, 2, 3, 0, 0, 4, 5, 6]) 

413 >>> np.r_[-1:1:6j, [0]*3, 5, 6] 

414 array([-1. , -0.6, -0.2, 0.2, 0.6, 1. , 0. , 0. , 0. , 5. , 6. ]) 

415 

416 String integers specify the axis to concatenate along or the minimum 

417 number of dimensions to force entries into. 

418 

419 >>> a = np.array([[0, 1, 2], [3, 4, 5]]) 

420 >>> np.r_['-1', a, a] # concatenate along last axis 

421 array([[0, 1, 2, 0, 1, 2], 

422 [3, 4, 5, 3, 4, 5]]) 

423 >>> np.r_['0,2', [1,2,3], [4,5,6]] # concatenate along first axis, dim>=2 

424 array([[1, 2, 3], 

425 [4, 5, 6]]) 

426 

427 >>> np.r_['0,2,0', [1,2,3], [4,5,6]] 

428 array([[1], 

429 [2], 

430 [3], 

431 [4], 

432 [5], 

433 [6]]) 

434 >>> np.r_['1,2,0', [1,2,3], [4,5,6]] 

435 array([[1, 4], 

436 [2, 5], 

437 [3, 6]]) 

438 

439 Using 'r' or 'c' as a first string argument creates a matrix. 

440 

441 >>> np.r_['r',[1,2,3], [4,5,6]] 

442 matrix([[1, 2, 3, 4, 5, 6]]) 

443 

444 """

685 that of the array a.

693 -------- 

694 diag_indices, diag_indices_from 

695 

696 Notes 

697 ----- 

698 .. versionadded:: 1.4.0

700 This functionality can be obtained via `diag_indices`, but internally 

701 this version uses a much faster implementation that never constructs the 

702 indices and uses simple slicing. 

703 

704 Examples 

705 -------- 

706 >>> a = np.zeros((3, 3), int) 

707 >>> np.fill_diagonal(a, 5) 

708 >>> a 

709 array([[5, 0, 0], 

710 [0, 5, 0], 

711 [0, 0, 5]]) 

712 

713 The same function can operate on a 4-D array: 

714 

715 >>> a = np.zeros((3, 3, 3, 3), int) 

716 >>> np.fill_diagonal(a, 4) 

717 

718 We only show a few blocks for clarity: 

719 

720 >>> a[0, 0] 

721 array([[4, 0, 0], 

722 [0, 0, 0], 

723 [0, 0, 0]]) 

724 >>> a[1, 1] 

725 array([[0, 0, 0], 

726 [0, 4, 0], 

727 [0, 0, 0]]) 

728 >>> a[2, 2] 

729 array([[0, 0, 0], 

730 [0, 0, 0], 

731 [0, 0, 4]]) 

732 

733 The wrap option affects only tall matrices: 

734 

735 >>> # tall matrices no wrap 

736 >>> a = np.zeros((5, 3),int) 

737 >>> fill_diagonal(a, 4) 

738 >>> a 

739 array([[4, 0, 0], 

740 [0, 4, 0], 

741 [0, 0, 4], 

742 [0, 0, 0], 

743 [0, 0, 0]]) 

744 

745 >>> # tall matrices wrap 

746 >>> a = np.zeros((5, 3),int) 

747 >>> fill_diagonal(a, 4, wrap=True) 

748 >>> a 

749 array([[4, 0, 0], 

750 [0, 4, 0], 

751 [0, 0, 4], 

752 [0, 0, 0], 

753 [4, 0, 0]]) 

754 

755 >>> # wide matrices 

756 >>> a = np.zeros((3, 5),int) 

757 >>> fill_diagonal(a, 4, wrap=True) 

758 >>> a 

759 array([[4, 0, 0, 0, 0], 

760 [0, 4, 0, 0, 0], 

761 [0, 0, 4, 0, 0]]) 

762 

763 """ 

764 if a.ndim < 2: 

765 raise ValueError("array must be at least 2-d") 

766 end = None 

767 if a.ndim == 2: 

768 # Explicit, fast formula for the common case. For 2-d arrays, we 

769 # accept rectangular ones. 

770 step = a.shape[1] + 1 

771 #This is needed to don't have tall matrix have the diagonal wrap. 

772 if not wrap: 

773 end = a.shape[1] * a.shape[1] 

774 else: 

775 # For more than d=2, the strided formula is only valid for arrays with 

776 # all dimensions equal, so we check first. 

777 if not alltrue(diff(a.shape) == 0): 

778 raise ValueError("All dimensions of input must be of equal length") 

779 step = 1 + (cumprod(a.shape[:-1])).sum() 

780 

781 # Write the value out into the diagonal. 

782 a.flat[:end:step] = val

785def diag_indices(n, ndim=2): 

786 """ 

787 Return the indices to access the main diagonal of an array. 

788 

789 This returns a tuple of indices that can be used to access the main 

790 diagonal of an array `a` with ``a.ndim >= 2`` dimensions and shape 

791 (n, n, ..., n). For ``a.ndim = 2`` this is the usual diagonal, for 

792 ``a.ndim > 2`` this is the set of indices to access ``a[i, i, ..., i]`` 

793 for ``i = [0..n-1]``. 

794 

795 Parameters 

796 ---------- 

797 n : int 

798 The size, along each dimension, of the arrays for which the returned 

799 indices can be used. 

800 

801 ndim : int, optional 

802 The number of dimensions. 

803 

804 See also 

805 -------- 

806 diag_indices_from 

807 

808 Notes 

809 ----- 

810 .. versionadded:: 1.4.0 

811 

812 Examples 

813 -------- 

814 Create a set of indices to access the diagonal of a (4, 4) array: 

815 

816 >>> di = np.diag_indices(4) 

817 >>> di 

818 (array([0, 1, 2, 3]), array([0, 1, 2, 3])) 

819 >>> a = np.arange(16).reshape(4, 4) 

820 >>> a 

821 array([[ 0, 1, 2, 3], 

822 [ 4, 5, 6, 7], 

823 [ 8, 9, 10, 11], 

824 [12, 13, 14, 15]]) 

825 >>> a[di] = 100 

826 >>> a 

827 array([[100, 1, 2, 3], 

828 [ 4, 100, 6, 7], 

829 [ 8, 9, 100, 11], 

830 [ 12, 13, 14, 100]]) 

831 

832 Now, we create indices to manipulate a 3-D array: 

833 

834 >>> d3 = np.diag_indices(2, 3) 

835 >>> d3 

836 (array([0, 1]), array([0, 1]), array([0, 1])) 

837 

838 And use it to set the diagonal of an array of zeros to 1: 

839 

840 >>> a = np.zeros((2, 2, 2), dtype=int) 

841 >>> a[d3] = 1 

842 >>> a 

843 array([[[1, 0], 

844 [0, 0]], 

845 [[0, 0], 

846 [0, 1]]]) 

847 

848 """ 

849 idx = arange(n) 

850 return (idx,) * ndim
