1""" 

2A module for converting numbers or color arguments to *RGB* or *RGBA* 

3 

4*RGB* and *RGBA* are sequences of, respectively, 3 or 4 floats in the 

5range 0-1. 

6 

7This module includes functions and classes for color specification 

8conversions, and for mapping numbers to colors in a 1-D array of colors called 

9a colormap. 

10 

11Mapping data onto colors using a colormap typically involves two steps: 

12a data array is first mapped onto the range 0-1 using a subclass of 

13:class:`Normalize`, then this number is mapped to a color using 

14a subclass of :class:`Colormap`. Two are provided here: 

15:class:`LinearSegmentedColormap`, which uses piecewise-linear interpolation 

16to define colormaps, and :class:`ListedColormap`, which makes a colormap 

17from a list of colors. 

18 

19.. seealso:: 

20 

21 :doc:`/tutorials/colors/colormap-manipulation` for examples of how to 

22 make colormaps and 

23 

24 :doc:`/tutorials/colors/colormaps` for a list of built-in colormaps. 

25 

26 :doc:`/tutorials/colors/colormapnorms` for more details about data 

27 normalization 

28 

29 More colormaps are available at palettable_ 

30 

31The module also provides functions for checking whether an object can be 

32interpreted as a color (:func:`is_color_like`), for converting such an object 

33to an RGBA tuple (:func:`to_rgba`) or to an HTML-like hex string in the 

34`#rrggbb` format (:func:`to_hex`), and a sequence of colors to an `(n, 4)` 

35RGBA array (:func:`to_rgba_array`). Caching is used for efficiency. 

36 

37Matplotlib recognizes the following formats to specify a color: 

38 

39* an RGB or RGBA tuple of float values in ``[0, 1]`` (e.g., ``(0.1, 0.2, 0.5)`` 

40 or ``(0.1, 0.2, 0.5, 0.3)``); 

41* a hex RGB or RGBA string (e.g., ``'#0F0F0F'`` or ``'#0F0F0F0F'``); 

42* a string representation of a float value in ``[0, 1]`` inclusive for gray 

43 level (e.g., ``'0.5'``); 

44* one of ``{'b', 'g', 'r', 'c', 'm', 'y', 'k', 'w'}``; 

45* a X11/CSS4 color name; 

46* a name from the `xkcd color survey <https://xkcd.com/color/rgb/>`__; 

47 prefixed with ``'xkcd:'`` (e.g., ``'xkcd:sky blue'``); 

48* one of ``{'tab:blue', 'tab:orange', 'tab:green', 

49 'tab:red', 'tab:purple', 'tab:brown', 'tab:pink', 

50 'tab:gray', 'tab:olive', 'tab:cyan'}`` which are the Tableau Colors from the 

51 'T10' categorical palette (which is the default color cycle); 

52* a "CN" color spec, i.e. `'C'` followed by a single digit, which is an index 

53 into the default property cycle (``matplotlib.rcParams['axes.prop_cycle']``); 

54 the indexing occurs at artist creation time and defaults to black if the 

55 cycle does not include color. 

56 

57All string specifications of color, other than "CN", are case-insensitive. 

58

150def to_rgba(c, alpha=None): 

151 """ 

152 Convert *c* to an RGBA color. 

153 

154 Parameters 

155 ---------- 

156 c : Matplotlib color 

157 

158 alpha : scalar, optional 

159 If *alpha* is not ``None``, it forces the alpha value, except if *c* is 

160 ``"none"`` (case-insensitive), which always maps to ``(0, 0, 0, 0)``. 

161 

162 Returns 

163 ------- 

164 tuple 

165 Tuple of ``(r, g, b, a)`` scalars. 

166 """ 

167 # Special-case nth color syntax because it should not be cached. 

168 if _is_nth_color(c): 

169 from matplotlib import rcParams 

170 prop_cycler = rcParams['axes.prop_cycle'] 

171 colors = prop_cycler.by_key().get('color', ['k']) 

172 c = colors[int(c[1]) % len(colors)] 

173 try: 

174 rgba = _colors_full_map.cache[c, alpha] 

175 except (KeyError, TypeError): # Not in cache, or unhashable. 

176 rgba = _to_rgba_no_colorcycle(c, alpha) 

177 try: 

178 _colors_full_map.cache[c, alpha] = rgba 

179 except TypeError: 

180 pass 

181 return rgba

184def _to_rgba_no_colorcycle(c, alpha=None): 

185 """Convert *c* to an RGBA color, with no support for color-cycle syntax. 

186 

187 If *alpha* is not ``None``, it forces the alpha value, except if *c* is 

188 ``"none"`` (case-insensitive), which always maps to ``(0, 0, 0, 0)``. 

189 """ 

190 orig_c = c 

191 if isinstance(c, str): 

192 if c.lower() == "none": 

193 return (0., 0., 0., 0.) 

194 # Named color. 

195 try: 

196 # This may turn c into a non-string, so we check again below. 

197 c = _colors_full_map[c.lower()] 

198 except KeyError: 

199 pass 

200 if isinstance(c, str): 

201 # hex color with no alpha. 

202 match = re.match(r"\A#[a-fA-F0-9]{6}\Z", c) 

203 if match: 

204 return (tuple(int(n, 16) / 255 

205 for n in [c[1:3], c[3:5], c[5:7]]) 

206 + (alpha if alpha is not None else 1.,)) 

207 # hex color with alpha. 

208 match = re.match(r"\A#[a-fA-F0-9]{8}\Z", c) 

209 if match: 

210 color = [int(n, 16) / 255 

211 for n in [c[1:3], c[3:5], c[5:7], c[7:9]]] 

212 if alpha is not None: 

213 color[-1] = alpha 

214 return tuple(color) 

215 # string gray. 

216 try: 

217 return (float(c),) * 3 + (alpha if alpha is not None else 1.,) 

218 except ValueError: 

219 pass 

220 raise ValueError("Invalid RGBA argument: {!r}".format(orig_c)) 

221 # tuple color. 

222 c = np.array(c) 

223 if not np.can_cast(c.dtype, float, "same_kind") or c.ndim != 1: 

224 # Test the dtype explicitly as `map(float, ...)`, `np.array(..., 

225 # float)` and `np.array(...).astype(float)` all convert "0.5" to 0.5. 

226 # Test dimensionality to reject single floats. 

227 raise ValueError("Invalid RGBA argument: {!r}".format(orig_c)) 

228 # Return a tuple to prevent the cached value from being modified. 

229 c = tuple(c.astype(float)) 

230 if len(c) not in [3, 4]: 

231 raise ValueError("RGBA sequence should have length 3 or 4") 

232 if len(c) == 3 and alpha is None: 

233 alpha = 1 

234 if alpha is not None: 

235 c = c[:3] + (alpha,) 

236 if any(elem < 0 or elem > 1 for elem in c): 

237 raise ValueError("RGBA values should be within 0-1 range") 

238 return c

241def to_rgba_array(c, alpha=None): 

242 """Convert *c* to a (n, 4) array of RGBA colors. 

243 

244 If *alpha* is not ``None``, it forces the alpha value. If *c* is 

245 ``"none"`` (case-insensitive) or an empty list, an empty array is returned. 

246 """ 

247 # Special-case inputs that are already arrays, for performance. (If the 

248 # array has the wrong kind or shape, raise the error during one-at-a-time 

249 # conversion.) 

250 if (isinstance(c, np.ndarray) and c.dtype.kind in "if" 

251 and c.ndim == 2 and c.shape[1] in [3, 4]): 

252 if c.shape[1] == 3: 

253 result = np.column_stack([c, np.zeros(len(c))]) 

254 result[:, -1] = alpha if alpha is not None else 1. 

255 elif c.shape[1] == 4: 

256 result = c.copy() 

257 if alpha is not None: 

258 result[:, -1] = alpha 

259 if np.any((result < 0) | (result > 1)): 

260 raise ValueError("RGBA values should be within 0-1 range") 

261 return result 

262 # Handle single values. 

263 # Note that this occurs *after* handling inputs that are already arrays, as 

264 # `to_rgba(c, alpha)` (below) is expensive for such inputs, due to the need 

265 # to format the array in the ValueError message(!). 

266 if cbook._str_lower_equal(c, "none"): 

267 return np.zeros((0, 4), float) 

268 try: 

269 return np.array([to_rgba(c, alpha)], float) 

270 except (ValueError, TypeError): 

271 pass 

272 # Convert one at a time. 

273 result = np.empty((len(c), 4), float) 

274 for i, cc in enumerate(c): 

275 result[i] = to_rgba(cc, alpha) 

276 return result

370def makeMappingArray(N, data, gamma=1.0): 

371 """Create an *N* -element 1-d lookup table 

372 

373 *data* represented by a list of x,y0,y1 mapping correspondences. 

374 Each element in this list represents how a value between 0 and 1 

375 (inclusive) represented by x is mapped to a corresponding value 

376 between 0 and 1 (inclusive). The two values of y are to allow 

377 for discontinuous mapping functions (say as might be found in a 

378 sawtooth) where y0 represents the value of y for values of x 

379 <= to that given, and y1 is the value to be used for x > than 

380 that given). The list must start with x=0, end with x=1, and 

381 all values of x must be in increasing order. Values between 

382 the given mapping points are determined by simple linear interpolation. 

383 

384 Alternatively, data can be a function mapping values between 0 - 1 

385 to 0 - 1. 

386 

387 The function returns an array "result" where ``result[x*(N-1)]`` 

388 gives the closest value for values of x between 0 and 1. 

389 """ 

390 

391 if callable(data): 

392 xind = np.linspace(0, 1, N) ** gamma 

393 lut = np.clip(np.array(data(xind), dtype=float), 0, 1) 

394 return lut 

395 

396 try: 

397 adata = np.array(data) 

398 except Exception: 

399 raise TypeError("data must be convertible to an array") 

400 shape = adata.shape 

401 if len(shape) != 2 or shape[1] != 3: 

402 raise ValueError("data must be nx3 format") 

403 

404 x = adata[:, 0] 

405 y0 = adata[:, 1] 

406 y1 = adata[:, 2] 

407 

408 if x[0] != 0. or x[-1] != 1.0: 

409 raise ValueError( 

410 "data mapping points must start with x=0 and end with x=1") 

411 if (np.diff(x) < 0).any(): 

412 raise ValueError("data mapping points must have x in increasing order") 

413 # begin generation of lookup table 

414 x = x * (N - 1) 

415 lut = np.zeros((N,), float) 

416 xind = (N - 1) * np.linspace(0, 1, N) ** gamma 

417 ind = np.searchsorted(x, xind)[1:-1] 

418 

419 distance = (xind[1:-1] - x[ind - 1]) / (x[ind] - x[ind - 1]) 

420 lut[1:-1] = distance * (y0[ind] - y1[ind - 1]) + y1[ind - 1] 

421 lut[0] = y1[0] 

422 lut[-1] = y0[-1] 

423 # ensure that the lut is confined to values between 0 and 1 by clipping it 

424 return np.clip(lut, 0.0, 1.0)

427class Colormap(object): 

428 """ 

429 Baseclass for all scalar to RGBA mappings. 

430 

431 Typically Colormap instances are used to convert data values (floats) from 

432 the interval ``[0, 1]`` to the RGBA color that the respective Colormap 

433 represents. For scaling of data into the ``[0, 1]`` interval see 

434 :class:`matplotlib.colors.Normalize`. It is worth noting that 

435 :class:`matplotlib.cm.ScalarMappable` subclasses make heavy use of this 

436 ``data->normalize->map-to-color`` processing chain. 

437 

438 """ 

439 def __init__(self, name, N=256): 

440 """ 

441 Parameters 

442 ---------- 

443 name : str 

444 The name of the colormap. 

445 N : int 

446 The number of rgb quantization levels. 

447 

448 """ 

449 self.name = name 

450 self.N = int(N) # ensure that N is always int 

451 self._rgba_bad = (0.0, 0.0, 0.0, 0.0) # If bad, don't paint anything. 

452 self._rgba_under = None 

453 self._rgba_over = None 

454 self._i_under = self.N 

455 self._i_over = self.N + 1 

456 self._i_bad = self.N + 2 

457 self._isinit = False 

458 

459 #: When this colormap exists on a scalar mappable and colorbar_extend 

460 #: is not False, colorbar creation will pick up ``colorbar_extend`` as 

461 #: the default value for the ``extend`` keyword in the 

462 #: :class:`matplotlib.colorbar.Colorbar` constructor. 

463 self.colorbar_extend = False

465 def __call__(self, X, alpha=None, bytes=False): 

466 """ 

467 Parameters 

468 ---------- 

469 X : scalar, ndarray 

470 The data value(s) to convert to RGBA. 

471 For floats, X should be in the interval ``[0.0, 1.0]`` to 

472 return the RGBA values ``X*100`` percent along the Colormap line. 

473 For integers, X should be in the interval ``[0, Colormap.N)`` to 

474 return RGBA values *indexed* from the Colormap with index ``X``. 

475 alpha : float, None 

476 Alpha must be a scalar between 0 and 1, or None. 

477 bytes : bool 

478 If False (default), the returned RGBA values will be floats in the 

479 interval ``[0, 1]`` otherwise they will be uint8s in the interval 

480 ``[0, 255]``. 

481 

482 Returns 

483 ------- 

484 Tuple of RGBA values if X is scalar, otherwise an array of 

485 RGBA values with a shape of ``X.shape + (4, )``. 

486 

487 """ 

488 # See class docstring for arg/kwarg documentation. 

489 if not self._isinit: 

490 self._init() 

491 mask_bad = None 

492 if not cbook.iterable(X): 

493 vtype = 'scalar' 

494 xa = np.array([X]) 

495 else: 

496 vtype = 'array' 

497 xma = np.ma.array(X, copy=True) # Copy here to avoid side effects. 

498 mask_bad = xma.mask # Mask will be used below. 

499 xa = xma.filled() # Fill to avoid infs, etc. 

500 del xma 

501 

502 # Calculations with native byteorder are faster, and avoid a 

503 # bug that otherwise can occur with putmask when the last 

504 # argument is a numpy scalar. 

505 if not xa.dtype.isnative: 

506 xa = xa.byteswap().newbyteorder() 

507 

508 if xa.dtype.kind == "f": 

509 xa *= self.N 

510 # Negative values are out of range, but astype(int) would truncate 

511 # them towards zero. 

512 xa[xa < 0] = -1 

513 # xa == 1 (== N after multiplication) is not out of range. 

514 xa[xa == self.N] = self.N - 1 

515 # Avoid converting large positive values to negative integers. 

516 np.clip(xa, -1, self.N, out=xa) 

517 xa = xa.astype(int) 

518 # Set the over-range indices before the under-range; 

519 # otherwise the under-range values get converted to over-range. 

520 xa[xa > self.N - 1] = self._i_over 

521 xa[xa < 0] = self._i_under 

522 if mask_bad is not None: 

523 if mask_bad.shape == xa.shape: 

524 np.copyto(xa, self._i_bad, where=mask_bad) 

525 elif mask_bad: 

526 xa.fill(self._i_bad) 

527 if bytes: 

528 lut = (self._lut * 255).astype(np.uint8) 

529 else: 

530 lut = self._lut.copy() # Don't let alpha modify original _lut. 

531 

532 if alpha is not None: 

533 alpha = np.clip(alpha, 0, 1) 

534 if bytes: 

535 alpha = int(alpha * 255) 

536 if (lut[-1] == 0).all(): 

537 lut[:-1, -1] = alpha 

538 # All zeros is taken as a flag for the default bad 

539 # color, which is no color--fully transparent. We 

540 # don't want to override this. 

541 else: 

542 lut[:, -1] = alpha 

543 # If the bad value is set to have a color, then we 

544 # override its alpha just as for any other value. 

545 

546 rgba = np.empty(shape=xa.shape + (4,), dtype=lut.dtype) 

547 lut.take(xa, axis=0, mode='clip', out=rgba) 

548 if vtype == 'scalar': 

549 rgba = tuple(rgba[0, :]) 

550 return rgba

632class LinearSegmentedColormap(Colormap): 

633 """Colormap objects based on lookup tables using linear segments. 

634 

635 The lookup table is generated using linear interpolation for each 

636 primary color, with the 0-1 domain divided into any number of 

637 segments. 

638 """ 

639 def __init__(self, name, segmentdata, N=256, gamma=1.0): 

640 """Create color map from linear mapping segments 

641 

642 segmentdata argument is a dictionary with a red, green and blue 

643 entries. Each entry should be a list of *x*, *y0*, *y1* tuples, 

644 forming rows in a table. Entries for alpha are optional. 

645 

646 Example: suppose you want red to increase from 0 to 1 over 

647 the bottom half, green to do the same over the middle half, 

648 and blue over the top half. Then you would use:: 

649 

650 cdict = {'red': [(0.0, 0.0, 0.0), 

651 (0.5, 1.0, 1.0), 

652 (1.0, 1.0, 1.0)], 

653 

654 'green': [(0.0, 0.0, 0.0), 

655 (0.25, 0.0, 0.0), 

656 (0.75, 1.0, 1.0), 

657 (1.0, 1.0, 1.0)], 

658 

659 'blue': [(0.0, 0.0, 0.0), 

660 (0.5, 0.0, 0.0), 

661 (1.0, 1.0, 1.0)]} 

662 

663 Each row in the table for a given color is a sequence of 

664 *x*, *y0*, *y1* tuples. In each sequence, *x* must increase 

665 monotonically from 0 to 1. For any input value *z* falling 

666 between *x[i]* and *x[i+1]*, the output value of a given color 

667 will be linearly interpolated between *y1[i]* and *y0[i+1]*:: 

668 

669 row i: x y0 y1 

670 / 

671 / 

672 row i+1: x y0 y1 

673 

674 Hence y0 in the first row and y1 in the last row are never used. 

675 

676 

677 .. seealso:: 

678 

679 :meth:`LinearSegmentedColormap.from_list` 

680 Static method; factory function for generating a 

681 smoothly-varying LinearSegmentedColormap. 

682 

683 :func:`makeMappingArray` 

684 For information about making a mapping array. 

685 """ 

686 # True only if all colors in map are identical; needed for contouring. 

687 self.monochrome = False 

688 Colormap.__init__(self, name, N) 

689 self._segmentdata = segmentdata 

690 self._gamma = gamma 

691 

692 def _init(self): 

693 self._lut = np.ones((self.N + 3, 4), float) 

694 self._lut[:-3, 0] = makeMappingArray( 

695 self.N, self._segmentdata['red'], self._gamma) 

696 self._lut[:-3, 1] = makeMappingArray( 

697 self.N, self._segmentdata['green'], self._gamma) 

698 self._lut[:-3, 2] = makeMappingArray( 

699 self.N, self._segmentdata['blue'], self._gamma) 

700 if 'alpha' in self._segmentdata: 

701 self._lut[:-3, 3] = makeMappingArray( 

702 self.N, self._segmentdata['alpha'], 1) 

703 self._isinit = True 

704 self._set_extremes() 

705 

706 def set_gamma(self, gamma): 

707 """ 

708 Set a new gamma value and regenerate color map. 

709 """ 

710 self._gamma = gamma 

711 self._init()

870class Normalize(object): 

871 """ 

872 A class which, when called, can normalize data into 

873 the ``[0.0, 1.0]`` interval. 

874 

875 """ 

876 def __init__(self, vmin=None, vmax=None, clip=False): 

877 """ 

878 If *vmin* or *vmax* is not given, they are initialized from the 

879 minimum and maximum value respectively of the first input 

880 processed. That is, *__call__(A)* calls *autoscale_None(A)*. 

881 If *clip* is *True* and the given value falls outside the range, 

882 the returned value will be 0 or 1, whichever is closer. 

883 Returns 0 if:: 

884 

885 vmin==vmax 

886 

887 Works with scalars or arrays, including masked arrays. If 

888 *clip* is *True*, masked values are set to 1; otherwise they 

889 remain masked. Clipping silently defeats the purpose of setting 

890 the over, under, and masked colors in the colormap, so it is 

891 likely to lead to surprises; therefore the default is 

892 *clip* = *False*. 

893 """ 

894 self.vmin = _sanitize_extrema(vmin) 

895 self.vmax = _sanitize_extrema(vmax) 

896 self.clip = clip 

897 

898 @staticmethod 

899 def process_value(value): 

900 """ 

901 Homogenize the input *value* for easy and efficient normalization. 

902 

903 *value* can be a scalar or sequence. 

904 

905 Returns *result*, *is_scalar*, where *result* is a 

906 masked array matching *value*. Float dtypes are preserved; 

907 integer types with two bytes or smaller are converted to 

908 np.float32, and larger types are converted to np.float64. 

909 Preserving float32 when possible, and using in-place operations, 

910 can greatly improve speed for large arrays. 

911 

912 Experimental; we may want to add an option to force the 

913 use of float32. 

914 """ 

915 is_scalar = not cbook.iterable(value) 

916 if is_scalar: 

917 value = [value] 

918 dtype = np.min_scalar_type(value) 

919 if np.issubdtype(dtype, np.integer) or dtype.type is np.bool_: 

920 # bool_/int8/int16 -> float32; int32/int64 -> float64 

921 dtype = np.promote_types(dtype, np.float32) 

922 # ensure data passed in as an ndarray subclass are interpreted as 

923 # an ndarray. See issue #6622. 

924 mask = np.ma.getmask(value) 

925 data = np.asarray(np.ma.getdata(value)) 

926 result = np.ma.array(data, mask=mask, dtype=dtype, copy=True) 

927 return result, is_scalar 

928 

929 def __call__(self, value, clip=None): 

930 """ 

931 Normalize *value* data in the ``[vmin, vmax]`` interval into 

932 the ``[0.0, 1.0]`` interval and return it. *clip* defaults 

933 to *self.clip* (which defaults to *False*). If not already 

934 initialized, *vmin* and *vmax* are initialized using 

935 *autoscale_None(value)*. 

936 """ 

937 if clip is None: 

938 clip = self.clip 

939 

940 result, is_scalar = self.process_value(value) 

941 

942 self.autoscale_None(result) 

943 # Convert at least to float, without losing precision. 

944 (vmin,), _ = self.process_value(self.vmin) 

945 (vmax,), _ = self.process_value(self.vmax) 

946 if vmin == vmax: 

947 result.fill(0) # Or should it be all masked? Or 0.5? 

948 elif vmin > vmax: 

949 raise ValueError("minvalue must be less than or equal to maxvalue") 

950 else: 

951 if clip: 

952 mask = np.ma.getmask(result) 

953 result = np.ma.array(np.clip(result.filled(vmax), vmin, vmax), 

954 mask=mask) 

955 # ma division is very slow; we can take a shortcut 

956 resdat = result.data 

957 resdat -= vmin 

958 resdat /= (vmax - vmin) 

959 result = np.ma.array(resdat, mask=result.mask, copy=False) 

960 if is_scalar: 

961 result = result[0] 

962 return result 

963 

964 def inverse(self, value): 

965 if not self.scaled(): 

966 raise ValueError("Not invertible until scaled") 

967 (vmin,), _ = self.process_value(self.vmin) 

968 (vmax,), _ = self.process_value(self.vmax) 

969 

970 if cbook.iterable(value): 

971 val = np.ma.asarray(value) 

972 return vmin + val * (vmax - vmin) 

973 else: 

974 return vmin + value * (vmax - vmin)

976 def autoscale(self, A): 

977 """ 

978 Set *vmin*, *vmax* to min, max of *A*. 

979 """ 

980 A = np.asanyarray(A) 

981 self.vmin = A.min() 

982 self.vmax = A.max()

984 def autoscale_None(self, A): 

985 """autoscale only None-valued vmin or vmax.""" 

986 A = np.asanyarray(A) 

987 if self.vmin is None and A.size: 

988 self.vmin = A.min() 

989 if self.vmax is None and A.size: 

990 self.vmax = A.max()

992 def scaled(self): 

993 'return true if vmin and vmax set' 

994 return (self.vmin is not None and self.vmax is not None)

1068class SymLogNorm(Normalize): 

1069 """ 

1070 The symmetrical logarithmic scale is logarithmic in both the 

1071 positive and negative directions from the origin. 

1072 

1073 Since the values close to zero tend toward infinity, there is a 

1074 need to have a range around zero that is linear. The parameter 

1075 *linthresh* allows the user to specify the size of this range 

1076 (-*linthresh*, *linthresh*). 

1077 """ 

1078 def __init__(self, linthresh, linscale=1.0, 

1079 vmin=None, vmax=None, clip=False): 

1080 """ 

1081 *linthresh*: 

1082 The range within which the plot is linear (to 

1083 avoid having the plot go to infinity around zero). 

1084 

1085 *linscale*: 

1086 This allows the linear range (-*linthresh* to *linthresh*) 

1087 to be stretched relative to the logarithmic range. Its 

1088 value is the number of decades to use for each half of the 

1089 linear range. For example, when *linscale* == 1.0 (the 

1090 default), the space used for the positive and negative 

1091 halves of the linear range will be equal to one decade in 

1092 the logarithmic range. Defaults to 1. 

1093 """ 

1094 Normalize.__init__(self, vmin, vmax, clip) 

1095 self.linthresh = float(linthresh) 

1096 self._linscale_adj = (linscale / (1.0 - np.e ** -1)) 

1097 if vmin is not None and vmax is not None: 

1098 self._transform_vmin_vmax() 

1099 

1100 def __call__(self, value, clip=None): 

1101 if clip is None: 

1102 clip = self.clip 

1103 

1104 result, is_scalar = self.process_value(value) 

1105 self.autoscale_None(result) 

1106 vmin, vmax = self.vmin, self.vmax 

1107 

1108 if vmin > vmax: 

1109 raise ValueError("minvalue must be less than or equal to maxvalue") 

1110 elif vmin == vmax: 

1111 result.fill(0) 

1112 else: 

1113 if clip: 

1114 mask = np.ma.getmask(result) 

1115 result = np.ma.array(np.clip(result.filled(vmax), vmin, vmax), 

1116 mask=mask) 

1117 # in-place equivalent of above can be much faster 

1118 resdat = self._transform(result.data) 

1119 resdat -= self._lower 

1120 resdat /= (self._upper - self._lower) 

1121 

1122 if is_scalar: 

1123 result = result[0] 

1124 return result 

1125 

1126 def _transform(self, a): 

1127 """ 

1128 Inplace transformation. 

1129 """ 

1130 with np.errstate(invalid="ignore"): 

1131 masked = np.abs(a) > self.linthresh 

1132 sign = np.sign(a[masked]) 

1133 log = (self._linscale_adj + np.log(np.abs(a[masked]) / self.linthresh)) 

1134 log *= sign * self.linthresh 

1135 a[masked] = log 

1136 a[~masked] *= self._linscale_adj 

1137 return a 

1138 

1139 def _inv_transform(self, a): 

1140 """ 

1141 Inverse inplace Transformation. 

1142 """ 

1143 masked = np.abs(a) > (self.linthresh * self._linscale_adj) 

1144 sign = np.sign(a[masked]) 

1145 exp = np.exp(sign * a[masked] / self.linthresh - self._linscale_adj) 

1146 exp *= sign * self.linthresh 

1147 a[masked] = exp 

1148 a[~masked] /= self._linscale_adj 

1149 return a 

1150 

1151 def _transform_vmin_vmax(self): 

1152 """ 

1153 Calculates vmin and vmax in the transformed system. 

1154 """ 

1155 vmin, vmax = self.vmin, self.vmax 

1156 arr = np.array([vmax, vmin]).astype(float) 

1157 self._upper, self._lower = self._transform(arr) 

1158 

1159 def inverse(self, value): 

1160 if not self.scaled(): 

1161 raise ValueError("Not invertible until scaled") 

1162 val = np.ma.asarray(value) 

1163 val = val * (self._upper - self._lower) + self._lower 

1164 return self._inv_transform(val)

1167 """ 

1168 Set *vmin*, *vmax* to min, max of *A*. 

1169 """ 

1170 self.vmin = np.ma.min(A) 

1171 self.vmax = np.ma.max(A) 

1172 self._transform_vmin_vmax()

1177 pass

1252class BoundaryNorm(Normalize): 

1253 """ 

1254 Generate a colormap index based on discrete intervals. 

1255 

1256 Unlike :class:`Normalize` or :class:`LogNorm`, 

1257 :class:`BoundaryNorm` maps values to integers instead of to the 

1258 interval 0-1. 

1259 

1260 Mapping to the 0-1 interval could have been done via 

1261 piece-wise linear interpolation, but using integers seems 

1262 simpler, and reduces the number of conversions back and forth 

1263 between integer and floating point. 

1264 """ 

1265 def __init__(self, boundaries, ncolors, clip=False): 

1266 """ 

1267 Parameters 

1268 ---------- 

1269 boundaries : array-like 

1270 Monotonically increasing sequence of boundaries 

1271 ncolors : int 

1272 Number of colors in the colormap to be used 

1273 clip : bool, optional 

1274 If clip is ``True``, out of range values are mapped to 0 if they 

1275 are below ``boundaries[0]`` or mapped to ncolors - 1 if they are 

1276 above ``boundaries[-1]``. 

1277 

1278 If clip is ``False``, out of range values are mapped to -1 if 

1279 they are below ``boundaries[0]`` or mapped to ncolors if they are 

1280 above ``boundaries[-1]``. These are then converted to valid indices 

1281 by :meth:`Colormap.__call__`. 

1282 

1283 Notes 

1284 ----- 

1285 *boundaries* defines the edges of bins, and data falling within a bin 

1286 is mapped to the color with the same index. 

1287 

1288 If the number of bins doesn't equal *ncolors*, the color is chosen 

1289 by linear interpolation of the bin number onto color numbers. 

1290 """ 

1291 self.clip = clip 

1292 self.vmin = boundaries[0] 

1293 self.vmax = boundaries[-1] 

1294 self.boundaries = np.asarray(boundaries) 

1295 self.N = len(self.boundaries) 

1296 self.Ncmap = ncolors 

1297 if self.N - 1 == self.Ncmap: 

1298 self._interp = False 

1299 else: 

1300 self._interp = True 

1301 

1302 def __call__(self, value, clip=None): 

1303 if clip is None: 

1304 clip = self.clip 

1305 

1306 xx, is_scalar = self.process_value(value) 

1307 mask = np.ma.getmaskarray(xx) 

1308 xx = np.atleast_1d(xx.filled(self.vmax + 1)) 

1309 if clip: 

1310 np.clip(xx, self.vmin, self.vmax, out=xx) 

1311 max_col = self.Ncmap - 1 

1312 else: 

1313 max_col = self.Ncmap 

1314 iret = np.zeros(xx.shape, dtype=np.int16) 

1315 for i, b in enumerate(self.boundaries): 

1316 iret[xx >= b] = i 

1317 if self._interp: 

1318 scalefac = (self.Ncmap - 1) / (self.N - 2) 

1319 iret = (iret * scalefac).astype(np.int16) 

1320 iret[xx < self.vmin] = -1 

1321 iret[xx >= self.vmax] = max_col 

1322 ret = np.ma.array(iret, mask=mask) 

1323 if is_scalar: 

1324 ret = int(ret[0]) # assume python scalar 

1325 return ret

1351def rgb_to_hsv(arr): 

1352 """ 

1353 convert float rgb values (in the range [0, 1]), in a numpy array to hsv 

1354 values. 

1355 

1356 Parameters 

1357 ---------- 

1358 arr : (..., 3) array-like 

1359 All values must be in the range [0, 1] 

1360 

1361 Returns 

1362 ------- 

1363 hsv : (..., 3) ndarray 

1364 Colors converted to hsv values in range [0, 1] 

1365 """ 

1366 # make sure it is an ndarray 

1367 arr = np.asarray(arr) 

1368 

1369 # check length of the last dimension, should be _some_ sort of rgb 

1370 if arr.shape[-1] != 3: 

1371 raise ValueError("Last dimension of input array must be 3; " 

1372 "shape {} was found.".format(arr.shape)) 

1373 

1374 in_ndim = arr.ndim 

1375 if arr.ndim == 1: 

1376 arr = np.array(arr, ndmin=2) 

1377 

1378 # make sure we don't have an int image 

1379 arr = arr.astype(np.promote_types(arr.dtype, np.float32)) 

1380 

1381 out = np.zeros_like(arr) 

1382 arr_max = arr.max(-1) 

1383 ipos = arr_max > 0 

1384 delta = arr.ptp(-1) 

1385 s = np.zeros_like(delta) 

1386 s[ipos] = delta[ipos] / arr_max[ipos] 

1387 ipos = delta > 0 

1388 # red is max 

1389 idx = (arr[..., 0] == arr_max) & ipos 

1390 out[idx, 0] = (arr[idx, 1] - arr[idx, 2]) / delta[idx] 

1391 # green is max 

1392 idx = (arr[..., 1] == arr_max) & ipos 

1393 out[idx, 0] = 2. + (arr[idx, 2] - arr[idx, 0]) / delta[idx] 

1394 # blue is max 

1395 idx = (arr[..., 2] == arr_max) & ipos 

1396 out[idx, 0] = 4. + (arr[idx, 0] - arr[idx, 1]) / delta[idx] 

1397 

1398 out[..., 0] = (out[..., 0] / 6.0) % 1.0 

1399 out[..., 1] = s 

1400 out[..., 2] = arr_max 

1401 

1402 if in_ndim == 1: 

1403 out.shape = (3,) 

1404 

1405 return out

1408def hsv_to_rgb(hsv): 

1409 """ 

1410 convert hsv values in a numpy array to rgb values 

1411 all values assumed to be in range [0, 1] 

1412 

1413 Parameters 

1414 ---------- 

1415 hsv : (..., 3) array-like 

1416 All values assumed to be in range [0, 1] 

1417 

1418 Returns 

1419 ------- 

1420 rgb : (..., 3) ndarray 

1421 Colors converted to RGB values in range [0, 1] 

1422 """ 

1423 hsv = np.asarray(hsv) 

1424 

1425 # check length of the last dimension, should be _some_ sort of rgb 

1426 if hsv.shape[-1] != 3: 

1427 raise ValueError("Last dimension of input array must be 3; " 

1428 "shape {shp} was found.".format(shp=hsv.shape)) 

1429 

1430 # if we got passed a 1D array, try to treat as 

1431 # a single color and reshape as needed 

1432 in_ndim = hsv.ndim 

1433 if in_ndim == 1: 

1434 hsv = np.array(hsv, ndmin=2) 

1435 

1436 # make sure we don't have an int image 

1437 hsv = hsv.astype(np.promote_types(hsv.dtype, np.float32)) 

1438 

1439 h = hsv[..., 0] 

1440 s = hsv[..., 1] 

1441 v = hsv[..., 2] 

1442 

1443 r = np.empty_like(h) 

1444 g = np.empty_like(h) 

1445 b = np.empty_like(h) 

1446 

1447 i = (h * 6.0).astype(int) 

1448 f = (h * 6.0) - i 

1449 p = v * (1.0 - s) 

1450 q = v * (1.0 - s * f) 

1451 t = v * (1.0 - s * (1.0 - f)) 

1452 

1453 idx = i % 6 == 0 

1454 r[idx] = v[idx] 

1455 g[idx] = t[idx] 

1456 b[idx] = p[idx] 

1457 

1458 idx = i == 1 

1459 r[idx] = q[idx] 

1460 g[idx] = v[idx] 

1461 b[idx] = p[idx] 

1462 

1463 idx = i == 2 

1464 r[idx] = p[idx] 

1465 g[idx] = v[idx] 

1466 b[idx] = t[idx] 

1467 

1468 idx = i == 3 

1469 r[idx] = p[idx] 

1470 g[idx] = q[idx] 

1471 b[idx] = v[idx] 

1472 

1473 idx = i == 4 

1474 r[idx] = t[idx] 

1475 g[idx] = p[idx] 

1476 b[idx] = v[idx] 

1477 

1478 idx = i == 5 

1479 r[idx] = v[idx] 

1480 g[idx] = p[idx] 

1481 b[idx] = q[idx] 

1482 

1483 idx = s == 0 

1484 r[idx] = v[idx] 

1485 g[idx] = v[idx] 

1486 b[idx] = v[idx] 

1487 

1488 rgb = np.stack([r, g, b], axis=-1) 

1489 

1490 if in_ndim == 1: 

1491 rgb.shape = (3,) 

1492 

1493 return rgb

1567 def hillshade(self, elevation, vert_exag=1, dx=1, dy=1, fraction=1.): 

1568 """ 

1569 Calculates the illumination intensity for a surface using the defined 

1570 azimuth and elevation for the light source. 

1571 

1572 This computes the normal vectors for the surface, and then passes them 

1573 on to `shade_normals` 

1574 

1575 Parameters 

1576 ---------- 

1577 elevation : array-like 

1578 A 2d array (or equivalent) of the height values used to generate an 

1579 illumination map 

1580 vert_exag : number, optional 

1581 The amount to exaggerate the elevation values by when calculating 

1582 illumination. This can be used either to correct for differences in 

1583 units between the x-y coordinate system and the elevation 

1584 coordinate system (e.g. decimal degrees vs meters) or to exaggerate 

1585 or de-emphasize topographic effects. 

1586 dx : number, optional 

1587 The x-spacing (columns) of the input *elevation* grid. 

1588 dy : number, optional 

1589 The y-spacing (rows) of the input *elevation* grid. 

1590 fraction : number, optional 

1591 Increases or decreases the contrast of the hillshade. Values 

1592 greater than one will cause intermediate values to move closer to 

1593 full illumination or shadow (and clipping any values that move 

1594 beyond 0 or 1). Note that this is not visually or mathematically 

1595 the same as vertical exaggeration. 

1596 Returns 

1597 ------- 

1598 intensity : ndarray 

1599 A 2d array of illumination values between 0-1, where 0 is 

1600 completely in shadow and 1 is completely illuminated. 

1601 """ 

1602 

1603 # Because most image and raster GIS data has the first row in the array 

1604 # as the "top" of the image, dy is implicitly negative. This is 

1605 # consistent to what `imshow` assumes, as well. 

1606 dy = -dy 

1607 

1608 # compute the normal vectors from the partial derivatives 

1609 e_dy, e_dx = np.gradient(vert_exag * elevation, dy, dx) 

1610 

1611 # .view is to keep subclasses 

1612 normal = np.empty(elevation.shape + (3,)).view(type(elevation)) 

1613 normal[..., 0] = -e_dx 

1614 normal[..., 1] = -e_dy 

1615 normal[..., 2] = 1 

1616 normal /= _vector_magnitude(normal) 

1617 

1618 return self.shade_normals(normal, fraction)

1620 def shade_normals(self, normals, fraction=1.): 

1621 """ 

1622 Calculates the illumination intensity for the normal vectors of a 

1623 surface using the defined azimuth and elevation for the light source. 

1624 

1625 Imagine an artificial sun placed at infinity in some azimuth and 

1626 elevation position illuminating our surface. The parts of the surface 

1627 that slope toward the sun should brighten while those sides facing away 

1628 should become darker. 

1629 

1630 Parameters 

1631 ---------- 

1632 fraction : number, optional 

1633 Increases or decreases the contrast of the hillshade. Values 

1634 greater than one will cause intermediate values to move closer to 

1635 full illumination or shadow (and clipping any values that move 

1636 beyond 0 or 1). Note that this is not visually or mathematically 

1637 the same as vertical exaggeration. 

1638 

1639 Returns 

1640 ------- 

1641 intensity : ndarray 

1642 A 2d array of illumination values between 0-1, where 0 is 

1643 completely in shadow and 1 is completely illuminated. 

1644 """ 

1645 

1646 intensity = normals.dot(self.direction) 

1647 

1648 # Apply contrast stretch 

1649 imin, imax = intensity.min(), intensity.max() 

1650 intensity *= fraction 

1651 

1652 # Rescale to 0-1, keeping range before contrast stretch 

1653 # If constant slope, keep relative scaling (i.e. flat should be 0.5, 

1654 # fully occluded 0, etc.) 

1655 if (imax - imin) > 1e-6: 

1656 # Strictly speaking, this is incorrect. Negative values should be 

1657 # clipped to 0 because they're fully occluded. However, rescaling 

1658 # in this manner is consistent with the previous implementation and 

1659 # visually appears better than a "hard" clip. 

1660 intensity -= imin 

1661 intensity /= (imax - imin) 

1662 intensity = np.clip(intensity, 0, 1, intensity) 

1663 

1664 return intensity

1775 illumination. This can be used either to correct for differences in

1817 def blend_hsv(self, rgb, intensity, hsv_max_sat=None, hsv_max_val=None, 

1818 hsv_min_val=None, hsv_min_sat=None): 

1819 """ 

1820 Take the input data array, convert to HSV values in the given colormap, 

1821 then adjust those color values to give the impression of a shaded 

1822 relief map with a specified light source. RGBA values are returned, 

1823 which can then be used to plot the shaded image with imshow. 

1824 

1825 The color of the resulting image will be darkened by moving the (s,v) 

1826 values (in hsv colorspace) toward (hsv_min_sat, hsv_min_val) in the 

1827 shaded regions, or lightened by sliding (s,v) toward (hsv_max_sat 

1828 hsv_max_val) in regions that are illuminated. The default extremes are 

1829 chose so that completely shaded points are nearly black (s = 1, v = 0) 

1830 and completely illuminated points are nearly white (s = 0, v = 1). 

1831 

1832 Parameters 

1833 ---------- 

1834 rgb : ndarray 

1835 An MxNx3 RGB array of floats ranging from 0 to 1 (color image). 

1836 intensity : ndarray 

1837 An MxNx1 array of floats ranging from 0 to 1 (grayscale image). 

1838 hsv_max_sat : number, optional 

1839 The maximum saturation value that the *intensity* map can shift the 

1840 output image to. Defaults to 1. 

1841 hsv_min_sat : number, optional 

1842 The minimum saturation value that the *intensity* map can shift the 

1843 output image to. Defaults to 0. 

1844 hsv_max_val : number, optional 

1845 The maximum value ("v" in "hsv") that the *intensity* map can shift 

1846 the output image to. Defaults to 1. 

1847 hsv_min_val: number, optional 

1848 The minimum value ("v" in "hsv") that the *intensity* map can shift 

1849 the output image to. Defaults to 0.

1854 An MxNx3 RGB array representing the combined images.

1857 if hsv_max_sat is None: 

1858 hsv_max_sat = self.hsv_max_sat 

1859 if hsv_max_val is None: 

1860 hsv_max_val = self.hsv_max_val 

1861 if hsv_min_sat is None: 

1862 hsv_min_sat = self.hsv_min_sat 

1863 if hsv_min_val is None: 

1864 hsv_min_val = self.hsv_min_val 

1865 

1866 # Expects a 2D intensity array scaled between -1 to 1... 

1867 intensity = intensity[..., 0] 

1868 intensity = 2 * intensity - 1 

1869 

1870 # convert to rgb, then rgb to hsv 

1871 hsv = rgb_to_hsv(rgb[:, :, 0:3]) 

1872 

1873 # modify hsv values to simulate illumination. 

1874 hsv[:, :, 1] = np.where(np.logical_and(np.abs(hsv[:, :, 1]) > 1.e-10, 

1875 intensity > 0), 

1876 ((1. - intensity) * hsv[:, :, 1] + 

1877 intensity * hsv_max_sat), 

1878 hsv[:, :, 1]) 

1879 

1880 hsv[:, :, 2] = np.where(intensity > 0, 

1881 ((1. - intensity) * hsv[:, :, 2] + 

1882 intensity * hsv_max_val), 

1883 hsv[:, :, 2]) 

1884 

1885 hsv[:, :, 1] = np.where(np.logical_and(np.abs(hsv[:, :, 1]) > 1.e-10, 

1886 intensity < 0), 

1887 ((1. + intensity) * hsv[:, :, 1] - 

1888 intensity * hsv_min_sat), 

1889 hsv[:, :, 1]) 

1890 hsv[:, :, 2] = np.where(intensity < 0, 

1891 ((1. + intensity) * hsv[:, :, 2] - 

1892 intensity * hsv_min_val), 

1893 hsv[:, :, 2])

1895 hsv[:, :, 1:] = np.where(hsv[:, :, 1:] > 1., 1, hsv[:, :, 1:])

1899 def blend_soft_light(self, rgb, intensity): 

1900 """ 

1901 Combines an rgb image with an intensity map using "soft light" 

1902 blending. Uses the "pegtop" formula. 

1903 

1904 Parameters 

1905 ---------- 

1906 rgb : ndarray 

1907 An MxNx3 RGB array of floats ranging from 0 to 1 (color image).

1910 

1911 Returns

1916 return 2 * intensity * rgb + (1 - 2 * intensity) * rgb**2

1918 def blend_overlay(self, rgb, intensity): 

1919 """ 

1920 Combines an rgb image with an intensity map using "overlay" blending. 

1921 

1922 Parameters 

1923 ---------- 

1924 rgb : ndarray 

1925 An MxNx3 RGB array of floats ranging from 0 to 1 (color image).

1940 """

1942 behave similar to contourf's levels and colors arguments.

1961 :class:`Normalize` instance

1964 colors_i1 = None

1967 colors_i0 = 1 

1968 colors_i1 = -1 

1969 extra_colors = 2 

1970 elif extend == 'min': 

1971 colors_i0 = 1 

1972 extra_colors = 1

1975 extra_colors = 1

1977 extra_colors = 0

1998 else:
