371 def _parse_thead_tbody_tfoot(self, table_html): 

372 """ 

373 Given a table, return parsed header, body, and foot. 

374 

375 Parameters 

376 ---------- 

377 table_html : node-like 

378 

379 Returns 

380 ------- 

381 tuple of (header, body, footer), each a list of list-of-text rows. 

382 

383 Notes 

384 ----- 

385 Header and body are lists-of-lists. Top level list is a list of 

386 rows. Each row is a list of str text. 

387 

388 Logic: Use <thead>, <tbody>, <tfoot> elements to identify 

389 header, body, and footer, otherwise: 

390 - Put all rows into body 

391 - Move rows from top of body to header only if 

392 all elements inside row are <th> 

393 - Move rows from bottom of body to footer only if 

394 all elements inside row are <th> 

395 """ 

396 

397 header_rows = self._parse_thead_tr(table_html) 

398 body_rows = self._parse_tbody_tr(table_html) 

399 footer_rows = self._parse_tfoot_tr(table_html) 

400 

401 def row_is_all_th(row): 

402 return all(self._equals_tag(t, "th") for t in self._parse_td(row)) 

403 

404 if not header_rows: 

405 # The table has no <thead>. Move the top all-<th> rows from 

406 # body_rows to header_rows. (This is a common case because many 

407 # tables in the wild have no <thead> or <tfoot> 

408 while body_rows and row_is_all_th(body_rows[0]): 

409 header_rows.append(body_rows.pop(0)) 

410 

411 header = self._expand_colspan_rowspan(header_rows) 

412 body = self._expand_colspan_rowspan(body_rows) 

413 footer = self._expand_colspan_rowspan(footer_rows) 

414 

415 return header, body, footer

417 def _expand_colspan_rowspan(self, rows): 

418 """ 

419 Given a list of <tr>s, return a list of text rows. 

420 

421 Parameters 

422 ---------- 

423 rows : list of node-like 

424 List of <tr>s 

425 

426 Returns 

427 ------- 

428 list of list 

429 Each returned row is a list of str text. 

430 

431 Notes 

432 ----- 

433 Any cell with ``rowspan`` or ``colspan`` will have its contents copied 

434 to subsequent cells. 

435 """ 

436 

437 all_texts = [] # list of rows, each a list of str 

438 remainder = [] # list of (index, text, nrows) 

439 

440 for tr in rows: 

441 texts = [] # the output for this row 

442 next_remainder = [] 

443 

444 index = 0 

445 tds = self._parse_td(tr) 

446 for td in tds: 

447 # Append texts from previous rows with rowspan>1 that come 

448 # before this <td> 

449 while remainder and remainder[0][0] <= index: 

450 prev_i, prev_text, prev_rowspan = remainder.pop(0) 

451 texts.append(prev_text) 

452 if prev_rowspan > 1: 

453 next_remainder.append((prev_i, prev_text, prev_rowspan - 1)) 

454 index += 1 

455 

456 # Append the text from this <td>, colspan times 

457 text = _remove_whitespace(self._text_getter(td)) 

458 rowspan = int(self._attr_getter(td, "rowspan") or 1) 

459 colspan = int(self._attr_getter(td, "colspan") or 1) 

460 

461 for _ in range(colspan): 

462 texts.append(text) 

463 if rowspan > 1: 

464 next_remainder.append((index, text, rowspan - 1)) 

465 index += 1 

466 

467 # Append texts from previous rows at the final position 

468 for prev_i, prev_text, prev_rowspan in remainder: 

469 texts.append(prev_text) 

470 if prev_rowspan > 1: 

471 next_remainder.append((prev_i, prev_text, prev_rowspan - 1)) 

472 

473 all_texts.append(texts) 

474 remainder = next_remainder 

475 

476 # Append rows that only appear because the previous row had non-1 

477 # rowspan 

478 while remainder: 

479 next_remainder = [] 

480 texts = [] 

481 for prev_i, prev_text, prev_rowspan in remainder: 

482 texts.append(prev_text) 

483 if prev_rowspan > 1: 

484 next_remainder.append((prev_i, prev_text, prev_rowspan - 1)) 

485 all_texts.append(texts) 

486 remainder = next_remainder 

487 

488 return all_texts

657 # 2. go up the tree until we find a table

806def _parser_dispatch(flavor): 

807 """Choose the parser based on the input flavor. 

808 

809 Parameters 

810 ---------- 

811 flavor : str 

812 The type of parser to use. This must be a valid backend. 

813 

814 Returns 

815 ------- 

816 cls : _HtmlFrameParser subclass 

817 The parser class based on the requested input flavor. 

818 

819 Raises 

820 ------ 

821 ValueError 

822 * If `flavor` is not a valid backend. 

823 ImportError 

824 * If you do not have the requested `flavor` 

825 """ 

826 valid_parsers = list(_valid_parsers.keys()) 

827 if flavor not in valid_parsers: 

828 raise ValueError( 

829 "{invalid!r} is not a valid flavor, valid flavors " 

830 "are {valid}".format(invalid=flavor, valid=valid_parsers) 

831 ) 

832 

833 if flavor in ("bs4", "html5lib"): 

834 if not _HAS_HTML5LIB: 

835 raise ImportError("html5lib not found, please install it") 

836 if not _HAS_BS4: 

837 raise ImportError("BeautifulSoup4 (bs4) not found, please install it") 

838 # Although we call this above, we want to raise here right before use. 

839 bs4 = import_optional_dependency("bs4") # noqa:F841 

840 

841 else: 

842 if not _HAS_LXML: 

843 raise ImportError("lxml not found, please install it") 

844 return _valid_parsers[flavor]

847def _print_as_set(s): 

848 return "{" + "{arg}".format(arg=", ".join(pprint_thing(el) for el in s)) + "}"

851def _validate_flavor(flavor): 

852 if flavor is None: 

853 flavor = "lxml", "bs4" 

854 elif isinstance(flavor, str): 

855 flavor = (flavor,) 

856 elif isinstance(flavor, abc.Iterable): 

857 if not all(isinstance(flav, str) for flav in flavor): 

858 raise TypeError( 

859 "Object of type {typ!r} is not an iterable of " 

860 "strings".format(typ=type(flavor).__name__) 

861 ) 

862 else: 

863 fmt = "{flavor!r}" if isinstance(flavor, str) else "{flavor}" 

864 fmt += " is not a valid flavor" 

865 raise ValueError(fmt.format(flavor=flavor)) 

866 

867 flavor = tuple(flavor) 

868 valid_flavors = set(_valid_parsers) 

869 flavor_set = set(flavor) 

870 

871 if not flavor_set & valid_flavors: 

872 raise ValueError( 

873 "{invalid} is not a valid set of flavors, valid " 

874 "flavors are {valid}".format( 

875 invalid=_print_as_set(flavor_set), valid=_print_as_set(valid_flavors) 

876 ) 

877 ) 

878 return flavor

881def _parse(flavor, io, match, attrs, encoding, displayed_only, **kwargs): 

882 flavor = _validate_flavor(flavor) 

883 compiled_match = re.compile(match) # you can pass a compiled regex here 

884 

885 # hack around python 3 deleting the exception variable 

886 retained = None 

887 for flav in flavor: 

888 parser = _parser_dispatch(flav) 

889 p = parser(io, compiled_match, attrs, encoding, displayed_only) 

890 

891 try: 

892 tables = p.parse_tables() 

893 except Exception as caught: 

894 # if `io` is an io-like object, check if it's seekable 

895 # and try to rewind it before trying the next parser 

896 if hasattr(io, "seekable") and io.seekable(): 

897 io.seek(0) 

898 elif hasattr(io, "seekable") and not io.seekable(): 

899 # if we couldn't rewind it, let the user know 

900 raise ValueError( 

901 "The flavor {} failed to parse your input. " 

902 "Since you passed a non-rewindable file " 

903 "object, we can't rewind it to try " 

904 "another parser. Try read_html() with a " 

905 "different flavor.".format(flav) 

906 ) 

907 

908 retained = caught 

909 else: 

910 break 

911 else: 

912 raise_with_traceback(retained) 

913 

914 ret = [] 

915 for table in tables: 

916 try: 

917 ret.append(_data_to_frame(data=table, **kwargs)) 

918 except EmptyDataError: # empty table 

919 continue 

920 return ret

941

943 ----------

948 

949 match : str or compiled regular expression, optional 

950 The set of tables containing text matching this regex or string will be 

951 returned. Unless the HTML is extremely simple you will probably need to 

952 pass a non-empty string here. Defaults to '.+' (match any non-empty 

953 string). The default value will return all tables contained on a page. 

954 This value is converted to a regular expression so that there is 

955 consistent behavior between Beautiful Soup and lxml. 

956 

957 flavor : str or None, container of strings 

958 The parsing engine to use. 'bs4' and 'html5lib' are synonymous with 

959 each other, they are both there for backwards compatibility. The

964 The row (or list of rows for a :class:`~pandas.MultiIndex`) to use to

966

972 sequence of integers or a slice is given, will skip the rows indexed by 

973 that sequence. Note that a single element sequence means 'skip the nth 

974 row' whereas an integer means 'skip n rows'.

979 passed to lxml or Beautiful Soup. However, these attributes must be 

980 valid HTML table attributes to work correctly. For example, :: 

981 

982 attrs = {'id': 'table'} 

983

985 a valid HTML attribute for *any* HTML tag as per `this document 

986 <http://www.w3.org/TR/html-markup/global-attributes.html>`__. :: 

987 

988 attrs = {'asdf': 'table'} 

989 

990 is *not* a valid attribute dictionary because 'asdf' is not a valid 

991 HTML attribute even if it is a valid XML attribute. Valid HTML 4.01 

992 table attributes can be found `here 

993 <http://www.w3.org/TR/REC-html40/struct/tables.html#h-11.2>`__. A 

994 working draft of the HTML 5 spec can be found `here 

995 <http://www.w3.org/TR/html-markup/table.html>`__. It contains the 

996 latest information on table attributes for the modern web.

998 parse_dates : bool, optional 

999 See :func:`~read_csv` for more details.

1003

1006 preserves the previous encoding behavior, which depends on the 

1007 underlying parser library (e.g., the parser library will try to use

1015

1018 either be integers or column labels, values are functions that take one 

1019 input argument, the cell (not column) content, and return the 

1020 transformed content. 

1021

1025 Custom NA values 

1026 

1027 .. versionadded:: 0.19.0 

1028

1033 .. versionadded:: 0.19.0

1036 Whether elements with "display: none" should be parsed

1043

1046 read_csv 

1047

1049 ----- 

1050 Before using this function you should read the :ref:`gotchas about the 

1051 HTML parsing libraries <io.html.gotchas>`. 

1052 

1053 Expect to do some cleanup after you call this function. For example, you 

1054 might need to manually assign column names if the column names are 

1055 converted to NaN when you pass the `header=0` argument. We try to assume as 

1056 little as possible about the structure of the table and push the
